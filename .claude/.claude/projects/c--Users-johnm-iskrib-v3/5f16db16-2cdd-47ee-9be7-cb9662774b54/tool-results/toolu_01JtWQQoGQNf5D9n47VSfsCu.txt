   246→const parseMediaTimestamp = (item = {}) => {
   247→    const fileName = item?.name || '';
   248→    const timestampPrefix = fileName.match(/^(\d+)_/);
   249→
   250→    if(timestampPrefix?.[1]){

   253→            return parsedPrefix;
   254→        }

   258→    return Number.isNaN(parsedDate) ? 0 : parsedDate;

   261→const createInitialMediaCursorState = () => ({
   262→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   263→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   264→    pending: []
   265→});
   266→
   267→const parseMediaCursor = (cursor) => {
   268→    if(!cursor){
   269→        return createInitialMediaCursorState();
   270→    }
   271→
   272→    try {
   273→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   274→        const parsed = JSON.parse(decoded);
   275→        const initial = createInitialMediaCursorState();
   276→
   277→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   278→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   279→                const value = Number(parsed.offsets[bucket]);
   280→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   281→            });
   282→        }
   283→
   284→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   285→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   286→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   287→            });
   288→        }
   289→
   290→        if(Array.isArray(parsed?.pending)){
   291→            initial.pending = parsed.pending
   292→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   293→                .map((item) => ({
   294→                    bucket: item.bucket,
   295→                    path: item.path,
   296→                    name: item.name,
   297→                    createdAt: item.createdAt || null,
   298→                    timestamp: Number(item.timestamp) || 0
   299→                }));
   300→        }
   301→
   302→        return initial;
   303→    } catch {
   304→        return createInitialMediaCursorState();
   305→    }
   306→};

   317→

   324→        .from(bucket)
   325→        .createSignedUrl(path, 60 * 60);
   326→
   327→    if(!signedUrlError && signedUrlData?.signedUrl){

   331→    return null;

   334→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   335→    const folderPath = `user_id_${userId}`;
   336→    const listLimit = Math.max(limit * 4, limit);
   337→
   338→    const {data: bucketItems, error: bucketError} = await supabase.storage
   339→        .from(bucket)
   340→        .list(folderPath, {
   341→            limit: listLimit,
   342→            offset: offset,
   343→            sortBy: {column: 'name', order: 'desc'}
   344→        });
   345→
   346→    if(bucketError){
   347→        return {
   348→            bucket: bucket,
   349→            nextOffset: offset,
   350→            exhausted: true,
   351→            items: [],
   352→            error: bucketError.message
   353→        };
   354→    }
   355→
   356→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   357→    const normalizedItems = listedItems
   358→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name) && isPrimaryListableFileName(item.name))
   359→        .map((item) => {
   360→            const path = `${folderPath}/${item.name}`;
   361→            const createdAt = item.created_at || item.updated_at || null;
   362→
   363→            return {
   364→                bucket: bucket,
   365→                path: path,
   366→                name: item.name,
   367→                createdAt: createdAt,
   368→                timestamp: parseMediaTimestamp(item)
   369→            };
   370→        });
   371→
   372→    return {
   373→        bucket: bucket,
   374→        nextOffset: offset + listedItems.length,
   375→        exhausted: listedItems.length < listLimit,
   376→        items: normalizedItems,
   377→        error: null
   378→    };
   379→};

   412→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   413→    if(!userId){
   414→        throw {status: 400, error: 'userId is undefined'};
   415→    }
   416→
   417→    const parsedLimit = Number(limit);
   418→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   419→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   420→    }
   421→
   422→    const cursorState = parseMediaCursor(cursor);
   423→    const offsets = {...cursorState.offsets};
   424→    const exhausted = {...cursorState.exhausted};
   425→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   426→    const unavailableBuckets = [];
   427→    const maxIterations = 6;
   428→    let iterationCount = 0;
   429→
   430→    while(
   431→        pending.length < parsedLimit &&
   432→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   433→        iterationCount < maxIterations
   434→    ){
   435→        iterationCount += 1;
   436→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   437→        const chunkResults = await Promise.all(
   438→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   439→        );
   440→
   441→        chunkResults.forEach((result) => {
   442→            offsets[result.bucket] = result.nextOffset;
   443→            exhausted[result.bucket] = result.exhausted;
   444→
   445→            if(result.error){
   446→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   447→                unavailableBuckets.push(result.bucket);
   448→                return;
   449→            }
   450→
   451→            pending.push(...result.items);
   452→        });
   453→
   454→        if(chunkResults.every((result) => result.items.length === 0)){
   455→            break;
   456→        }
   457→    }
   458→
   459→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   460→
   461→    const selectedItems = pending.slice(0, parsedLimit);
   462→    const remainingItems = pending.slice(parsedLimit);
   463→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   464→
   465→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   466→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   467→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   468→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   469→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   470→        hydratedSelectedItems.push(...hydratedFallbackItems);
   471→    }
   472→
   473→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   474→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   475→
   476→    const nextCursor = hasMore
   477→        ? encodeMediaCursor({
   478→            offsets: offsets,
   479→            exhausted: exhausted,
   480→            pending: remainingItems
   481→        })
   482→        : null;
   483→
   484→    return {
   485→        data: hydratedSelectedItems,
   486→        hasMore: hasMore,
   487→        nextCursor: nextCursor,
   488→        unavailableBuckets: dedupedUnavailableBuckets
   489→    };
   490→}

   521→            data: [],
   522→            hasMore: false
   523→        }
   524→        return journalData
   525→    }
   526→
   527→    // If userId is provided, fetch personalization (likes/bookmarks/reactions)

   529→    let userHasLikedSet = new Set();
   530→    let userHasBookmarkedSet = new Set();
   531→    let userReactionMap = new Map();

   535→
   536→        const [userLikes, userBookmarks, userReactions] = await Promise.all([
   537→            supabase
   538→            .from('likes')
   539→            .select('journal_id')
   540→            .in('journal_id', journalIds)
   541→            .eq('user_id', userId),
   542→
   543→            supabase
   544→            .from('bookmarks')
   545→            .select('journal_id')
   546→            .in('journal_id', journalIds)
   547→            .eq('user_id', userId),
   548→
   549→            supabase
   550→            .from('reactions')
   551→            .select('journal_id, reaction_type')
   552→            .in('journal_id', journalIds)
   553→            .eq('user_id', userId)
   554→        ]);
   555→
   556→        const {data: userLikesResult, error: errorUserLikeResult} = userLikes;
   557→        const {data: userBookmarksResult, error: errorUserBookmarksResult} = userBookmarks;
   558→        const {data: userReactionsResult} = userReactions;
   559→
   560→        if(errorUserLikeResult || errorUserBookmarksResult) {
   561→            console.error('supabase error:', errorUserLikeResult?.message || errorUserBookmarksResult?.message);
   562→            throw {status: 500, error: 'supabase error on fetching user likes or user bookmarks'}
   563→        }
   564→
   565→        userHasLikedSet = new Set(userLikesResult?.map((j) => j.journal_id) || []);
   566→        userHasBookmarkedSet = new Set(userBookmarksResult?.map((j) => j.journal_id)|| []);
   567→        userReactionMap = new Map((userReactionsResult || []).map((r) => [r.journal_id, r.reaction_type]));
   568→    }
   569→
   570→    await attachRepostSources(data);
   571→
   572→    const formattedData = data?.map((journal) => ({
   573→        ...journal,
   574→        has_liked: userHasLikedSet.has(journal.id),
   575→        has_bookmarked: userHasBookmarkedSet.has(journal.id),
   576→        user_reaction: userReactionMap.get(journal.id) || null
   577→    }));
   578→    const optimizedData = decorateJournalCollection(formattedData, JOURNAL_MEDIA_USAGE.feed);
   579→
   580→    const hasMore = data?.length > parsedLimit;
   581→    const slicedData = hasMore ? optimizedData.slice(0, parsedLimit) : optimizedData;
   582→
   583→    const journalData = {
   584→        data: slicedData,
   585→        hasMore: hasMore
   586→    }
   587→    return journalData;
   588→}

   810→        .from('journals')

   813→        .eq('privacy', 'public')
   814→        .maybeSingle();
   815→
   816→    if (journalError) {
   817→        console.error('supabase error while fetching journal by id:', journalError.message);
   818→        throw { status: 500, error: 'supabase error while fetching journal by id' };
   819→    }
   820→
   821→    if (!journal) {
   822→        return null;
   823→    }
   824→
   825→    await attachRepostSources(journal, { includeContent: includeRepostContent });
   826→
   827→    let hasLiked = false;
   828→    let hasBookmarked = false;
   829→    let userReaction = null;
   830→
   831→    if (userId) {
   832→        const [likeResult, bookmarkResult, reactionResult] = await Promise.all([
   833→            supabase
   834→                .from('likes')

   836→                .eq('journal_id', journalId)

   839→                .from('bookmarks')
   840→                .select('journal_id', { count: 'exact', head: true })

   844→                .from('reactions')
   845→                .select('reaction_type')
   846→                .eq('journal_id', journalId)
   847→                .eq('user_id', userId)
   848→                .maybeSingle()
   849→        ]);
   850→
   851→        if (likeResult.error || bookmarkResult.error) {
   852→            console.error(
   853→                'supabase error while fetching journal interactions:',
   854→                likeResult.error?.message || bookmarkResult.error?.message
   855→            );
   856→            throw { status: 500, error: 'supabase error while fetching journal interactions' };
   857→        }
   858→
   859→        hasLiked = (likeResult.count || 0) > 0;

   861→        userReaction = reactionResult.data?.reaction_type || null;
   862→    }
   863→
   864→    return decorateJournalMedia({
   865→        ...journal,
   866→        has_liked: hasLiked,
   867→        has_bookmarked: hasBookmarked,
   868→        user_reaction: userReaction
   869→    }, JOURNAL_MEDIA_USAGE.detail);

  1425→export const searchJournalsService = async(query, limit, userId) => {
  1426→    const normalizedQuery = normalizeSearchQuery(query);
  1427→    if(!normalizedQuery || normalizedQuery.length < SEARCH_QUERY_MIN_LENGTH){
  1428→        throw {status: 400, error: `query should be at least ${SEARCH_QUERY_MIN_LENGTH} characters`};
  1429→    }
  1430→
  1431→    if(isNaN(limit) || limit < 1 || limit > SEARCH_LIMIT_MAX){
  1432→        throw {status: 400, error: `limit should be an integer between 1 and ${SEARCH_LIMIT_MAX}`};
  1433→    }
  1434→
  1435→    const parsedLimit = parseInt(limit);
  1436→    const fetchLimit = Math.min(parsedLimit * 3, 60);
  1437→
  1438→    const selectColumns = JOURNAL_METADATA_WITH_COUNTS_SELECT;
  1439→
  1440→    // Prefer semantic retrieval. If pgvector RPC is unavailable or errors,
  1441→    // fallback to keyword search so endpoint still returns useful data.
  1442→    try {
  1443→        const queryEmbedding = await GenerateEmbeddings(normalizedQuery, '');
  1444→
  1445→        if(Array.isArray(queryEmbedding) && queryEmbedding.length > 0){
  1446→            const {data: matches, error: matchError} = await supabase.rpc('match_public_journals', {
  1447→                query_embedding: queryEmbedding,
  1448→                match_count: fetchLimit,
  1449→                similarity_threshold: 0.35
  1450→            });
  1451→
  1452→            if(matchError){
  1453→                console.error('semantic search rpc error:', matchError.message);
  1454→            } else if(Array.isArray(matches) && matches.length > 0){
  1455→                const matchIds = matches.map((row) => row.id);
  1456→                const similarityMap = new Map(matches.map((row) => [row.id, row.similarity]));
  1457→
  1458→                const {data: journals, error: errorJournals} = await supabase
  1459→                    .from('journals')
  1460→                    .select(selectColumns)
  1461→                    .in('id', matchIds)
  1462→                    .eq('privacy', 'public');
  1463→
  1464→                if(errorJournals){
  1465→                    console.error('supabase error while fetching semantic search journals:', errorJournals.message);
  1466→                    throw {status: 500, error: 'supabase error while fetching semantic search journals'};
  1467→                }
  1468→
  1469→                const journalById = new Map((journals || []).map((journal) => [journal.id, journal]));
  1470→                const orderedSemantic = matchIds
  1471→                    .map((id) => journalById.get(id))
  1472→                    .filter(Boolean)
  1473→                    .map((journal) => ({
  1474→                        ...journal,
  1475→                        similarity: similarityMap.get(journal.id) || null
  1476→                    }));
  1477→
  1478→                const withInteraction = await attachUserInteractionFlags(orderedSemantic, userId);
  1479→                const hasMore = withInteraction.length > parsedLimit;
  1480→
  1481→                const optimizedData = decorateJournalCollection(withInteraction, JOURNAL_MEDIA_USAGE.feed);
  1482→
  1483→                return {
  1484→                    data: hasMore ? optimizedData.slice(0, parsedLimit) : optimizedData,
  1485→                    hasMore: hasMore,
  1486→                    mode: 'semantic'
  1487→                };
  1488→            }
  1489→        }

  1491→        console.error('semantic embedding/search fallback error:', error?.message || error);

  1493→
  1494→    const escapedQuery = normalizedQuery.replace(/[%_]/g, (match) => `\\${match}`);
