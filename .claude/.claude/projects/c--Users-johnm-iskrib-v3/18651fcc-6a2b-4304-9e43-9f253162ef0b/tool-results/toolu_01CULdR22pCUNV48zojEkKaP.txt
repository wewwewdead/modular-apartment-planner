    29→const extractBearerToken = (authHeader = "") => {
    30→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    31→    if (!trimmed) return "";
    32→
    33→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    34→    if (bearerMatch?.[1]) {
    35→        return bearerMatch[1].trim();
    36→    }
    37→
    38→    return trimmed;
    39→};
    40→
    41→const resolveAuthUser = async (token) => {
    42→    if (!token) {
    43→        return { user: null, error: null };
    44→    }
    45→
    46→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    47→    if (authError || !authData?.user?.id) {
    48→        return { user: null, error: authError || new Error("missing user id") };
    49→    }
    50→
    51→    return { user: authData.user, error: null };
    52→};
    53→
    54→const requireAuth = async (req, res, next) => {
    55→    const token = extractBearerToken(req.headers?.authorization);
    56→    if (!token) {
    57→        return res.status(401).json({ error: 'not authorized' });
    58→    }
    59→
    60→    const { user, error } = await resolveAuthUser(token);
    61→    if (error || !user?.id) {
    62→        console.error('auth middleware error:', error?.message || 'missing user id');
    63→        return res.status(401).json({ error: 'not authorized' });
    64→    }
    65→
    66→    req.userId = user.id;
    67→    req.authUser = user;
    68→    return next();
    69→};
    70→
    71→const optionalAuth = async (req, _res, next) => {
    72→    const token = extractBearerToken(req.headers?.authorization);
    73→    if (!token) {
    74→        return next();
    75→    }
    76→
    77→    const { user } = await resolveAuthUser(token);
    78→    if (user?.id) {
    79→        req.userId = user.id;
    80→        req.authUser = user;
    81→    }
    82→
    83→    return next();
    84→};

   192→export const imageUploader = async(file, userId, bucket) =>{
   193→    if(!file){
   194→        throw new Error('no file received');
   195→    }
   196→
   197→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   198→    if(!sourceBuffer){
   199→        throw new Error('invalid file payload');
   200→    }
   201→
   202→    let img_buffer = null
   203→    let img_url = null
   204→    img_buffer = await sharp(sourceBuffer)
   205→    .webp({quality: 80})
   206→    .toBuffer();
   207→
   208→    const folderName = `user_id_${userId}`;
   209→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   210→    const filePath = `${folderName}/${fileName}`;
   211→
   212→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   213→    .from(bucket)
   214→    .upload(filePath, img_buffer, {
   215→        contentType: 'image/webp',
   216→        cacheControl: '31536000',
   217→        upsert: true
   218→    })
   219→    if(errorUploadImage){
   220→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   221→        throw new Error('error uploading image into supabase bucket');
   222→    }
   223→    const {data: data_url} = supabase.storage
   224→    .from(bucket)
   225→    .getPublicUrl(filePath)
   226→    
   227→    if(data_url){
   228→        img_url = data_url.publicUrl;
   229→        return img_url;
   230→    } else {
   231→        throw new Error('Error uploading the image');
   232→    }
   233→}

   376→router.get('/getNotifications', requireAuth, async(req, res) =>{
   377→    const {before} = req.query;
   378→    const userId = req.userId;
   379→    const parsedLimit = parseLimitWithinRange(
   380→        req.query.limit,
   381→        NOTIFICATION_LIMIT_MIN,
   382→        NOTIFICATION_LIMIT_MAX,
   383→        NOTIFICATION_DEFAULT_LIMIT

   385→
   386→    if(parsedLimit == null){
   387→        return res.status(400).json({error: `limit must be between ${NOTIFICATION_LIMIT_MIN} and ${NOTIFICATION_LIMIT_MAX}`});
   388→    }
   389→
   390→    // Fetch a generous amount from both tables, then merge and paginate
   391→    const fetchLimit = parsedLimit + 1;
   392→
   393→    let journalQuery = supabase
   394→    .from('notifications')
   395→    .select(NOTIFICATION_JOURNAL_SELECT)
   396→    .eq('receiver_id', userId)
   397→    .order('created_at', {ascending: false})
   398→    .order('id', {ascending: false})
   399→    .limit(fetchLimit)
   400→
   401→    let opinionQuery = supabase
   402→    .from('notification_opinions')
   403→    .select(NOTIFICATION_OPINION_SELECT)
   404→    .eq('receiver_id', userId)
   405→    .order('created_at', {ascending: false})
   406→    .order('id', {ascending: false})
   407→    .limit(fetchLimit)
   408→
   409→    if(before){
   410→        journalQuery = journalQuery.lt('created_at', before);
   411→        opinionQuery = opinionQuery.lt('created_at', before);
   412→    }
   413→
   414→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   415→
   416→    if(journalResult.error){
   417→        console.error('error while fetching data from notification table:', journalResult.error);
   418→        return res.status(500).json({error: 'error fetching data from notification table'});
   419→    }
   420→    if(opinionResult.error){
   421→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   422→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   423→    }
   424→
   425→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   426→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   427→
   428→    // Merge and sort by created_at descending
   429→    const merged = [...journalNotifs, ...opinionNotifs]
   430→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
   431→
   432→    // Likes/bookmarks lookup only for journal notifications
   433→    const journalIds = [...new Set(
   434→        merged
   435→            .filter((n) => n.source === 'journal')
   436→            .map((n) => n.journal_id)
   437→            .filter(Boolean)
   438→    )];
   439→
   440→    let hasLikedResult = [];
   441→    let hasBookMarkedResult = [];
   442→    if(journalIds.length > 0){
   443→        const [hasLiked, hasBookMarked] = await Promise.all([
   444→            supabase.from('likes').select('journal_id').in('journal_id', journalIds).eq('user_id', userId),
   445→            supabase.from('bookmarks').select('journal_id').in('journal_id', journalIds).eq('user_id', userId)
   446→        ]);

   449→            console.error('supabase error while fetching data:', hasLiked.error || hasBookMarked.error);
   450→        }

   452→        hasBookMarkedResult = hasBookMarked.data || [];
   453→    }
   454→
   455→    const userHasLikedSet = new Set(hasLikedResult.map((likes) => likes.journal_id));
   456→    const userHasBookmarkedSet = new Set(hasBookMarkedResult.map((bookmarks) => bookmarks.journal_id));
   457→
   458→    const formatted = merged.map((notif) => ({
   459→        ...notif,
   460→        hasLiked: notif.source === 'journal' ? userHasLikedSet.has(notif.journal_id) : false,
   461→        hasBookMarked: notif.source === 'journal' ? userHasBookmarkedSet.has(notif.journal_id) : false
   462→    }))
   463→
   464→    const hasMore = formatted.length > parsedLimit;
   465→    const slicedData = hasMore ? formatted.slice(0, parsedLimit) : formatted;
   466→

   469→            hasMore: hasMore,
