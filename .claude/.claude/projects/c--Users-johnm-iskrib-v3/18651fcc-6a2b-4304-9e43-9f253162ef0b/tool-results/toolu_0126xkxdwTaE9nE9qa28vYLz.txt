   204→    const [
   205→        {data: settledData, error: settledError},
   206→        {data: unsettledData, error: unsettledError}
   207→    ] = await Promise.all([settledQuery, unsettledQuery]);
   208→
   209→    if(settledError || unsettledError){
   210→        console.error(
   211→            'supabase fallback error while fetching universe posts:',
   212→            settledError?.message || unsettledError?.message
   213→        );
   214→        throw {status: 500, error: 'failed to fetch universe posts'};
   215→    }

   218→    const mergedRows = [...(settledData || []), ...(unsettledData || [])];
   219→    for (const row of mergedRows) {
   220→        if(!row?.id || mergedById.has(row.id)){
   221→            continue;
   222→        }
   223→
   224→        mergedById.set(row.id, normalizeUniversePost(row));
   225→        if(mergedById.size >= limit){

   384→const createInitialMediaCursorState = () => ({
   385→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   386→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   387→    pending: []
   388→});
   389→
   390→const parseMediaCursor = (cursor) => {
   391→    if(!cursor){
   392→        return createInitialMediaCursorState();
   393→    }
   394→
   395→    try {
   396→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   397→        const parsed = JSON.parse(decoded);
   398→        const initial = createInitialMediaCursorState();
   399→
   400→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   401→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   402→                const value = Number(parsed.offsets[bucket]);
   403→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   404→            });
   405→        }
   406→
   407→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   408→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   409→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   410→            });
   411→        }
   412→
   413→        if(Array.isArray(parsed?.pending)){
   414→            initial.pending = parsed.pending
   415→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   416→                .map((item) => ({
   417→                    bucket: item.bucket,
   418→                    path: item.path,
   419→                    name: item.name,
   420→                    createdAt: item.createdAt || null,
   421→                    timestamp: Number(item.timestamp) || 0
   422→                }));
   423→        }
   424→
   425→        return initial;
   426→    } catch {
   427→        return createInitialMediaCursorState();
   428→    }
   429→};

   436→const parseCanvasGalleryCursor = (cursor, expectedSort = 'hottest') => {
   437→    const fallback = {
   438→        offset: 0,
   439→        sort: expectedSort
   440→    };
   441→
   442→    if(!cursor){
   443→        return fallback;
   444→    }
   445→
   446→    try {
   447→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   448→        const parsed = JSON.parse(decoded);
   449→        const parsedSort = typeof parsed?.sort === 'string' && parsed.sort.toLowerCase().trim() === 'newest'
   450→            ? 'newest'
   451→            : 'hottest';
   452→
   453→        if(parsedSort !== expectedSort){
   454→            return fallback;
   455→        }
   456→
   457→        const parsedOffset = Number(parsed?.offset);
   458→        const normalizedOffset = Number.isNaN(parsedOffset) || parsedOffset < 0
   459→            ? 0
   460→            : Math.floor(parsedOffset);
   461→
   462→        return {
   463→            offset: normalizedOffset,
   464→            sort: parsedSort
   465→        };
   466→    } catch {
   467→        return fallback;
   468→    }
   469→};

   500→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   501→    const folderPath = `user_id_${userId}`;
   502→
   503→    const {data: bucketItems, error: bucketError} = await supabase.storage
   504→        .from(bucket)
   505→        .list(folderPath, {
   506→            limit: limit,
   507→            offset: offset,
   508→            sortBy: {column: 'name', order: 'desc'}
   509→        });
   510→
   511→    if(bucketError){
   512→        return {
   513→            bucket: bucket,
   514→            nextOffset: offset,
   515→            exhausted: true,
   516→            items: [],
   517→            error: bucketError.message
   518→        };
   519→    }
   520→
   521→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   522→    const normalizedItems = listedItems
   523→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name))
   524→        .map((item) => {
   525→            const path = `${folderPath}/${item.name}`;
   526→            const createdAt = item.created_at || item.updated_at || null;
   527→
   528→            return {
   529→                bucket: bucket,
   530→                path: path,
   531→                name: item.name,
   532→                createdAt: createdAt,
   533→                timestamp: parseMediaTimestamp(item)
   534→            };
   535→        });
   536→
   537→    return {
   538→        bucket: bucket,
   539→        nextOffset: offset + listedItems.length,
   540→        exhausted: listedItems.length < limit,
   541→        items: normalizedItems,
   542→        error: null
   543→    };
   544→};

   571→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   572→    if(!userId){
   573→        throw {status: 400, error: 'userId is undefined'};
   574→    }
   575→
   576→    const parsedLimit = Number(limit);
   577→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   578→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   579→    }
   580→
   581→    const cursorState = parseMediaCursor(cursor);
   582→    const offsets = {...cursorState.offsets};
   583→    const exhausted = {...cursorState.exhausted};
   584→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   585→    const unavailableBuckets = [];
   586→    const maxIterations = 6;
   587→    let iterationCount = 0;
   588→
   589→    while(
   590→        pending.length < parsedLimit &&
   591→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   592→        iterationCount < maxIterations
   593→    ){
   594→        iterationCount += 1;
   595→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   596→        const chunkResults = await Promise.all(
   597→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   598→        );
   599→
   600→        chunkResults.forEach((result) => {
   601→            offsets[result.bucket] = result.nextOffset;
   602→            exhausted[result.bucket] = result.exhausted;
   603→
   604→            if(result.error){
   605→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   606→                unavailableBuckets.push(result.bucket);
   607→                return;
   608→            }
   609→
   610→            pending.push(...result.items);
   611→        });
   612→
   613→        if(chunkResults.every((result) => result.items.length === 0)){
   614→            break;
   615→        }
   616→    }
   617→
   618→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   619→
   620→    const selectedItems = pending.slice(0, parsedLimit);
   621→    const remainingItems = pending.slice(parsedLimit);
   622→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   623→
   624→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   625→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   626→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   627→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   628→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   629→        hydratedSelectedItems.push(...hydratedFallbackItems);
   630→    }
   631→
   632→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   633→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   634→
   635→    const nextCursor = hasMore
   636→        ? encodeMediaCursor({
   637→            offsets: offsets,
   638→            exhausted: exhausted,
   639→            pending: remainingItems
   640→        })
   641→        : null;
   642→
   643→    return {
   644→        data: hydratedSelectedItems,
   645→        hasMore: hasMore,
   646→        nextCursor: nextCursor,
   647→        unavailableBuckets: dedupedUnavailableBuckets

  1422→export const searchJournalsService = async(query, limit, userId) => {
  1423→    const normalizedQuery = normalizeSearchQuery(query);
  1424→    if(!normalizedQuery || normalizedQuery.length < SEARCH_QUERY_MIN_LENGTH){
  1425→        throw {status: 400, error: `query should be at least ${SEARCH_QUERY_MIN_LENGTH} characters`};
  1426→    }
  1427→
  1428→    if(isNaN(limit) || limit < 1 || limit > SEARCH_LIMIT_MAX){
  1429→        throw {status: 400, error: `limit should be an integer between 1 and ${SEARCH_LIMIT_MAX}`};
  1430→    }
  1431→
  1432→    const parsedLimit = parseInt(limit);
  1433→    const fetchLimit = Math.min(parsedLimit * 3, 60);
  1434→
  1435→    const selectColumns = JOURNAL_WITH_COUNTS_SELECT;
  1436→
  1437→    // Prefer semantic retrieval. If pgvector RPC is unavailable or errors,
  1438→    // fallback to keyword search so endpoint still returns useful data.
  1439→    try {
  1440→        const queryEmbedding = await GenerateEmbeddings(normalizedQuery, '');
  1441→
  1442→        if(Array.isArray(queryEmbedding) && queryEmbedding.length > 0){
  1443→            const {data: matches, error: matchError} = await supabase.rpc('match_public_journals', {
  1444→                query_embedding: queryEmbedding,
  1445→                match_count: fetchLimit,
  1446→                similarity_threshold: 0.35
  1447→            });
  1448→
  1449→            if(matchError){
  1450→                console.error('semantic search rpc error:', matchError.message);
  1451→            } else if(Array.isArray(matches) && matches.length > 0){
  1452→                const matchIds = matches.map((row) => row.id);
  1453→                const similarityMap = new Map(matches.map((row) => [row.id, row.similarity]));
  1454→
  1455→                const {data: journals, error: errorJournals} = await supabase
  1456→                    .from('journals')
  1457→                    .select(selectColumns)

  1459→                    .eq('privacy', 'public');
  1460→
  1461→                if(errorJournals){
  1462→                    console.error('supabase error while fetching semantic search journals:', errorJournals.message);
  1463→                    throw {status: 500, error: 'supabase error while fetching semantic search journals'};
  1464→                }
  1465→
  1466→                const journalById = new Map((journals || []).map((journal) => [journal.id, journal]));
  1467→                const orderedSemantic = matchIds
  1468→                    .map((id) => journalById.get(id))
  1469→                    .filter(Boolean)
  1470→                    .map((journal) => ({
  1471→                        ...journal,
  1472→                        similarity: similarityMap.get(journal.id) || null
  1473→                    }));
  1474→
  1475→                const withInteraction = await attachUserInteractionFlags(orderedSemantic, userId);
  1476→                const hasMore = withInteraction.length > parsedLimit;

  1478→                return {

  1480→                    hasMore: hasMore,
  1481→                    mode: 'semantic'
  1482→                };

  1490→

  1492→        supabase
  1493→            .from('journals')
  1494→            .select(selectColumns)
  1495→            .eq('privacy', 'public')
  1496→            .ilike('title', `%${escapedQuery}%`)
  1497→            .order('created_at', {ascending: false})
  1498→            .order('id', {ascending: false})
  1499→            .limit(parsedLimit + 1),
  1500→        supabase
  1501→            .from('journals')
  1502→            .select(selectColumns)
  1503→            .eq('privacy', 'public')
  1504→            .ilike('content', `%${escapedQuery}%`)
  1505→            .order('created_at', {ascending: false})
  1506→            .order('id', {ascending: false})
  1507→            .limit(parsedLimit + 1)
  1508→    ]);
  1509→
  1510→    if(titleResult.error || contentResult.error){
  1511→        console.error('keyword search error:', titleResult.error?.message || contentResult.error?.message);
  1512→        throw {status: 500, error: 'supabase error while searching journals'};
  1513→    }
  1514→
