    33→const extractBearerToken = (authHeader = "") => {
    34→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    35→    if (!trimmed) return "";
    36→
    37→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    38→    if (bearerMatch?.[1]) {
    39→        return bearerMatch[1].trim();
    40→    }
    41→
    42→    return trimmed;
    43→};
    44→
    45→const resolveAuthUser = async (token) => {
    46→    if (!token) {
    47→        return { user: null, error: null };
    48→    }
    49→
    50→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    51→    if (authError || !authData?.user?.id) {
    52→        return { user: null, error: authError || new Error("missing user id") };
    53→    }
    54→
    55→    return { user: authData.user, error: null };
    56→};
    57→
    58→const requireAuth = async (req, res, next) => {
    59→    const token = extractBearerToken(req.headers?.authorization);
    60→    if (!token) {
    61→        return res.status(401).json({ error: 'not authorized' });
    62→    }
    63→
    64→    const { user, error } = await resolveAuthUser(token);
    65→    if (error || !user?.id) {
    66→        console.error('auth middleware error:', error?.message || 'missing user id');
    67→        return res.status(401).json({ error: 'not authorized' });
    68→    }
    69→
    70→    req.userId = user.id;
    71→    req.authUser = user;
    72→    return next();
    73→};
    74→
    75→const optionalAuth = async (req, _res, next) => {
    76→    const token = extractBearerToken(req.headers?.authorization);
    77→    if (!token) {
    78→        return next();
    79→    }
    80→
    81→    const { user } = await resolveAuthUser(token);
    82→    if (user?.id) {
    83→        req.userId = user.id;
    84→        req.authUser = user;
    85→    }
    86→
    87→    return next();
    88→};

   196→export const imageUploader = async(file, userId, bucket) =>{
   197→    if(!file){
   198→        throw new Error('no file received');
   199→    }
   200→
   201→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   202→    if(!sourceBuffer){
   203→        throw new Error('invalid file payload');
   204→    }
   205→
   206→    let img_buffer = null
   207→    let img_url = null
   208→    img_buffer = await sharp(sourceBuffer)
   209→    .webp({quality: 80})
   210→    .toBuffer();
   211→
   212→    const folderName = `user_id_${userId}`;
   213→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   214→    const filePath = `${folderName}/${fileName}`;
   215→
   216→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   217→    .from(bucket)
   218→    .upload(filePath, img_buffer, {
   219→        contentType: 'image/webp',
   220→        cacheControl: '31536000',
   221→        upsert: true
   222→    })
   223→    if(errorUploadImage){
   224→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   225→        throw new Error('error uploading image into supabase bucket');
   226→    }
   227→    const {data: data_url} = supabase.storage
   228→    .from(bucket)
   229→    .getPublicUrl(filePath)
   230→    
   231→    if(data_url){
   232→        img_url = data_url.publicUrl;
   233→        return img_url;
   234→    } else {
   235→        throw new Error('Error uploading the image');
   236→    }
   237→}

   380→router.get('/getNotifications', requireAuth, async(req, res) =>{
   381→    const {before} = req.query;
   382→    const userId = req.userId;
   383→    const parsedLimit = parseLimitWithinRange(
   384→        req.query.limit,
   385→        NOTIFICATION_LIMIT_MIN,
   386→        NOTIFICATION_LIMIT_MAX,
   387→        NOTIFICATION_DEFAULT_LIMIT
   388→    );
   389→
   390→    if(parsedLimit == null){
   391→        return res.status(400).json({error: `limit must be between ${NOTIFICATION_LIMIT_MIN} and ${NOTIFICATION_LIMIT_MAX}`});
   392→    }
   393→
   394→    // Fetch a generous amount from both tables, then merge and paginate
   395→    const fetchLimit = parsedLimit + 1;
   396→
   397→    let journalQuery = supabase
   398→    .from('notifications')
   399→    .select(NOTIFICATION_JOURNAL_SELECT)
   400→    .eq('receiver_id', userId)
   401→    .order('created_at', {ascending: false})
   402→    .order('id', {ascending: false})
   403→    .limit(fetchLimit)
   404→
   405→    let opinionQuery = supabase
   406→    .from('notification_opinions')
   407→    .select(NOTIFICATION_OPINION_SELECT)
   408→    .eq('receiver_id', userId)
   409→    .order('created_at', {ascending: false})
   410→    .order('id', {ascending: false})
   411→    .limit(fetchLimit)
   412→
   413→    if(before){
   414→        journalQuery = journalQuery.lt('created_at', before);
   415→        opinionQuery = opinionQuery.lt('created_at', before);
   416→    }
   417→
   418→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   419→
   420→    if(journalResult.error){
   421→        console.error('error while fetching data from notification table:', journalResult.error);
   422→        return res.status(500).json({error: 'error fetching data from notification table'});
   423→    }
   424→    if(opinionResult.error){
   425→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   426→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   427→    }
   428→
   429→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   430→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   431→
   432→    // Merge and sort by created_at descending
   433→    const merged = [...journalNotifs, ...opinionNotifs]
   434→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
   435→
   436→    // Likes/bookmarks lookup only for journal notifications
   437→    const journalIds = [...new Set(
   438→        merged
   439→            .filter((n) => n.source === 'journal')
   440→            .map((n) => n.journal_id)
   441→            .filter(Boolean)
   442→    )];
   443→
   444→    let hasLikedResult = [];
   445→    let hasBookMarkedResult = [];
   446→    if(journalIds.length > 0){
   447→        const [hasLiked, hasBookMarked] = await Promise.all([
   448→            supabase.from('likes').select('journal_id').in('journal_id', journalIds).eq('user_id', userId),
   449→            supabase.from('bookmarks').select('journal_id').in('journal_id', journalIds).eq('user_id', userId)
   450→        ]);
   451→
   452→        if(hasLiked.error || hasBookMarked.error){
   453→            console.error('supabase error while fetching data:', hasLiked.error || hasBookMarked.error);
   454→        }
   455→        hasLikedResult = hasLiked.data || [];
   456→        hasBookMarkedResult = hasBookMarked.data || [];
   457→    }
   458→
   459→    const userHasLikedSet = new Set(hasLikedResult.map((likes) => likes.journal_id));
   460→    const userHasBookmarkedSet = new Set(hasBookMarkedResult.map((bookmarks) => bookmarks.journal_id));
   461→
   462→    const formatted = merged.map((notif) => ({
   463→        ...notif,
   464→        hasLiked: notif.source === 'journal' ? userHasLikedSet.has(notif.journal_id) : false,
   465→        hasBookMarked: notif.source === 'journal' ? userHasBookmarkedSet.has(notif.journal_id) : false
   466→    }))
   467→
   468→    const hasMore = formatted.length > parsedLimit;
   469→    const slicedData = hasMore ? formatted.slice(0, parsedLimit) : formatted;
   470→
   471→    return res.status(200).json(
   472→        {
   473→            hasMore: hasMore,
   474→            data: slicedData,
   475→        }
   476→    )
   477→
   478→})
