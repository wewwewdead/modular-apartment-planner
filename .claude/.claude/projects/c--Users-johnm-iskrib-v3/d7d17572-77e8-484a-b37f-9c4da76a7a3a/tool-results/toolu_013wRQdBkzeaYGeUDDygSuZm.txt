    28→const extractBearerToken = (authHeader = "") => {
    29→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    30→    if (!trimmed) return "";
    31→
    32→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    33→    if (bearerMatch?.[1]) {
    34→        return bearerMatch[1].trim();
    35→    }
    36→
    37→    return trimmed;
    38→};
    39→
    40→const resolveAuthUser = async (token) => {
    41→    if (!token) {
    42→        return { user: null, error: null };
    43→    }
    44→
    45→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    46→    if (authError || !authData?.user?.id) {
    47→        return { user: null, error: authError || new Error("missing user id") };
    48→    }
    49→
    50→    return { user: authData.user, error: null };
    51→};
    52→
    53→const requireAuth = async (req, res, next) => {
    54→    const token = extractBearerToken(req.headers?.authorization);
    55→    if (!token) {
    56→        return res.status(401).json({ error: 'not authorized' });
    57→    }
    58→
    59→    const { user, error } = await resolveAuthUser(token);
    60→    if (error || !user?.id) {
    61→        console.error('auth middleware error:', error?.message || 'missing user id');
    62→        return res.status(401).json({ error: 'not authorized' });
    63→    }
    64→
    65→    req.userId = user.id;
    66→    req.authUser = user;
    67→    return next();
    68→};
    69→
    70→const optionalAuth = async (req, _res, next) => {
    71→    const token = extractBearerToken(req.headers?.authorization);
    72→    if (!token) {
    73→        return next();
    74→    }
    75→
    76→    const { user } = await resolveAuthUser(token);
    77→    if (user?.id) {
    78→        req.userId = user.id;
    79→        req.authUser = user;
    80→    }
    81→
    82→    return next();
    83→};

   191→export const imageUploader = async(file, userId, bucket) =>{
   192→    if(!file){
   193→        throw new Error('no file received');
   194→    }
   195→
   196→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   197→    if(!sourceBuffer){
   198→        throw new Error('invalid file payload');
   199→    }
   200→
   201→    let img_buffer = null
   202→    let img_url = null
   203→    img_buffer = await sharp(sourceBuffer)
   204→    .webp({quality: 80})
   205→    .toBuffer();
   206→
   207→    const folderName = `user_id_${userId}`;
   208→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   209→    const filePath = `${folderName}/${fileName}`;
   210→
   211→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   212→    .from(bucket)
   213→    .upload(filePath, img_buffer, {
   214→        contentType: 'image/webp',
   215→        cacheControl: '31536000',
   216→        upsert: true
   217→    })
   218→    if(errorUploadImage){
   219→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   220→        throw new Error('error uploading image into supabase bucket');
   221→    }
   222→    const {data: data_url} = supabase.storage
   223→    .from(bucket)
   224→    .getPublicUrl(filePath)
   225→    
   226→    if(data_url){
   227→        img_url = data_url.publicUrl;
   228→        return img_url;
   229→    } else {
   230→        throw new Error('Error uploading the image');
   231→    }
   232→}

   374→router.get('/getNotifications', requireAuth, async(req, res) =>{
   375→    const {before} = req.query;
   376→    const userId = req.userId;
   377→    const parsedLimit = parseLimitWithinRange(
   378→        req.query.limit,
   379→        NOTIFICATION_LIMIT_MIN,
   380→        NOTIFICATION_LIMIT_MAX,
   381→        NOTIFICATION_DEFAULT_LIMIT
   382→    );
   383→
   384→    if(parsedLimit == null){
   385→        return res.status(400).json({error: `limit must be between ${NOTIFICATION_LIMIT_MIN} and ${NOTIFICATION_LIMIT_MAX}`});
   386→    }
   387→
   388→    // Fetch a generous amount from both tables, then merge and paginate
   389→    const fetchLimit = parsedLimit + 1;
   390→
   391→    let journalQuery = supabase
   392→    .from('notifications')
   393→    .select(NOTIFICATION_JOURNAL_SELECT)
   394→    .eq('receiver_id', userId)
   395→    .order('created_at', {ascending: false})
   396→    .order('id', {ascending: false})
   397→    .limit(fetchLimit)
   398→
   399→    let opinionQuery = supabase
   400→    .from('notification_opinions')
   401→    .select(NOTIFICATION_OPINION_SELECT)
   402→    .eq('receiver_id', userId)
   403→    .order('created_at', {ascending: false})
   404→    .order('id', {ascending: false})
   405→    .limit(fetchLimit)
   406→
   407→    if(before){
   408→        journalQuery = journalQuery.lt('created_at', before);
   409→        opinionQuery = opinionQuery.lt('created_at', before);
   410→    }
   411→
   412→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   413→
   414→    if(journalResult.error){
   415→        console.error('error while fetching data from notification table:', journalResult.error);
   416→        return res.status(500).json({error: 'error fetching data from notification table'});
   417→    }
   418→    if(opinionResult.error){
   419→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   420→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   421→    }
   422→
   423→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   424→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   425→
   426→    // Merge and sort by created_at descending
   427→    const merged = [...journalNotifs, ...opinionNotifs]
   428→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
   429→
   430→    // Likes/bookmarks lookup only for journal notifications
   431→    const journalIds = [...new Set(
   432→        merged
   433→            .filter((n) => n.source === 'journal')
   434→            .map((n) => n.journal_id)
   435→            .filter(Boolean)
   436→    )];
   437→
   438→    let hasLikedResult = [];
   439→    let hasBookMarkedResult = [];
   440→    if(journalIds.length > 0){
   441→        const [hasLiked, hasBookMarked] = await Promise.all([
   442→            supabase.from('likes').select('journal_id').in('journal_id', journalIds).eq('user_id', userId),
   443→            supabase.from('bookmarks').select('journal_id').in('journal_id', journalIds).eq('user_id', userId)
   444→        ]);
   445→
   446→        if(hasLiked.error || hasBookMarked.error){
   447→            console.error('supabase error while fetching data:', hasLiked.error || hasBookMarked.error);
   448→        }
   449→        hasLikedResult = hasLiked.data || [];
   450→        hasBookMarkedResult = hasBookMarked.data || [];
   451→    }
   452→
   453→    const userHasLikedSet = new Set(hasLikedResult.map((likes) => likes.journal_id));
   454→    const userHasBookmarkedSet = new Set(hasBookMarkedResult.map((bookmarks) => bookmarks.journal_id));
   455→
   456→    const formatted = merged.map((notif) => ({
   457→        ...notif,
   458→        hasLiked: notif.source === 'journal' ? userHasLikedSet.has(notif.journal_id) : false,
   459→        hasBookMarked: notif.source === 'journal' ? userHasBookmarkedSet.has(notif.journal_id) : false
   460→    }))
   461→
   462→    const hasMore = formatted.length > parsedLimit;
   463→    const slicedData = hasMore ? formatted.slice(0, parsedLimit) : formatted;
   464→
   465→    return res.status(200).json(
   466→        {
   467→            hasMore: hasMore,
   468→            data: slicedData,
   469→        }
   470→    )
   471→
   472→})
