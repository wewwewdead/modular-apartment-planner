   363→const createInitialMediaCursorState = () => ({
   364→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   365→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   366→    pending: []
   367→});
   368→
   369→const parseMediaCursor = (cursor) => {
   370→    if(!cursor){
   371→        return createInitialMediaCursorState();
   372→    }
   373→
   374→    try {
   375→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   376→        const parsed = JSON.parse(decoded);
   377→        const initial = createInitialMediaCursorState();
   378→
   379→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   380→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   381→                const value = Number(parsed.offsets[bucket]);
   382→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   383→            });
   384→        }
   385→
   386→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   387→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   388→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   389→            });
   390→        }
   391→
   392→        if(Array.isArray(parsed?.pending)){
   393→            initial.pending = parsed.pending
   394→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   395→                .map((item) => ({
   396→                    bucket: item.bucket,
   397→                    path: item.path,
   398→                    name: item.name,
   399→                    createdAt: item.createdAt || null,
   400→                    timestamp: Number(item.timestamp) || 0
   401→                }));
   402→        }
   403→
   404→        return initial;
   405→    } catch {
   406→        return createInitialMediaCursorState();
   407→    }
   408→};

   415→const parseCanvasGalleryCursor = (cursor, expectedSort = 'hottest') => {
   416→    const fallback = {
   417→        offset: 0,
   418→        sort: expectedSort
   419→    };
   420→
   421→    if(!cursor){
   422→        return fallback;
   423→    }
   424→
   425→    try {
   426→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   427→        const parsed = JSON.parse(decoded);
   428→        const parsedSort = typeof parsed?.sort === 'string' && parsed.sort.toLowerCase().trim() === 'newest'
   429→            ? 'newest'
   430→            : 'hottest';
   431→
   432→        if(parsedSort !== expectedSort){
   433→            return fallback;
   434→        }
   435→
   436→        const parsedOffset = Number(parsed?.offset);
   437→        const normalizedOffset = Number.isNaN(parsedOffset) || parsedOffset < 0
   438→            ? 0
   439→            : Math.floor(parsedOffset);
   440→
   441→        return {
   442→            offset: normalizedOffset,
   443→            sort: parsedSort
   444→        };
   445→    } catch {
   446→        return fallback;
   447→    }
   448→};

   479→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   480→    const folderPath = `user_id_${userId}`;
   481→
   482→    const {data: bucketItems, error: bucketError} = await supabase.storage
   483→        .from(bucket)
   484→        .list(folderPath, {
   485→            limit: limit,
   486→            offset: offset,
   487→            sortBy: {column: 'name', order: 'desc'}
   488→        });
   489→
   490→    if(bucketError){
   491→        return {
   492→            bucket: bucket,
   493→            nextOffset: offset,
   494→            exhausted: true,
   495→            items: [],
   496→            error: bucketError.message
   497→        };
   498→    }
   499→
   500→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   501→    const normalizedItems = listedItems
   502→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name))
   503→        .map((item) => {
   504→            const path = `${folderPath}/${item.name}`;
   505→            const createdAt = item.created_at || item.updated_at || null;
   506→
   507→            return {
   508→                bucket: bucket,
   509→                path: path,
   510→                name: item.name,
   511→                createdAt: createdAt,
   512→                timestamp: parseMediaTimestamp(item)
   513→            };
   514→        });
   515→
   516→    return {
   517→        bucket: bucket,
   518→        nextOffset: offset + listedItems.length,
   519→        exhausted: listedItems.length < limit,
   520→        items: normalizedItems,
   521→        error: null
   522→    };
   523→};

   525→const hydrateMediaItems = async(items) => {
   526→    if(!Array.isArray(items) || items.length === 0){

   531→        const url = await resolveMediaUrl(item.bucket, item.path);
   532→        if(!url){
   533→            return null;
   534→        }
   535→
   536→        return {
   537→            id: `${item.bucket}-${item.path}`,
   538→            bucket: item.bucket,
   539→            path: item.path,
   540→            name: item.name,
   541→            createdAt: item.createdAt || null,
   542→            timestamp: item.timestamp || 0,
   543→            url: url
   544→        };
   545→    }));
   546→
   547→    return hydrated.filter(Boolean);
   548→};

   550→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   551→    if(!userId){
   552→        throw {status: 400, error: 'userId is undefined'};
   553→    }
   554→
   555→    const parsedLimit = Number(limit);
   556→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   557→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   558→    }
   559→
   560→    const cursorState = parseMediaCursor(cursor);
   561→    const offsets = {...cursorState.offsets};
   562→    const exhausted = {...cursorState.exhausted};
   563→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   564→    const unavailableBuckets = [];
   565→    const maxIterations = 6;
   566→    let iterationCount = 0;
   567→
   568→    while(
   569→        pending.length < parsedLimit &&
   570→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   571→        iterationCount < maxIterations
   572→    ){
   573→        iterationCount += 1;
   574→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   575→        const chunkResults = await Promise.all(
   576→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   577→        );
   578→
   579→        chunkResults.forEach((result) => {
   580→            offsets[result.bucket] = result.nextOffset;
   581→            exhausted[result.bucket] = result.exhausted;
   582→
   583→            if(result.error){
   584→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   585→                unavailableBuckets.push(result.bucket);
   586→                return;
   587→            }
   588→
   589→            pending.push(...result.items);
   590→        });
   591→
   592→        if(chunkResults.every((result) => result.items.length === 0)){
   593→            break;
   594→        }
   595→    }
   596→
   597→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   598→
   599→    const selectedItems = pending.slice(0, parsedLimit);
   600→    const remainingItems = pending.slice(parsedLimit);
   601→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   602→
   603→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   604→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   605→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   606→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   607→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   608→        hydratedSelectedItems.push(...hydratedFallbackItems);
   609→    }
   610→

   781→    const nextMonthStartUtc = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth() + 1, 1, 0, 0, 0, 0));

   816→        if(dateDiff !== 0){
   817→            return dateDiff;
   818→        }
   819→
   820→        return (b?.id || 0) - (a?.id || 0);

   822→
   823→    return {
   824→        data: scored.slice(0, parsedLimit),
   825→        period: {
   826→            startUtc: monthStartUtc.toISOString(),
   827→            endUtc: nextMonthStartUtc.toISOString(),
   828→            timezone: 'UTC'

  1391→export const searchJournalsService = async(query, limit, userId) => {
  1392→    const normalizedQuery = normalizeSearchQuery(query);
  1393→    if(!normalizedQuery || normalizedQuery.length < SEARCH_QUERY_MIN_LENGTH){
  1394→        throw {status: 400, error: `query should be at least ${SEARCH_QUERY_MIN_LENGTH} characters`};
  1395→    }
  1396→
  1397→    if(isNaN(limit) || limit < 1 || limit > SEARCH_LIMIT_MAX){
  1398→        throw {status: 400, error: `limit should be an integer between 1 and ${SEARCH_LIMIT_MAX}`};
  1399→    }
  1400→
  1401→    const parsedLimit = parseInt(limit);
  1402→    const fetchLimit = Math.min(parsedLimit * 3, 60);
  1403→
  1404→    const selectColumns = JOURNAL_WITH_COUNTS_SELECT;
  1405→
  1406→    // Prefer semantic retrieval. If pgvector RPC is unavailable or errors,
  1407→    // fallback to keyword search so endpoint still returns useful data.
  1408→    try {
  1409→        const queryEmbedding = await GenerateEmbeddings(normalizedQuery, '');
  1410→
  1411→        if(Array.isArray(queryEmbedding) && queryEmbedding.length > 0){
  1412→            const {data: matches, error: matchError} = await supabase.rpc('match_public_journals', {
  1413→                query_embedding: queryEmbedding,
  1414→                match_count: fetchLimit,
  1415→                similarity_threshold: 0.35
  1416→            });
  1417→
  1418→            if(matchError){
  1419→                console.error('semantic search rpc error:', matchError.message);
  1420→            } else if(Array.isArray(matches) && matches.length > 0){
  1421→                const matchIds = matches.map((row) => row.id);
  1422→                const similarityMap = new Map(matches.map((row) => [row.id, row.similarity]));
  1423→
  1424→                const {data: journals, error: errorJournals} = await supabase
  1425→                    .from('journals')
  1426→                    .select(selectColumns)
  1427→                    .in('id', matchIds)
  1428→                    .eq('privacy', 'public');
  1429→
  1430→                if(errorJournals){
  1431→                    console.error('supabase error while fetching semantic search journals:', errorJournals.message);
  1432→                    throw {status: 500, error: 'supabase error while fetching semantic search journals'};
  1433→                }
  1434→
  1435→                const journalById = new Map((journals || []).map((journal) => [journal.id, journal]));
  1436→                const orderedSemantic = matchIds
  1437→                    .map((id) => journalById.get(id))
  1438→                    .filter(Boolean)
  1439→                    .map((journal) => ({
  1440→                        ...journal,
  1441→                        similarity: similarityMap.get(journal.id) || null
  1442→                    }));
  1443→
  1444→                const withInteraction = await attachUserInteractionFlags(orderedSemantic, userId);
  1445→                const hasMore = withInteraction.length > parsedLimit;
  1446→
  1447→                return {
  1448→                    data: hasMore ? withInteraction.slice(0, parsedLimit) : withInteraction,
  1449→                    hasMore: hasMore,
  1450→                    mode: 'semantic'
  1451→                };
  1452→            }
  1453→        }
  1454→    } catch (error) {
  1455→        console.error('semantic embedding/search fallback error:', error?.message || error);
  1456→    }
  1457→
  1458→    const escapedQuery = normalizedQuery.replace(/[%_]/g, (match) => `\\${match}`);
  1459→
  1460→    const [titleResult, contentResult] = await Promise.all([
  1461→        supabase
  1462→            .from('journals')
  1463→            .select(selectColumns)
  1464→            .eq('privacy', 'public')
  1465→            .ilike('title', `%${escapedQuery}%`)
  1466→            .order('created_at', {ascending: false})
  1467→            .order('id', {ascending: false})
  1468→            .limit(parsedLimit + 1),
  1469→        supabase
  1470→            .from('journals')
  1471→            .select(selectColumns)
  1472→            .eq('privacy', 'public')
  1473→            .ilike('content', `%${escapedQuery}%`)
  1474→            .order('created_at', {ascending: false})
  1475→            .order('id', {ascending: false})
  1476→            .limit(parsedLimit + 1)
  1477→    ]);
  1478→
  1479→    if(titleResult.error || contentResult.error){
  1480→        console.error('keyword search error:', titleResult.error?.message || contentResult.error?.message);
  1481→        throw {status: 500, error: 'supabase error while searching journals'};
  1482→    }
  1483→
  1484→    const keywordMap = new Map();
  1485→    (titleResult.data || []).forEach((journal) => {
  1486→        keywordMap.set(journal.id, journal);
  1487→    });
  1488→    (contentResult.data || []).forEach((journal) => {
  1489→        if(!keywordMap.has(journal.id)){
  1490→            keywordMap.set(journal.id, journal);
  1491→        }
  1492→    });
  1493→
  1494→    const keywordJournals = [...keywordMap.values()];
  1495→
  1496→    const withInteraction = await attachUserInteractionFlags(keywordJournals || [], userId);
  1497→    const hasMore = withInteraction.length > parsedLimit;
  1498→
  1499→    return {
  1500→        data: hasMore ? withInteraction.slice(0, parsedLimit) : withInteraction,
  1501→        hasMore: hasMore,
  1502→        mode: 'keyword'
  1503→    };
  1504→}
