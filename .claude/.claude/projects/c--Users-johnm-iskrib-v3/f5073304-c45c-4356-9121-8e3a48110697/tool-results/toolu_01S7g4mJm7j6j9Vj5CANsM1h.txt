   254→const createInitialMediaCursorState = () => ({
   255→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   256→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   257→    pending: []
   258→});
   259→
   260→const parseMediaCursor = (cursor) => {
   261→    if(!cursor){
   262→        return createInitialMediaCursorState();
   263→    }
   264→
   265→    try {
   266→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   267→        const parsed = JSON.parse(decoded);
   268→        const initial = createInitialMediaCursorState();
   269→
   270→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   271→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   272→                const value = Number(parsed.offsets[bucket]);
   273→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   274→            });
   275→        }
   276→
   277→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   278→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   279→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   280→            });
   281→        }
   282→
   283→        if(Array.isArray(parsed?.pending)){
   284→            initial.pending = parsed.pending
   285→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   286→                .map((item) => ({
   287→                    bucket: item.bucket,
   288→                    path: item.path,
   289→                    name: item.name,
   290→                    createdAt: item.createdAt || null,
   291→                    timestamp: Number(item.timestamp) || 0
   292→                }));
   293→        }
   294→
   295→        return initial;
   296→    } catch {
   297→        return createInitialMediaCursorState();
   298→    }
   299→};

   317→        .from(bucket)

   319→
   320→    if(!signedUrlError && signedUrlData?.signedUrl){

   327→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   328→    const folderPath = `user_id_${userId}`;
   329→    const listLimit = Math.max(limit * 4, limit);
   330→
   331→    const {data: bucketItems, error: bucketError} = await supabase.storage
   332→        .from(bucket)
   333→        .list(folderPath, {
   334→            limit: listLimit,
   335→            offset: offset,
   336→            sortBy: {column: 'name', order: 'desc'}
   337→        });
   338→
   339→    if(bucketError){
   340→        return {
   341→            bucket: bucket,
   342→            nextOffset: offset,
   343→            exhausted: true,
   344→            items: [],
   345→            error: bucketError.message
   346→        };
   347→    }
   348→
   349→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   350→    const normalizedItems = listedItems
   351→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name) && isPrimaryListableFileName(item.name))
   352→        .map((item) => {
   353→            const path = `${folderPath}/${item.name}`;
   354→            const createdAt = item.created_at || item.updated_at || null;
   355→
   356→            return {
   357→                bucket: bucket,
   358→                path: path,
   359→                name: item.name,
   360→                createdAt: createdAt,
   361→                timestamp: parseMediaTimestamp(item)
   362→            };
   363→        });
   364→
   365→    return {
   366→        bucket: bucket,
   367→        nextOffset: offset + listedItems.length,
   368→        exhausted: listedItems.length < listLimit,
   369→        items: normalizedItems,
   370→        error: null
   371→    };
   372→};

   405→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   406→    if(!userId){
   407→        throw {status: 400, error: 'userId is undefined'};
   408→    }
   409→
   410→    const parsedLimit = Number(limit);
   411→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   412→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   413→    }
   414→
   415→    const cursorState = parseMediaCursor(cursor);
   416→    const offsets = {...cursorState.offsets};
   417→    const exhausted = {...cursorState.exhausted};
   418→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   419→    const unavailableBuckets = [];
   420→    const maxIterations = 6;
   421→    let iterationCount = 0;
   422→
   423→    while(
   424→        pending.length < parsedLimit &&
   425→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   426→        iterationCount < maxIterations
   427→    ){
   428→        iterationCount += 1;
   429→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   430→        const chunkResults = await Promise.all(
   431→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   432→        );
   433→
   434→        chunkResults.forEach((result) => {
   435→            offsets[result.bucket] = result.nextOffset;
   436→            exhausted[result.bucket] = result.exhausted;
   437→
   438→            if(result.error){
   439→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   440→                unavailableBuckets.push(result.bucket);
   441→                return;
   442→            }
   443→
   444→            pending.push(...result.items);
   445→        });
   446→
   447→        if(chunkResults.every((result) => result.items.length === 0)){
   448→            break;
   449→        }
   450→    }
   451→
   452→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   453→
   454→    const selectedItems = pending.slice(0, parsedLimit);
   455→    const remainingItems = pending.slice(parsedLimit);
   456→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   457→
   458→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   459→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   460→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   461→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   462→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   463→        hydratedSelectedItems.push(...hydratedFallbackItems);
   464→    }
   465→
   466→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   467→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   468→
   469→    const nextCursor = hasMore
   470→        ? encodeMediaCursor({
   471→            offsets: offsets,
   472→            exhausted: exhausted,
   473→            pending: remainingItems
   474→        })
   475→        : null;
   476→
   477→    return {
   478→        data: hydratedSelectedItems,
   479→        hasMore: hasMore,
   480→        nextCursor: nextCursor,
   481→        unavailableBuckets: dedupedUnavailableBuckets
   482→    };
   483→}

  1418→export const searchJournalsService = async(query, limit, userId) => {
  1419→    const normalizedQuery = normalizeSearchQuery(query);
  1420→    if(!normalizedQuery || normalizedQuery.length < SEARCH_QUERY_MIN_LENGTH){
  1421→        throw {status: 400, error: `query should be at least ${SEARCH_QUERY_MIN_LENGTH} characters`};
  1422→    }
  1423→
  1424→    if(isNaN(limit) || limit < 1 || limit > SEARCH_LIMIT_MAX){
  1425→        throw {status: 400, error: `limit should be an integer between 1 and ${SEARCH_LIMIT_MAX}`};
  1426→    }
  1427→
  1428→    const parsedLimit = parseInt(limit);
  1429→    const fetchLimit = Math.min(parsedLimit * 3, 60);
  1430→
  1431→    const selectColumns = JOURNAL_METADATA_WITH_COUNTS_SELECT;
  1432→
  1433→    // Prefer semantic retrieval. If pgvector RPC is unavailable or errors,
  1434→    // fallback to keyword search so endpoint still returns useful data.
  1435→    try {
  1436→        const queryEmbedding = await GenerateEmbeddings(normalizedQuery, '');
  1437→
  1438→        if(Array.isArray(queryEmbedding) && queryEmbedding.length > 0){
  1439→            const {data: matches, error: matchError} = await supabase.rpc('match_public_journals', {
  1440→                query_embedding: queryEmbedding,
  1441→                match_count: fetchLimit,
  1442→                similarity_threshold: 0.35
  1443→            });
  1444→
  1445→            if(matchError){
  1446→                console.error('semantic search rpc error:', matchError.message);
  1447→            } else if(Array.isArray(matches) && matches.length > 0){
  1448→                const matchIds = matches.map((row) => row.id);
  1449→                const similarityMap = new Map(matches.map((row) => [row.id, row.similarity]));
  1450→
  1451→                const {data: journals, error: errorJournals} = await supabase
  1452→                    .from('journals')
  1453→                    .select(selectColumns)
  1454→                    .in('id', matchIds)
  1455→                    .eq('privacy', 'public');

  1458→                    console.error('supabase error while fetching semantic search journals:', errorJournals.message);

  1460→                }
  1461→
  1462→                const journalById = new Map((journals || []).map((journal) => [journal.id, journal]));
  1463→                const orderedSemantic = matchIds
  1464→                    .map((id) => journalById.get(id))
  1465→                    .filter(Boolean)
  1466→                    .map((journal) => ({
  1467→                        ...journal,
  1468→                        similarity: similarityMap.get(journal.id) || null
  1469→                    }));
  1470→
  1471→                const withInteraction = await attachUserInteractionFlags(orderedSemantic, userId);
  1472→                const hasMore = withInteraction.length > parsedLimit;
  1473→
  1474→                const optimizedData = decorateJournalCollection(withInteraction, 'card');
  1475→
  1476→                return {
  1477→                    data: hasMore ? optimizedData.slice(0, parsedLimit) : optimizedData,
  1478→                    hasMore: hasMore,
  1479→                    mode: 'semantic'
  1480→                };
  1481→            }
  1482→        }
  1483→    } catch (error) {
  1484→        console.error('semantic embedding/search fallback error:', error?.message || error);
  1485→    }
  1486→
  1487→    const escapedQuery = normalizedQuery.replace(/[%_]/g, (match) => `\\${match}`);
  1488→
  1489→    const [titleResult, contentResult] = await Promise.all([
  1490→        supabase

  1495→            .order('created_at', {ascending: false})
