   195→const parseMediaTimestamp = (item = {}) => {
   196→    const fileName = item?.name || '';
   197→    const timestampPrefix = fileName.match(/^(\d+)_/);
   198→
   199→    if(timestampPrefix?.[1]){
   200→        const parsedPrefix = Number(timestampPrefix[1]);
   201→        if(!Number.isNaN(parsedPrefix)){
   202→            return parsedPrefix;
   203→        }
   204→    }
   205→
   206→    const parsedDate = Date.parse(item?.created_at || item?.updated_at || '');
   207→    return Number.isNaN(parsedDate) ? 0 : parsedDate;

   210→const createInitialMediaCursorState = () => ({
   211→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   212→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   213→    pending: []
   214→});
   215→
   216→const parseMediaCursor = (cursor) => {
   217→    if(!cursor){
   218→        return createInitialMediaCursorState();
   219→    }
   220→
   221→    try {
   222→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   223→        const parsed = JSON.parse(decoded);
   224→        const initial = createInitialMediaCursorState();
   225→
   226→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   227→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   228→                const value = Number(parsed.offsets[bucket]);
   229→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   230→            });
   231→        }
   232→
   233→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   234→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   235→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   236→            });
   237→        }
   238→
   239→        if(Array.isArray(parsed?.pending)){
   240→            initial.pending = parsed.pending
   241→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   242→                .map((item) => ({
   243→                    bucket: item.bucket,
   244→                    path: item.path,
   245→                    name: item.name,
   246→                    createdAt: item.createdAt || null,
   247→                    timestamp: Number(item.timestamp) || 0
   248→                }));
   249→        }
   250→
   251→        return initial;
   252→    } catch {
   253→        return createInitialMediaCursorState();
   254→    }
   255→};

   262→const parseCanvasGalleryCursor = (cursor, expectedSort = 'hottest') => {
   263→    const fallback = {
   264→        offset: 0,
   265→        sort: expectedSort
   266→    };
   267→
   268→    if(!cursor){
   269→        return fallback;
   270→    }
   271→
   272→    try {
   273→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   274→        const parsed = JSON.parse(decoded);
   275→        const parsedSort = typeof parsed?.sort === 'string' && parsed.sort.toLowerCase().trim() === 'newest'
   276→            ? 'newest'
   277→            : 'hottest';
   278→
   279→        if(parsedSort !== expectedSort){
   280→            return fallback;
   281→        }
   282→
   283→        const parsedOffset = Number(parsed?.offset);
   284→        const normalizedOffset = Number.isNaN(parsedOffset) || parsedOffset < 0
   285→            ? 0
   286→            : Math.floor(parsedOffset);
   287→
   288→        return {
   289→            offset: normalizedOffset,
   290→            sort: parsedSort
   291→        };
   292→    } catch {
   293→        return fallback;
   294→    }
   295→};

   297→const encodeCanvasGalleryCursor = (offset, sort) => {
   298→    const payload = JSON.stringify({
   299→        offset: offset,
   300→        sort: sort
   301→    });
   302→    return Buffer.from(payload, 'utf8').toString('base64');

   316→        .from(bucket)

   318→
   319→    if(!signedUrlError && signedUrlData?.signedUrl){

   321→    }
   322→
   323→    return null;

   326→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   327→    const folderPath = `user_id_${userId}`;
   328→
   329→    const {data: bucketItems, error: bucketError} = await supabase.storage
   330→        .from(bucket)
   331→        .list(folderPath, {
   332→            limit: limit,
   333→            offset: offset,
   334→            sortBy: {column: 'name', order: 'desc'}
   335→        });
   336→
   337→    if(bucketError){
   338→        return {
   339→            bucket: bucket,
   340→            nextOffset: offset,
   341→            exhausted: true,
   342→            items: [],
   343→            error: bucketError.message
   344→        };
   345→    }
   346→
   347→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   348→    const normalizedItems = listedItems
   349→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name))
   350→        .map((item) => {
   351→            const path = `${folderPath}/${item.name}`;
   352→            const createdAt = item.created_at || item.updated_at || null;
   353→
   354→            return {
   355→                bucket: bucket,
   356→                path: path,
   357→                name: item.name,
   358→                createdAt: createdAt,
   359→                timestamp: parseMediaTimestamp(item)
   360→            };
   361→        });
   362→
   363→    return {
   364→        bucket: bucket,
   365→        nextOffset: offset + listedItems.length,
   366→        exhausted: listedItems.length < limit,
   367→        items: normalizedItems,
   368→        error: null
   369→    };
   370→};
   371→
   372→const hydrateMediaItems = async(items) => {
   373→    if(!Array.isArray(items) || items.length === 0){
   374→        return [];
   375→    }
   376→
   377→    const hydrated = await Promise.all(items.map(async(item) => {
   378→        const url = await resolveMediaUrl(item.bucket, item.path);
   379→        if(!url){
   380→            return null;
   381→        }
   382→
   383→        return {
   384→            id: `${item.bucket}-${item.path}`,
   385→            bucket: item.bucket,
   386→            path: item.path,
   387→            name: item.name,
   388→            createdAt: item.createdAt || null,
   389→            timestamp: item.timestamp || 0,
   390→            url: url
   391→        };
   392→    }));
   393→
   394→    return hydrated.filter(Boolean);
   395→};

   397→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   398→    if(!userId){
   399→        throw {status: 400, error: 'userId is undefined'};
   400→    }
   401→
   402→    const parsedLimit = Number(limit);
   403→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   404→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   405→    }
   406→
   407→    const cursorState = parseMediaCursor(cursor);
   408→    const offsets = {...cursorState.offsets};
   409→    const exhausted = {...cursorState.exhausted};
   410→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   411→    const unavailableBuckets = [];
   412→    const maxIterations = 6;
   413→    let iterationCount = 0;
   414→
   415→    while(
   416→        pending.length < parsedLimit &&
   417→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   418→        iterationCount < maxIterations
   419→    ){
   420→        iterationCount += 1;
   421→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   422→        const chunkResults = await Promise.all(
   423→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   424→        );
   425→
   426→        chunkResults.forEach((result) => {
   427→            offsets[result.bucket] = result.nextOffset;
   428→            exhausted[result.bucket] = result.exhausted;
   429→
   430→            if(result.error){
   431→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   432→                unavailableBuckets.push(result.bucket);
   433→                return;
   434→            }
   435→
   436→            pending.push(...result.items);
   437→        });
   438→
   439→        if(chunkResults.every((result) => result.items.length === 0)){
   440→            break;
   441→        }
   442→    }
   443→
   444→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   445→
   446→    const selectedItems = pending.slice(0, parsedLimit);
   447→    const remainingItems = pending.slice(parsedLimit);
   448→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   449→
   450→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   451→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   452→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   453→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   454→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   455→        hydratedSelectedItems.push(...hydratedFallbackItems);
   456→    }
   457→
   458→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   459→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   460→
   461→    const nextCursor = hasMore
   462→        ? encodeMediaCursor({
   463→            offsets: offsets,
   464→            exhausted: exhausted,
   465→            pending: remainingItems
   466→        })
   467→        : null;
   468→
   469→    return {
   470→        data: hydratedSelectedItems,
   471→        hasMore: hasMore,
   472→        nextCursor: nextCursor,
   473→        unavailableBuckets: dedupedUnavailableBuckets
   474→    };
   475→}

   568→        data: slicedData,

   574→export const getFollowingFeedService = async(limit, userId, before) => {
   575→    if (!userId) {
   576→        throw { status: 401, error: 'authentication required' };
   577→    }
   578→    if (isNaN(limit) || limit > 20 || limit < 1) {
   579→        throw { status: 400, error: 'limit should be integer, not below 1 and higher than 20' };
   580→    }
   581→
   582→    const parsedLimit = parseInt(limit);
   583→
   584→    // Step 1: Fetch journals via RPC (subquery stays inside Postgres — scales to 100k+ followings)
   585→    const { data: rpcData, error } = await supabase.rpc('get_following_feed', {
   586→        p_user_id: userId,
   587→        p_limit: parsedLimit + 1,
   588→        p_before: before || null
   589→    });
   590→
   591→    if (error) {
   592→        console.error('supabase error fetching following feed:', error.message);
   593→        throw { status: 500, error: 'supabase error fetching following feed' };
   594→    }
   595→
   596→    if (!rpcData || rpcData.length === 0) {
   597→        return { data: [], hasMore: false };
   598→    }
   599→
   600→    // Reshape RPC rows to match PostgREST format the client expects
   601→    const data = rpcData.map(row => ({
   602→        id: row.id,
   603→        user_id: row.user_id,
   604→        title: row.title,
   605→        content: row.content,

  1321→    }

  1329→    // fallback to keyword search so endpoint still returns useful data.
