    33→const extractBearerToken = (authHeader = "") => {
    34→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    35→    if (!trimmed) return "";
    36→
    37→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    38→    if (bearerMatch?.[1]) {
    39→        return bearerMatch[1].trim();
    40→    }
    41→
    42→    return trimmed;
    43→};
    44→
    45→const resolveAuthUser = async (token) => {
    46→    if (!token) {
    47→        return { user: null, error: null };
    48→    }
    49→
    50→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    51→    if (authError || !authData?.user?.id) {
    52→        return { user: null, error: authError || new Error("missing user id") };
    53→    }
    54→
    55→    return { user: authData.user, error: null };
    56→};
    57→
    58→const requireAuth = async (req, res, next) => {
    59→    const token = extractBearerToken(req.headers?.authorization);
    60→    if (!token) {
    61→        return res.status(401).json({ error: 'not authorized' });
    62→    }
    63→
    64→    const { user, error } = await resolveAuthUser(token);
    65→    if (error || !user?.id) {
    66→        console.error('auth middleware error:', error?.message || 'missing user id');
    67→        return res.status(401).json({ error: 'not authorized' });
    68→    }
    69→
    70→    req.userId = user.id;
    71→    req.authUser = user;
    72→    return next();
    73→};

    75→const optionalAuth = async (req, _res, next) => {

    77→    if (!token) {
    78→        return next();
    79→    }
    80→
    81→    const { user } = await resolveAuthUser(token);
    82→    if (user?.id) {
    83→        req.userId = user.id;
    84→        req.authUser = user;
    85→    }
    86→
    87→    return next();
    88→};

   197→export const imageUploader = async(file, userId, bucket) =>{
   198→    if(!file){
   199→        throw new Error('no file received');
   200→    }
   201→
   202→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   203→    if(!sourceBuffer){
   204→        throw new Error('invalid file payload');
   205→    }
   206→
   207→    let img_buffer = null
   208→    let img_url = null
   209→    img_buffer = await sharp(sourceBuffer)
   210→    .webp({quality: 80})
   211→    .toBuffer();
   212→
   213→    const folderName = `user_id_${userId}`;
   214→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   215→    const filePath = `${folderName}/${fileName}`;
   216→
   217→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   218→    .from(bucket)
   219→    .upload(filePath, img_buffer, {
   220→        contentType: 'image/webp',
   221→        cacheControl: '31536000',
   222→        upsert: true
   223→    })
   224→    if(errorUploadImage){
   225→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   226→        throw new Error('error uploading image into supabase bucket');
   227→    }
   228→    const {data: data_url} = supabase.storage
   229→    .from(bucket)
   230→    .getPublicUrl(filePath)
   231→    
   232→    if(data_url){
   233→        img_url = data_url.publicUrl;
   234→        return img_url;
   235→    } else {
   236→        throw new Error('Error uploading the image');
   237→    }
   238→}

   381→router.get('/getNotifications', requireAuth, async(req, res) =>{
   382→    const {before} = req.query;
   383→    const userId = req.userId;
   384→    const parsedLimit = parseLimitWithinRange(
   385→        req.query.limit,
   386→        NOTIFICATION_LIMIT_MIN,
   387→        NOTIFICATION_LIMIT_MAX,
   388→        NOTIFICATION_DEFAULT_LIMIT
   389→    );
   390→
   391→    if(parsedLimit == null){
   392→        return res.status(400).json({error: `limit must be between ${NOTIFICATION_LIMIT_MIN} and ${NOTIFICATION_LIMIT_MAX}`});
   393→    }
   394→
   395→    // Fetch a generous amount from both tables, then merge and paginate
   396→    const fetchLimit = parsedLimit + 1;
   397→
   398→    let journalQuery = supabase
   399→    .from('notifications')
   400→    .select(NOTIFICATION_JOURNAL_SELECT)
   401→    .eq('receiver_id', userId)
   402→    .order('created_at', {ascending: false})
   403→    .order('id', {ascending: false})
   404→    .limit(fetchLimit)
   405→
   406→    let opinionQuery = supabase
   407→    .from('notification_opinions')
   408→    .select(NOTIFICATION_OPINION_SELECT)
   409→    .eq('receiver_id', userId)
   410→    .order('created_at', {ascending: false})
   411→    .order('id', {ascending: false})
   412→    .limit(fetchLimit)
   413→
   414→    if(before){
   415→        journalQuery = journalQuery.lt('created_at', before);
   416→        opinionQuery = opinionQuery.lt('created_at', before);
   417→    }
   418→
   419→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   420→
   421→    if(journalResult.error){
   422→        console.error('error while fetching data from notification table:', journalResult.error);
   423→        return res.status(500).json({error: 'error fetching data from notification table'});
   424→    }
   425→    if(opinionResult.error){
   426→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   427→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   428→    }
   429→
   430→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   431→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   432→
   433→    // Merge and sort by created_at descending
   434→    const merged = [...journalNotifs, ...opinionNotifs]
   435→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
   436→
   437→    // Likes/bookmarks lookup only for journal notifications
   438→    const journalIds = [...new Set(
   439→        merged
   440→            .filter((n) => n.source === 'journal')
   441→            .map((n) => n.journal_id)
   442→            .filter(Boolean)
   443→    )];
   444→
   445→    let hasLikedResult = [];
   446→    let hasBookMarkedResult = [];
   447→    if(journalIds.length > 0){
   448→        const [hasLiked, hasBookMarked] = await Promise.all([
   449→            supabase.from('likes').select('journal_id').in('journal_id', journalIds).eq('user_id', userId),
   450→            supabase.from('bookmarks').select('journal_id').in('journal_id', journalIds).eq('user_id', userId)
   451→        ]);

   453→        if(hasLiked.error || hasBookMarked.error){

   457→        hasBookMarkedResult = hasBookMarked.data || [];
   458→    }
   459→
   460→    const userHasLikedSet = new Set(hasLikedResult.map((likes) => likes.journal_id));
   461→    const userHasBookmarkedSet = new Set(hasBookMarkedResult.map((bookmarks) => bookmarks.journal_id));
   462→
   463→    const formatted = merged.map((notif) => ({
   464→        ...notif,
   465→        hasLiked: notif.source === 'journal' ? userHasLikedSet.has(notif.journal_id) : false,
   466→        hasBookMarked: notif.source === 'journal' ? userHasBookmarkedSet.has(notif.journal_id) : false
   467→    }))
   468→
   469→    const hasMore = formatted.length > parsedLimit;
   470→    const slicedData = hasMore ? formatted.slice(0, parsedLimit) : formatted;
   471→
   472→    return res.status(200).json(
   473→        {
   474→            hasMore: hasMore,
