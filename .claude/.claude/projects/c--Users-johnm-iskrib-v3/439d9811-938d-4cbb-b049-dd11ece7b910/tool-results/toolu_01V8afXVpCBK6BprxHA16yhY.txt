   397→const createInitialMediaCursorState = () => ({
   398→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   399→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   400→    pending: []
   401→});
   402→
   403→const parseMediaCursor = (cursor) => {
   404→    if(!cursor){
   405→        return createInitialMediaCursorState();
   406→    }
   407→
   408→    try {
   409→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   410→        const parsed = JSON.parse(decoded);
   411→        const initial = createInitialMediaCursorState();
   412→
   413→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   414→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   415→                const value = Number(parsed.offsets[bucket]);
   416→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   417→            });
   418→        }
   419→
   420→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   421→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   422→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   423→            });
   424→        }
   425→
   426→        if(Array.isArray(parsed?.pending)){
   427→            initial.pending = parsed.pending
   428→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   429→                .map((item) => ({
   430→                    bucket: item.bucket,
   431→                    path: item.path,
   432→                    name: item.name,
   433→                    createdAt: item.createdAt || null,
   434→                    timestamp: Number(item.timestamp) || 0
   435→                }));
   436→        }
   437→
   438→        return initial;
   439→    } catch {
   440→        return createInitialMediaCursorState();
   441→    }
   442→};

   449→const parseCanvasGalleryCursor = (cursor, expectedSort = 'hottest') => {
   450→    const fallback = {
   451→        offset: 0,
   452→        sort: expectedSort
   453→    };
   454→
   455→    if(!cursor){
   456→        return fallback;
   457→    }
   458→
   459→    try {
   460→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   461→        const parsed = JSON.parse(decoded);
   462→        const parsedSort = typeof parsed?.sort === 'string' && parsed.sort.toLowerCase().trim() === 'newest'
   463→            ? 'newest'
   464→            : 'hottest';
   465→
   466→        if(parsedSort !== expectedSort){
   467→            return fallback;
   468→        }
   469→
   470→        const parsedOffset = Number(parsed?.offset);
   471→        const normalizedOffset = Number.isNaN(parsedOffset) || parsedOffset < 0
   472→            ? 0
   473→            : Math.floor(parsedOffset);
   474→
   475→        return {
   476→            offset: normalizedOffset,
   477→            sort: parsedSort
   478→        };
   479→    } catch {
   480→        return fallback;
   481→    }
   482→};

   523→
   524→    if(bucketError){

   526→            bucket: bucket,
   527→            nextOffset: offset,
   528→            exhausted: true,

   530→            error: bucketError.message
   531→        };
   532→    }
   533→
   534→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   535→    const normalizedItems = listedItems
   536→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name))
   537→        .map((item) => {
   538→            const path = `${folderPath}/${item.name}`;

   540→

   542→                bucket: bucket,
   543→                path: path,
   544→                name: item.name,
   545→                createdAt: createdAt,
   546→                timestamp: parseMediaTimestamp(item)
   547→            };
   548→        });
   549→
   550→    return {
   551→        bucket: bucket,
   552→        nextOffset: offset + listedItems.length,
   553→        exhausted: listedItems.length < limit,
   554→        items: normalizedItems,
   555→        error: null

   584→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   585→    if(!userId){
   586→        throw {status: 400, error: 'userId is undefined'};
   587→    }
   588→
   589→    const parsedLimit = Number(limit);
   590→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   591→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   592→    }
   593→
   594→    const cursorState = parseMediaCursor(cursor);
   595→    const offsets = {...cursorState.offsets};
   596→    const exhausted = {...cursorState.exhausted};
   597→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   598→    const unavailableBuckets = [];
   599→    const maxIterations = 6;
   600→    let iterationCount = 0;
   601→
   602→    while(
   603→        pending.length < parsedLimit &&
   604→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   605→        iterationCount < maxIterations
   606→    ){
   607→        iterationCount += 1;
   608→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   609→        const chunkResults = await Promise.all(
   610→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   611→        );
   612→
   613→        chunkResults.forEach((result) => {
   614→            offsets[result.bucket] = result.nextOffset;
   615→            exhausted[result.bucket] = result.exhausted;
   616→
   617→            if(result.error){
   618→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   619→                unavailableBuckets.push(result.bucket);
   620→                return;
   621→            }
   622→
   623→            pending.push(...result.items);
   624→        });
   625→
   626→        if(chunkResults.every((result) => result.items.length === 0)){
   627→            break;
   628→        }
   629→    }
   630→
   631→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   632→
   633→    const selectedItems = pending.slice(0, parsedLimit);
   634→    const remainingItems = pending.slice(parsedLimit);
   635→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   636→
   637→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   638→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   639→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   640→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   641→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   642→        hydratedSelectedItems.push(...hydratedFallbackItems);
   643→    }
   644→
   645→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   646→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   647→
   648→    const nextCursor = hasMore
   649→        ? encodeMediaCursor({
   650→            offsets: offsets,
   651→            exhausted: exhausted,
   652→            pending: remainingItems
   653→        })
   654→        : null;
   655→
   656→    return {
   657→        data: hydratedSelectedItems,
   658→        hasMore: hasMore,
   659→        nextCursor: nextCursor,
   660→        unavailableBuckets: dedupedUnavailableBuckets
   661→    };
   662→}

  1454→export const searchJournalsService = async(query, limit, userId) => {
  1455→    const normalizedQuery = normalizeSearchQuery(query);
  1456→    if(!normalizedQuery || normalizedQuery.length < SEARCH_QUERY_MIN_LENGTH){
  1457→        throw {status: 400, error: `query should be at least ${SEARCH_QUERY_MIN_LENGTH} characters`};
  1458→    }
  1459→
  1460→    if(isNaN(limit) || limit < 1 || limit > SEARCH_LIMIT_MAX){
  1461→        throw {status: 400, error: `limit should be an integer between 1 and ${SEARCH_LIMIT_MAX}`};
  1462→    }
  1463→
  1464→    const parsedLimit = parseInt(limit);
  1465→    const fetchLimit = Math.min(parsedLimit * 3, 60);
  1466→
  1467→    const selectColumns = JOURNAL_WITH_COUNTS_SELECT;
  1468→
  1469→    // Prefer semantic retrieval. If pgvector RPC is unavailable or errors,
  1470→    // fallback to keyword search so endpoint still returns useful data.
  1471→    try {
  1472→        const queryEmbedding = await GenerateEmbeddings(normalizedQuery, '');
  1473→
  1474→        if(Array.isArray(queryEmbedding) && queryEmbedding.length > 0){
  1475→            const {data: matches, error: matchError} = await supabase.rpc('match_public_journals', {
  1476→                query_embedding: queryEmbedding,
  1477→                match_count: fetchLimit,
  1478→                similarity_threshold: 0.35
  1479→            });
  1480→
  1481→            if(matchError){
  1482→                console.error('semantic search rpc error:', matchError.message);
  1483→            } else if(Array.isArray(matches) && matches.length > 0){
  1484→                const matchIds = matches.map((row) => row.id);
  1485→                const similarityMap = new Map(matches.map((row) => [row.id, row.similarity]));
  1486→
  1487→                const {data: journals, error: errorJournals} = await supabase
  1488→                    .from('journals')
  1489→                    .select(selectColumns)
  1490→                    .in('id', matchIds)
  1491→                    .eq('privacy', 'public');
  1492→
  1493→                if(errorJournals){
  1494→                    console.error('supabase error while fetching semantic search journals:', errorJournals.message);
  1495→                    throw {status: 500, error: 'supabase error while fetching semantic search journals'};
  1496→                }
  1497→
  1498→                const journalById = new Map((journals || []).map((journal) => [journal.id, journal]));
  1499→                const orderedSemantic = matchIds
  1500→                    .map((id) => journalById.get(id))
  1501→                    .filter(Boolean)
  1502→                    .map((journal) => ({
  1503→                        ...journal,
  1504→                        similarity: similarityMap.get(journal.id) || null
  1505→                    }));
  1506→
  1507→                const withInteraction = await attachUserInteractionFlags(orderedSemantic, userId);
  1508→                const hasMore = withInteraction.length > parsedLimit;
  1509→
  1510→                return {
  1511→                    data: hasMore ? withInteraction.slice(0, parsedLimit) : withInteraction,
  1512→                    hasMore: hasMore,
  1513→                    mode: 'semantic'
  1514→                };
  1515→            }
  1516→        }
  1517→    } catch (error) {
  1518→        console.error('semantic embedding/search fallback error:', error?.message || error);
  1519→    }
  1520→
  1521→    const escapedQuery = normalizedQuery.replace(/[%_]/g, (match) => `\\${match}`);
  1522→
  1523→    const [titleResult, contentResult] = await Promise.all([
  1524→        supabase
  1525→            .from('journals')
  1526→            .select(selectColumns)
  1527→            .eq('privacy', 'public')
  1528→            .ilike('title', `%${escapedQuery}%`)
  1529→            .order('created_at', {ascending: false})
  1530→            .order('id', {ascending: false})
  1531→            .limit(parsedLimit + 1),
  1532→        supabase
  1533→            .from('journals')
  1534→            .select(selectColumns)
  1535→            .eq('privacy', 'public')
  1536→            .ilike('content', `%${escapedQuery}%`)
  1537→            .order('created_at', {ascending: false})
  1538→            .order('id', {ascending: false})
  1539→            .limit(parsedLimit + 1)
  1540→    ]);
  1541→
  1542→    if(titleResult.error || contentResult.error){
  1543→        console.error('keyword search error:', titleResult.error?.message || contentResult.error?.message);
  1544→        throw {status: 500, error: 'supabase error while searching journals'};
  1545→    }
  1546→
  1547→    const keywordMap = new Map();
  1548→    (titleResult.data || []).forEach((journal) => {
  1549→        keywordMap.set(journal.id, journal);
  1550→    });
  1551→    (contentResult.data || []).forEach((journal) => {
  1552→        if(!keywordMap.has(journal.id)){
  1553→            keywordMap.set(journal.id, journal);
  1554→        }
  1555→    });
  1556→
  1557→    const keywordJournals = [...keywordMap.values()];
  1558→
  1559→    const withInteraction = await attachUserInteractionFlags(keywordJournals || [], userId);
  1560→    const hasMore = withInteraction.length > parsedLimit;
  1561→
  1562→    return {
  1563→        data: hasMore ? withInteraction.slice(0, parsedLimit) : withInteraction,
  1564→        hasMore: hasMore,
  1565→        mode: 'keyword'
