    31→const extractBearerToken = (authHeader = "") => {
    32→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    33→    if (!trimmed) return "";
    34→
    35→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    36→    if (bearerMatch?.[1]) {
    37→        return bearerMatch[1].trim();
    38→    }
    39→
    40→    return trimmed;
    41→};
    42→
    43→const resolveAuthUser = async (token) => {
    44→    if (!token) {
    45→        return { user: null, error: null };
    46→    }
    47→
    48→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    49→    if (authError || !authData?.user?.id) {
    50→        return { user: null, error: authError || new Error("missing user id") };
    51→    }
    52→
    53→    return { user: authData.user, error: null };
    54→};
    55→
    56→const requireAuth = async (req, res, next) => {
    57→    const token = extractBearerToken(req.headers?.authorization);
    58→    if (!token) {
    59→        return res.status(401).json({ error: 'not authorized' });
    60→    }
    61→
    62→    const { user, error } = await resolveAuthUser(token);
    63→    if (error || !user?.id) {
    64→        console.error('auth middleware error:', error?.message || 'missing user id');
    65→        return res.status(401).json({ error: 'not authorized' });
    66→    }
    67→
    68→    req.userId = user.id;
    69→    req.authUser = user;
    70→    return next();
    71→};
    72→
    73→const optionalAuth = async (req, _res, next) => {
    74→    const token = extractBearerToken(req.headers?.authorization);
    75→    if (!token) {
    76→        return next();
    77→    }
    78→
    79→    const { user } = await resolveAuthUser(token);
    80→    if (user?.id) {
    81→        req.userId = user.id;
    82→        req.authUser = user;
    83→    }
    84→
    85→    return next();
    86→};

   193→export const imageUploader = async(file, userId, bucket) =>{
   194→    if(!file){
   195→        throw new Error('no file received');
   196→    }
   197→
   198→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   199→    if(!sourceBuffer){
   200→        throw new Error('invalid file payload');
   201→    }
   202→
   203→    let img_buffer = null
   204→    let img_url = null
   205→    img_buffer = await sharp(sourceBuffer)
   206→    .webp({quality: 80})
   207→    .toBuffer();
   208→
   209→    const folderName = `user_id_${userId}`;
   210→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   211→    const filePath = `${folderName}/${fileName}`;
   212→
   213→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   214→    .from(bucket)
   215→    .upload(filePath, img_buffer, {
   216→        contentType: 'image/webp',
   217→        cacheControl: '31536000',
   218→        upsert: true
   219→    })
   220→    if(errorUploadImage){
   221→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   222→        throw new Error('error uploading image into supabase bucket');
   223→    }
   224→    const {data: data_url} = supabase.storage
   225→    .from(bucket)
   226→    .getPublicUrl(filePath)
   227→    
   228→    if(data_url){
   229→        img_url = data_url.publicUrl;
   230→        return img_url;
   231→    } else {
   232→        throw new Error('Error uploading the image');
   233→    }
   234→}

   370→    const parsedLimit = parseLimitWithinRange(

   381→    // Fetch a generous amount from both tables, then merge and paginate
   382→    const fetchLimit = parsedLimit + 1;
   383→
   384→    let journalQuery = supabase
   385→    .from('notifications')
   386→    .select(NOTIFICATION_JOURNAL_SELECT)
   387→    .eq('receiver_id', userId)
   388→    .order('created_at', {ascending: false})

   393→    .from('notification_opinions')
   394→    .select(NOTIFICATION_OPINION_SELECT)
   395→    .eq('receiver_id', userId)
   396→    .order('created_at', {ascending: false})
   397→    .order('id', {ascending: false})
   398→    .limit(fetchLimit)
   399→
   400→    if(before){
   401→        journalQuery = journalQuery.lt('created_at', before);
   402→        opinionQuery = opinionQuery.lt('created_at', before);
   403→    }
   404→
   405→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   406→
   407→    if(journalResult.error){
   408→        console.error('error while fetching data from notification table:', journalResult.error);
   409→        return res.status(500).json({error: 'error fetching data from notification table'});
   410→    }
   411→    if(opinionResult.error){
   412→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   413→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   414→    }

   418→

   448→
   449→    const formatted = merged.map((notif) => ({
   450→        ...notif,

   453→    }))
