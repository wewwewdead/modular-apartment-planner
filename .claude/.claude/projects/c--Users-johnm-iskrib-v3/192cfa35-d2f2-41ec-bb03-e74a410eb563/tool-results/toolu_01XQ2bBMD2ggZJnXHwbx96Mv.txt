   593→   593→            break;
   594→   594→        }

   605→   605→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;

   619→   619→        })
   620→   620→        : null;
   621→   621→
   622→   622→    return {
   623→   623→        data: hydratedSelectedItems,
   624→   624→        hasMore: hasMore,
   625→   625→        nextCursor: nextCursor,
   626→   626→        unavailableBuckets: dedupedUnavailableBuckets
