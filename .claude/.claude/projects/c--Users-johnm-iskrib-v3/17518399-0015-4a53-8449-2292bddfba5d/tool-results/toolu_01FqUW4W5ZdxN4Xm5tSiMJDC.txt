   334→   334→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   335→   335→    const folderPath = `user_id_${userId}`;

   340→   340→        .list(folderPath, {
   341→   341→            limit: listLimit,
   342→   342→            offset: offset,
   343→   343→            sortBy: {column: 'name', order: 'desc'}
   344→   344→        });
   345→   345→
   346→   346→    if(bucketError){
   347→   347→        return {
   348→   348→            bucket: bucket,
   349→   349→            nextOffset: offset,
   350→   350→            exhausted: true,

   372→   372→    return {
   373→   373→        bucket: bucket,
   374→   374→        nextOffset: offset + listedItems.length,
   375→   375→        exhausted: listedItems.length < listLimit,
   376→   376→        items: normalizedItems,

   412→   412→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   413→   413→    if(!userId){
   414→   414→        throw {status: 400, error: 'userId is undefined'};
   415→   415→    }
   416→   416→
   417→   417→    const parsedLimit = Number(limit);
   418→   418→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   419→   419→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   420→   420→    }
   421→   421→
   422→   422→    const cursorState = parseMediaCursor(cursor);
   423→   423→    const offsets = {...cursorState.offsets};
   424→   424→    const exhausted = {...cursorState.exhausted};
   425→   425→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   426→   426→    const unavailableBuckets = [];
   427→   427→    const maxIterations = 6;
   428→   428→    let iterationCount = 0;
   429→   429→
   430→   430→    while(
   431→   431→        pending.length < parsedLimit &&
   432→   432→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   433→   433→        iterationCount < maxIterations
   434→   434→    ){
   435→   435→        iterationCount += 1;
   436→   436→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   437→   437→        const chunkResults = await Promise.all(
   438→   438→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   439→   439→        );
   440→   440→
   441→   441→        chunkResults.forEach((result) => {
   442→   442→            offsets[result.bucket] = result.nextOffset;
   443→   443→            exhausted[result.bucket] = result.exhausted;
   444→   444→
   445→   445→            if(result.error){
   446→   446→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   447→   447→                unavailableBuckets.push(result.bucket);
   448→   448→                return;
   449→   449→            }
   450→   450→
   451→   451→            pending.push(...result.items);
   452→   452→        });
   453→   453→
   454→   454→        if(chunkResults.every((result) => result.items.length === 0)){
   455→   455→            break;
   456→   456→        }
   457→   457→    }
   458→   458→
   459→   459→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   460→   460→
   461→   461→    const selectedItems = pending.slice(0, parsedLimit);
   462→   462→    const remainingItems = pending.slice(parsedLimit);
   463→   463→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   464→   464→
   465→   465→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   466→   466→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   467→   467→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   468→   468→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   469→   469→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   470→   470→        hydratedSelectedItems.push(...hydratedFallbackItems);
   471→   471→    }
   472→   472→
   473→   473→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   474→   474→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   475→   475→
   476→   476→    const nextCursor = hasMore
   477→   477→        ? encodeMediaCursor({
   478→   478→            offsets: offsets,
   479→   479→            exhausted: exhausted,
   480→   480→            pending: remainingItems
   481→   481→        })
   482→   482→        : null;
   483→   483→
   484→   484→    return {
   485→   485→        data: hydratedSelectedItems,
   486→   486→        hasMore: hasMore,
   487→   487→        nextCursor: nextCursor,
   488→   488→        unavailableBuckets: dedupedUnavailableBuckets
   489→   489→    };
   490→   490→}
