   246→const parseMediaTimestamp = (item = {}) => {

   248→    const timestampPrefix = fileName.match(/^(\d+)_/);

   261→const createInitialMediaCursorState = () => ({
   262→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   263→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   264→    pending: []
   265→});
   266→
   267→const parseMediaCursor = (cursor) => {
   268→    if(!cursor){
   269→        return createInitialMediaCursorState();
   270→    }
   271→
   272→    try {
   273→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   274→        const parsed = JSON.parse(decoded);
   275→        const initial = createInitialMediaCursorState();
   276→
   277→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   278→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   279→                const value = Number(parsed.offsets[bucket]);
   280→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   281→            });
   282→        }
   283→
   284→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   285→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   286→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   287→            });
   288→        }
   289→
   290→        if(Array.isArray(parsed?.pending)){
   291→            initial.pending = parsed.pending
   292→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   293→                .map((item) => ({
   294→                    bucket: item.bucket,
   295→                    path: item.path,
   296→                    name: item.name,
   297→                    createdAt: item.createdAt || null,
   298→                    timestamp: Number(item.timestamp) || 0
   299→                }));
   300→        }
   301→
   302→        return initial;
   303→    } catch {
   304→        return createInitialMediaCursorState();
   305→    }
   306→};

   314→    const {data: publicUrlData} = supabase.storage
   315→        .from(bucket)

   317→

   324→        .from(bucket)
   325→        .createSignedUrl(path, 60 * 60);
   326→
   327→    if(!signedUrlError && signedUrlData?.signedUrl){

   334→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   335→    const folderPath = `user_id_${userId}`;
   336→    const listLimit = Math.max(limit * 4, limit);
   337→
   338→    const {data: bucketItems, error: bucketError} = await supabase.storage
   339→        .from(bucket)
   340→        .list(folderPath, {
   341→            limit: listLimit,
   342→            offset: offset,
   343→            sortBy: {column: 'name', order: 'desc'}
   344→        });
   345→
   346→    if(bucketError){
   347→        return {
   348→            bucket: bucket,
   349→            nextOffset: offset,
   350→            exhausted: true,
   351→            items: [],
   352→            error: bucketError.message
   353→        };
   354→    }
   355→
   356→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   357→    const normalizedItems = listedItems
   358→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name) && isPrimaryListableFileName(item.name))
   359→        .map((item) => {
   360→            const path = `${folderPath}/${item.name}`;
   361→            const createdAt = item.created_at || item.updated_at || null;
   362→
   363→            return {
   364→                bucket: bucket,
   365→                path: path,
   366→                name: item.name,
   367→                createdAt: createdAt,
   368→                timestamp: parseMediaTimestamp(item)
   369→            };
   370→        });
   371→
   372→    return {
   373→        bucket: bucket,
   374→        nextOffset: offset + listedItems.length,
   375→        exhausted: listedItems.length < listLimit,
   376→        items: normalizedItems,
   377→        error: null
   378→    };
   379→};

   395→            id: `${item.bucket}-${mediaPayload?.path || item.path}`,
   396→            bucket: item.bucket,
   397→            path: mediaPayload?.path || item.path,
   398→            name: item.name,
   399→            createdAt: item.createdAt || null,
   400→            timestamp: item.timestamp || 0,
   401→            url: mediaPayload?.preferred_url || url,
   402→            thumbnailUrl: mediaPayload?.thumbnail_url || url,
   403→            cardUrl: mediaPayload?.card_url || url,
   404→            detailUrl: mediaPayload?.detail_url || url,
   405→            originalUrl: mediaPayload?.original_url || url
   406→        };
   407→    }));
   408→
   409→    return hydrated.filter(Boolean);

   412→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   413→    if(!userId){
   414→        throw {status: 400, error: 'userId is undefined'};
   415→    }
   416→
   417→    const parsedLimit = Number(limit);
   418→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   419→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   420→    }
   421→
   422→    const cursorState = parseMediaCursor(cursor);
   423→    const offsets = {...cursorState.offsets};
   424→    const exhausted = {...cursorState.exhausted};
   425→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   426→    const unavailableBuckets = [];
   427→    const maxIterations = 6;
   428→    let iterationCount = 0;
   429→
   430→    while(
   431→        pending.length < parsedLimit &&
   432→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   433→        iterationCount < maxIterations
   434→    ){
   435→        iterationCount += 1;
   436→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   437→        const chunkResults = await Promise.all(
   438→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   439→        );
   440→
   441→        chunkResults.forEach((result) => {
   442→            offsets[result.bucket] = result.nextOffset;
   443→            exhausted[result.bucket] = result.exhausted;
   444→
   445→            if(result.error){
   446→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   447→                unavailableBuckets.push(result.bucket);
   448→                return;
   449→            }
   450→
   451→            pending.push(...result.items);
   452→        });
   453→
   454→        if(chunkResults.every((result) => result.items.length === 0)){
   455→            break;
   456→        }
   457→    }
   458→
   459→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   460→
   461→    const selectedItems = pending.slice(0, parsedLimit);
   462→    const remainingItems = pending.slice(parsedLimit);
   463→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   464→
   465→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   466→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   467→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   468→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   469→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   470→        hydratedSelectedItems.push(...hydratedFallbackItems);
   471→    }
   472→
   473→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   474→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   475→
   476→    const nextCursor = hasMore
   477→        ? encodeMediaCursor({
   478→            offsets: offsets,
   479→            exhausted: exhausted,
   480→            pending: remainingItems
   481→        })
   482→        : null;
   483→
   484→    return {
   485→        data: hydratedSelectedItems,
   486→        hasMore: hasMore,
   487→        nextCursor: nextCursor,
   488→        unavailableBuckets: dedupedUnavailableBuckets
   489→    };
   490→}

   522→            data: [],
   523→            hasMore: false
   524→        }
   525→        return journalData
   526→    }
   527→
   528→    // If userId is provided, fetch personalization (likes/bookmarks/reactions)
   529→    // Otherwise return journals with has_liked/has_bookmarked/user_reaction defaulting to false/null
   530→    let userHasLikedSet = new Set();
   531→    let userHasBookmarkedSet = new Set();
   532→    let userReactionMap = new Map();

   536→

   538→            supabase
   539→            .from('likes')

   541→            .in('journal_id', journalIds)
   542→            .eq('user_id', userId),
   543→
   544→            supabase
   545→            .from('bookmarks')
   546→            .select('journal_id')
   547→            .in('journal_id', journalIds)
   548→            .eq('user_id', userId),
   549→
   550→            supabase
   551→            .from('reactions')
   552→            .select('journal_id, reaction_type')
   553→            .in('journal_id', journalIds)
   554→            .eq('user_id', userId)
   555→        ]);
   556→
   557→        const {data: userLikesResult, error: errorUserLikeResult} = userLikes;
   558→        const {data: userBookmarksResult, error: errorUserBookmarksResult} = userBookmarks;
   559→        const {data: userReactionsResult} = userReactions;
   560→
   561→        if(errorUserLikeResult || errorUserBookmarksResult) {
   562→            console.error('supabase error:', errorUserLikeResult?.message || errorUserBookmarksResult?.message);
   563→            throw {status: 500, error: 'supabase error on fetching user likes or user bookmarks'}
   564→        }
   565→
   566→        userHasLikedSet = new Set(userLikesResult?.map((j) => j.journal_id) || []);
   567→        userHasBookmarkedSet = new Set(userBookmarksResult?.map((j) => j.journal_id)|| []);
   568→        userReactionMap = new Map((userReactionsResult || []).map((r) => [r.journal_id, r.reaction_type]));
   569→    }
   570→
   571→    await attachRepostSources(data);
   572→
   573→    const formattedData = data?.map((journal) => ({
   574→        ...journal,
   575→        has_liked: userHasLikedSet.has(journal.id),
   576→        has_bookmarked: userHasBookmarkedSet.has(journal.id),
   577→        user_reaction: userReactionMap.get(journal.id) || null
   578→    }));
   579→    const optimizedData = decorateJournalCollection(formattedData, JOURNAL_MEDIA_USAGE.feed);
   580→
   581→    const hasMore = data?.length > parsedLimit;
   582→    const slicedData = hasMore ? optimizedData.slice(0, parsedLimit) : optimizedData;
   583→
   584→    const journalData = {
   585→        data: slicedData,
   586→        hasMore: hasMore
   587→    }
   588→    return journalData;
   589→}

  1429→export const searchJournalsService = async(query, limit, userId) => {
  1430→    const normalizedQuery = normalizeSearchQuery(query);
  1431→    if(!normalizedQuery || normalizedQuery.length < SEARCH_QUERY_MIN_LENGTH){
  1432→        throw {status: 400, error: `query should be at least ${SEARCH_QUERY_MIN_LENGTH} characters`};
  1433→    }
  1434→
  1435→    if(isNaN(limit) || limit < 1 || limit > SEARCH_LIMIT_MAX){
  1436→        throw {status: 400, error: `limit should be an integer between 1 and ${SEARCH_LIMIT_MAX}`};
  1437→    }
  1438→
  1439→    const parsedLimit = parseInt(limit);
  1440→    const fetchLimit = Math.min(parsedLimit * 3, 60);
  1441→
  1442→    const selectColumns = JOURNAL_METADATA_WITH_COUNTS_SELECT;
  1443→
  1444→    // Prefer semantic retrieval. If pgvector RPC is unavailable or errors,
  1445→    // fallback to keyword search so endpoint still returns useful data.
  1446→    try {
  1447→        const queryEmbedding = await GenerateEmbeddings(normalizedQuery, '');
  1448→
  1449→        if(Array.isArray(queryEmbedding) && queryEmbedding.length > 0){
  1450→            const {data: matches, error: matchError} = await supabase.rpc('match_public_journals', {
  1451→                query_embedding: queryEmbedding,
  1452→                match_count: fetchLimit,
  1453→                similarity_threshold: 0.35
  1454→            });
  1455→
  1456→            if(matchError){
  1457→                console.error('semantic search rpc error:', matchError.message);
  1458→            } else if(Array.isArray(matches) && matches.length > 0){
  1459→                const matchIds = matches.map((row) => row.id);
  1460→                const similarityMap = new Map(matches.map((row) => [row.id, row.similarity]));
  1461→
  1462→                const {data: journals, error: errorJournals} = await supabase
  1463→                    .from('journals')
  1464→                    .select(selectColumns)
  1465→                    .in('id', matchIds)
  1466→                    .eq('privacy', 'public');
  1467→
  1468→                if(errorJournals){
  1469→                    console.error('supabase error while fetching semantic search journals:', errorJournals.message);
  1470→                    throw {status: 500, error: 'supabase error while fetching semantic search journals'};
  1471→                }
  1472→
  1473→                const journalById = new Map((journals || []).map((journal) => [journal.id, journal]));
  1474→                const orderedSemantic = matchIds
  1475→                    .map((id) => journalById.get(id))
  1476→                    .filter(Boolean)
  1477→                    .map((journal) => ({
  1478→                        ...journal,
  1479→                        similarity: similarityMap.get(journal.id) || null
  1480→                    }));
  1481→
  1482→                const withInteraction = await attachUserInteractionFlags(orderedSemantic, userId);
  1483→                const hasMore = withInteraction.length > parsedLimit;
  1484→
  1485→                const optimizedData = decorateJournalCollection(withInteraction, JOURNAL_MEDIA_USAGE.feed);
  1486→
  1487→                return {
  1488→                    data: hasMore ? optimizedData.slice(0, parsedLimit) : optimizedData,
  1489→                    hasMore: hasMore,
  1490→                    mode: 'semantic'
  1491→                };
  1492→            }
  1493→        }
  1494→    } catch (error) {
  1495→        console.error('semantic embedding/search fallback error:', error?.message || error);

  1497→
  1498→    const escapedQuery = normalizedQuery.replace(/[%_]/g, (match) => `\\${match}`);

  1501→        supabase

  1505→            .eq('status', 'published')

  1507→            .order('created_at', {ascending: false})
  1508→            .order('id', {ascending: false})
  1509→            .limit(parsedLimit + 1),
  1510→        supabase
  1511→            .from('journals')
  1512→            .select(selectColumns)
  1513→            .eq('privacy', 'public')
  1514→            .eq('status', 'published')

  1516→            .order('created_at', {ascending: false})
  1517→            .order('id', {ascending: false})

  1520→
  1521→    if(titleResult.error || contentResult.error){
