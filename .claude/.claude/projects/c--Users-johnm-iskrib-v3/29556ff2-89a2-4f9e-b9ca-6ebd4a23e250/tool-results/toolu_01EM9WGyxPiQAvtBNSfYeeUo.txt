   187→const hasSupportedMediaExtension = (fileName = '') => {

   192→const parseMediaTimestamp = (item = {}) => {
   193→    const fileName = item?.name || '';
   194→    const timestampPrefix = fileName.match(/^(\d+)_/);
   195→
   196→    if(timestampPrefix?.[1]){
   197→        const parsedPrefix = Number(timestampPrefix[1]);
   198→        if(!Number.isNaN(parsedPrefix)){
   199→            return parsedPrefix;
   200→        }
   201→    }
   202→
   203→    const parsedDate = Date.parse(item?.created_at || item?.updated_at || '');
   204→    return Number.isNaN(parsedDate) ? 0 : parsedDate;
   205→};

   207→const createInitialMediaCursorState = () => ({
   208→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   209→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   210→    pending: []
   211→});
   212→
   213→const parseMediaCursor = (cursor) => {
   214→    if(!cursor){
   215→        return createInitialMediaCursorState();
   216→    }
   217→
   218→    try {
   219→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   220→        const parsed = JSON.parse(decoded);
   221→        const initial = createInitialMediaCursorState();
   222→
   223→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   224→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   225→                const value = Number(parsed.offsets[bucket]);
   226→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   227→            });
   228→        }
   229→
   230→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   231→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   232→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   233→            });
   234→        }
   235→
   236→        if(Array.isArray(parsed?.pending)){
   237→            initial.pending = parsed.pending
   238→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   239→                .map((item) => ({
   240→                    bucket: item.bucket,
   241→                    path: item.path,
   242→                    name: item.name,
   243→                    createdAt: item.createdAt || null,
   244→                    timestamp: Number(item.timestamp) || 0
   245→                }));
   246→        }
   247→
   248→        return initial;
   249→    } catch {
   250→        return createInitialMediaCursorState();
   251→    }
   252→};

   254→const encodeMediaCursor = (state) => {
   255→    const payload = JSON.stringify(state);
   256→    return Buffer.from(payload, 'utf8').toString('base64');

   259→const resolveMediaUrl = async(bucket, path) => {
   260→    const {data: publicUrlData} = supabase.storage
   261→        .from(bucket)
   262→        .getPublicUrl(path);
   263→
   264→    const publicUrl = publicUrlData?.publicUrl || null;
   265→    if(publicUrl){
   266→        return publicUrl;
   267→    }
   268→
   269→    const {data: signedUrlData, error: signedUrlError} = await supabase.storage
   270→        .from(bucket)
   271→        .createSignedUrl(path, 60 * 60);
   272→
   273→    if(!signedUrlError && signedUrlData?.signedUrl){
   274→        return signedUrlData.signedUrl;
   275→    }
   276→
   277→    return null;
   278→};

   280→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   281→    const folderPath = `user_id_${userId}`;
   282→
   283→    const {data: bucketItems, error: bucketError} = await supabase.storage
   284→        .from(bucket)
   285→        .list(folderPath, {
   286→            limit: limit,
   287→            offset: offset,
   288→            sortBy: {column: 'name', order: 'desc'}
   289→        });
   290→
   291→    if(bucketError){
   292→        return {
   293→            bucket: bucket,
   294→            nextOffset: offset,
   295→            exhausted: true,
   296→            items: [],
   297→            error: bucketError.message
   298→        };
   299→    }
   300→
   301→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   302→    const normalizedItems = listedItems
   303→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name))
   304→        .map((item) => {
   305→            const path = `${folderPath}/${item.name}`;
   306→            const createdAt = item.created_at || item.updated_at || null;
   307→
   308→            return {
   309→                bucket: bucket,
   310→                path: path,
   311→                name: item.name,
   312→                createdAt: createdAt,
   313→                timestamp: parseMediaTimestamp(item)
   314→            };
   315→        });
   316→
   317→    return {
   318→        bucket: bucket,
   319→        nextOffset: offset + listedItems.length,
   320→        exhausted: listedItems.length < limit,
   321→        items: normalizedItems,
   322→        error: null
   323→    };
   324→};
   325→
   326→const hydrateMediaItems = async(items) => {
   327→    if(!Array.isArray(items) || items.length === 0){
   328→        return [];
   329→    }
   330→
   331→    const hydrated = await Promise.all(items.map(async(item) => {
   332→        const url = await resolveMediaUrl(item.bucket, item.path);
   333→        if(!url){
   334→            return null;
   335→        }
   336→
   337→        return {
   338→            id: `${item.bucket}-${item.path}`,
   339→            bucket: item.bucket,
   340→            path: item.path,
   341→            name: item.name,
   342→            createdAt: item.createdAt || null,
   343→            timestamp: item.timestamp || 0,
   344→            url: url
   345→        };
   346→    }));
   347→
   348→    return hydrated.filter(Boolean);
   349→};

   351→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   352→    if(!userId){
   353→        throw {status: 400, error: 'userId is undefined'};
   354→    }
   355→
   356→    const parsedLimit = Number(limit);
   357→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   358→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   359→    }
   360→
   361→    const cursorState = parseMediaCursor(cursor);
   362→    const offsets = {...cursorState.offsets};
   363→    const exhausted = {...cursorState.exhausted};
   364→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   365→    const unavailableBuckets = [];
   366→    const maxIterations = 6;
   367→    let iterationCount = 0;
   368→
   369→    while(
   370→        pending.length < parsedLimit &&
   371→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   372→        iterationCount < maxIterations
   373→    ){
   374→        iterationCount += 1;
   375→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   376→        const chunkResults = await Promise.all(
   377→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   378→        );
   379→
   380→        chunkResults.forEach((result) => {
   381→            offsets[result.bucket] = result.nextOffset;
   382→            exhausted[result.bucket] = result.exhausted;
   383→
   384→            if(result.error){
   385→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   386→                unavailableBuckets.push(result.bucket);
   387→                return;
   388→            }
   389→
   390→            pending.push(...result.items);
   391→        });
   392→
   393→        if(chunkResults.every((result) => result.items.length === 0)){
   394→            break;
   395→        }
   396→    }
   397→
   398→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   399→
   400→    const selectedItems = pending.slice(0, parsedLimit);
   401→    const remainingItems = pending.slice(parsedLimit);
   402→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   403→
   404→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   405→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   406→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   407→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   408→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   409→        hydratedSelectedItems.push(...hydratedFallbackItems);
   410→    }
   411→
   412→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   413→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   414→
   415→    const nextCursor = hasMore
   416→        ? encodeMediaCursor({
   417→            offsets: offsets,
   418→            exhausted: exhausted,
   419→            pending: remainingItems
   420→        })
   421→        : null;
   422→
   423→    return {
   424→        data: hydratedSelectedItems,
   425→        hasMore: hasMore,
   426→        nextCursor: nextCursor,
   427→        unavailableBuckets: dedupedUnavailableBuckets
   428→    };
   429→}
