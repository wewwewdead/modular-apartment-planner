   177→   177→const loadUniversePostsFallback = async(minX, maxX, minY, maxY, limit) => {

   197→   197→        .gte('universe_x', minX)
   198→   198→        .lte('universe_x', maxX)
   199→   199→        .gte('universe_y', minY)
   200→   200→        .lte('universe_y', maxY)
   201→   201→        .order('created_at', {ascending: false})
   202→   202→        .limit(limit);
   203→   203→

   205→   205→        {data: settledData, error: settledError},
   206→   206→        {data: unsettledData, error: unsettledError}

   208→   208→
   209→   209→    if(settledError || unsettledError){
   210→   210→        console.error(
   211→   211→            'supabase fallback error while fetching universe posts:',
   212→   212→            settledError?.message || unsettledError?.message
   213→   213→        );
   214→   214→        throw {status: 500, error: 'failed to fetch universe posts'};
   215→   215→    }
   216→   216→
   217→   217→    const mergedById = new Map();
   218→   218→    const mergedRows = [...(settledData || []), ...(unsettledData || [])];
   219→   219→    for (const row of mergedRows) {
   220→   220→        if(!row?.id || mergedById.has(row.id)){
   221→   221→            continue;
   222→   222→        }
   223→   223→
   224→   224→        mergedById.set(row.id, normalizeUniversePost(row));
   225→   225→        if(mergedById.size >= limit){
   226→   226→            break;
   227→   227→        }
   228→   228→    }
   229→   229→
   230→   230→    return Array.from(mergedById.values());

   450→   450→            ? 'newest'

   461→   461→
   462→   462→        return {
   463→   463→            offset: normalizedOffset,
   464→   464→            sort: parsedSort

   571→   571→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   572→   572→    if(!userId){
   573→   573→        throw {status: 400, error: 'userId is undefined'};
   574→   574→    }
   575→   575→
   576→   576→    const parsedLimit = Number(limit);
   577→   577→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   578→   578→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   579→   579→    }
   580→   580→
   581→   581→    const cursorState = parseMediaCursor(cursor);
   582→   582→    const offsets = {...cursorState.offsets};
   583→   583→    const exhausted = {...cursorState.exhausted};
   584→   584→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   585→   585→    const unavailableBuckets = [];
   586→   586→    const maxIterations = 6;
   587→   587→    let iterationCount = 0;
   588→   588→
   589→   589→    while(
   590→   590→        pending.length < parsedLimit &&
   591→   591→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   592→   592→        iterationCount < maxIterations
   593→   593→    ){
   594→   594→        iterationCount += 1;
   595→   595→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   596→   596→        const chunkResults = await Promise.all(
   597→   597→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   598→   598→        );
   599→   599→
   600→   600→        chunkResults.forEach((result) => {
   601→   601→            offsets[result.bucket] = result.nextOffset;
   602→   602→            exhausted[result.bucket] = result.exhausted;
   603→   603→
   604→   604→            if(result.error){
   605→   605→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   606→   606→                unavailableBuckets.push(result.bucket);
   607→   607→                return;
   608→   608→            }
   609→   609→
   610→   610→            pending.push(...result.items);
   611→   611→        });
   612→   612→
   613→   613→        if(chunkResults.every((result) => result.items.length === 0)){
   614→   614→            break;
   615→   615→        }
   616→   616→    }
   617→   617→
   618→   618→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   619→   619→
   620→   620→    const selectedItems = pending.slice(0, parsedLimit);
   621→   621→    const remainingItems = pending.slice(parsedLimit);
   622→   622→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   623→   623→
   624→   624→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   625→   625→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   626→   626→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   627→   627→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   628→   628→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   629→   629→        hydratedSelectedItems.push(...hydratedFallbackItems);
   630→   630→    }
   631→   631→
   632→   632→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   633→   633→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   634→   634→
   635→   635→    const nextCursor = hasMore
   636→   636→        ? encodeMediaCursor({
   637→   637→            offsets: offsets,
   638→   638→            exhausted: exhausted,
   639→   639→            pending: remainingItems
   640→   640→        })
   641→   641→        : null;
   642→   642→
   643→   643→    return {
   644→   644→        data: hydratedSelectedItems,
   645→   645→        hasMore: hasMore,
   646→   646→        nextCursor: nextCursor,
   647→   647→        unavailableBuckets: dedupedUnavailableBuckets
   648→   648→    };
   649→   649→}
