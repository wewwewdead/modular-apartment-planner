    28→    28→const extractBearerToken = (authHeader = "") => {
    29→    29→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    30→    30→    if (!trimmed) return "";
    31→    31→
    32→    32→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    33→    33→    if (bearerMatch?.[1]) {
    34→    34→        return bearerMatch[1].trim();
    35→    35→    }
    36→    36→
    37→    37→    return trimmed;
    38→    38→};

    40→    40→const resolveAuthUser = async (token) => {
    41→    41→    if (!token) {
    42→    42→        return { user: null, error: null };
    43→    43→    }
    44→    44→
    45→    45→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    46→    46→    if (authError || !authData?.user?.id) {
    47→    47→        return { user: null, error: authError || new Error("missing user id") };
    48→    48→    }
    49→    49→
    50→    50→    return { user: authData.user, error: null };
    51→    51→};

    53→    53→const requireAuth = async (req, res, next) => {
    54→    54→    const token = extractBearerToken(req.headers?.authorization);
    55→    55→    if (!token) {
    56→    56→        return res.status(401).json({ error: 'not authorized' });
    57→    57→    }
    58→    58→
    59→    59→    const { user, error } = await resolveAuthUser(token);
    60→    60→    if (error || !user?.id) {
    61→    61→        console.error('auth middleware error:', error?.message || 'missing user id');
    62→    62→        return res.status(401).json({ error: 'not authorized' });
    63→    63→    }
    64→    64→
    65→    65→    req.userId = user.id;
    66→    66→    req.authUser = user;
    67→    67→    return next();
    68→    68→};

   191→   191→export const imageUploader = async(file, userId, bucket) =>{
   192→   192→    if(!file){
   193→   193→        throw new Error('no file received');
   194→   194→    }
   195→   195→
   196→   196→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   197→   197→    if(!sourceBuffer){
   198→   198→        throw new Error('invalid file payload');
   199→   199→    }
   200→   200→
   201→   201→    let img_buffer = null
   202→   202→    let img_url = null
   203→   203→    img_buffer = await sharp(sourceBuffer)
   204→   204→    .webp({quality: 80})
   205→   205→    .toBuffer();
   206→   206→
   207→   207→    const folderName = `user_id_${userId}`;
   208→   208→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   209→   209→    const filePath = `${folderName}/${fileName}`;
   210→   210→
   211→   211→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   212→   212→    .from(bucket)
   213→   213→    .upload(filePath, img_buffer, {
   214→   214→        contentType: 'image/webp',
   215→   215→        cacheControl: '31536000',
   216→   216→        upsert: true
   217→   217→    })
   218→   218→    if(errorUploadImage){
   219→   219→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   220→   220→        throw new Error('error uploading image into supabase bucket');
   221→   221→    }
   222→   222→    const {data: data_url} = supabase.storage
   223→   223→    .from(bucket)
   224→   224→    .getPublicUrl(filePath)
   225→   225→    
   226→   226→    if(data_url){
   227→   227→        img_url = data_url.publicUrl;
   228→   228→        return img_url;
   229→   229→    } else {
   230→   230→        throw new Error('Error uploading the image');
   231→   231→    }
   232→   232→}
