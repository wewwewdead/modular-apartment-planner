    31→const extractBearerToken = (authHeader = "") => {
    32→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    33→    if (!trimmed) return "";
    34→
    35→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    36→    if (bearerMatch?.[1]) {
    37→        return bearerMatch[1].trim();
    38→    }
    39→
    40→    return trimmed;
    41→};
    42→
    43→const resolveAuthUser = async (token) => {
    44→    if (!token) {
    45→        return { user: null, error: null };
    46→    }
    47→
    48→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    49→    if (authError || !authData?.user?.id) {
    50→        return { user: null, error: authError || new Error("missing user id") };
    51→    }
    52→
    53→    return { user: authData.user, error: null };
    54→};
    55→
    56→const requireAuth = async (req, res, next) => {
    57→    const token = extractBearerToken(req.headers?.authorization);
    58→    if (!token) {
    59→        return res.status(401).json({ error: 'not authorized' });
    60→    }
    61→
    62→    const { user, error } = await resolveAuthUser(token);
    63→    if (error || !user?.id) {
    64→        console.error('auth middleware error:', error?.message || 'missing user id');
    65→        return res.status(401).json({ error: 'not authorized' });
    66→    }
    67→
    68→    req.userId = user.id;
    69→    req.authUser = user;
    70→    return next();
    71→};
    72→
    73→const optionalAuth = async (req, _res, next) => {
    74→    const token = extractBearerToken(req.headers?.authorization);
    75→    if (!token) {
    76→        return next();
    77→    }
    78→
    79→    const { user } = await resolveAuthUser(token);
    80→    if (user?.id) {
    81→        req.userId = user.id;
    82→        req.authUser = user;
    83→    }
    84→
    85→    return next();
    86→};

   170→export const imageUploader = async(file, userId, bucket) =>{
   171→    if(!file){
   172→        throw new Error('no file received');
   173→    }
   174→
   175→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   176→    if(!sourceBuffer){
   177→        throw new Error('invalid file payload');
   178→    }
   179→
   180→    let img_buffer = null
   181→    let img_url = null
   182→    img_buffer = await sharp(sourceBuffer)
   183→    .webp({quality: 80})
   184→    .toBuffer();
   185→
   186→    const folderName = `user_id_${userId}`;
   187→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   188→    const filePath = `${folderName}/${fileName}`;
   189→
   190→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   191→    .from(bucket)
   192→    .upload(filePath, img_buffer, {
   193→        contentType: 'image/webp',
   194→        cacheControl: '31536000',
   195→        upsert: true
   196→    })
   197→    if(errorUploadImage){
   198→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   199→        throw new Error('error uploading image into supabase bucket');
   200→    }
   201→    const {data: data_url} = supabase.storage
   202→    .from(bucket)
   203→    .getPublicUrl(filePath)
   204→    
   205→    if(data_url){
   206→        img_url = data_url.publicUrl;
   207→        return img_url;
   208→    } else {
   209→        throw new Error('Error uploading the image');
   210→    }
   211→}

   344→router.get('/getNotifications', requireAuth, async(req, res) =>{
   345→    const {before} = req.query;

   347→    const parsedLimit = parseLimitWithinRange(
   348→        req.query.limit,

   356→    }
   357→
   358→    // Fetch a generous amount from both tables, then merge and paginate
   359→    const fetchLimit = parsedLimit + 1;
   360→
   361→    let journalQuery = supabase
   362→    .from('notifications')
   363→    .select(NOTIFICATION_JOURNAL_SELECT)
   364→    .eq('receiver_id', userId)
   365→    .order('created_at', {ascending: false})

   369→    let opinionQuery = supabase
   370→    .from('notification_opinions')
   371→    .select(NOTIFICATION_OPINION_SELECT)
   372→    .eq('receiver_id', userId)
   373→    .order('created_at', {ascending: false})
   374→    .order('id', {ascending: false})
   375→    .limit(fetchLimit)
   376→
   377→    if(before){
   378→        journalQuery = journalQuery.lt('created_at', before);
   379→        opinionQuery = opinionQuery.lt('created_at', before);
   380→    }
   381→
   382→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   383→
   384→    if(journalResult.error){
   385→        console.error('error while fetching data from notification table:', journalResult.error);
   386→        return res.status(500).json({error: 'error fetching data from notification table'});
   387→    }
   388→    if(opinionResult.error){
   389→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   390→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   391→    }
   392→
   393→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   394→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   395→
   396→    // Merge and sort by created_at descending
   397→    const merged = [...journalNotifs, ...opinionNotifs]
   398→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
   399→
   400→    // Likes/bookmarks lookup only for journal notifications

   402→        merged
   403→            .filter((n) => n.source === 'journal')

   405→            .filter(Boolean)
   406→    )];
   407→
   408→    let hasLikedResult = [];

   412→            supabase.from('likes').select('journal_id').in('journal_id', journalIds).eq('user_id', userId),
   413→            supabase.from('bookmarks').select('journal_id').in('journal_id', journalIds).eq('user_id', userId)
   414→        ]);

   420→        hasBookMarkedResult = hasBookMarked.data || [];

   425→
   426→    const formatted = merged.map((notif) => ({
