    33→const extractBearerToken = (authHeader = "") => {
    34→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    35→    if (!trimmed) return "";
    36→
    37→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    38→    if (bearerMatch?.[1]) {
    39→        return bearerMatch[1].trim();
    40→    }
    41→
    42→    return trimmed;
    43→};
    44→
    45→const resolveAuthUser = async (token) => {
    46→    if (!token) {
    47→        return { user: null, error: null };
    48→    }
    49→
    50→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    51→    if (authError || !authData?.user?.id) {
    52→        return { user: null, error: authError || new Error("missing user id") };
    53→    }
    54→
    55→    return { user: authData.user, error: null };
    56→};
    57→
    58→const requireAuth = async (req, res, next) => {
    59→    const token = extractBearerToken(req.headers?.authorization);
    60→    if (!token) {
    61→        return res.status(401).json({ error: 'not authorized' });
    62→    }
    63→
    64→    const { user, error } = await resolveAuthUser(token);
    65→    if (error || !user?.id) {
    66→        console.error('auth middleware error:', error?.message || 'missing user id');
    67→        return res.status(401).json({ error: 'not authorized' });
    68→    }
    69→
    70→    req.userId = user.id;
    71→    req.authUser = user;
    72→    return next();
    73→};

    82→    if (user?.id) {
    83→        req.userId = user.id;
    84→        req.authUser = user;
    85→    }
    86→
    87→    return next();

   197→export const imageUploader = async(file, userId, bucket) =>{
   198→    if(!file){
   199→        throw new Error('no file received');
   200→    }
   201→
   202→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   203→    if(!sourceBuffer){
   204→        throw new Error('invalid file payload');
   205→    }
   206→
   207→    let img_buffer = null
   208→    let img_url = null
   209→    img_buffer = await sharp(sourceBuffer)
   210→    .webp({quality: 80})
   211→    .toBuffer();
   212→
   213→    const folderName = `user_id_${userId}`;
   214→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   215→    const filePath = `${folderName}/${fileName}`;
   216→
   217→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   218→    .from(bucket)
   219→    .upload(filePath, img_buffer, {
   220→        contentType: 'image/webp',
   221→        cacheControl: '31536000',
   222→        upsert: true
   223→    })
   224→    if(errorUploadImage){
   225→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   226→        throw new Error('error uploading image into supabase bucket');
   227→    }
   228→    const {data: data_url} = supabase.storage
   229→    .from(bucket)
   230→    .getPublicUrl(filePath)
   231→    
   232→    if(data_url){
   233→        img_url = data_url.publicUrl;
   234→        return img_url;
   235→    } else {
   236→        throw new Error('Error uploading the image');
   237→    }
   238→}

   382→router.get('/getNotifications', requireAuth, async(req, res) =>{
   383→    const {before} = req.query;
   384→    const userId = req.userId;
   385→    const parsedLimit = parseLimitWithinRange(
   386→        req.query.limit,
   387→        NOTIFICATION_LIMIT_MIN,
   388→        NOTIFICATION_LIMIT_MAX,
   389→        NOTIFICATION_DEFAULT_LIMIT
   390→    );
   391→
   392→    if(parsedLimit == null){
   393→        return res.status(400).json({error: `limit must be between ${NOTIFICATION_LIMIT_MIN} and ${NOTIFICATION_LIMIT_MAX}`});
   394→    }
   395→
   396→    // Fetch a generous amount from both tables, then merge and paginate
   397→    const fetchLimit = parsedLimit + 1;
   398→
   399→    let journalQuery = supabase
   400→    .from('notifications')
   401→    .select(NOTIFICATION_JOURNAL_SELECT)
   402→    .eq('receiver_id', userId)
   403→    .order('created_at', {ascending: false})
   404→    .order('id', {ascending: false})
   405→    .limit(fetchLimit)
   406→
   407→    let opinionQuery = supabase
   408→    .from('notification_opinions')
   409→    .select(NOTIFICATION_OPINION_SELECT)
   410→    .eq('receiver_id', userId)
   411→    .order('created_at', {ascending: false})
   412→    .order('id', {ascending: false})
   413→    .limit(fetchLimit)
   414→
   415→    if(before){
   416→        journalQuery = journalQuery.lt('created_at', before);
   417→        opinionQuery = opinionQuery.lt('created_at', before);
   418→    }
   419→
   420→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   421→
   422→    if(journalResult.error){
   423→        console.error('error while fetching data from notification table:', journalResult.error);
   424→        return res.status(500).json({error: 'error fetching data from notification table'});
   425→    }
   426→    if(opinionResult.error){
   427→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   428→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   429→    }
   430→
   431→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   432→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   433→
   434→    // Merge and sort by created_at descending
   435→    const merged = [...journalNotifs, ...opinionNotifs]
   436→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
   437→
   438→    // Likes/bookmarks lookup only for journal notifications
   439→    const journalIds = [...new Set(
   440→        merged
   441→            .filter((n) => n.source === 'journal')
   442→            .map((n) => n.journal_id)
   443→            .filter(Boolean)
   444→    )];
   445→
   446→    let hasLikedResult = [];
   447→    let hasBookMarkedResult = [];
   448→    if(journalIds.length > 0){
   449→        const [hasLiked, hasBookMarked] = await Promise.all([
   450→            supabase.from('likes').select('journal_id').in('journal_id', journalIds).eq('user_id', userId),
   451→            supabase.from('bookmarks').select('journal_id').in('journal_id', journalIds).eq('user_id', userId)
   452→        ]);
   453→
   454→        if(hasLiked.error || hasBookMarked.error){
   455→            console.error('supabase error while fetching data:', hasLiked.error || hasBookMarked.error);
   456→        }
   457→        hasLikedResult = hasLiked.data || [];
   458→        hasBookMarkedResult = hasBookMarked.data || [];
   459→    }
   460→
   461→    const userHasLikedSet = new Set(hasLikedResult.map((likes) => likes.journal_id));
   462→    const userHasBookmarkedSet = new Set(hasBookMarkedResult.map((bookmarks) => bookmarks.journal_id));
   463→
   464→    const formatted = merged.map((notif) => ({
   465→        ...notif,
   466→        hasLiked: notif.source === 'journal' ? userHasLikedSet.has(notif.journal_id) : false,
   467→        hasBookMarked: notif.source === 'journal' ? userHasBookmarkedSet.has(notif.journal_id) : false
   468→    }))
   469→
   470→    const hasMore = formatted.length > parsedLimit;
   471→    const slicedData = hasMore ? formatted.slice(0, parsedLimit) : formatted;
   472→
   473→    return res.status(200).json(
   474→        {
   475→            hasMore: hasMore,
   476→            data: slicedData,
   477→        }
   478→    )
