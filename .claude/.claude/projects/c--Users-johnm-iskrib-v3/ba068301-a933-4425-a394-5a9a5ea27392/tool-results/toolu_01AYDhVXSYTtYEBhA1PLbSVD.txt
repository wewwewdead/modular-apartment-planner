   189→const hasSupportedMediaExtension = (fileName = '') => {

   194→const parseMediaTimestamp = (item = {}) => {
   195→    const fileName = item?.name || '';
   196→    const timestampPrefix = fileName.match(/^(\d+)_/);
   197→
   198→    if(timestampPrefix?.[1]){
   199→        const parsedPrefix = Number(timestampPrefix[1]);
   200→        if(!Number.isNaN(parsedPrefix)){
   201→            return parsedPrefix;
   202→        }
   203→    }
   204→
   205→    const parsedDate = Date.parse(item?.created_at || item?.updated_at || '');
   206→    return Number.isNaN(parsedDate) ? 0 : parsedDate;
   207→};
   208→
   209→const createInitialMediaCursorState = () => ({
   210→    offsets: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, 0])),
   211→    exhausted: Object.fromEntries(PROFILE_MEDIA_BUCKETS.map((bucket) => [bucket, false])),
   212→    pending: []
   213→});
   214→
   215→const parseMediaCursor = (cursor) => {
   216→    if(!cursor){
   217→        return createInitialMediaCursorState();
   218→    }
   219→
   220→    try {
   221→        const decoded = Buffer.from(cursor, 'base64').toString('utf8');
   222→        const parsed = JSON.parse(decoded);
   223→        const initial = createInitialMediaCursorState();
   224→
   225→        if(parsed?.offsets && typeof parsed.offsets === 'object'){
   226→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   227→                const value = Number(parsed.offsets[bucket]);
   228→                initial.offsets[bucket] = Number.isNaN(value) || value < 0 ? 0 : value;
   229→            });
   230→        }
   231→
   232→        if(parsed?.exhausted && typeof parsed.exhausted === 'object'){
   233→            PROFILE_MEDIA_BUCKETS.forEach((bucket) => {
   234→                initial.exhausted[bucket] = Boolean(parsed.exhausted[bucket]);
   235→            });
   236→        }
   237→
   238→        if(Array.isArray(parsed?.pending)){
   239→            initial.pending = parsed.pending
   240→                .filter((item) => PROFILE_MEDIA_BUCKETS.includes(item?.bucket) && typeof item?.path === 'string' && typeof item?.name === 'string')
   241→                .map((item) => ({
   242→                    bucket: item.bucket,
   243→                    path: item.path,
   244→                    name: item.name,
   245→                    createdAt: item.createdAt || null,
   246→                    timestamp: Number(item.timestamp) || 0
   247→                }));
   248→        }
   249→
   250→        return initial;
   251→    } catch {
   252→        return createInitialMediaCursorState();
   253→    }
   254→};
   255→
   256→const encodeMediaCursor = (state) => {
   257→    const payload = JSON.stringify(state);
   258→    return Buffer.from(payload, 'utf8').toString('base64');

   261→const resolveMediaUrl = async(bucket, path) => {
   262→    const {data: publicUrlData} = supabase.storage
   263→        .from(bucket)
   264→        .getPublicUrl(path);
   265→
   266→    const publicUrl = publicUrlData?.publicUrl || null;
   267→    if(publicUrl){
   268→        return publicUrl;
   269→    }
   270→
   271→    const {data: signedUrlData, error: signedUrlError} = await supabase.storage
   272→        .from(bucket)
   273→        .createSignedUrl(path, 60 * 60);
   274→
   275→    if(!signedUrlError && signedUrlData?.signedUrl){
   276→        return signedUrlData.signedUrl;
   277→    }
   278→
   279→    return null;
   280→};

   282→const listMediaChunkForBucket = async(bucket, userId, offset, limit) => {
   283→    const folderPath = `user_id_${userId}`;
   284→
   285→    const {data: bucketItems, error: bucketError} = await supabase.storage
   286→        .from(bucket)
   287→        .list(folderPath, {
   288→            limit: limit,
   289→            offset: offset,
   290→            sortBy: {column: 'name', order: 'desc'}
   291→        });
   292→
   293→    if(bucketError){
   294→        return {
   295→            bucket: bucket,
   296→            nextOffset: offset,
   297→            exhausted: true,
   298→            items: [],
   299→            error: bucketError.message
   300→        };
   301→    }
   302→
   303→    const listedItems = Array.isArray(bucketItems) ? bucketItems : [];
   304→    const normalizedItems = listedItems
   305→        .filter((item) => item?.name && hasSupportedMediaExtension(item.name))
   306→        .map((item) => {
   307→            const path = `${folderPath}/${item.name}`;
   308→            const createdAt = item.created_at || item.updated_at || null;
   309→
   310→            return {
   311→                bucket: bucket,
   312→                path: path,
   313→                name: item.name,
   314→                createdAt: createdAt,
   315→                timestamp: parseMediaTimestamp(item)
   316→            };
   317→        });
   318→
   319→    return {
   320→        bucket: bucket,
   321→        nextOffset: offset + listedItems.length,
   322→        exhausted: listedItems.length < limit,
   323→        items: normalizedItems,
   324→        error: null
   325→    };
   326→};
   327→
   328→const hydrateMediaItems = async(items) => {
   329→    if(!Array.isArray(items) || items.length === 0){
   330→        return [];
   331→    }
   332→
   333→    const hydrated = await Promise.all(items.map(async(item) => {
   334→        const url = await resolveMediaUrl(item.bucket, item.path);
   335→        if(!url){
   336→            return null;
   337→        }
   338→
   339→        return {
   340→            id: `${item.bucket}-${item.path}`,
   341→            bucket: item.bucket,
   342→            path: item.path,
   343→            name: item.name,
   344→            createdAt: item.createdAt || null,
   345→            timestamp: item.timestamp || 0,
   346→            url: url
   347→        };
   348→    }));
   349→
   350→    return hydrated.filter(Boolean);
   351→};

   353→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   354→    if(!userId){
   355→        throw {status: 400, error: 'userId is undefined'};
   356→    }
   357→
   358→    const parsedLimit = Number(limit);
   359→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   360→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   361→    }
   362→
   363→    const cursorState = parseMediaCursor(cursor);
   364→    const offsets = {...cursorState.offsets};
   365→    const exhausted = {...cursorState.exhausted};
   366→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   367→    const unavailableBuckets = [];
   368→    const maxIterations = 6;
   369→    let iterationCount = 0;
   370→
   371→    while(
   372→        pending.length < parsedLimit &&
   373→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   374→        iterationCount < maxIterations
   375→    ){
   376→        iterationCount += 1;
   377→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   378→        const chunkResults = await Promise.all(
   379→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   380→        );
   381→
   382→        chunkResults.forEach((result) => {
   383→            offsets[result.bucket] = result.nextOffset;
   384→            exhausted[result.bucket] = result.exhausted;
   385→
   386→            if(result.error){
   387→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   388→                unavailableBuckets.push(result.bucket);
   389→                return;
   390→            }
   391→
   392→            pending.push(...result.items);
   393→        });
   394→
   395→        if(chunkResults.every((result) => result.items.length === 0)){
   396→            break;
   397→        }
   398→    }
   399→
   400→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   401→
   402→    const selectedItems = pending.slice(0, parsedLimit);
   403→    const remainingItems = pending.slice(parsedLimit);
   404→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   405→
   406→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   407→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   408→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   409→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   410→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   411→        hydratedSelectedItems.push(...hydratedFallbackItems);
   412→    }
   413→
   414→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   415→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   416→
   417→    const nextCursor = hasMore
   418→        ? encodeMediaCursor({
   419→            offsets: offsets,
   420→            exhausted: exhausted,
   421→            pending: remainingItems
   422→        })
   423→        : null;
   424→
   425→    return {
   426→        data: hydratedSelectedItems,
   427→        hasMore: hasMore,
   428→        nextCursor: nextCursor,
   429→        unavailableBuckets: dedupedUnavailableBuckets
   430→    };
   431→}
