   293→   293→            bucket: bucket,

   315→   315→        });
   316→   316→
   317→   317→    return {
   318→   318→        bucket: bucket,
   319→   319→        nextOffset: offset + listedItems.length,
   320→   320→        exhausted: listedItems.length < limit,
   321→   321→        items: normalizedItems,
   322→   322→        error: null
   323→   323→    };

   351→   351→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   352→   352→    if(!userId){
   353→   353→        throw {status: 400, error: 'userId is undefined'};
   354→   354→    }
   355→   355→
   356→   356→    const parsedLimit = Number(limit);
   357→   357→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   358→   358→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   359→   359→    }
   360→   360→
   361→   361→    const cursorState = parseMediaCursor(cursor);
   362→   362→    const offsets = {...cursorState.offsets};
   363→   363→    const exhausted = {...cursorState.exhausted};
   364→   364→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   365→   365→    const unavailableBuckets = [];
   366→   366→    const maxIterations = 6;
   367→   367→    let iterationCount = 0;
   368→   368→
   369→   369→    while(
   370→   370→        pending.length < parsedLimit &&
   371→   371→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   372→   372→        iterationCount < maxIterations
   373→   373→    ){
   374→   374→        iterationCount += 1;
   375→   375→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   376→   376→        const chunkResults = await Promise.all(
   377→   377→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   378→   378→        );
   379→   379→
   380→   380→        chunkResults.forEach((result) => {
   381→   381→            offsets[result.bucket] = result.nextOffset;
   382→   382→            exhausted[result.bucket] = result.exhausted;
   383→   383→
   384→   384→            if(result.error){
   385→   385→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   386→   386→                unavailableBuckets.push(result.bucket);
   387→   387→                return;
   388→   388→            }
   389→   389→
   390→   390→            pending.push(...result.items);
   391→   391→        });
   392→   392→
   393→   393→        if(chunkResults.every((result) => result.items.length === 0)){
   394→   394→            break;
   395→   395→        }
   396→   396→    }
   397→   397→
   398→   398→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   399→   399→
   400→   400→    const selectedItems = pending.slice(0, parsedLimit);
   401→   401→    const remainingItems = pending.slice(parsedLimit);
   402→   402→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   403→   403→
   404→   404→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   405→   405→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   406→   406→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   407→   407→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   408→   408→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   409→   409→        hydratedSelectedItems.push(...hydratedFallbackItems);
   410→   410→    }
   411→   411→
   412→   412→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   413→   413→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   414→   414→
   415→   415→    const nextCursor = hasMore
   416→   416→        ? encodeMediaCursor({
   417→   417→            offsets: offsets,
   418→   418→            exhausted: exhausted,
   419→   419→            pending: remainingItems
   420→   420→        })
   421→   421→        : null;
   422→   422→
   423→   423→    return {
   424→   424→        data: hydratedSelectedItems,
   425→   425→        hasMore: hasMore,
   426→   426→        nextCursor: nextCursor,
   427→   427→        unavailableBuckets: dedupedUnavailableBuckets
   428→   428→    };
   429→   429→}
