    32→const extractBearerToken = (authHeader = "") => {
    33→    const trimmed = typeof authHeader === "string" ? authHeader.trim() : "";
    34→    if (!trimmed) return "";
    35→
    36→    const bearerMatch = trimmed.match(/^Bearer\s+(.+)$/i);
    37→    if (bearerMatch?.[1]) {
    38→        return bearerMatch[1].trim();
    39→    }
    40→
    41→    return trimmed;
    42→};
    43→
    44→const resolveAuthUser = async (token) => {
    45→    if (!token) {
    46→        return { user: null, error: null };
    47→    }
    48→
    49→    const { data: authData, error: authError } = await supabase.auth.getUser(token);
    50→    if (authError || !authData?.user?.id) {
    51→        return { user: null, error: authError || new Error("missing user id") };
    52→    }
    53→
    54→    return { user: authData.user, error: null };
    55→};
    56→
    57→const requireAuth = async (req, res, next) => {
    58→    const token = extractBearerToken(req.headers?.authorization);
    59→    if (!token) {
    60→        return res.status(401).json({ error: 'not authorized' });
    61→    }
    62→
    63→    const { user, error } = await resolveAuthUser(token);
    64→    if (error || !user?.id) {
    65→        console.error('auth middleware error:', error?.message || 'missing user id');
    66→        return res.status(401).json({ error: 'not authorized' });
    67→    }
    68→
    69→    req.userId = user.id;
    70→    req.authUser = user;
    71→    return next();
    72→};
    73→
    74→const optionalAuth = async (req, _res, next) => {
    75→    const token = extractBearerToken(req.headers?.authorization);
    76→    if (!token) {
    77→        return next();
    78→    }
    79→
    80→    const { user } = await resolveAuthUser(token);
    81→    if (user?.id) {
    82→        req.userId = user.id;
    83→        req.authUser = user;
    84→    }
    85→
    86→    return next();
    87→};

   196→export const imageUploader = async(file, userId, bucket) =>{
   197→    if(!file){
   198→        throw new Error('no file received');
   199→    }
   200→
   201→    const sourceBuffer = Buffer.isBuffer(file) ? file : file?.buffer;
   202→    if(!sourceBuffer){
   203→        throw new Error('invalid file payload');
   204→    }
   205→
   206→    let img_buffer = null
   207→    let img_url = null
   208→    img_buffer = await sharp(sourceBuffer)
   209→    .webp({quality: 80})
   210→    .toBuffer();
   211→
   212→    const folderName = `user_id_${userId}`;
   213→    const fileName = `${Date.now()}_${crypto.randomUUID()}.webp`;
   214→    const filePath = `${folderName}/${fileName}`;
   215→
   216→    const {data: uploadImage, error: errorUploadImage} = await supabase.storage
   217→    .from(bucket)
   218→    .upload(filePath, img_buffer, {
   219→        contentType: 'image/webp',
   220→        cacheControl: '31536000',
   221→        upsert: true
   222→    })
   223→    if(errorUploadImage){
   224→        console.error('supabase error while uploading image to supabase bucket', errorUploadImage)
   225→        throw new Error('error uploading image into supabase bucket');
   226→    }
   227→    const {data: data_url} = supabase.storage
   228→    .from(bucket)
   229→    .getPublicUrl(filePath)
   230→    
   231→    if(data_url){
   232→        img_url = data_url.publicUrl;
   233→        return img_url;
   234→    } else {
   235→        throw new Error('Error uploading the image');
   236→    }
   237→}

   371→router.get('/getNotifications', requireAuth, async(req, res) =>{
   372→    const {before} = req.query;
   373→    const userId = req.userId;
   374→    const parsedLimit = parseLimitWithinRange(
   375→        req.query.limit,

   382→        return res.status(400).json({error: `limit must be between ${NOTIFICATION_LIMIT_MIN} and ${NOTIFICATION_LIMIT_MAX}`});
   383→    }
   384→
   385→    // Fetch a generous amount from both tables, then merge and paginate
   386→    const fetchLimit = parsedLimit + 1;
   387→
   388→    let journalQuery = supabase
   389→    .from('notifications')
   390→    .select(NOTIFICATION_JOURNAL_SELECT)
   391→    .eq('receiver_id', userId)
   392→    .order('created_at', {ascending: false})
   393→    .order('id', {ascending: false})

   395→
   396→    let opinionQuery = supabase
   397→    .from('notification_opinions')
   398→    .select(NOTIFICATION_OPINION_SELECT)
   399→    .eq('receiver_id', userId)
   400→    .order('created_at', {ascending: false})
   401→    .order('id', {ascending: false})
   402→    .limit(fetchLimit)
   403→
   404→    if(before){
   405→        journalQuery = journalQuery.lt('created_at', before);
   406→        opinionQuery = opinionQuery.lt('created_at', before);
   407→    }
   408→
   409→    const [journalResult, opinionResult] = await Promise.all([journalQuery, opinionQuery]);
   410→
   411→    if(journalResult.error){
   412→        console.error('error while fetching data from notification table:', journalResult.error);
   413→        return res.status(500).json({error: 'error fetching data from notification table'});
   414→    }
   415→    if(opinionResult.error){
   416→        console.error('error while fetching data from notification_opinions table:', opinionResult.error);
   417→        return res.status(500).json({error: 'error fetching data from notification_opinions table'});
   418→    }

   420→    const journalNotifs = (journalResult.data || []).map(n => ({...n, source: 'journal'}));
   421→    const opinionNotifs = (opinionResult.data || []).map(n => ({...n, source: 'opinion'}));
   422→

   424→    const merged = [...journalNotifs, ...opinionNotifs]
   425→        .sort((a, b) => new Date(b.created_at) - new Date(a.created_at));

   427→    // Likes/bookmarks lookup only for journal notifications

   429→        merged

   432→            .filter(Boolean)

   447→        hasBookMarkedResult = hasBookMarked.data || [];

   453→    const formatted = merged.map((notif) => ({

   455→        hasLiked: notif.source === 'journal' ? userHasLikedSet.has(notif.journal_id) : false,
   456→        hasBookMarked: notif.source === 'journal' ? userHasBookmarkedSet.has(notif.journal_id) : false
   457→    }))
