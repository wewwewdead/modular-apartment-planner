    24→    24→function useUnits() {
    25→    25→  const [unit, setUnit] = useState('mm');
    26→    26→  const isMm = unit === 'mm';
    27→    27→  return {
    28→    28→    unit,
    29→    29→    setUnit,
    30→    30→    suffix: isMm ? 'mm' : 'm',
    31→    31→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    32→    32→    fromDisplay: (v) => isMm ? v : v * 1000,
    33→    33→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    34→    34→  };
    35→    35→}

    37→    37→function SheetExportMenu({ sheet, editorDispatch }) {
    38→    38→  const [open, setOpen] = useState(false);
    39→    39→  const rootRef = useRef(null);
    40→    40→  const pngOptions = [
    41→    41→    { label: 'PNG 150 DPI', dpi: 150 },
    42→    42→    { label: 'PNG 300 DPI', dpi: 300 },
    43→    43→    { label: 'PNG 600 DPI', dpi: 600 },
    44→    44→  ];
    45→    45→
    46→    46→  useEffect(() => {
    47→    47→    if (!open) return undefined;
    48→    48→
    49→    49→    const handleMouseDown = (event) => {
    50→    50→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    51→    51→        setOpen(false);
    52→    52→      }
    53→    53→    };
    54→    54→
    55→    55→    document.addEventListener('mousedown', handleMouseDown);
    56→    56→    return () => document.removeEventListener('mousedown', handleMouseDown);
    57→    57→  }, [open]);
    58→    58→
    59→    59→  const handleExport = async (kind, options = {}) => {
    60→    60→    if (!sheet) return;
    61→    61→    try {
    62→    62→      if (kind === 'pdf') {
    63→    63→        await exportActiveSheetAsPdf(
    64→    64→          sheet.title || sheet.drawingName || 'sheet',
    65→    65→          sheet.paperSize
    66→    66→        );
    67→    67→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    68→    68→      } else {
    69→    69→        await exportActiveSheetAsPng(
    70→    70→          sheet.title || sheet.drawingName || 'sheet',
    71→    71→          options
    72→    72→        );
    73→    73→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    74→    74→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    75→    75→      }
    76→    76→    } catch (error) {
    77→    77→      editorDispatch({
    78→    78→        type: 'SET_STATUS_MESSAGE',
    79→    79→        message: error.message || `${kind.toUpperCase()} export failed.`,
    80→    80→      });
    81→    81→    } finally {
    82→    82→      setOpen(false);
    83→    83→    }
    84→    84→  };
    85→    85→
    86→    86→  return (
    87→    87→    <div className={styles.exportMenu} ref={rootRef}>
    88→    88→      <button
    89→    89→        className={styles.exportMenuTrigger}
    90→    90→        type="button"
    91→    91→        onClick={() => setOpen((value) => !value)}
    92→    92→      >
    93→    93→        Save sheet
    94→    94→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    95→    95→      </button>
    96→    96→      {open && (
    97→    97→        <div className={styles.exportMenuList}>
    98→    98→          <button
    99→    99→            className={styles.exportMenuItem}
   100→   100→            type="button"
   101→   101→            onClick={() => handleExport('pdf')}
   102→   102→          >
   103→   103→            PDF
   104→   104→          </button>
   105→   105→          {pngOptions.map((option) => (
   106→   106→            <button
   107→   107→              key={option.dpi}
   108→   108→              className={styles.exportMenuItem}
   109→   109→              type="button"
   110→   110→              onClick={() => handleExport('png', { dpi: option.dpi })}
   111→   111→            >
   112→   112→              {option.label}
   113→   113→            </button>
   114→   114→          ))}
   115→   115→        </div>
   116→   116→      )}
   117→   117→    </div>
   118→   118→  );
   119→   119→}
