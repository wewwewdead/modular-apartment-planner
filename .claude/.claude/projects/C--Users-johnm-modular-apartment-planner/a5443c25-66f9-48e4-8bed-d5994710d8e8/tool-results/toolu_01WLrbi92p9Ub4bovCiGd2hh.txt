    24→function useUnits() {
    25→  const [unit, setUnit] = useState('mm');
    26→  const isMm = unit === 'mm';
    27→  return {
    28→    unit,
    29→    setUnit,
    30→    suffix: isMm ? 'mm' : 'm',
    31→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    32→    fromDisplay: (v) => isMm ? v : v * 1000,
    33→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    34→  };
    35→}

    37→function SheetExportMenu({ sheet, editorDispatch }) {
    38→  const [open, setOpen] = useState(false);
    39→  const rootRef = useRef(null);
    40→  const pngOptions = [
    41→    { label: 'PNG 150 DPI', dpi: 150 },
    42→    { label: 'PNG 300 DPI', dpi: 300 },
    43→    { label: 'PNG 600 DPI', dpi: 600 },
    44→  ];
    45→
    46→  useEffect(() => {
    47→    if (!open) return undefined;
    48→
    49→    const handleMouseDown = (event) => {
    50→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    51→        setOpen(false);
    52→      }
    53→    };
    54→
    55→    document.addEventListener('mousedown', handleMouseDown);
    56→    return () => document.removeEventListener('mousedown', handleMouseDown);
    57→  }, [open]);
    58→
    59→  const handleExport = async (kind, options = {}) => {
    60→    if (!sheet) return;
    61→    try {
    62→      if (kind === 'pdf') {
    63→        await exportActiveSheetAsPdf(
    64→          sheet.title || sheet.drawingName || 'sheet',
    65→          sheet.paperSize
    66→        );
    67→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    68→      } else {
    69→        await exportActiveSheetAsPng(
    70→          sheet.title || sheet.drawingName || 'sheet',
    71→          options
    72→        );
    73→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    74→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    75→      }
    76→    } catch (error) {
    77→      editorDispatch({
    78→        type: 'SET_STATUS_MESSAGE',
    79→        message: error.message || `${kind.toUpperCase()} export failed.`,
    80→      });
    81→    } finally {
    82→      setOpen(false);
    83→    }
    84→  };
    85→
    86→  return (
    87→    <div className={styles.exportMenu} ref={rootRef}>
    88→      <button
    89→        className={styles.exportMenuTrigger}
    90→        type="button"
    91→        onClick={() => setOpen((value) => !value)}
    92→      >
    93→        Save sheet
    94→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    95→      </button>
    96→      {open && (
    97→        <div className={styles.exportMenuList}>
    98→          <button
    99→            className={styles.exportMenuItem}
   100→            type="button"
   101→            onClick={() => handleExport('pdf')}
   102→          >
   103→            PDF
   104→          </button>
   105→          {pngOptions.map((option) => (
   106→            <button
   107→              key={option.dpi}
   108→              className={styles.exportMenuItem}
   109→              type="button"
   110→              onClick={() => handleExport('png', { dpi: option.dpi })}
   111→            >
   112→              {option.label}
   113→            </button>
   114→          ))}
   115→        </div>
   116→      )}
   117→    </div>
   118→  );
   119→}

   914→    } else if (selectedType === 'slab') {

   925→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });

   931→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });

   933→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
   934→    }
   935→    editorDispatch({ type: 'DESELECT' });
   936→  };
   937→
   938→  let content = workspaceMode === 'sheet' && sheet
   939→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
   940→    : <ProjectSummary project={project} floor={floor} dispatch={dispatch} />;
   941→
   942→  if (selectedId && workspaceMode === 'sheet' && sheet) {
   943→    if (selectedType === 'sheet') {
   944→      if (sheet.id === selectedId) {
   945→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;

   950→        content = <SheetViewportProperties sheet={sheet} viewport={viewport} project={project} dispatch={dispatch} u={u} />;
   951→      }
   952→    }
   953→  } else if (selectedId && floor) {
   954→    if (selectedType === 'slab') {
   955→      if (floor.slab?.id === selectedId) {
   956→        content = <SlabProperties slab={floor.slab} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;

   964→      const beam = (floor.beams || []).find(b => b.id === selectedId);
   965→      if (beam) {
   966→        content = <BeamProperties beam={beam} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
   967→      }
   968→    } else if (selectedType === 'sectionCut') {
   969→      const sectionCut = floor.sectionCut?.id === selectedId ? floor.sectionCut : null;
   970→      if (sectionCut) {
   971→        content = <SectionCutProperties sectionCut={sectionCut} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
   972→      }
   973→    } else if (selectedType === 'annotation') {
   974→      const annotation = (floor.annotations || []).find((entry) => entry.id === selectedId);
   975→      if (annotation) {
   976→        content = <AnnotationProperties annotation={annotation} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
   977→      }
   978→    } else if (selectedType === 'stair') {
   979→      const stair = (floor.stairs || []).find(entry => entry.id === selectedId);
   980→      if (stair) {
   981→        content = <StairProperties stair={stair} project={project} dispatch={dispatch} floorId={activeFloorId} u={u} />;
   982→      }
   983→    } else if (selectedType === 'door') {
   984→      const door = floor.doors.find(d => d.id === selectedId);
   985→      const wall = door ? floor.walls.find(w => w.id === door.wallId) : null;
   986→      if (door) {
   987→        content = <DoorProperties door={door} wall={wall} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
   988→      }
   989→    } else if (selectedType === 'window') {
   990→      const win = floor.windows.find(w => w.id === selectedId);
   991→      const wall = win ? floor.walls.find(w => w.id === win.wallId) : null;
   992→      if (win) {
   993→        content = <WindowProperties window={win} wall={wall} dispatch={dispatch} floorId={activeFloorId} u={u} />;
   994→      }
   995→    } else if (selectedType === 'column') {
   996→      const column = (floor.columns || []).find(c => c.id === selectedId);
   997→      if (column) {
   998→        content = <ColumnProperties column={column} floor={floor} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
   999→      }
  1000→    } else if (selectedType === 'room') {
  1001→      const room = floor.rooms.find(r => r.id === selectedId);
  1002→      if (room) {
  1003→        content = <RoomProperties room={room} dispatch={dispatch} floorId={activeFloorId} />;
  1004→      }
  1005→    }
  1006→  }
  1007→
  1008→  return (
  1009→    <div className={styles.panel}>
  1010→      <div className={styles.unitToggle}>
  1011→        <span>Units:</span>
  1012→        <button
  1013→          className={`${styles.unitBtn} ${u.unit === 'mm' ? styles.unitBtnActive : ''}`}
  1014→          onClick={() => u.setUnit('mm')}
  1015→        >mm</button>
  1016→        <button
  1017→          className={`${styles.unitBtn} ${u.unit === 'm' ? styles.unitBtnActive : ''}`}
  1018→          onClick={() => u.setUnit('m')}
  1019→        >m</button>
  1020→      </div>
  1021→      {workspaceMode === 'sheet' && sheet && (
  1022→        <SheetExportMenu sheet={sheet} editorDispatch={editorDispatch} />
  1023→      )}
  1024→      {content}
  1025→      {selectedId && (
  1026→        <button className={styles.deleteBtn} onClick={handleDelete}>
  1027→          Delete {selectedType === 'sheetViewport' ? 'viewport' : selectedType}
  1028→        </button>
  1029→      )}
  1030→    </div>
  1031→  );
