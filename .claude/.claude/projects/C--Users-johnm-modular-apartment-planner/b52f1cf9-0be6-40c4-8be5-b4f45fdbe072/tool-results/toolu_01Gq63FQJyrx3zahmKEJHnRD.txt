    28→function useUnits() {
    29→  const [unit, setUnit] = useState('mm');
    30→  const isMm = unit === 'mm';
    31→  return {
    32→    unit,
    33→    setUnit,
    34→    suffix: isMm ? 'mm' : 'm',
    35→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    36→    fromDisplay: (v) => isMm ? v : v * 1000,
    37→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    38→  };
    39→}

    41→function SheetExportMenu({ sheet, editorDispatch }) {
    42→  const [open, setOpen] = useState(false);
    43→  const rootRef = useRef(null);
    44→  const pngOptions = [
    45→    { label: 'PNG 150 DPI', dpi: 150 },
    46→    { label: 'PNG 300 DPI', dpi: 300 },
    47→    { label: 'PNG 600 DPI', dpi: 600 },
    48→  ];
    49→
    50→  useEffect(() => {
    51→    if (!open) return undefined;
    52→
    53→    const handleMouseDown = (event) => {
    54→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    55→        setOpen(false);
    56→      }
    57→    };
    58→
    59→    document.addEventListener('mousedown', handleMouseDown);
    60→    return () => document.removeEventListener('mousedown', handleMouseDown);
    61→  }, [open]);
    62→
    63→  const handleExport = async (kind, options = {}) => {
    64→    if (!sheet) return;
    65→    try {
    66→      if (kind === 'pdf') {
    67→        await exportActiveSheetAsPdf(
    68→          sheet.title || sheet.drawingName || 'sheet',
    69→          sheet.paperSize
    70→        );
    71→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    72→      } else {
    73→        await exportActiveSheetAsPng(
    74→          sheet.title || sheet.drawingName || 'sheet',
    75→          options
    76→        );
    77→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    78→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    79→      }
    80→    } catch (error) {
    81→      editorDispatch({
    82→        type: 'SET_STATUS_MESSAGE',
    83→        message: error.message || `${kind.toUpperCase()} export failed.`,
    84→      });
    85→    } finally {
    86→      setOpen(false);
    87→    }
    88→  };
    89→
    90→  return (
    91→    <div className={styles.exportMenu} ref={rootRef}>
    92→      <button
    93→        className={styles.exportMenuTrigger}
    94→        type="button"
    95→        onClick={() => setOpen((value) => !value)}
    96→      >
    97→        Export sheet
    98→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    99→      </button>
   100→      {open && (
   101→        <div className={styles.exportMenuList}>
   102→          <button
   103→            className={styles.exportMenuItem}
   104→            type="button"
   105→            onClick={() => handleExport('pdf')}
   106→          >
   107→            PDF
   108→          </button>
   109→          {pngOptions.map((option) => (
   110→            <button
   111→              key={option.dpi}
   112→              className={styles.exportMenuItem}
   113→              type="button"
   114→              onClick={() => handleExport('png', { dpi: option.dpi })}
   115→            >
   116→              {option.label}
   117→            </button>
   118→          ))}
   119→        </div>
   120→      )}
   121→    </div>
   122→  );
   123→}

   501→function LandingProperties({ landing, floor, dispatch, floorId, u }) {
   502→  const updateLanding = (updates) => {
   503→    dispatch({ type: 'LANDING_UPDATE', floorId, landing: { id: landing.id, ...updates } });
   504→  };
   505→
   506→  const stairs = floor?.stairs || [];
