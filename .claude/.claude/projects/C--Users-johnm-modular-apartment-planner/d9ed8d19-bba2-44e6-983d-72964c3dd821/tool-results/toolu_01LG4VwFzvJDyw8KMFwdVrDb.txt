    39→function useUnits() {
    40→  const [unit, setUnit] = useState('mm');
    41→  const isMm = unit === 'mm';
    42→  return {
    43→    unit,
    44→    setUnit,
    45→    suffix: isMm ? 'mm' : 'm',
    46→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    47→    fromDisplay: (v) => isMm ? v : v * 1000,
    48→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    49→  };
    50→}

    52→function WallDrawingInput({ start, preview, dispatch, editorDispatch, activeFloorId, u, selectNewWall }) {
    53→  const [nsDir, setNsDir] = useState('N');
    54→  const [ewDir, setEwDir] = useState('E');
    55→  const [degrees, setDegrees] = useState('');
    56→  const [minutes, setMinutes] = useState('');
    57→  const [lengthVal, setLengthVal] = useState('');
    58→  const lengthRef = useRef(null);
    59→  const livePreviewLength = preview ? formatMeasurement(wallLength({ start, end: preview })) : '';
    60→  const livePreviewBearing = preview ? formatSurveyorBearing(pointsToSurveyorBearing(start, preview)) : '';
    61→
    62→  const handleConfirm = () => {
    63→    const deg = parseInt(degrees, 10) || 0;
    64→    const min = parseInt(minutes, 10) || 0;
    65→    const len = u.fromDisplay(parseFloat(lengthVal) || 0);
    66→
    67→    if (len < MIN_WALL_LENGTH) return;
    68→    if (deg < 0 || deg > 90 || min < 0 || min > 59) return;
    69→
    70→    const angle = surveyorBearingToSvgAngle(nsDir, deg, min, ewDir);
    71→    const endPt = endpointFromBearing(start, len, angle);
    72→    const wall = createWall(start, endPt);
    73→
    74→    dispatch({ type: 'WALL_ADD', floorId: activeFloorId, wall });
    75→    if (selectNewWall) {
    76→      editorDispatch({ type: 'SELECT_OBJECT', id: wall.id, objectType: 'wall' });
    77→    } else {
    78→      editorDispatch({
    79→        type: 'UPDATE_TOOL_STATE',
    80→        payload: { start: endPt, startAttachment: null, preview: null },
    81→      });
    82→    }
    83→
    84→    setDegrees('');
    85→    setMinutes('');
    86→    setLengthVal('');
    87→    setTimeout(() => lengthRef.current?.focus(), 0);
    88→  };
    89→
    90→  const handleKeyDown = (e) => {
    91→    if (e.key === 'Enter') {
    92→      e.preventDefault();
    93→      handleConfirm();
    94→    }
    95→  };
    96→
    97→  return (
    98→    <div>
    99→      {!selectNewWall && (
   100→        <>
   101→          <div className={styles.title}>Drawing Wall</div>
   102→          <div style={{ fontSize: 12, color: 'var(--color-text-secondary)', marginBottom: 'var(--space-sm)' }}>
   103→            Start: ({u.toDisplay(start.x)}, {u.toDisplay(start.y)}) {u.suffix}
   104→          </div>
   105→        </>
   106→      )}
   107→      {preview && (
   108→        <div className={styles.drawingMeta}>
   109→          Cursor: {livePreviewLength} · {livePreviewBearing}
   110→        </div>
   111→      )}
   112→      <div className={styles.drawingHint}>
   113→        Hold Shift while drawing to snap the wall bearing to 45° increments.
   114→      </div>
   115→
   116→      <div className={styles.subtitle}>Bearing</div>
   117→      <div className={styles.bearingRow}>
   118→        <select className={styles.bearingSelect} value={nsDir} onChange={(e) => setNsDir(e.target.value)}>
   119→          <option value="N">N</option>
   120→          <option value="S">S</option>
   121→        </select>
   122→        <input
   123→          className={styles.bearingInput}
   124→          type="number"
   125→          min={0}
   126→          max={90}
   127→          placeholder="0"
   128→          value={degrees}
   129→          onChange={(e) => setDegrees(e.target.value)}
   130→          onKeyDown={handleKeyDown}
   131→        />
   132→        <span className={styles.bearingSymbol}>°</span>
   133→        <input
   134→          className={styles.bearingInput}
   135→          type="number"
   136→          min={0}
   137→          max={59}
   138→          placeholder="0"
   139→          value={minutes}
   140→          onChange={(e) => setMinutes(e.target.value)}
   141→          onKeyDown={handleKeyDown}
   142→        />
   143→        <span className={styles.bearingSymbol}>′</span>
   144→        <select className={styles.bearingSelect} value={ewDir} onChange={(e) => setEwDir(e.target.value)}>
   145→          <option value="E">E</option>
   146→          <option value="W">W</option>
   147→        </select>
   148→      </div>
   149→
   150→      <div className={styles.subtitle}>Length</div>
   151→      <input
   152→        ref={lengthRef}
   153→        className={styles.bearingInput}
   154→        style={{ width: '100%', textAlign: 'left', padding: '0 8px', marginBottom: 'var(--space-sm)' }}
   155→        type="number"
   156→        min={0}
   157→        step={u.step(100)}
   158→        placeholder={`0 ${u.suffix}`}
   159→        value={lengthVal}
   160→        onChange={(e) => setLengthVal(e.target.value)}
   161→        onKeyDown={handleKeyDown}
   162→      />
   163→
   164→      <button className={styles.confirmBtn} onClick={handleConfirm}>
   165→        Create Wall (Enter)
   166→      </button>
   167→    </div>
   168→  );
   169→}

   171→function SheetExportMenu({ sheet, editorDispatch }) {
   172→  const [open, setOpen] = useState(false);
   173→  const rootRef = useRef(null);
   174→  const pngOptions = [
   175→    { label: 'PNG 150 DPI', dpi: 150 },
   176→    { label: 'PNG 300 DPI', dpi: 300 },
   177→    { label: 'PNG 600 DPI', dpi: 600 },
   178→  ];
   179→
   180→  useEffect(() => {
   181→    if (!open) return undefined;
   182→
   183→    const handleMouseDown = (event) => {
   184→      if (rootRef.current && !rootRef.current.contains(event.target)) {
   185→        setOpen(false);
   186→      }
   187→    };
   188→
   189→    document.addEventListener('mousedown', handleMouseDown);
   190→    return () => document.removeEventListener('mousedown', handleMouseDown);
   191→  }, [open]);
   192→
   193→  const handleExport = async (kind, options = {}) => {
   194→    if (!sheet) return;
   195→    try {
   196→      if (kind === 'pdf') {
   197→        await exportActiveSheetAsPdf(
   198→          sheet.title || sheet.drawingName || 'sheet',
   199→          sheet.paperSize
   200→        );
   201→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
   202→      } else {
   203→        await exportActiveSheetAsPng(
   204→          sheet.title || sheet.drawingName || 'sheet',
   205→          options
   206→        );
   207→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
   208→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
   209→      }
   210→    } catch (error) {
   211→      editorDispatch({
   212→        type: 'SET_STATUS_MESSAGE',
   213→        message: error.message || `${kind.toUpperCase()} export failed.`,
   214→      });
   215→    } finally {
   216→      setOpen(false);
   217→    }
   218→  };
   219→
   220→  return (
   221→    <div className={styles.exportMenu} ref={rootRef}>
   222→      <button
   223→        className={styles.exportMenuTrigger}
   224→        type="button"
   225→        onClick={() => setOpen((value) => !value)}
   226→      >
   227→        Export sheet
   228→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   229→      </button>
   230→      {open && (
   231→        <div className={styles.exportMenuList}>
   232→          <button
   233→            className={styles.exportMenuItem}
   234→            type="button"
   235→            onClick={() => handleExport('pdf')}
   236→          >
   237→            PDF
   238→          </button>
   239→          {pngOptions.map((option) => (
   240→            <button
   241→              key={option.dpi}
   242→              className={styles.exportMenuItem}
   243→              type="button"
   244→              onClick={() => handleExport('png', { dpi: option.dpi })}
   245→            >
   246→              {option.label}
   247→            </button>
   248→          ))}
   249→        </div>
   250→      )}
   251→    </div>
   252→  );
   253→}

   532→      <div className={styles.subtitle}>Properties</div>
   533→      <InputField
   534→        label="Offset" type="number" suffix={u.suffix} step={u.step(10)}
   535→        value={u.toDisplay(annotation.offset)}
   536→        onChange={(value) => updateAnnotation({ offset: u.fromDisplay(value) })}
   537→      />
   538→      <InputField
   539→        label="Text"
   540→        value={annotation.textOverride}
   541→        onChange={(value) => updateAnnotation({ textOverride: value })}
   542→      />
   543→      <InputField
   544→        label="Measured" type="number" suffix={u.suffix} step={u.step(100)}
   545→        value={u.toDisplay(measuredValue)}
   546→        onChange={(value) => {
   547→          const newLen = u.fromDisplay(value);
   548→          if (newLen <= 0) return;
   549→          const s = annotation.startPoint;
   550→          const e = annotation.endPoint;
   551→          const mode = annotation.mode;
   552→          let newEnd;
   553→          if (mode === 'horizontal') {
   554→            const sign = e.x >= s.x ? 1 : -1;
   555→            newEnd = { ...e, x: s.x + sign * newLen };
   556→          } else if (mode === 'vertical') {
   557→            const sign = e.y >= s.y ? 1 : -1;
   558→            newEnd = { ...e, y: s.y + sign * newLen };
   559→          } else {
   560→            const dir = normalize(subtract(e, s));
   561→            newEnd = add(s, scale(dir, newLen));
   562→          }
   563→          updateAnnotation({ endPoint: newEnd });
   564→        }}
   565→      />
   566→    </div>
   567→  );
   568→}

  1601→export default function PropertiesPanel() {
  1602→  const { project, dispatch, duplicateFloor } = useProject();
  1603→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, activeTool, toolState, dispatch: editorDispatch } = useEditor();
  1604→  const orderedFloors = getOrderedFloors(project);
  1605→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1606→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1607→  const u = useUnits();
  1608→
  1609→  const selectFloor = (floorId) => {
  1610→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1611→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1612→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1613→  };
  1614→
  1615→  const handleDuplicateFloor = (floorToDuplicate) => {
  1616→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1617→    if (!duplicatedFloor) return;
  1618→
  1619→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1620→    selectFloor(duplicatedFloor.id);
  1621→  };
  1622→
  1623→  const handleDeleteFloor = (floorToDelete) => {
  1624→    if (orderedFloors.length <= 1) return;
  1625→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1626→    if (!confirmed) return;
  1627→
  1628→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1629→    const fallbackFloor = remainingFloors[0] || null;
  1630→
  1631→    dispatch({
  1632→      type: 'FLOOR_DELETE',
  1633→      floorId: floorToDelete.id,
  1634→      fallbackFloorId: fallbackFloor?.id ?? null,
  1635→    });
  1636→
  1637→    if (fallbackFloor) {
  1638→      selectFloor(fallbackFloor.id);

  1640→      editorDispatch({ type: 'DESELECT' });

  1643→
  1644→  const handleDelete = () => {

  1658→    } else if (selectedType === 'annotation') {
  1659→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1660→    } else if (selectedType === 'stair') {
  1661→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1662→    } else if (selectedType === 'landing') {
  1663→      dispatch({ type: 'LANDING_DELETE', floorId: activeFloorId, landingId: selectedId });
  1664→    } else if (selectedType === 'door') {

  1667→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });

  1669→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });

  1675→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1676→    } else if (selectedType === 'floor') {
  1677→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1678→      if (targetFloor) {
  1679→        handleDeleteFloor(targetFloor);
  1680→        return;
