    25→function useUnits() {
    26→  const [unit, setUnit] = useState('mm');
    27→  const isMm = unit === 'mm';
    28→  return {
    29→    unit,
    30→    setUnit,
    31→    suffix: isMm ? 'mm' : 'm',
    32→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    33→    fromDisplay: (v) => isMm ? v : v * 1000,
    34→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    35→  };
    36→}

    38→function SheetExportMenu({ sheet, editorDispatch }) {
    39→  const [open, setOpen] = useState(false);
    40→  const rootRef = useRef(null);
    41→  const pngOptions = [
    42→    { label: 'PNG 150 DPI', dpi: 150 },
    43→    { label: 'PNG 300 DPI', dpi: 300 },
    44→    { label: 'PNG 600 DPI', dpi: 600 },
    45→  ];
    46→
    47→  useEffect(() => {
    48→    if (!open) return undefined;
    49→
    50→    const handleMouseDown = (event) => {
    51→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    52→        setOpen(false);
    53→      }
    54→    };
    55→
    56→    document.addEventListener('mousedown', handleMouseDown);
    57→    return () => document.removeEventListener('mousedown', handleMouseDown);
    58→  }, [open]);
    59→
    60→  const handleExport = async (kind, options = {}) => {
    61→    if (!sheet) return;
    62→    try {
    63→      if (kind === 'pdf') {
    64→        await exportActiveSheetAsPdf(
    65→          sheet.title || sheet.drawingName || 'sheet',
    66→          sheet.paperSize
    67→        );
    68→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    69→      } else {
    70→        await exportActiveSheetAsPng(
    71→          sheet.title || sheet.drawingName || 'sheet',
    72→          options
    73→        );
    74→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    75→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    76→      }
    77→    } catch (error) {
    78→      editorDispatch({
    79→        type: 'SET_STATUS_MESSAGE',
    80→        message: error.message || `${kind.toUpperCase()} export failed.`,
    81→      });
    82→    } finally {
    83→      setOpen(false);
    84→    }
    85→  };
    86→
    87→  return (
    88→    <div className={styles.exportMenu} ref={rootRef}>
    89→      <button
    90→        className={styles.exportMenuTrigger}
    91→        type="button"
    92→        onClick={() => setOpen((value) => !value)}
    93→      >
    94→        Save sheet
    95→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    96→      </button>
    97→      {open && (
    98→        <div className={styles.exportMenuList}>
    99→          <button
   100→            className={styles.exportMenuItem}
   101→            type="button"
   102→            onClick={() => handleExport('pdf')}
   103→          >
   104→            PDF
   105→          </button>
   106→          {pngOptions.map((option) => (
   107→            <button
   108→              key={option.dpi}
   109→              className={styles.exportMenuItem}
   110→              type="button"
   111→              onClick={() => handleExport('png', { dpi: option.dpi })}
   112→            >
   113→              {option.label}
   114→            </button>
   115→          ))}
   116→        </div>
   117→      )}
   118→    </div>
   119→  );
   120→}

   759→function SheetProperties({ sheet, project, dispatch, editorDispatch, activeFloorId, u }) {

   762→  };
   763→
   764→  const addViewport = () => {
   765→    try {
   766→      const sourceFloorId = getDefaultActiveFloorId(project, activeFloorId);
   767→      const sourceView = 'plan';
   768→      const viewportBase = createSheetViewport(sourceView, sourceFloorId, { scale: 100 });
   769→
   770→      let rect;
   771→      try {
   772→        const source = resolveSheetViewportSource(project, viewportBase);
   773→        rect = getDefaultViewportRect(sheet, source, viewportBase.scale);
   774→      } catch {
   775→        // Geometry resolution failed — use default dimensions
   776→        rect = { x: 20, y: 20, width: 160, height: 100 };
   777→      }
   778→
   779→      const viewport = { ...viewportBase, ...rect };
   780→      dispatch({ type: 'SHEET_VIEWPORT_ADD', sheetId: sheet.id, viewport });
   781→      editorDispatch({ type: 'SELECT_OBJECT', id: viewport.id, objectType: 'sheetViewport' });

   786→    }
   787→  };
   788→
   789→  return (
   790→    <div>
   791→      <div className={styles.title}>Sheet</div>
   792→      <InputField label="Title" value={sheet.title} onChange={(value) => updateSheet({ title: value })} />

   826→function SheetViewportProperties({ sheet, viewport, project, dispatch, u }) {

   829→    dispatch({ type: 'SHEET_VIEWPORT_UPDATE', sheetId: sheet.id, viewport: nextViewport });
   830→  };
   831→
   832→  const floorOptions = getOrderedFloors(project);
   833→  const updateSource = (nextSourceView, nextFloorId = viewport.sourceFloorId, nextRefId) => {
   834→    const floor = floorOptions.find((entry) => entry.id === nextFloorId) || null;
   835→    const refId = nextRefId !== undefined
   836→      ? nextRefId
   837→      : nextSourceView === 'section' ? (floor?.sectionCuts || [])[0]?.id || null : null;
   838→    updateViewport({
   839→      sourceView: nextSourceView,
   840→      sourceFloorId: nextFloorId,
   841→      sourceRefId: refId,
   842→    });
   843→  };
   844→
   845→  const currentFloor = floorOptions.find((entry) => entry.id === viewport.sourceFloorId) || null;
   846→  const sectionCutOptions = currentFloor?.sectionCuts || [];
   847→
   848→  const source = resolveSheetViewportSource(project, viewport);

   850→  return (
   851→    <div>
   852→      <div className={styles.title}>Sheet Viewport</div>

   975→        <p>Floors: {project.floors.length}</p>
   976→        <p>Sheets: {(project.sheets || []).length}</p>
   977→        {floor && (
   978→          <>
   979→            <p>Slabs: {(floor.slabs || []).length}</p>
   980→            <p>Walls: {floor.walls.length}</p>
   981→            <p>Annotations: {(floor.annotations || []).length}</p>
   982→            <p>Beams: {(floor.beams || []).length}</p>
   983→            <p>Stairs: {(floor.stairs || []).length}</p>
   984→            <p>Section Cuts: {(floor.sectionCuts || []).length}</p>
   985→            <p>Doors: {floor.doors.length}</p>
   986→            <p>Windows: {floor.windows.length}</p>
   987→            <p>Columns: {(floor.columns || []).length}</p>
   988→          </>
   989→        )}
   990→      </div>
   991→      {floor && <AnnotationSettingsProperties floor={floor} dispatch={dispatch} />}
   992→    </div>

   995→
   996→export default function PropertiesPanel() {
   997→  const { project, dispatch, duplicateFloor } = useProject();
   998→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, dispatch: editorDispatch } = useEditor();
   999→  const orderedFloors = getOrderedFloors(project);
  1000→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1001→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1002→  const u = useUnits();
  1003→
  1004→  const selectFloor = (floorId) => {
  1005→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1006→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1007→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1008→  };
  1009→
  1010→  const handleDuplicateFloor = (floorToDuplicate) => {
  1011→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1012→    if (!duplicatedFloor) return;
  1013→
  1014→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1015→    selectFloor(duplicatedFloor.id);
  1016→  };
  1017→
  1018→  const handleDeleteFloor = (floorToDelete) => {
  1019→    if (orderedFloors.length <= 1) return;
  1020→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1021→    if (!confirmed) return;
  1022→
  1023→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1024→    const fallbackFloor = remainingFloors[0] || null;
  1025→
  1026→    dispatch({
  1027→      type: 'FLOOR_DELETE',
  1028→      floorId: floorToDelete.id,
  1029→      fallbackFloorId: fallbackFloor?.id ?? null,
  1030→    });
  1031→
  1032→    if (fallbackFloor) {
  1033→      selectFloor(fallbackFloor.id);
  1034→    } else {
  1035→      editorDispatch({ type: 'DESELECT' });
  1036→    }
  1037→  };
  1038→
  1039→  const handleDelete = () => {
  1040→    if (!selectedId) return;
  1041→    if (selectedType === 'sheet') {
  1042→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1043→    } else if (selectedType === 'sheetViewport' && sheet) {
  1044→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1045→    } else if (selectedType === 'slab') {
  1046→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1047→    } else if (selectedType === 'wall') {
  1048→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1049→    } else if (selectedType === 'beam') {
  1050→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1051→    } else if (selectedType === 'sectionCut') {
  1052→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1053→    } else if (selectedType === 'annotation') {
  1054→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1055→    } else if (selectedType === 'stair') {
  1056→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1057→    } else if (selectedType === 'door') {
  1058→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1059→    } else if (selectedType === 'window') {
  1060→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1061→    } else if (selectedType === 'column') {
  1062→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1063→    } else if (selectedType === 'room') {
  1064→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1065→    } else if (selectedType === 'floor') {
  1066→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1067→      if (targetFloor) {
  1068→        handleDeleteFloor(targetFloor);
  1069→        return;
  1070→      }
  1071→    }
  1072→    editorDispatch({ type: 'DESELECT' });
  1073→  };
  1074→
  1075→  let content = workspaceMode === 'sheet' && sheet
  1076→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
  1077→    : (
  1078→        <>
  1079→          {floor && (
  1080→            <FloorProperties
  1081→              floor={floor}
  1082→              dispatch={dispatch}
  1083→              onDuplicate={handleDuplicateFloor}
  1084→              onDelete={handleDeleteFloor}
  1085→              u={u}
  1086→            />
  1087→          )}
  1088→          <ProjectSummary project={project} floor={floor} dispatch={dispatch} />
  1089→        </>
  1090→      );
  1091→
  1092→  if (selectedId && workspaceMode === 'sheet' && sheet) {
  1093→    if (selectedType === 'sheet') {
  1094→      if (sheet.id === selectedId) {
  1095→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;
  1096→      }
  1097→    } else if (selectedType === 'sheetViewport') {
  1098→      const viewport = (sheet.viewports || []).find((entry) => entry.id === selectedId);
  1099→      if (viewport) {
  1100→        content = <SheetViewportProperties sheet={sheet} viewport={viewport} project={project} dispatch={dispatch} u={u} />;
  1101→      }
  1102→    }
  1103→  } else if (selectedId && floor) {
  1104→    if (selectedType === 'floor') {
  1105→      const selectedFloor = orderedFloors.find((entry) => entry.id === selectedId) || floor;
  1106→      content = (
  1107→        <FloorProperties
  1108→          floor={selectedFloor}
  1109→          dispatch={dispatch}
  1110→          onDuplicate={handleDuplicateFloor}
  1111→          onDelete={handleDeleteFloor}
  1112→          u={u}
  1113→        />
  1114→      );
  1115→    } else if (selectedType === 'slab') {
  1116→      const slab = (floor.slabs || []).find(s => s.id === selectedId) || null;
  1117→      if (slab) {
  1118→        content = <SlabProperties slab={slab} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1119→      }
  1120→    } else if (selectedType === 'wall') {
  1121→      const wall = floor.walls.find(w => w.id === selectedId);
  1122→      if (wall) {
  1123→        content = <WallProperties wall={wall} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1124→      }
  1125→    } else if (selectedType === 'beam') {
  1126→      const beam = (floor.beams || []).find(b => b.id === selectedId);
  1127→      if (beam) {
  1128→        content = <BeamProperties beam={beam} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1129→      }
  1130→    } else if (selectedType === 'sectionCut') {
  1131→      const sectionCut = (floor.sectionCuts || []).find(s => s.id === selectedId) || null;
  1132→      if (sectionCut) {
  1133→        content = <SectionCutProperties sectionCut={sectionCut} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
  1134→      }
  1135→    } else if (selectedType === 'annotation') {
  1136→      const annotation = (floor.annotations || []).find((entry) => entry.id === selectedId);
  1137→      if (annotation) {
  1138→        content = <AnnotationProperties annotation={annotation} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1139→      }
  1140→    } else if (selectedType === 'stair') {
  1141→      const stair = (floor.stairs || []).find(entry => entry.id === selectedId);
  1142→      if (stair) {
  1143→        content = <StairProperties stair={stair} project={project} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1144→      }
  1145→    } else if (selectedType === 'door') {
  1146→      const door = floor.doors.find(d => d.id === selectedId);
  1147→      const wall = door ? floor.walls.find(w => w.id === door.wallId) : null;
  1148→      if (door) {
  1149→        content = <DoorProperties door={door} wall={wall} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
  1150→      }
  1151→    } else if (selectedType === 'window') {
  1152→      const win = floor.windows.find(w => w.id === selectedId);
  1153→      const wall = win ? floor.walls.find(w => w.id === win.wallId) : null;
  1154→      if (win) {
  1155→        content = <WindowProperties window={win} wall={wall} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1156→      }
  1157→    } else if (selectedType === 'column') {
  1158→      const column = (floor.columns || []).find(c => c.id === selectedId);
  1159→      if (column) {
  1160→        content = <ColumnProperties column={column} floor={floor} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
  1161→      }
  1162→    } else if (selectedType === 'room') {
  1163→      const room = floor.rooms.find(r => r.id === selectedId);
  1164→      if (room) {
  1165→        content = <RoomProperties room={room} dispatch={dispatch} floorId={activeFloorId} />;
  1166→      }

  1169→
  1170→  return (

  1173→        <span>Units:</span>

  1177→        >mm</button>

  1183→      {workspaceMode === 'sheet' && sheet && (
  1184→        <SheetExportMenu sheet={sheet} editorDispatch={editorDispatch} />
  1185→      )}
  1186→      {content}
  1187→      {selectedId && selectedType !== 'floor' && (
  1188→        <button className={styles.deleteBtn} onClick={handleDelete}>
  1189→          Delete {selectedType === 'sheetViewport' ? 'viewport' : selectedType}
  1190→        </button>
  1191→      )}
  1192→    </div>
  1193→  );
  1194→}
