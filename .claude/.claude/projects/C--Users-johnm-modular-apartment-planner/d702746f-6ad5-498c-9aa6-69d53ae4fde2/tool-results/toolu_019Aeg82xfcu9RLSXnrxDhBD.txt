    25→function useUnits() {
    26→  const [unit, setUnit] = useState('mm');
    27→  const isMm = unit === 'mm';
    28→  return {
    29→    unit,
    30→    setUnit,
    31→    suffix: isMm ? 'mm' : 'm',
    32→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    33→    fromDisplay: (v) => isMm ? v : v * 1000,
    34→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    35→  };
    36→}

    38→function SheetExportMenu({ sheet, editorDispatch }) {
    39→  const [open, setOpen] = useState(false);
    40→  const rootRef = useRef(null);
    41→  const pngOptions = [
    42→    { label: 'PNG 150 DPI', dpi: 150 },
    43→    { label: 'PNG 300 DPI', dpi: 300 },
    44→    { label: 'PNG 600 DPI', dpi: 600 },
    45→  ];
    46→
    47→  useEffect(() => {
    48→    if (!open) return undefined;
    49→
    50→    const handleMouseDown = (event) => {
    51→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    52→        setOpen(false);
    53→      }
    54→    };
    55→
    56→    document.addEventListener('mousedown', handleMouseDown);
    57→    return () => document.removeEventListener('mousedown', handleMouseDown);
    58→  }, [open]);
    59→
    60→  const handleExport = async (kind, options = {}) => {
    61→    if (!sheet) return;
    62→    try {
    63→      if (kind === 'pdf') {
    64→        await exportActiveSheetAsPdf(
    65→          sheet.title || sheet.drawingName || 'sheet',
    66→          sheet.paperSize
    67→        );
    68→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    69→      } else {
    70→        await exportActiveSheetAsPng(
    71→          sheet.title || sheet.drawingName || 'sheet',
    72→          options
    73→        );
    74→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    75→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    76→      }
    77→    } catch (error) {
    78→      editorDispatch({
    79→        type: 'SET_STATUS_MESSAGE',
    80→        message: error.message || `${kind.toUpperCase()} export failed.`,
    81→      });
    82→    } finally {
    83→      setOpen(false);
    84→    }
    85→  };
    86→
    87→  return (
    88→    <div className={styles.exportMenu} ref={rootRef}>
    89→      <button
    90→        className={styles.exportMenuTrigger}
    91→        type="button"
    92→        onClick={() => setOpen((value) => !value)}
    93→      >
    94→        Save sheet
    95→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    96→      </button>
    97→      {open && (
    98→        <div className={styles.exportMenuList}>
    99→          <button
   100→            className={styles.exportMenuItem}
   101→            type="button"
   102→            onClick={() => handleExport('pdf')}
   103→          >
   104→            PDF
   105→          </button>
   106→          {pngOptions.map((option) => (
   107→            <button
   108→              key={option.dpi}
   109→              className={styles.exportMenuItem}
   110→              type="button"
   111→              onClick={() => handleExport('png', { dpi: option.dpi })}
   112→            >
   113→              {option.label}
   114→            </button>
   115→          ))}
   116→        </div>
   117→      )}
   118→    </div>
   119→  );
   120→}

   766→    try {
   767→      const sourceFloorId = getDefaultActiveFloorId(project, activeFloorId);

   776→        // Geometry resolution failed — use default dimensions

   840→      sourceView: nextSourceView,
   841→      sourceFloorId: nextFloorId,
   842→      sourceRefId: refId,

   980→        {floor && (

   999→export default function PropertiesPanel() {
  1000→  const { project, dispatch, duplicateFloor } = useProject();
  1001→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, dispatch: editorDispatch } = useEditor();
  1002→  const orderedFloors = getOrderedFloors(project);
  1003→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1004→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1005→  const u = useUnits();
  1006→
  1007→  const selectFloor = (floorId) => {
  1008→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1009→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1010→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1011→  };
  1012→
  1013→  const handleDuplicateFloor = (floorToDuplicate) => {
  1014→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1015→    if (!duplicatedFloor) return;
  1016→
  1017→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1018→    selectFloor(duplicatedFloor.id);
  1019→  };
  1020→
  1021→  const handleDeleteFloor = (floorToDelete) => {
  1022→    if (orderedFloors.length <= 1) return;
  1023→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1024→    if (!confirmed) return;
  1025→
  1026→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1027→    const fallbackFloor = remainingFloors[0] || null;
  1028→
  1029→    dispatch({
  1030→      type: 'FLOOR_DELETE',
  1031→      floorId: floorToDelete.id,
  1032→      fallbackFloorId: fallbackFloor?.id ?? null,
  1033→    });
  1034→
  1035→    if (fallbackFloor) {
  1036→      selectFloor(fallbackFloor.id);
  1037→    } else {
  1038→      editorDispatch({ type: 'DESELECT' });
  1039→    }
  1040→  };
  1041→
  1042→  const handleDelete = () => {
  1043→    if (!selectedId) return;
  1044→    if (selectedType === 'sheet') {
  1045→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1046→    } else if (selectedType === 'sheetViewport' && sheet) {
  1047→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1048→    } else if (selectedType === 'slab') {
  1049→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1050→    } else if (selectedType === 'wall') {
  1051→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1052→    } else if (selectedType === 'beam') {
  1053→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1054→    } else if (selectedType === 'sectionCut') {
  1055→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1056→    } else if (selectedType === 'annotation') {
  1057→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1058→    } else if (selectedType === 'stair') {
  1059→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1060→    } else if (selectedType === 'door') {
  1061→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1062→    } else if (selectedType === 'window') {
  1063→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1064→    } else if (selectedType === 'column') {
  1065→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1066→    } else if (selectedType === 'room') {
  1067→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1068→    } else if (selectedType === 'floor') {
  1069→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1070→      if (targetFloor) {
  1071→        handleDeleteFloor(targetFloor);
  1072→        return;
  1073→      }
  1074→    }
  1075→    editorDispatch({ type: 'DESELECT' });
  1076→  };
  1077→
  1078→  let content = workspaceMode === 'sheet' && sheet
  1079→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
  1080→    : (
  1081→        <>
  1082→          {floor && (
  1083→            <FloorProperties
  1084→              floor={floor}
  1085→              dispatch={dispatch}
  1086→              onDuplicate={handleDuplicateFloor}
  1087→              onDelete={handleDeleteFloor}
  1088→              canDelete={orderedFloors.length > 1}
  1089→              u={u}
  1090→            />
  1091→          )}
  1092→          <ProjectSummary project={project} floor={floor} dispatch={dispatch} />
  1093→        </>
  1094→      );
  1095→
  1096→  if (selectedId && workspaceMode === 'sheet' && sheet) {
  1097→    if (selectedType === 'sheet') {
  1098→      if (sheet.id === selectedId) {
  1099→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;
  1100→      }
  1101→    } else if (selectedType === 'sheetViewport') {
  1102→      const viewport = (sheet.viewports || []).find((entry) => entry.id === selectedId);
  1103→      if (viewport) {
  1104→        content = <SheetViewportProperties sheet={sheet} viewport={viewport} project={project} dispatch={dispatch} u={u} />;

  1114→          onDuplicate={handleDuplicateFloor}
  1115→          onDelete={handleDeleteFloor}
