    72→function createBounds(points = []) {
    73→  if (!points.length) {
    74→    return {
    75→      minX: 0,
    76→      maxX: 1000,
    77→      minY: 0,
    78→      maxY: 1000,
    79→      minElevation: 0,
    80→      maxElevation: 1000,
    81→    };
    82→  }
    83→
    84→  return {
    85→    minX: Math.min(...points.map((point) => point.x)),
    86→    maxX: Math.max(...points.map((point) => point.x)),
    87→    minY: Math.min(...points.map((point) => point.y)),
    88→    maxY: Math.max(...points.map((point) => point.y)),
    89→    minElevation: Math.min(...points.map((point) => point.elevation ?? 0)),
    90→    maxElevation: Math.max(...points.map((point) => point.elevation ?? 0)),
    91→  };
    92→}

   101→function pointFromLocal(structuralStartPoint, spanDirection, localPoint, baseElevation) {
   102→  const planPoint = add(structuralStartPoint, scale(spanDirection, localPoint.x));
   103→  return {
   104→    x: planPoint.x,
   105→    y: planPoint.y,
   106→    elevation: baseElevation + localPoint.z,
   107→  };
   108→}

   110→function createCopyGeometry(trussSystem, instance, profile, layout, origin, index) {
   111→  const structuralStartPoint = add(origin, scale(layout.spanDirection, -(profile.metrics.span / 2)));
   112→  const structuralEndPoint = add(origin, scale(layout.spanDirection, profile.metrics.span / 2));
   113→  const overallStartPoint = add(structuralStartPoint, scale(layout.spanDirection, -profile.metrics.overhangStart));
   114→  const overallEndPoint = add(structuralEndPoint, scale(layout.spanDirection, profile.metrics.overhangEnd));
   115→  const nodeMap = new Map(profile.nodes.map((node) => [node.id, node]));
   116→  const baseElevation = trussSystem.baseElevation || 0;
   117→
   118→  const worldNodes = profile.nodes.map((node) => ({
   119→    ...node,
   120→    planPoint: add(structuralStartPoint, scale(layout.spanDirection, node.x)),
   121→    worldPoint: pointFromLocal(structuralStartPoint, layout.spanDirection, node, baseElevation),
   122→  }));
   123→  const members = profile.members.map((member) => {
   124→    const startNode = nodeMap.get(member.startNodeId);
   125→    const endNode = nodeMap.get(member.endNodeId);
   126→    return {
   127→      ...member,
   128→      startNode,
   129→      endNode,
   130→      startWorld: pointFromLocal(structuralStartPoint, layout.spanDirection, startNode, baseElevation),
   131→      endWorld: pointFromLocal(structuralStartPoint, layout.spanDirection, endNode, baseElevation),
   132→    };
   133→  });
   134→  const purlinAttachments = (profile.purlinAttachments || []).map((attachment) => ({
   135→    ...attachment,
   136→    planPoint: add(structuralStartPoint, scale(layout.spanDirection, attachment.localPoint.x)),
   137→    worldPoint: pointFromLocal(structuralStartPoint, layout.spanDirection, attachment.localPoint, baseElevation),
   138→  }));
   139→
   140→  return {
   141→    id: `${instance.id}_copy_${index}`,
   142→    index,
   143→    origin,
   144→    structuralStartPoint,
   145→    structuralEndPoint,
   146→    overallStartPoint,
   147→    overallEndPoint,
   148→    worldNodes,
   149→    members,
   150→    purlinAttachments,
   151→  };
   152→}

   154→function normalizePlanVector(vector) {
   155→  const magnitude = Math.sqrt((vector.x * vector.x) + (vector.y * vector.y));
   156→  if (magnitude <= 1e-6) {
   157→    return { x: 1, y: 0 };
   158→  }
   159→  return {
   160→    x: vector.x / magnitude,
   161→    y: vector.y / magnitude,
   162→  };
   163→}

   165→function createPlanMarker(centerPoint, axis, length = DEFAULT_PURLIN_MARKER_LENGTH) {
   166→  return {
   167→    start: add(centerPoint, scale(axis, -(length / 2))),
   168→    end: add(centerPoint, scale(axis, length / 2)),
   169→  };

   205→function offsetWorldPoint(worldPoint, axis, distance) {
   206→  if (!worldPoint || Math.abs(distance) <= 1e-6) return worldPoint;
   207→  return {
   208→    ...worldPoint,
   209→    x: worldPoint.x + (axis.x * distance),
   210→    y: worldPoint.y + (axis.y * distance),
   211→  };
   212→}

   214→function buildInstancePurlinGeometry(trussSystem, instanceGeometry, purlinSystem) {
   215→  if (!purlinSystem?.enabled) {
   216→    return {
   217→      purlinRuns: [],
   218→      purlinMarkers: [],
   219→      purlinSegments: [],
   220→    };
   221→  }
   222→
   223→  const attachmentGroups = new Map();
   224→  for (const copy of instanceGeometry.copies) {
   225→    for (const attachment of copy.purlinAttachments || []) {
   226→      if (!attachmentGroups.has(attachment.id)) {
   227→        attachmentGroups.set(attachment.id, []);
   228→      }
   229→      attachmentGroups.get(attachment.id).push({
   230→        ...attachment,
   231→        copyId: copy.id,
   232→        copyIndex: copy.index,
   233→      });
   234→    }
   235→  }
   236→
   237→  const layoutAxis = normalizePlanVector({
   238→    x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   239→    y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   240→  });
   241→  const purlinRuns = [];
   242→  const purlinMarkers = [];
   243→  const purlinSegments = [];
   244→  const overhangStart = Math.max(Number(purlinSystem?.overhangStart || 0), 0);
   245→  const overhangEnd = Math.max(Number(purlinSystem?.overhangEnd || 0), 0);
   246→
   247→  for (const [attachmentId, attachmentPoints] of attachmentGroups.entries()) {
   248→    const sortedPoints = [...attachmentPoints].sort((a, b) => a.copyIndex - b.copyIndex);
   249→    if (sortedPoints.length >= 2) {
   250→      const firstPoint = sortedPoints[0];
   251→      const lastPoint = sortedPoints[sortedPoints.length - 1];
   252→      const runPoints = [
   253→        ...(overhangStart > 1e-6
   254→          ? [add(firstPoint.planPoint, scale(layoutAxis, -overhangStart))]
   255→          : []),
   256→        ...sortedPoints.map((point) => point.planPoint),
   257→        ...(overhangEnd > 1e-6
   258→          ? [add(lastPoint.planPoint, scale(layoutAxis, overhangEnd))]
   259→          : []),
   260→      ];
   261→      const segments = sortedPoints.slice(0, -1).map((startPoint, index) => {
   262→        const endPoint = sortedPoints[index + 1];
   263→        const startWorld = index === 0
   264→          ? offsetWorldPoint(startPoint.worldPoint, layoutAxis, -overhangStart)
   265→          : startPoint.worldPoint;
   266→        const endWorld = index === sortedPoints.length - 2
   267→          ? offsetWorldPoint(endPoint.worldPoint, layoutAxis, overhangEnd)
   268→          : endPoint.worldPoint;
   269→
   270→        return createPurlinSegment(
   271→          `${instanceGeometry.instance.id}_${attachmentId}_${index}`,
   272→          trussSystem,
   273→          instanceGeometry.instance,
   274→          startWorld,
   275→          endWorld,
   276→          purlinSystem,
   277→          startPoint.side,
   278→          [startPoint.copyId, endPoint.copyId]
   279→        );
   280→      }).filter(Boolean);
   281→      purlinRuns.push({
   282→        id: `${instanceGeometry.instance.id}_${attachmentId}_run`,
   283→        side: firstPoint.side,
   284→        points: runPoints,
   285→        attachments: sortedPoints,
   286→        segments,
   287→      });
   288→      purlinSegments.push(...segments);
   289→      continue;
   290→    }
   291→
   292→    if (sortedPoints.length === 1) {
   293→      const marker = createPlanMarker(sortedPoints[0].planPoint, layoutAxis);
   294→      purlinMarkers.push({
   295→        id: `${instanceGeometry.instance.id}_${attachmentId}_marker`,
   296→        side: sortedPoints[0].side,
   297→        center: sortedPoints[0].planPoint,
   298→        ...marker,
   299→      });
   300→    }
   301→  }
   302→
   303→  return {
   304→    purlinRuns,
   305→    purlinMarkers,
   306→    purlinSegments,
   307→  };
   308→}

   319→function transformCopyGeometry(copy, transform) {
   320→  const worldNodes = copy.worldNodes.map((node) => ({
   321→    ...node,
   322→    planPoint: transformPlanPoint(node.planPoint, transform),
   323→    worldPoint: rotateWorldPoint(node.worldPoint, transform),
   324→  }));
   325→  const members = copy.members.map((member) => ({
   326→    ...member,
   327→    startWorld: rotateWorldPoint(member.startWorld, transform),
   328→    endWorld: rotateWorldPoint(member.endWorld, transform),
   329→  }));
   330→  const purlinAttachments = (copy.purlinAttachments || []).map((attachment) => ({
   331→    ...attachment,
   332→    planPoint: transformPlanPoint(attachment.planPoint, transform),
   333→    worldPoint: rotateWorldPoint(attachment.worldPoint, transform),
   334→  }));
   335→
   336→  return {
   337→    ...copy,
   338→    origin: transformPlanPoint(copy.origin, transform),
   339→    structuralStartPoint: transformPlanPoint(copy.structuralStartPoint, transform),
   340→    structuralEndPoint: transformPlanPoint(copy.structuralEndPoint, transform),
   341→    overallStartPoint: transformPlanPoint(copy.overallStartPoint, transform),
   342→    overallEndPoint: transformPlanPoint(copy.overallEndPoint, transform),
   343→    worldNodes,
   344→    members,
   345→    purlinAttachments,
   346→  };
   347→}

   349→function transformInstanceGeometry(instanceGeometry, transform, baseElevation) {
   350→  const copies = instanceGeometry.copies.map((copy) => transformCopyGeometry(copy, transform));
   351→  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   352→
   353→  return {
   354→    ...instanceGeometry,
   355→    copies,
   356→    layoutLineStartPoint: transformPlanPoint(instanceGeometry.instance.startPoint, transform),
   357→    layoutLineEndPoint: transformPlanPoint(instanceGeometry.instance.endPoint, transform),
   358→    planBounds: createBounds(planPoints.map((point) => ({
   359→      x: point.x,
   360→      y: point.y,
   361→      elevation: baseElevation,
   362→    }))),
   363→  };
   364→}

   366→function buildRawTrussInstanceGeometry(trussSystem, instance) {
   367→  const layout = resolveTrussLayout(instance);
   368→  const profile = buildTrussProfile(instance);
   369→  const purlinAttachments = buildTrussPurlinAttachments(profile, trussSystem?.purlinSystem);
   370→  const copies = buildTrussCopyOrigins(instance).map((copy) => {
   371→    const layoutOffset = (
   372→      ((copy.origin.x ?? 0) - (instance.startPoint?.x ?? 0)) * layout.axis.x
   373→    ) + (
   374→      ((copy.origin.y ?? 0) - (instance.startPoint?.y ?? 0)) * layout.axis.y
   375→    );
   376→    const copyProfile = buildTrussProfile(instance, undefined, {
   377→      layoutOffset,
   378→      supportLength: layout.supportLength,
   379→    });
   380→    const copyPurlinAttachments = buildTrussPurlinAttachments(copyProfile, trussSystem?.purlinSystem);
   381→
   382→    return createCopyGeometry(trussSystem, instance, {
   383→      ...copyProfile,
   384→      purlinAttachments: copyPurlinAttachments,
   385→    }, layout, copy.origin, copy.index);
   386→  });
   387→
   388→  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   389→  const detailPoints = [
   390→    ...profile.roofOutline.map((point) => ({ x: point.x, y: 0, elevation: point.z })),
   391→    ...profile.bottomOutline.map((point) => ({ x: point.x, y: 0, elevation: point.z })),
   392→    ...purlinAttachments.map((attachment) => ({
   393→      x: attachment.localPoint.x,
   394→      y: 0,
   395→      elevation: attachment.localPoint.z,
   396→    })),
   397→  ];
   398→
   399→  return {
   400→    instance,
   401→    trussType: profile.trussType,
   402→    metrics: profile.metrics,
   403→    nodes: profile.nodes,
   404→    members: profile.members,
   405→    roofOutline: profile.roofOutline,
   406→    bottomOutline: profile.bottomOutline,
   407→    topChordRuns: profile.topChordRuns || [],
   408→    purlinAttachments,
   409→    copies,
   410→    layoutLineStartPoint: instance.startPoint,
   411→    layoutLineEndPoint: instance.endPoint,
   412→    planBounds: createBounds(planPoints.map((point) => ({
   413→      x: point.x,
   414→      y: point.y,
   415→      elevation: trussSystem.baseElevation || 0,
   416→    }))),
   417→    detailBounds: {
   418→      minX: Math.min(...detailPoints.map((point) => point.x)) - DETAIL_PADDING_X,
   419→      maxX: Math.max(...detailPoints.map((point) => point.x)) + DETAIL_PADDING_X,
   420→      minZ: Math.min(...detailPoints.map((point) => point.elevation), 0) - DETAIL_PADDING_Z,
   421→      maxZ: Math.max(...detailPoints.map((point) => point.elevation), 0) + DETAIL_PADDING_Z,
   422→    },
   423→  };
   424→}

   426→function createSystemTransform(trussSystem, rawInstances, baseElevation) {
   427→  const rawPlanPoints = rawInstances.flatMap((instanceGeometry) => (
   428→    instanceGeometry.copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint])
   429→  ));
   430→  const rawBounds = createBounds(rawPlanPoints.map((point) => ({
   431→    x: point.x,
   432→    y: point.y,
   433→    elevation: baseElevation,
   434→  })));
   435→  const rawPivot = centerOfBounds(rawBounds);
   436→  const primaryAxis = getSafeAxis(rawInstances[0]
   437→    ? {
   438→        x: rawInstances[0].layoutLineEndPoint.x - rawInstances[0].layoutLineStartPoint.x,
   439→        y: rawInstances[0].layoutLineEndPoint.y - rawInstances[0].layoutLineStartPoint.y,
   440→      }
   441→    : { x: 1, y: 0 });
   442→  const projectedAlongs = rawPlanPoints.map((point) => projectAlongAxis(point, rawPivot, primaryAxis));
   443→  const startAlong = projectedAlongs.length ? Math.min(...projectedAlongs) : 0;
   444→  const endAlong = projectedAlongs.length ? Math.max(...projectedAlongs) : 0;
   445→  const rawLength = Math.max(endAlong - startAlong, 0);
   446→  const resizable = rawLength > 1e-6 && rawInstances.every((instanceGeometry) => {
   447→    const instanceAxis = getSafeAxis({
   448→      x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   449→      y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   450→    });
   451→    return Math.abs(Math.abs((instanceAxis.x * primaryAxis.x) + (instanceAxis.y * primaryAxis.y)) - 1) <= 1e-3;
   452→  });
   453→  const rotationDegrees = normalizeSignedRotationDegrees(trussSystem?.planRotationOffsetDegrees || 0);
   454→  const translation = normalizePlanOffset(trussSystem?.planOffset);
   455→  const lengthScale = normalizePlanLengthScale(trussSystem?.planLengthScale);
   456→
   457→  return {
   458→    rawPivot,
   459→    pivot: {
   460→      x: rawPivot.x + translation.x,
   461→      y: rawPivot.y + translation.y,
   462→    },
   463→    axis: primaryAxis,
   464→    displayAxis: getDisplayedAxis({ axis: primaryAxis, rotationDegrees }),
   465→    translation,
   466→    lengthScale,
   467→    rawLength,
   468→    currentLength: rawLength * lengthScale,
   469→    startAlong,
   470→    endAlong,
   471→    startHandlePoint: transformPlanPoint(
   472→      {
   473→        x: rawPivot.x + (primaryAxis.x * startAlong),
   474→        y: rawPivot.y + (primaryAxis.y * startAlong),
   475→      },
   476→      {
   477→        rawPivot,
   478→        axis: primaryAxis,
   479→        rotationDegrees,
   480→        translation,
   481→        lengthScale,
   482→      }
   483→    ),
   484→    endHandlePoint: transformPlanPoint(
   485→      {
   486→        x: rawPivot.x + (primaryAxis.x * endAlong),
   487→        y: rawPivot.y + (primaryAxis.y * endAlong),
   488→      },
   489→      {
   490→        rawPivot,
   491→        axis: primaryAxis,
   492→        rotationDegrees,
   493→        translation,
   494→        lengthScale,
   495→      }
   496→    ),
   497→    resizable,
   498→    rotationDegrees,
   499→  };
   500→}

   574→function detailPolylineLength(points = []) {
   575→  let total = 0;
   576→  for (let index = 0; index < points.length - 1; index += 1) {
   577→    total += Math.hypot(points[index + 1].x - points[index].x, points[index + 1].y - points[index].y);
   578→  }
   579→  return total;
   580→}

   582→function worldSegmentLength(startWorld, endWorld) {
   583→  if (!startWorld || !endWorld) return 0;
   584→  return Math.hypot(
   585→    endWorld.x - startWorld.x,
   586→    endWorld.y - startWorld.y,
   587→    (endWorld.elevation || 0) - (startWorld.elevation || 0)
   588→  );
   589→}

   594→    return run.segments.reduce((total, segment) => (

   596→    ), 0);
   597→  }

   601→function detailPolylineMidpoint(points = []) {
   602→  if (!points.length) return { x: 0, y: 0 };
   603→  if (points.length === 1) return points[0];
   604→
   605→  const totalLength = detailPolylineLength(points);
   606→  if (totalLength <= 1e-6) {
   607→    return {
   608→      x: (points[0].x + points[points.length - 1].x) / 2,
   609→      y: (points[0].y + points[points.length - 1].y) / 2,
   610→    };
   611→  }
   612→
   613→  let remaining = totalLength / 2;
   614→  for (let index = 0; index < points.length - 1; index += 1) {
   615→    const start = points[index];
   616→    const end = points[index + 1];
   617→    const segmentLength = Math.hypot(end.x - start.x, end.y - start.y);
   618→    if (remaining <= segmentLength || index === points.length - 2) {
   619→      const ratio = segmentLength <= 1e-6 ? 0 : remaining / segmentLength;
   620→      return {
   621→        x: start.x + ((end.x - start.x) * ratio),
   622→        y: start.y + ((end.y - start.y) * ratio),
   623→      };
   624→    }
   625→    remaining -= segmentLength;
   626→  }
   627→
   628→  return points[points.length - 1];
   629→}

   667→function estimateDimensionBounds(figure) {
   668→  if (!figure) return null;
   669→
   670→  const points = [
   671→    figure.lineStart,
   672→    figure.lineEnd,
   673→    ...figure.extensionLines.flatMap((line) => [line.start, line.end]),
   674→    ...figure.arrowheads.flatMap((arrow) => [arrow.left, arrow.tip, arrow.right]),
   675→  ].filter(Boolean);
   676→  const fontSize = DRAWING_GRAPHICS.annotation.textSize;
   677→  const width = estimateTextWidth(figure.text?.value, fontSize) + DRAWING_GRAPHICS.annotation.textMaskPaddingX;
   678→  const height = fontSize + DRAWING_GRAPHICS.annotation.textMaskPaddingY;
   679→  const textPosition = figure.text?.position || { x: 0, y: 0 };
   680→  const textRadius = Math.sqrt((width * width) + (height * height)) / 2;
   681→
   682→  points.push(
   683→    { x: textPosition.x - textRadius, y: textPosition.y - textRadius },
   684→    { x: textPosition.x + textRadius, y: textPosition.y + textRadius }
   685→  );
   686→
   687→  return {
   688→    minX: Math.min(...points.map((point) => point.x)),
   689→    maxX: Math.max(...points.map((point) => point.x)),
   690→    minY: Math.min(...points.map((point) => point.y)),
   691→    maxY: Math.max(...points.map((point) => point.y)),
   692→  };
   693→}

   695→function estimateTagBounds(tag) {
   696→  if (!tag) return null;
   697→
   698→  const fontSize = Number(tag.fontSize || DETAIL_TAG_FONT_SIZE);
   699→  const lineHeight = Number(tag.lineHeight || DETAIL_TAG_LINE_HEIGHT);
   700→  const maskPaddingX = Number(tag.maskPaddingX ?? DETAIL_TAG_MASK_PADDING_X);
   701→  const maskPaddingY = Number(tag.maskPaddingY ?? DETAIL_TAG_MASK_PADDING_Y);
   702→  const maxLineLength = Math.max(...tag.textLines.map((line) => String(line || '').length), 1);
   703→  const width = (maxLineLength * fontSize * 0.58) + Math.max(maskPaddingX, fontSize * 0.5);
   704→  const height = Math.max(lineHeight, tag.textLines.length * lineHeight) + maskPaddingY;
   705→  const minX = tag.textAnchor === 'start'
   706→    ? tag.position.x
   707→    : (tag.textAnchor === 'end' ? tag.position.x - width : tag.position.x - (width / 2));
   708→  const maxX = tag.textAnchor === 'start'
   709→    ? tag.position.x + width
   710→    : (tag.textAnchor === 'end' ? tag.position.x : tag.position.x + (width / 2));
   711→  const minY = tag.position.y - (height / 2);
   712→  const maxY = tag.position.y + (height / 2);
   713→  const angle = Number(tag.angle || 0);
   714→  const corners = [
   715→    { x: minX, y: minY },
   716→    { x: maxX, y: minY },
   717→    { x: maxX, y: maxY },
   718→    { x: minX, y: maxY },
   719→  ].map((corner) => rotate(corner, tag.position, angle));
   720→
   721→  return {
   722→    minX: Math.min(...corners.map((point) => point.x)),
   723→    maxX: Math.max(...corners.map((point) => point.x)),
   724→    minY: Math.min(...corners.map((point) => point.y)),
   725→    maxY: Math.max(...corners.map((point) => point.y)),
   726→  };
   727→}

   729→function selectOutwardNormal(normal, midpoint, spanMidpoint) {
   730→  const preferred = midpoint.x < spanMidpoint
   731→    ? { x: -1, y: 0 }
   732→    : { x: 1, y: 0 };
   733→  const alignment = (normal.x * preferred.x) + (normal.y * preferred.y);
   734→
   735→  return alignment >= 0
   736→    ? normal
   737→    : { x: -normal.x, y: -normal.y };
   738→}

   740→function extendDetailBounds(detailBounds, dimensions = [], tags = []) {
   741→  const bounds = {
   742→    minX: detailBounds.minX,
   743→    maxX: detailBounds.maxX,
   744→    minY: -detailBounds.maxZ,
   745→    maxY: -detailBounds.minZ,
   746→  };
   747→
   748→  for (const figure of dimensions) {
   749→    const figureBounds = estimateDimensionBounds(figure);
   750→    if (!figureBounds) continue;
   751→    bounds.minX = Math.min(bounds.minX, figureBounds.minX);
   752→    bounds.maxX = Math.max(bounds.maxX, figureBounds.maxX);
   753→    bounds.minY = Math.min(bounds.minY, figureBounds.minY);
   754→    bounds.maxY = Math.max(bounds.maxY, figureBounds.maxY);
   755→  }
   756→
   757→  for (const tag of tags) {
   758→    const tagBounds = estimateTagBounds(tag);
   759→    if (!tagBounds) continue;
   760→    bounds.minX = Math.min(bounds.minX, tagBounds.minX);
   761→    bounds.maxX = Math.max(bounds.maxX, tagBounds.maxX);
   762→    bounds.minY = Math.min(bounds.minY, tagBounds.minY);
   763→    bounds.maxY = Math.max(bounds.maxY, tagBounds.maxY);
   764→  }
   765→
   766→  return {
   767→    minX: bounds.minX,
   768→    maxX: bounds.maxX,
   769→    minZ: -bounds.maxY,
   770→    maxZ: -bounds.minY,
   771→  };
   772→}

   774→function boundsOverlap(a, b, gap) {
   775→  if (!a || !b) return false;
   776→  const halfGap = gap / 2;
   777→  return (
   778→    a.minX - halfGap < b.maxX + halfGap &&
   779→    a.maxX + halfGap > b.minX - halfGap &&
   780→    a.minY - halfGap < b.maxY + halfGap &&
   781→    a.maxY + halfGap > b.minY - halfGap
   782→  );
   783→}

   785→function applyDetailTagCollisionOffsets(tags = [], fixedObstacles = []) {
   786→  const placed = fixedObstacles.filter(Boolean);
   787→
   788→  return [...tags]
   789→    .sort((a, b) => (b.priority || 0) - (a.priority || 0))
   790→    .map((tag) => {
   791→      for (const offset of DETAIL_TAG_OFFSETS) {
   792→        const candidate = {
   793→          ...tag,
   794→          position: {
   795→            x: tag.position.x + offset.x,
   796→            y: tag.position.y + offset.y,
   797→          },
   798→          leaderLine: tag.leaderLine
   799→            ? {
   800→                start: tag.leaderLine.start,
   801→                end: {
   802→                  x: tag.leaderLine.end.x + offset.x,
   803→                  y: tag.leaderLine.end.y + offset.y,
   804→                },
   805→              }
   806→            : null,
   807→        };
   808→        const candidateBounds = estimateTagBounds(candidate);
   809→        if (!candidateBounds) continue;
   810→        if (!placed.some((obs) => boundsOverlap(candidateBounds, obs, DETAIL_TAG_GAP))) {
   811→          placed.push(candidateBounds);
   812→          return candidate;
   813→        }
   814→      }
   815→
   816→      const bounds = estimateTagBounds(tag);
   817→      if (bounds) placed.push(bounds);
   818→      return tag;
   819→    });
   820→}

   822→function resolveDetailCrestPoint(roofOutline = []) {
   823→  if (!roofOutline.length) {
   824→    return { x: 0, z: 0 };
   825→  }
   826→
   827→  const maxZ = Math.max(...roofOutline.map((point) => point.z));
   828→  const crestPoints = roofOutline.filter((point) => Math.abs(point.z - maxZ) <= 1e-6);
   829→  const averageX = crestPoints.reduce((sum, point) => sum + point.x, 0) / Math.max(crestPoints.length, 1);
   830→
   831→  return {
   832→    x: averageX,
   833→    z: maxZ,
   834→  };
   835→}

  1010→function buildTrussDetailWebTags(lineSegments, instanceGeometry) {
  1011→  const spanMidpoint = (instanceGeometry?.metrics?.span || 0) / 2;
  1012→  let webIndex = 0;
  1013→
  1014→  return lineSegments
  1015→    .filter((segment) => segment.memberType === 'web' && segment.start && segment.end)
  1016→    .map((segment) => {
  1017→      webIndex += 1;
  1018→
  1019→      const start = createDetailPoint(segment.start);
  1020→      const end = createDetailPoint(segment.end);
  1021→      const dx = end.x - start.x;
  1022→      const dy = end.y - start.y;
  1023→      const length = Math.hypot(dx, dy);
  1024→      if (length <= 1) return null;
  1025→
  1026→      const midpoint = {
  1027→        x: (start.x + end.x) / 2,
  1028→        y: (start.y + end.y) / 2,
  1029→      };
  1030→      let offset = { x: 0, y: 0 };
  1031→
  1032→      if (Math.abs(dx) <= 1e-6) {
  1033→        offset = {
  1034→          x: midpoint.x < spanMidpoint ? -DETAIL_WEB_TAG_OFFSET : DETAIL_WEB_TAG_OFFSET,
  1035→          y: 0,
  1036→        };
  1037→      } else {
  1038→        const normal = { x: -dy / length, y: dx / length };
  1039→        const tangent = { x: dx / length, y: dy / length };
  1040→        const outwardNormal = selectOutwardNormal(normal, midpoint, spanMidpoint);
  1041→        const tangentShift = (webIndex % 2 === 0 ? 1 : -1) * DETAIL_WEB_TAG_TANGENT_SHIFT;
  1042→        offset = {
  1043→          x: (outwardNormal.x * DETAIL_WEB_TAG_OFFSET) + (tangent.x * tangentShift),
  1044→          y: (outwardNormal.y * DETAIL_WEB_TAG_OFFSET) + (tangent.y * tangentShift),
  1045→        };
  1046→      }
  1047→
  1048→      return createDetailTag(
  1049→        `${instanceGeometry.instance.id}_detail_web_${webIndex}`,
  1050→        [`W${webIndex} ${formatMeasurement(length)}`],
  1051→        {
  1052→          x: midpoint.x + offset.x,
  1053→          y: midpoint.y + offset.y,
  1054→        },
  1055→        {
  1056→          angle: 0,
  1057→          sourceType: 'truss_web',
  1058→          sourceId: segment.id,
  1059→          measurementValue: length,
  1060→          leaderLine: {
  1061→            start: midpoint,
  1062→            end: {
  1063→              x: midpoint.x + (offset.x * 0.72),
  1064→              y: midpoint.y + (offset.y * 0.72),
  1065→            },
  1066→          },
  1067→        }
  1068→      );
  1069→    })
  1070→    .filter(Boolean);
  1071→}

  1073→function buildTopChordRuns(instanceGeometry) {

  1091→function buildTrussDetailTopChordTags(instanceGeometry) {
  1092→  return buildTopChordRuns(instanceGeometry)
  1093→    .map((run, index) => {
  1094→      const midpointSample = sampleTopChordRunMidpoint(run);
  1095→      if (!midpointSample?.point || !midpointSample?.tangent) return null;
  1096→
  1097→      const detailTangent = {
  1098→        x: midpointSample.tangent.x,
  1099→        y: -midpointSample.tangent.z,
  1100→      };
  1101→      const tangentLength = Math.hypot(detailTangent.x, detailTangent.y);
  1102→      const length = getTopChordRunLength(run);
  1103→      if (length <= 1 || tangentLength <= 1e-6) return null;
  1104→
  1105→      const midpoint = {
  1106→        x: midpointSample.point.x,
  1107→        y: -midpointSample.point.z,
  1108→      };
  1109→      const outwardNormal = {
  1110→        x: detailTangent.y / tangentLength,
  1111→        y: -detailTangent.x / tangentLength,
  1112→      };
  1113→
  1114→      return createDetailTag(
  1115→        `${instanceGeometry.instance.id}_detail_top_chord_${run.id}`,
  1116→        [`TC${index + 1} ${formatMeasurement(length)}`],
  1117→        {
  1118→          x: midpoint.x + (outwardNormal.x * DETAIL_TOP_CHORD_TAG_OFFSET),
  1119→          y: midpoint.y + (outwardNormal.y * DETAIL_TOP_CHORD_TAG_OFFSET),
  1120→        },
  1121→        {
  1122→          angle: 0,
  1123→          sourceType: 'truss_top_chord',
  1124→          sourceId: run.id,
  1125→          measurementValue: length,
  1126→          priority: 1,
  1127→          leaderLine: {
  1128→            start: midpoint,
  1129→            end: {
  1130→              x: midpoint.x + (outwardNormal.x * DETAIL_TOP_CHORD_TAG_OFFSET * 0.78),
  1131→              y: midpoint.y + (outwardNormal.y * DETAIL_TOP_CHORD_TAG_OFFSET * 0.78),
  1132→            },
  1133→          },
  1134→        }
  1135→      );
  1136→    })
  1137→    .filter(Boolean);
  1138→}

  1204→export function buildTrussDetailScene(trussSystem, trussInstanceId = null) {

  1209→  if (!activeInstance) return null;
  1210→
  1211→  const nodeMap = new Map(activeInstance.nodes.map((node) => [node.id, node]));
  1212→  const lineSegments = activeInstance.members.map((member) => ({

  1225→  const dimensionObstacles = detailAnnotations.dimensions
  1226→    .map((fig) => estimateDimensionBounds(fig))
