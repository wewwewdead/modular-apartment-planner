    40→function useUnits() {
    41→  const [unit, setUnit] = useState('mm');
    42→  const isMm = unit === 'mm';
    43→  return {
    44→    unit,
    45→    setUnit,
    46→    suffix: isMm ? 'mm' : 'm',
    47→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    48→    fromDisplay: (v) => isMm ? v : v * 1000,
    49→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    50→  };
    51→}

    53→function WallDrawingInput({ start, preview, dispatch, editorDispatch, activeFloorId, u, selectNewWall }) {
    54→  const [nsDir, setNsDir] = useState('N');
    55→  const [ewDir, setEwDir] = useState('E');
    56→  const [degrees, setDegrees] = useState('');
    57→  const [minutes, setMinutes] = useState('');
    58→  const [lengthVal, setLengthVal] = useState('');
    59→  const lengthRef = useRef(null);
    60→  const livePreviewLength = preview ? formatMeasurement(wallLength({ start, end: preview })) : '';
    61→  const livePreviewBearing = preview ? formatSurveyorBearing(pointsToSurveyorBearing(start, preview)) : '';
    62→
    63→  const handleConfirm = () => {
    64→    const deg = parseInt(degrees, 10) || 0;
    65→    const min = parseInt(minutes, 10) || 0;
    66→    const len = u.fromDisplay(parseFloat(lengthVal) || 0);
    67→
    68→    if (len < MIN_WALL_LENGTH) return;
    69→    if (deg < 0 || deg > 90 || min < 0 || min > 59) return;
    70→
    71→    const angle = surveyorBearingToSvgAngle(nsDir, deg, min, ewDir);
    72→    const endPt = endpointFromBearing(start, len, angle);
    73→    const wall = createWall(start, endPt);
    74→
    75→    dispatch({ type: 'WALL_ADD', floorId: activeFloorId, wall });
    76→    if (selectNewWall) {
    77→      editorDispatch({ type: 'SELECT_OBJECT', id: wall.id, objectType: 'wall' });
    78→    } else {
    79→      editorDispatch({
    80→        type: 'UPDATE_TOOL_STATE',
    81→        payload: { start: endPt, startAttachment: null, preview: null },
    82→      });
    83→    }
    84→
    85→    setDegrees('');
    86→    setMinutes('');
    87→    setLengthVal('');
    88→    setTimeout(() => lengthRef.current?.focus(), 0);
    89→  };
    90→
    91→  const handleKeyDown = (e) => {
    92→    if (e.key === 'Enter') {
    93→      e.preventDefault();
    94→      handleConfirm();
    95→    }
    96→  };
    97→
    98→  return (
    99→    <div>
   100→      {!selectNewWall && (
   101→        <>
   102→          <div className={styles.title}>Drawing Wall</div>
   103→          <div style={{ fontSize: 12, color: 'var(--color-text-secondary)', marginBottom: 'var(--space-sm)' }}>
   104→            Start: ({u.toDisplay(start.x)}, {u.toDisplay(start.y)}) {u.suffix}
   105→          </div>
   106→        </>
   107→      )}
   108→      {preview && (
   109→        <div className={styles.drawingMeta}>
   110→          Cursor: {livePreviewLength} · {livePreviewBearing}
   111→        </div>
   112→      )}
   113→      <div className={styles.drawingHint}>
   114→        Hold Shift while drawing to snap the wall bearing to 45° increments.
   115→      </div>
   116→
   117→      <div className={styles.subtitle}>Bearing</div>
   118→      <div className={styles.bearingRow}>
   119→        <select className={styles.bearingSelect} value={nsDir} onChange={(e) => setNsDir(e.target.value)}>
   120→          <option value="N">N</option>
   121→          <option value="S">S</option>
   122→        </select>
   123→        <input
   124→          className={styles.bearingInput}
   125→          type="number"
   126→          min={0}
   127→          max={90}
   128→          placeholder="0"
   129→          value={degrees}
   130→          onChange={(e) => setDegrees(e.target.value)}
   131→          onKeyDown={handleKeyDown}
   132→        />
   133→        <span className={styles.bearingSymbol}>°</span>
   134→        <input
   135→          className={styles.bearingInput}
   136→          type="number"
   137→          min={0}
   138→          max={59}
   139→          placeholder="0"
   140→          value={minutes}
   141→          onChange={(e) => setMinutes(e.target.value)}
   142→          onKeyDown={handleKeyDown}
   143→        />
   144→        <span className={styles.bearingSymbol}>′</span>
   145→        <select className={styles.bearingSelect} value={ewDir} onChange={(e) => setEwDir(e.target.value)}>
   146→          <option value="E">E</option>
   147→          <option value="W">W</option>
   148→        </select>
   149→      </div>
   150→
   151→      <div className={styles.subtitle}>Length</div>
   152→      <input
   153→        ref={lengthRef}
   154→        className={styles.bearingInput}
   155→        style={{ width: '100%', textAlign: 'left', padding: '0 8px', marginBottom: 'var(--space-sm)' }}
   156→        type="number"
   157→        min={0}
   158→        step={u.step(100)}
   159→        placeholder={`0 ${u.suffix}`}
   160→        value={lengthVal}
   161→        onChange={(e) => setLengthVal(e.target.value)}
   162→        onKeyDown={handleKeyDown}
   163→      />
   164→
   165→      <button className={styles.confirmBtn} onClick={handleConfirm}>
   166→        Create Wall (Enter)
   167→      </button>
   168→    </div>
   169→  );
   170→}

   172→function SheetExportMenu({ sheet, editorDispatch }) {
   173→  const [open, setOpen] = useState(false);
   174→  const rootRef = useRef(null);
   175→  const pngOptions = [
   176→    { label: 'PNG 150 DPI', dpi: 150 },
   177→    { label: 'PNG 300 DPI', dpi: 300 },
   178→    { label: 'PNG 600 DPI', dpi: 600 },
   179→  ];
   180→
   181→  useEffect(() => {
   182→    if (!open) return undefined;
   183→
   184→    const handleMouseDown = (event) => {
   185→      if (rootRef.current && !rootRef.current.contains(event.target)) {
   186→        setOpen(false);
   187→      }
   188→    };
   189→
   190→    document.addEventListener('mousedown', handleMouseDown);
   191→    return () => document.removeEventListener('mousedown', handleMouseDown);
   192→  }, [open]);
   193→
   194→  const handleExport = async (kind, options = {}) => {
   195→    if (!sheet) return;
   196→    try {
   197→      if (kind === 'pdf') {
   198→        await exportActiveSheetAsPdf(
   199→          sheet.title || sheet.drawingName || 'sheet',
   200→          sheet.paperSize
   201→        );
   202→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
   203→      } else {
   204→        await exportActiveSheetAsPng(
   205→          sheet.title || sheet.drawingName || 'sheet',
   206→          options
   207→        );
   208→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
   209→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
   210→      }
   211→    } catch (error) {
   212→      editorDispatch({
   213→        type: 'SET_STATUS_MESSAGE',
   214→        message: error.message || `${kind.toUpperCase()} export failed.`,
   215→      });
   216→    } finally {
   217→      setOpen(false);
   218→    }
   219→  };
   220→
   221→  return (
   222→    <div className={styles.exportMenu} ref={rootRef}>
   223→      <button
   224→        className={styles.exportMenuTrigger}
   225→        type="button"
   226→        onClick={() => setOpen((value) => !value)}
   227→      >
   228→        Export sheet
   229→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   230→      </button>
   231→      {open && (
   232→        <div className={styles.exportMenuList}>
   233→          <button
   234→            className={styles.exportMenuItem}
   235→            type="button"
   236→            onClick={() => handleExport('pdf')}
   237→          >
   238→            PDF
   239→          </button>
   240→          {pngOptions.map((option) => (
   241→            <button
   242→              key={option.dpi}
   243→              className={styles.exportMenuItem}
   244→              type="button"
   245→              onClick={() => handleExport('png', { dpi: option.dpi })}
   246→            >
   247→              {option.label}
   248→            </button>
   249→          ))}
   250→        </div>
   251→      )}
   252→    </div>
   253→  );
   254→}

   273→function PhaseProperties({ phase, project, dispatch }) {
   274→  const updatePhase = (updates) => {
   275→    dispatch({ type: 'PHASE_UPDATE', phase: { id: phase.id, ...updates } });
   276→  };
   277→
   278→  const countObjectsInPhase = () => {
   279→    let count = 0;
   280→    const keys = ['walls', 'doors', 'windows', 'columns', 'beams', 'slabs',
   281→      'stairs', 'landings', 'fixtures', 'rooms', 'railings'];
   282→    for (const floor of project.floors) {
   283→      for (const key of keys) {
   284→        for (const obj of floor[key] || []) {
   285→          if (obj.phaseId === phase.id) count++;
   286→        }
   287→      }
   288→    }
   289→    return count;
   290→  };
   291→
   292→  return (
   293→    <div>
   294→      <div className={styles.title}>Phase</div>
   295→      <InputField
   296→        label="Name"
   297→        value={phase.name}
   298→        onChange={(value) => updatePhase({ name: value })}
   299→      />
   300→      <div className={styles.colorField}>
   301→        <label className={styles.colorLabel}>Color</label>
   302→        <div className={styles.colorControls}>
   303→          <input
   304→            className={styles.colorPicker}
   305→            type="color"
   306→            value={phase.color}
   307→            onChange={(e) => updatePhase({ color: e.target.value })}
   308→            aria-label="Phase color"
   309→          />
   310→          <input
   311→            className={styles.colorHexInput}
   312→            type="text"
   313→            value={phase.color}
   314→            onChange={(e) => {
   315→              const hex = e.target.value;
   316→              if (/^#[0-9a-fA-F]{6}$/.test(hex)) updatePhase({ color: hex });
   317→            }}
   318→            onBlur={(e) => {
   319→              let hex = e.target.value.trim();
   320→              if (!hex.startsWith('#')) hex = '#' + hex;
   321→              if (/^#[0-9a-fA-F]{6}$/.test(hex)) updatePhase({ color: hex });
   322→            }}
   323→          />
   324→        </div>
   325→      </div>
   326→      <InputField label="Order" value={phase.order} readOnly />
   327→      <InputField label="Objects" value={countObjectsInPhase()} readOnly />
   328→    </div>
   329→  );
   330→}

   532→      <InputField
   533→        label="Start"
   534→        value={startColumn ? getColumnListLabel(startColumn, floor.columns || []) : beam.startRef?.id || '\u2014'}
   535→        readOnly

   611→        onChange={(value) => updateAnnotation({ endPoint: { ...annotation.endPoint, y: u.fromDisplay(value) } })}
   612→      />
   613→      <div className={styles.subtitle}>Properties</div>
   614→      <InputField
   615→        label="Offset" type="number" suffix={u.suffix} step={u.step(10)}
   616→        value={u.toDisplay(annotation.offset)}
   617→        onChange={(value) => updateAnnotation({ offset: u.fromDisplay(value) })}
   618→      />
   619→      <InputField
   620→        label="Text"
   621→        value={annotation.textOverride}
   622→        onChange={(value) => updateAnnotation({ textOverride: value })}
   623→      />
   624→      <InputField
   625→        label="Measured" type="number" suffix={u.suffix} step={u.step(100)}
   626→        value={u.toDisplay(measuredValue)}
   627→        onChange={(value) => {
   628→          const newLen = u.fromDisplay(value);
   629→          if (newLen <= 0) return;
   630→          const s = annotation.startPoint;
   631→          const e = annotation.endPoint;
   632→          const mode = annotation.mode;
   633→          let newEnd;
   634→          if (mode === 'horizontal') {
   635→            const sign = e.x >= s.x ? 1 : -1;
   636→            newEnd = { ...e, x: s.x + sign * newLen };
   637→          } else if (mode === 'vertical') {
   638→            const sign = e.y >= s.y ? 1 : -1;
   639→            newEnd = { ...e, y: s.y + sign * newLen };
   640→          } else {
   641→            const dir = normalize(subtract(e, s));
   642→            newEnd = add(s, scale(dir, newLen));
   643→          }
   644→          updateAnnotation({ endPoint: newEnd });
   645→        }}
   646→      />
   647→    </div>
   648→  );
