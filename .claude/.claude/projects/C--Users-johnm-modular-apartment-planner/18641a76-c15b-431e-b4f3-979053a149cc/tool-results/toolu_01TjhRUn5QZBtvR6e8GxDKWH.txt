    36→function useUnits() {
    37→  const [unit, setUnit] = useState('mm');
    38→  const isMm = unit === 'mm';
    39→  return {
    40→    unit,
    41→    setUnit,
    42→    suffix: isMm ? 'mm' : 'm',
    43→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    44→    fromDisplay: (v) => isMm ? v : v * 1000,
    45→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    46→  };
    47→}

    49→function WallDrawingInput({ start, dispatch, editorDispatch, activeFloorId, u }) {
    50→  const [nsDir, setNsDir] = useState('N');
    51→  const [ewDir, setEwDir] = useState('E');
    52→  const [degrees, setDegrees] = useState('');
    53→  const [minutes, setMinutes] = useState('');
    54→  const [lengthVal, setLengthVal] = useState('');
    55→  const lengthRef = useRef(null);
    56→
    57→  const handleConfirm = () => {
    58→    const deg = parseInt(degrees, 10) || 0;
    59→    const min = parseInt(minutes, 10) || 0;
    60→    const len = u.fromDisplay(parseFloat(lengthVal) || 0);
    61→
    62→    if (len < MIN_WALL_LENGTH) return;
    63→    if (deg < 0 || deg > 90 || min < 0 || min > 59) return;
    64→
    65→    const angle = surveyorBearingToSvgAngle(nsDir, deg, min, ewDir);
    66→    const endPt = endpointFromBearing(start, len, angle);
    67→    const wall = createWall(start, endPt);
    68→
    69→    dispatch({ type: 'WALL_ADD', floorId: activeFloorId, wall });
    70→    editorDispatch({
    71→      type: 'UPDATE_TOOL_STATE',
    72→      payload: { start: endPt, startAttachment: null, preview: null },
    73→    });
    74→
    75→    setDegrees('');
    76→    setMinutes('');
    77→    setLengthVal('');
    78→    setTimeout(() => lengthRef.current?.focus(), 0);
    79→  };
    80→
    81→  const handleKeyDown = (e) => {
    82→    if (e.key === 'Enter') {
    83→      e.preventDefault();
    84→      handleConfirm();
    85→    }
    86→  };
    87→
    88→  return (
    89→    <div>
    90→      <div className={styles.title}>Drawing Wall</div>
    91→      <div style={{ fontSize: 12, color: 'var(--color-text-secondary)', marginBottom: 'var(--space-sm)' }}>
    92→        Start: ({u.toDisplay(start.x)}, {u.toDisplay(start.y)}) {u.suffix}
    93→      </div>
    94→
    95→      <div className={styles.subtitle}>Bearing</div>
    96→      <div className={styles.bearingRow}>
    97→        <select className={styles.bearingSelect} value={nsDir} onChange={(e) => setNsDir(e.target.value)}>
    98→          <option value="N">N</option>
    99→          <option value="S">S</option>
   100→        </select>
   101→        <input
   102→          className={styles.bearingInput}
   103→          type="number"
   104→          min={0}
   105→          max={90}
   106→          placeholder="0"
   107→          value={degrees}
   108→          onChange={(e) => setDegrees(e.target.value)}
   109→          onKeyDown={handleKeyDown}
   110→        />
   111→        <span className={styles.bearingSymbol}>°</span>
   112→        <input
   113→          className={styles.bearingInput}
   114→          type="number"
   115→          min={0}
   116→          max={59}
   117→          placeholder="0"
   118→          value={minutes}
   119→          onChange={(e) => setMinutes(e.target.value)}
   120→          onKeyDown={handleKeyDown}
   121→        />
   122→        <span className={styles.bearingSymbol}>′</span>
   123→        <select className={styles.bearingSelect} value={ewDir} onChange={(e) => setEwDir(e.target.value)}>
   124→          <option value="E">E</option>
   125→          <option value="W">W</option>
   126→        </select>
   127→      </div>
   128→
   129→      <div className={styles.subtitle}>Length</div>
   130→      <input
   131→        ref={lengthRef}
   132→        className={styles.bearingInput}
   133→        style={{ width: '100%', textAlign: 'left', padding: '0 8px', marginBottom: 'var(--space-sm)' }}
   134→        type="number"
   135→        min={0}
   136→        step={u.step(100)}
   137→        placeholder={`0 ${u.suffix}`}
   138→        value={lengthVal}
   139→        onChange={(e) => setLengthVal(e.target.value)}
   140→        onKeyDown={handleKeyDown}
   141→      />
   142→
   143→      <button className={styles.confirmBtn} onClick={handleConfirm}>
   144→        Create Wall (Enter)
   145→      </button>
   146→    </div>
   147→  );
   148→}

   150→function SheetExportMenu({ sheet, editorDispatch }) {
   151→  const [open, setOpen] = useState(false);
   152→  const rootRef = useRef(null);
   153→  const pngOptions = [
   154→    { label: 'PNG 150 DPI', dpi: 150 },
   155→    { label: 'PNG 300 DPI', dpi: 300 },
   156→    { label: 'PNG 600 DPI', dpi: 600 },
   157→  ];
   158→
   159→  useEffect(() => {
   160→    if (!open) return undefined;
   161→
   162→    const handleMouseDown = (event) => {
   163→      if (rootRef.current && !rootRef.current.contains(event.target)) {
   164→        setOpen(false);
   165→      }
   166→    };
   167→
   168→    document.addEventListener('mousedown', handleMouseDown);
   169→    return () => document.removeEventListener('mousedown', handleMouseDown);
   170→  }, [open]);
   171→
   172→  const handleExport = async (kind, options = {}) => {
   173→    if (!sheet) return;
   174→    try {
   175→      if (kind === 'pdf') {
   176→        await exportActiveSheetAsPdf(
   177→          sheet.title || sheet.drawingName || 'sheet',
   178→          sheet.paperSize
   179→        );
   180→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
   181→      } else {
   182→        await exportActiveSheetAsPng(
   183→          sheet.title || sheet.drawingName || 'sheet',
   184→          options
   185→        );
   186→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
   187→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
   188→      }
   189→    } catch (error) {
   190→      editorDispatch({
   191→        type: 'SET_STATUS_MESSAGE',
   192→        message: error.message || `${kind.toUpperCase()} export failed.`,
   193→      });
   194→    } finally {
   195→      setOpen(false);
   196→    }
   197→  };
   198→
   199→  return (
   200→    <div className={styles.exportMenu} ref={rootRef}>
   201→      <button
   202→        className={styles.exportMenuTrigger}
   203→        type="button"
   204→        onClick={() => setOpen((value) => !value)}
   205→      >
   206→        Export sheet
   207→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   208→      </button>
   209→      {open && (
   210→        <div className={styles.exportMenuList}>
   211→          <button
   212→            className={styles.exportMenuItem}
   213→            type="button"
   214→            onClick={() => handleExport('pdf')}
   215→          >
   216→            PDF
   217→          </button>
   218→          {pngOptions.map((option) => (
   219→            <button
   220→              key={option.dpi}
   221→              className={styles.exportMenuItem}
   222→              type="button"
   223→              onClick={() => handleExport('png', { dpi: option.dpi })}
   224→            >
   225→              {option.label}
   226→            </button>
   227→          ))}
   228→        </div>
   229→      )}
   230→    </div>
   231→  );
   232→}
