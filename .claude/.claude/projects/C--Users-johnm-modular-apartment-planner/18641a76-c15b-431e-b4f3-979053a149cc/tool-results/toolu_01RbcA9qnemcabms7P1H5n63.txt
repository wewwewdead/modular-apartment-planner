    34→function useUnits() {
    35→  const [unit, setUnit] = useState('mm');
    36→  const isMm = unit === 'mm';
    37→  return {
    38→    unit,
    39→    setUnit,
    40→    suffix: isMm ? 'mm' : 'm',
    41→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    42→    fromDisplay: (v) => isMm ? v : v * 1000,
    43→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    44→  };
    45→}

    47→function SheetExportMenu({ sheet, editorDispatch }) {
    48→  const [open, setOpen] = useState(false);
    49→  const rootRef = useRef(null);
    50→  const pngOptions = [
    51→    { label: 'PNG 150 DPI', dpi: 150 },
    52→    { label: 'PNG 300 DPI', dpi: 300 },
    53→    { label: 'PNG 600 DPI', dpi: 600 },
    54→  ];
    55→
    56→  useEffect(() => {
    57→    if (!open) return undefined;
    58→
    59→    const handleMouseDown = (event) => {
    60→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    61→        setOpen(false);
    62→      }
    63→    };
    64→
    65→    document.addEventListener('mousedown', handleMouseDown);
    66→    return () => document.removeEventListener('mousedown', handleMouseDown);
    67→  }, [open]);
    68→
    69→  const handleExport = async (kind, options = {}) => {
    70→    if (!sheet) return;
    71→    try {
    72→      if (kind === 'pdf') {
    73→        await exportActiveSheetAsPdf(
    74→          sheet.title || sheet.drawingName || 'sheet',
    75→          sheet.paperSize
    76→        );
    77→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    78→      } else {
    79→        await exportActiveSheetAsPng(
    80→          sheet.title || sheet.drawingName || 'sheet',
    81→          options
    82→        );
    83→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    84→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    85→      }
    86→    } catch (error) {
    87→      editorDispatch({
    88→        type: 'SET_STATUS_MESSAGE',
    89→        message: error.message || `${kind.toUpperCase()} export failed.`,
    90→      });
    91→    } finally {
    92→      setOpen(false);
    93→    }
    94→  };
    95→
    96→  return (
    97→    <div className={styles.exportMenu} ref={rootRef}>
    98→      <button
    99→        className={styles.exportMenuTrigger}
   100→        type="button"
   101→        onClick={() => setOpen((value) => !value)}
   102→      >
   103→        Export sheet
   104→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   105→      </button>
   106→      {open && (
   107→        <div className={styles.exportMenuList}>
   108→          <button
   109→            className={styles.exportMenuItem}
   110→            type="button"
   111→            onClick={() => handleExport('pdf')}
   112→          >
   113→            PDF
   114→          </button>
   115→          {pngOptions.map((option) => (
   116→            <button
   117→              key={option.dpi}
   118→              className={styles.exportMenuItem}
   119→              type="button"
   120→              onClick={() => handleExport('png', { dpi: option.dpi })}
   121→            >
   122→              {option.label}
   123→            </button>
   124→          ))}
   125→        </div>
   126→      )}
   127→    </div>
   128→  );
   129→}

   131→function WallProperties({ wall, dispatch, floorId, u }) {
   132→  const len = wallLength(wall);
   133→
   134→  const updateWall = (updates) => {
   135→    dispatch({ type: 'WALL_UPDATE', floorId, wall: { id: wall.id, ...updates } });
   136→  };
   137→
   138→  const updateWallLength = (value) => {
   139→    const resizedWall = resizeWallFromStart(wall, u.fromDisplay(value), MIN_WALL_LENGTH);
   140→    updateWall({ end: resizedWall.end });
   141→  };
   142→
   143→  return (
   144→    <div>
   145→      <div className={styles.title}>Wall</div>
   146→      <div className={styles.subtitle}>Start Point</div>
   147→      <InputField
   148→        label="X" type="number" suffix={u.suffix}
   149→        value={u.toDisplay(wall.start.x)}
   150→        onChange={(v) => updateWall({ start: { ...wall.start, x: u.fromDisplay(v) } })}
   151→      />
   152→      <InputField
   153→        label="Y" type="number" suffix={u.suffix}
   154→        value={u.toDisplay(wall.start.y)}
   155→        onChange={(v) => updateWall({ start: { ...wall.start, y: u.fromDisplay(v) } })}
   156→      />
   157→      <div className={styles.subtitle}>End Point</div>
   158→      <InputField
   159→        label="X" type="number" suffix={u.suffix}
   160→        value={u.toDisplay(wall.end.x)}
   161→        onChange={(v) => updateWall({ end: { ...wall.end, x: u.fromDisplay(v) } })}
   162→      />
   163→      <InputField
   164→        label="Y" type="number" suffix={u.suffix}
   165→        value={u.toDisplay(wall.end.y)}
   166→        onChange={(v) => updateWall({ end: { ...wall.end, y: u.fromDisplay(v) } })}
   167→      />
   168→      <div className={styles.subtitle}>Properties</div>

   170→        label="Thickness" type="number" suffix={u.suffix} step={u.step(10)}

   175→        label="Height" type="number" suffix={u.suffix} step={u.step(100)}

   906→function SheetProperties({ sheet, project, dispatch, editorDispatch, activeFloorId, u }) {

   916→        type: 'SHEET_UPDATE',

   925→        ...viewport,
   926→        lockAutoLayout: false,
   927→      })),
   928→    });
   929→  };
   930→
   931→  const updateTitleBlock = (updates) => {
   932→    updateSheet({
   933→      titleBlock: {
   934→        ...(sheet.titleBlock || {}),
   935→        ...updates,
   936→      },
   937→    });
   938→  };
   939→
   940→  const updateRevisions = (nextRevisions) => {

   946→      const sourceFloorId = getDefaultActiveFloorId(project, activeFloorId);

  1381→export default function PropertiesPanel() {
  1382→  const { project, dispatch, duplicateFloor } = useProject();
  1383→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, dispatch: editorDispatch } = useEditor();
  1384→  const orderedFloors = getOrderedFloors(project);
  1385→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1386→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1387→  const u = useUnits();
  1388→
  1389→  const selectFloor = (floorId) => {
  1390→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1391→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1392→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1393→  };
  1394→
  1395→  const handleDuplicateFloor = (floorToDuplicate) => {
  1396→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1397→    if (!duplicatedFloor) return;
  1398→
  1399→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1400→    selectFloor(duplicatedFloor.id);
  1401→  };
  1402→
  1403→  const handleDeleteFloor = (floorToDelete) => {
  1404→    if (orderedFloors.length <= 1) return;
  1405→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1406→    if (!confirmed) return;
  1407→
  1408→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1409→    const fallbackFloor = remainingFloors[0] || null;
  1410→
  1411→    dispatch({
  1412→      type: 'FLOOR_DELETE',
  1413→      floorId: floorToDelete.id,
  1414→      fallbackFloorId: fallbackFloor?.id ?? null,
  1415→    });
  1416→
  1417→    if (fallbackFloor) {
  1418→      selectFloor(fallbackFloor.id);
  1419→    } else {
  1420→      editorDispatch({ type: 'DESELECT' });
  1421→    }
  1422→  };
  1423→
  1424→  const handleDelete = () => {
  1425→    if (!selectedId) return;
  1426→    if (selectedType === 'sheet') {
  1427→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1428→    } else if (selectedType === 'sheetViewport' && sheet) {
  1429→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1430→    } else if (selectedType === 'slab') {
  1431→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1432→    } else if (selectedType === 'wall') {
  1433→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1434→    } else if (selectedType === 'beam') {
  1435→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1436→    } else if (selectedType === 'sectionCut') {
  1437→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1438→    } else if (selectedType === 'annotation') {
  1439→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1440→    } else if (selectedType === 'stair') {
  1441→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1442→    } else if (selectedType === 'landing') {
  1443→      dispatch({ type: 'LANDING_DELETE', floorId: activeFloorId, landingId: selectedId });
  1444→    } else if (selectedType === 'door') {
  1445→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1446→    } else if (selectedType === 'window') {
  1447→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1448→    } else if (selectedType === 'column') {
  1449→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1450→    } else if (selectedType === 'fixture') {
  1451→      dispatch({ type: 'FIXTURE_DELETE', floorId: activeFloorId, fixtureId: selectedId });
  1452→    } else if (selectedType === 'room') {
  1453→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1454→    } else if (selectedType === 'floor') {
  1455→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1456→      if (targetFloor) {
  1457→        handleDeleteFloor(targetFloor);
  1458→        return;
  1459→      }
  1460→    }
  1461→    editorDispatch({ type: 'DESELECT' });
  1462→  };
  1463→
  1464→  let content = workspaceMode === 'sheet' && sheet
