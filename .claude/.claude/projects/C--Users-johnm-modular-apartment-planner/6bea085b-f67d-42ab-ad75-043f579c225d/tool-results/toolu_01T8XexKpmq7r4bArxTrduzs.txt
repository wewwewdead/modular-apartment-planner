    27→    27→function useUnits() {
    28→    28→  const [unit, setUnit] = useState('mm');
    29→    29→  const isMm = unit === 'mm';
    30→    30→  return {
    31→    31→    unit,
    32→    32→    setUnit,
    33→    33→    suffix: isMm ? 'mm' : 'm',
    34→    34→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    35→    35→    fromDisplay: (v) => isMm ? v : v * 1000,
    36→    36→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    37→    37→  };
    38→    38→}

    40→    40→function SheetExportMenu({ sheet, editorDispatch }) {
    41→    41→  const [open, setOpen] = useState(false);
    42→    42→  const rootRef = useRef(null);
    43→    43→  const pngOptions = [
    44→    44→    { label: 'PNG 150 DPI', dpi: 150 },
    45→    45→    { label: 'PNG 300 DPI', dpi: 300 },
    46→    46→    { label: 'PNG 600 DPI', dpi: 600 },
    47→    47→  ];
    48→    48→
    49→    49→  useEffect(() => {
    50→    50→    if (!open) return undefined;
    51→    51→
    52→    52→    const handleMouseDown = (event) => {
    53→    53→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    54→    54→        setOpen(false);
    55→    55→      }
    56→    56→    };
    57→    57→
    58→    58→    document.addEventListener('mousedown', handleMouseDown);
    59→    59→    return () => document.removeEventListener('mousedown', handleMouseDown);
    60→    60→  }, [open]);
    61→    61→
    62→    62→  const handleExport = async (kind, options = {}) => {
    63→    63→    if (!sheet) return;
    64→    64→    try {
    65→    65→      if (kind === 'pdf') {
    66→    66→        await exportActiveSheetAsPdf(
    67→    67→          sheet.title || sheet.drawingName || 'sheet',
    68→    68→          sheet.paperSize
    69→    69→        );
    70→    70→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    71→    71→      } else {
    72→    72→        await exportActiveSheetAsPng(
    73→    73→          sheet.title || sheet.drawingName || 'sheet',
    74→    74→          options
    75→    75→        );
    76→    76→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    77→    77→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    78→    78→      }
    79→    79→    } catch (error) {
    80→    80→      editorDispatch({
    81→    81→        type: 'SET_STATUS_MESSAGE',
    82→    82→        message: error.message || `${kind.toUpperCase()} export failed.`,
    83→    83→      });
    84→    84→    } finally {
    85→    85→      setOpen(false);
    86→    86→    }
    87→    87→  };
    88→    88→
    89→    89→  return (
    90→    90→    <div className={styles.exportMenu} ref={rootRef}>
    91→    91→      <button
    92→    92→        className={styles.exportMenuTrigger}
    93→    93→        type="button"
    94→    94→        onClick={() => setOpen((value) => !value)}
    95→    95→      >
    96→    96→        Save sheet
    97→    97→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    98→    98→      </button>
    99→    99→      {open && (
   100→   100→        <div className={styles.exportMenuList}>
   101→   101→          <button
   102→   102→            className={styles.exportMenuItem}
   103→   103→            type="button"
   104→   104→            onClick={() => handleExport('pdf')}
   105→   105→          >
   106→   106→            PDF
   107→   107→          </button>
   108→   108→          {pngOptions.map((option) => (
   109→   109→            <button
   110→   110→              key={option.dpi}
   111→   111→              className={styles.exportMenuItem}
   112→   112→              type="button"
   113→   113→              onClick={() => handleExport('png', { dpi: option.dpi })}
   114→   114→            >
   115→   115→              {option.label}
   116→   116→            </button>
   117→   117→          ))}
   118→   118→        </div>
   119→   119→      )}
   120→   120→    </div>
   121→   121→  );
   122→   122→}

   829→   829→function SheetProperties({ sheet, project, dispatch, editorDispatch, activeFloorId, u }) {

   835→   835→    try {
   836→   836→      const sourceFloorId = getDefaultActiveFloorId(project, activeFloorId);
   837→   837→      const sourceView = 'plan';
   838→   838→      const viewportBase = createSheetViewport(sourceView, sourceFloorId, { scale: 100 });
   839→   839→
   840→   840→      let rect;
   841→   841→      try {

   843→   843→        rect = getDefaultViewportRect(sheet, source, viewportBase.scale);
   844→   844→      } catch {
