    28→function useUnits() {
    29→  const [unit, setUnit] = useState('mm');
    30→  const isMm = unit === 'mm';
    31→  return {
    32→    unit,
    33→    setUnit,
    34→    suffix: isMm ? 'mm' : 'm',
    35→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    36→    fromDisplay: (v) => isMm ? v : v * 1000,
    37→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    38→  };
    39→}

    41→function SheetExportMenu({ sheet, editorDispatch }) {
    42→  const [open, setOpen] = useState(false);
    43→  const rootRef = useRef(null);
    44→  const pngOptions = [
    45→    { label: 'PNG 150 DPI', dpi: 150 },
    46→    { label: 'PNG 300 DPI', dpi: 300 },
    47→    { label: 'PNG 600 DPI', dpi: 600 },
    48→  ];
    49→
    50→  useEffect(() => {
    51→    if (!open) return undefined;
    52→
    53→    const handleMouseDown = (event) => {
    54→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    55→        setOpen(false);
    56→      }
    57→    };
    58→
    59→    document.addEventListener('mousedown', handleMouseDown);
    60→    return () => document.removeEventListener('mousedown', handleMouseDown);
    61→  }, [open]);
    62→
    63→  const handleExport = async (kind, options = {}) => {
    64→    if (!sheet) return;
    65→    try {
    66→      if (kind === 'pdf') {
    67→        await exportActiveSheetAsPdf(
    68→          sheet.title || sheet.drawingName || 'sheet',
    69→          sheet.paperSize
    70→        );
    71→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    72→      } else {
    73→        await exportActiveSheetAsPng(
    74→          sheet.title || sheet.drawingName || 'sheet',
    75→          options
    76→        );
    77→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    78→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    79→      }
    80→    } catch (error) {
    81→      editorDispatch({
    82→        type: 'SET_STATUS_MESSAGE',
    83→        message: error.message || `${kind.toUpperCase()} export failed.`,
    84→      });
    85→    } finally {
    86→      setOpen(false);
    87→    }
    88→  };
    89→
    90→  return (
    91→    <div className={styles.exportMenu} ref={rootRef}>
    92→      <button
    93→        className={styles.exportMenuTrigger}
    94→        type="button"
    95→        onClick={() => setOpen((value) => !value)}
    96→      >
    97→        Export sheet
    98→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    99→      </button>
   100→      {open && (
   101→        <div className={styles.exportMenuList}>
   102→          <button
   103→            className={styles.exportMenuItem}
   104→            type="button"
   105→            onClick={() => handleExport('pdf')}
   106→          >
   107→            PDF
   108→          </button>
   109→          {pngOptions.map((option) => (
   110→            <button
   111→              key={option.dpi}
   112→              className={styles.exportMenuItem}
   113→              type="button"
   114→              onClick={() => handleExport('png', { dpi: option.dpi })}
   115→            >
   116→              {option.label}
   117→            </button>
   118→          ))}
   119→        </div>
   120→      )}
   121→    </div>
   122→  );
   123→}

   501→function LandingProperties({ landing, floor, dispatch, floorId, u }) {
   502→  const updateLanding = (updates) => {
   503→    dispatch({ type: 'LANDING_UPDATE', floorId, landing: { id: landing.id, ...updates } });
   504→  };
   505→
   506→  const stairs = floor?.stairs || [];

  1350→export default function PropertiesPanel() {

  1353→  const orderedFloors = getOrderedFloors(project);
  1354→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1355→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1356→  const u = useUnits();
  1357→
  1358→  const selectFloor = (floorId) => {
  1359→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1360→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1361→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1362→  };
  1363→
  1364→  const handleDuplicateFloor = (floorToDuplicate) => {
  1365→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1366→    if (!duplicatedFloor) return;
  1367→
  1368→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1369→    selectFloor(duplicatedFloor.id);
  1370→  };
  1371→
  1372→  const handleDeleteFloor = (floorToDelete) => {
  1373→    if (orderedFloors.length <= 1) return;
  1374→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1375→    if (!confirmed) return;
  1376→
  1377→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1378→    const fallbackFloor = remainingFloors[0] || null;
  1379→
  1380→    dispatch({
  1381→      type: 'FLOOR_DELETE',
  1382→      floorId: floorToDelete.id,
  1383→      fallbackFloorId: fallbackFloor?.id ?? null,
  1384→    });
  1385→
  1386→    if (fallbackFloor) {
  1387→      selectFloor(fallbackFloor.id);
  1388→    } else {
  1389→      editorDispatch({ type: 'DESELECT' });
  1390→    }
  1391→  };
  1392→
  1393→  const handleDelete = () => {
  1394→    if (!selectedId) return;
  1395→    if (selectedType === 'sheet') {
  1396→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1397→    } else if (selectedType === 'sheetViewport' && sheet) {
  1398→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1399→    } else if (selectedType === 'slab') {
  1400→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1401→    } else if (selectedType === 'wall') {
  1402→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1403→    } else if (selectedType === 'beam') {
  1404→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1405→    } else if (selectedType === 'sectionCut') {
  1406→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1407→    } else if (selectedType === 'annotation') {
  1408→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1409→    } else if (selectedType === 'stair') {
  1410→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1411→    } else if (selectedType === 'landing') {
  1412→      dispatch({ type: 'LANDING_DELETE', floorId: activeFloorId, landingId: selectedId });
  1413→    } else if (selectedType === 'door') {
  1414→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1415→    } else if (selectedType === 'window') {
  1416→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1417→    } else if (selectedType === 'column') {
  1418→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1419→    } else if (selectedType === 'fixture') {
  1420→      dispatch({ type: 'FIXTURE_DELETE', floorId: activeFloorId, fixtureId: selectedId });
  1421→    } else if (selectedType === 'room') {
  1422→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1423→    } else if (selectedType === 'floor') {
  1424→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1425→      if (targetFloor) {
  1426→        handleDeleteFloor(targetFloor);
  1427→        return;
  1428→      }
  1429→    }
  1430→    editorDispatch({ type: 'DESELECT' });
  1431→  };
  1432→
  1433→  let content = workspaceMode === 'sheet' && sheet
  1434→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
  1435→    : (
  1436→        <>
  1437→          {floor && (
  1438→            <FloorProperties
  1439→              floor={floor}
  1440→              dispatch={dispatch}
  1441→              onDuplicate={handleDuplicateFloor}
  1442→              onDelete={handleDeleteFloor}
  1443→              canDelete={orderedFloors.length > 1}
  1444→              u={u}
  1445→            />
  1446→          )}
  1447→          <ProjectSummary project={project} floor={floor} dispatch={dispatch} />
  1448→        </>
  1449→      );
  1450→
  1451→  if (selectedId && workspaceMode === 'sheet' && sheet) {
  1452→    if (selectedType === 'sheet') {
  1453→      if (sheet.id === selectedId) {
  1454→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;
  1455→      }
