    91→function createBounds(points = []) {
    92→  if (!points.length) {
    93→    return {
    94→      minX: 0,
    95→      maxX: 1000,
    96→      minY: 0,
    97→      maxY: 1000,
    98→      minElevation: 0,
    99→      maxElevation: 1000,
   100→    };
   101→  }
   102→
   103→  return {
   104→    minX: Math.min(...points.map((point) => point.x)),
   105→    maxX: Math.max(...points.map((point) => point.x)),
   106→    minY: Math.min(...points.map((point) => point.y)),
   107→    maxY: Math.max(...points.map((point) => point.y)),
   108→    minElevation: Math.min(...points.map((point) => point.elevation ?? 0)),
   109→    maxElevation: Math.max(...points.map((point) => point.elevation ?? 0)),
   110→  };
   111→}

   120→function pointFromLocal(structuralStartPoint, spanDirection, localPoint, baseElevation) {
   121→  const planPoint = add(structuralStartPoint, scale(spanDirection, localPoint.x));
   122→  return {
   123→    x: planPoint.x,
   124→    y: planPoint.y,
   125→    elevation: baseElevation + localPoint.z,
   126→  };
   127→}

   129→function createCopyGeometry(trussSystem, instance, profile, layout, origin, index) {
   130→  const structuralStartPoint = add(origin, scale(layout.spanDirection, -(profile.metrics.span / 2)));
   131→  const structuralEndPoint = add(origin, scale(layout.spanDirection, profile.metrics.span / 2));
   132→  const overallStartPoint = add(structuralStartPoint, scale(layout.spanDirection, -profile.metrics.overhangStart));
   133→  const overallEndPoint = add(structuralEndPoint, scale(layout.spanDirection, profile.metrics.overhangEnd));
   134→  const nodeMap = new Map(profile.nodes.map((node) => [node.id, node]));
   135→  const baseElevation = trussSystem.baseElevation || 0;
   136→
   137→  const worldNodes = profile.nodes.map((node) => ({
   138→    ...node,
   139→    planPoint: add(structuralStartPoint, scale(layout.spanDirection, node.x)),
   140→    worldPoint: pointFromLocal(structuralStartPoint, layout.spanDirection, node, baseElevation),
   141→  }));
   142→  const members = profile.members.map((member) => {
   143→    const startNode = nodeMap.get(member.startNodeId);
   144→    const endNode = nodeMap.get(member.endNodeId);
   145→    return {
   146→      ...member,
   147→      startNode,
   148→      endNode,
   149→      startWorld: pointFromLocal(structuralStartPoint, layout.spanDirection, startNode, baseElevation),
   150→      endWorld: pointFromLocal(structuralStartPoint, layout.spanDirection, endNode, baseElevation),
   151→    };
   152→  });
   153→  const purlinAttachments = (profile.purlinAttachments || []).map((attachment) => ({
   154→    ...attachment,
   155→    planPoint: add(structuralStartPoint, scale(layout.spanDirection, attachment.localPoint.x)),
   156→    worldPoint: pointFromLocal(structuralStartPoint, layout.spanDirection, attachment.localPoint, baseElevation),
   157→  }));
   158→
   159→  return {
   160→    id: `${instance.id}_copy_${index}`,
   161→    index,
   162→    origin,
   163→    structuralStartPoint,
   164→    structuralEndPoint,
   165→    overallStartPoint,
   166→    overallEndPoint,
   167→    worldNodes,
   168→    members,
   169→    purlinAttachments,
   170→  };
   171→}

   173→function normalizePlanVector(vector) {
   174→  const magnitude = Math.sqrt((vector.x * vector.x) + (vector.y * vector.y));
   175→  if (magnitude <= 1e-6) {
   176→    return { x: 1, y: 0 };
   177→  }
   178→  return {
   179→    x: vector.x / magnitude,
   180→    y: vector.y / magnitude,
   181→  };
   182→}

   184→function createPlanMarker(centerPoint, axis, length = DEFAULT_PURLIN_MARKER_LENGTH) {
   185→  return {
   186→    start: add(centerPoint, scale(axis, -(length / 2))),
   187→    end: add(centerPoint, scale(axis, length / 2)),
   188→  };

   224→function offsetWorldPoint(worldPoint, axis, distance) {
   225→  if (!worldPoint || Math.abs(distance) <= 1e-6) return worldPoint;
   226→  return {
   227→    ...worldPoint,
   228→    x: worldPoint.x + (axis.x * distance),
   229→    y: worldPoint.y + (axis.y * distance),
   230→  };
   231→}

   233→function buildInstancePurlinGeometry(trussSystem, instanceGeometry, purlinSystem) {
   234→  if (!purlinSystem?.enabled) {
   235→    return {
   236→      purlinRuns: [],
   237→      purlinMarkers: [],
   238→      purlinSegments: [],
   239→    };
   240→  }
   241→
   242→  const attachmentGroups = new Map();
   243→  for (const copy of instanceGeometry.copies) {
   244→    for (const attachment of copy.purlinAttachments || []) {
   245→      if (!attachmentGroups.has(attachment.id)) {
   246→        attachmentGroups.set(attachment.id, []);
   247→      }
   248→      attachmentGroups.get(attachment.id).push({
   249→        ...attachment,
   250→        copyId: copy.id,
   251→        copyIndex: copy.index,
   252→      });
   253→    }
   254→  }
   255→
   256→  const layoutAxis = normalizePlanVector({
   257→    x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   258→    y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   259→  });
   260→  const purlinRuns = [];
   261→  const purlinMarkers = [];
   262→  const purlinSegments = [];
   263→  const overhangStart = Math.max(Number(purlinSystem?.overhangStart || 0), 0);
   264→  const overhangEnd = Math.max(Number(purlinSystem?.overhangEnd || 0), 0);
   265→
   266→  for (const [attachmentId, attachmentPoints] of attachmentGroups.entries()) {
   267→    const sortedPoints = [...attachmentPoints].sort((a, b) => a.copyIndex - b.copyIndex);
   268→    if (sortedPoints.length >= 2) {
   269→      const firstPoint = sortedPoints[0];
   270→      const lastPoint = sortedPoints[sortedPoints.length - 1];
   271→      const runPoints = [
   272→        ...(overhangStart > 1e-6
   273→          ? [add(firstPoint.planPoint, scale(layoutAxis, -overhangStart))]
   274→          : []),
   275→        ...sortedPoints.map((point) => point.planPoint),
   276→        ...(overhangEnd > 1e-6
   277→          ? [add(lastPoint.planPoint, scale(layoutAxis, overhangEnd))]
   278→          : []),
   279→      ];
   280→      const segments = sortedPoints.slice(0, -1).map((startPoint, index) => {
   281→        const endPoint = sortedPoints[index + 1];
   282→        const startWorld = index === 0
   283→          ? offsetWorldPoint(startPoint.worldPoint, layoutAxis, -overhangStart)
   284→          : startPoint.worldPoint;
   285→        const endWorld = index === sortedPoints.length - 2
   286→          ? offsetWorldPoint(endPoint.worldPoint, layoutAxis, overhangEnd)
   287→          : endPoint.worldPoint;
   288→
   289→        return createPurlinSegment(
   290→          `${instanceGeometry.instance.id}_${attachmentId}_${index}`,
   291→          trussSystem,
   292→          instanceGeometry.instance,
   293→          startWorld,
   294→          endWorld,
   295→          purlinSystem,
   296→          startPoint.side,
   297→          [startPoint.copyId, endPoint.copyId]
   298→        );
   299→      }).filter(Boolean);
   300→      purlinRuns.push({
   301→        id: `${instanceGeometry.instance.id}_${attachmentId}_run`,
   302→        side: firstPoint.side,
   303→        points: runPoints,
   304→        attachments: sortedPoints,
   305→        segments,
   306→      });
   307→      purlinSegments.push(...segments);
   308→      continue;
   309→    }
   310→
   311→    if (sortedPoints.length === 1) {
   312→      const marker = createPlanMarker(sortedPoints[0].planPoint, layoutAxis);
   313→      purlinMarkers.push({
   314→        id: `${instanceGeometry.instance.id}_${attachmentId}_marker`,
   315→        side: sortedPoints[0].side,
   316→        center: sortedPoints[0].planPoint,
   317→        ...marker,
   318→      });
   319→    }
   320→  }
   321→
   322→  return {
   323→    purlinRuns,
   324→    purlinMarkers,
   325→    purlinSegments,
   326→  };
   327→}

   338→function transformCopyGeometry(copy, transform) {
   339→  const worldNodes = copy.worldNodes.map((node) => ({
   340→    ...node,
   341→    planPoint: transformPlanPoint(node.planPoint, transform),
   342→    worldPoint: rotateWorldPoint(node.worldPoint, transform),
   343→  }));
   344→  const members = copy.members.map((member) => ({
   345→    ...member,
   346→    startWorld: rotateWorldPoint(member.startWorld, transform),
   347→    endWorld: rotateWorldPoint(member.endWorld, transform),
   348→  }));
   349→  const purlinAttachments = (copy.purlinAttachments || []).map((attachment) => ({
   350→    ...attachment,
   351→    planPoint: transformPlanPoint(attachment.planPoint, transform),
   352→    worldPoint: rotateWorldPoint(attachment.worldPoint, transform),
   353→  }));
   354→
   355→  return {
   356→    ...copy,
   357→    origin: transformPlanPoint(copy.origin, transform),
   358→    structuralStartPoint: transformPlanPoint(copy.structuralStartPoint, transform),
   359→    structuralEndPoint: transformPlanPoint(copy.structuralEndPoint, transform),
   360→    overallStartPoint: transformPlanPoint(copy.overallStartPoint, transform),
   361→    overallEndPoint: transformPlanPoint(copy.overallEndPoint, transform),
   362→    worldNodes,
   363→    members,
   364→    purlinAttachments,
   365→  };
   366→}

   368→function transformInstanceGeometry(instanceGeometry, transform, baseElevation) {
   369→  const copies = instanceGeometry.copies.map((copy) => transformCopyGeometry(copy, transform));
   370→  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   371→
   372→  return {
   373→    ...instanceGeometry,
   374→    copies,
   375→    layoutLineStartPoint: transformPlanPoint(instanceGeometry.instance.startPoint, transform),
   376→    layoutLineEndPoint: transformPlanPoint(instanceGeometry.instance.endPoint, transform),
   377→    planBounds: createBounds(planPoints.map((point) => ({
   378→      x: point.x,
   379→      y: point.y,
   380→      elevation: baseElevation,
   381→    }))),
   382→  };
   383→}

   385→function buildRawTrussInstanceGeometry(trussSystem, instance) {
   386→  const layout = resolveTrussLayout(instance);
   387→  const profile = buildTrussProfile(instance);
   388→  const purlinAttachments = buildTrussPurlinAttachments(profile, trussSystem?.purlinSystem);
   389→  const copies = buildTrussCopyOrigins(instance).map((copy) => {
   390→    const layoutOffset = (
   391→      ((copy.origin.x ?? 0) - (instance.startPoint?.x ?? 0)) * layout.axis.x
   392→    ) + (
   393→      ((copy.origin.y ?? 0) - (instance.startPoint?.y ?? 0)) * layout.axis.y
   394→    );
   395→    const copyProfile = buildTrussProfile(instance, undefined, {
   396→      layoutOffset,
   397→      supportLength: layout.supportLength,
   398→    });
   399→    const copyPurlinAttachments = buildTrussPurlinAttachments(copyProfile, trussSystem?.purlinSystem);
   400→
   401→    return createCopyGeometry(trussSystem, instance, {
   402→      ...copyProfile,
   403→      purlinAttachments: copyPurlinAttachments,
   404→    }, layout, copy.origin, copy.index);
   405→  });
   406→
   407→  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   408→  const detailPoints = [
   409→    ...profile.roofOutline.map((point) => ({ x: point.x, y: 0, elevation: point.z })),
   410→    ...profile.bottomOutline.map((point) => ({ x: point.x, y: 0, elevation: point.z })),
   411→    ...purlinAttachments.map((attachment) => ({
   412→      x: attachment.localPoint.x,
   413→      y: 0,
   414→      elevation: attachment.localPoint.z,
   415→    })),
   416→  ];
   417→
   418→  return {
   419→    instance,
   420→    trussType: profile.trussType,
   421→    metrics: profile.metrics,
   422→    nodes: profile.nodes,
   423→    members: profile.members,
   424→    roofOutline: profile.roofOutline,
   425→    bottomOutline: profile.bottomOutline,
   426→    topChordRuns: profile.topChordRuns || [],
   427→    purlinAttachments,
   428→    copies,
   429→    layoutLineStartPoint: instance.startPoint,
   430→    layoutLineEndPoint: instance.endPoint,
   431→    planBounds: createBounds(planPoints.map((point) => ({
   432→      x: point.x,
   433→      y: point.y,
   434→      elevation: trussSystem.baseElevation || 0,
   435→    }))),
   436→    detailBounds: {
   437→      minX: Math.min(...detailPoints.map((point) => point.x)) - DETAIL_PADDING_X,
   438→      maxX: Math.max(...detailPoints.map((point) => point.x)) + DETAIL_PADDING_X,
   439→      minZ: Math.min(...detailPoints.map((point) => point.elevation), 0) - DETAIL_PADDING_Z,
   440→      maxZ: Math.max(...detailPoints.map((point) => point.elevation), 0) + DETAIL_PADDING_Z,
   441→    },
   442→  };
   443→}

   445→function createSystemTransform(trussSystem, rawInstances, baseElevation) {
   446→  const rawPlanPoints = rawInstances.flatMap((instanceGeometry) => (
   447→    instanceGeometry.copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint])
   448→  ));
   449→  const rawBounds = createBounds(rawPlanPoints.map((point) => ({
   450→    x: point.x,
   451→    y: point.y,
   452→    elevation: baseElevation,
   453→  })));
   454→  const rawPivot = centerOfBounds(rawBounds);
   455→  const primaryAxis = getSafeAxis(rawInstances[0]
   456→    ? {
   457→        x: rawInstances[0].layoutLineEndPoint.x - rawInstances[0].layoutLineStartPoint.x,
   458→        y: rawInstances[0].layoutLineEndPoint.y - rawInstances[0].layoutLineStartPoint.y,
   459→      }
   460→    : { x: 1, y: 0 });
   461→  const projectedAlongs = rawPlanPoints.map((point) => projectAlongAxis(point, rawPivot, primaryAxis));
   462→  const startAlong = projectedAlongs.length ? Math.min(...projectedAlongs) : 0;
   463→  const endAlong = projectedAlongs.length ? Math.max(...projectedAlongs) : 0;
   464→  const rawLength = Math.max(endAlong - startAlong, 0);
   465→  const resizable = rawLength > 1e-6 && rawInstances.every((instanceGeometry) => {
   466→    const instanceAxis = getSafeAxis({
   467→      x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   468→      y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   469→    });
   470→    return Math.abs(Math.abs((instanceAxis.x * primaryAxis.x) + (instanceAxis.y * primaryAxis.y)) - 1) <= 1e-3;
   471→  });
   472→  const rotationDegrees = normalizeSignedRotationDegrees(trussSystem?.planRotationOffsetDegrees || 0);
   473→  const translation = normalizePlanOffset(trussSystem?.planOffset);
   474→  const lengthScale = normalizePlanLengthScale(trussSystem?.planLengthScale);
   475→
   476→  return {
   477→    rawPivot,
   478→    pivot: {
   479→      x: rawPivot.x + translation.x,
   480→      y: rawPivot.y + translation.y,
   481→    },
   482→    axis: primaryAxis,
   483→    displayAxis: getDisplayedAxis({ axis: primaryAxis, rotationDegrees }),
   484→    translation,
   485→    lengthScale,
   486→    rawLength,
   487→    currentLength: rawLength * lengthScale,
   488→    startAlong,
   489→    endAlong,
   490→    startHandlePoint: transformPlanPoint(
   491→      {
   492→        x: rawPivot.x + (primaryAxis.x * startAlong),
   493→        y: rawPivot.y + (primaryAxis.y * startAlong),
   494→      },
   495→      {
   496→        rawPivot,
   497→        axis: primaryAxis,
   498→        rotationDegrees,
   499→        translation,
   500→        lengthScale,
   501→      }
   502→    ),
   503→    endHandlePoint: transformPlanPoint(
   504→      {
   505→        x: rawPivot.x + (primaryAxis.x * endAlong),
   506→        y: rawPivot.y + (primaryAxis.y * endAlong),
   507→      },
   508→      {
   509→        rawPivot,
   510→        axis: primaryAxis,
   511→        rotationDegrees,
   512→        translation,
   513→        lengthScale,
   514→      }
   515→    ),
   516→    resizable,
   517→    rotationDegrees,
   518→  };
   519→}

   521→function createSegmentDescriptor(id, trussSystem, instance, member, copy) {
   522→  const material = normalizeMaterial(instance?.material || 'timber');
   523→  return {
   524→    id,
   525→    kind: 'trussInstance',
   526→    geometry: 'segment3d',
   527→    start: {
   528→      x: member.startWorld.x,
   529→      y: member.startWorld.elevation,
   530→      z: member.startWorld.y,
   531→    },
   532→    end: {
   533→      x: member.endWorld.x,
   534→      y: member.endWorld.elevation,
   535→      z: member.endWorld.y,
   536→    },
   537→    thickness: member.memberType === 'web' ? DEFAULT_WEB_THICKNESS : DEFAULT_CHORD_THICKNESS,
   538→    materialKey: getTrussMemberMaterialKey(member.memberType, material),
   539→    metadata: {
   540→      sourceId: instance.id,
   541→      trussSystemId: trussSystem.id,
   542→      floorId: trussSystem.floorId,
   543→      copyId: copy.id,
   544→      memberType: member.memberType,
   545→      material,
   546→    },
   547→    bounds: {
   548→      minX: Math.min(member.startWorld.x, member.endWorld.x),

   593→function detailPolylineLength(points = []) {
   594→  let total = 0;
   595→  for (let index = 0; index < points.length - 1; index += 1) {
   596→    total += Math.hypot(points[index + 1].x - points[index].x, points[index + 1].y - points[index].y);
   597→  }
   598→  return total;
   599→}

   601→function worldSegmentLength(startWorld, endWorld) {
   602→  if (!startWorld || !endWorld) return 0;
   603→  return Math.hypot(
   604→    endWorld.x - startWorld.x,
   605→    endWorld.y - startWorld.y,
   606→    (endWorld.elevation || 0) - (startWorld.elevation || 0)
   607→  );
   608→}

   611→  if (!run) return 0;
   612→  if (run.segments?.length) {
   613→    return run.segments.reduce((total, segment) => (
   614→      total + worldSegmentLength(segment.startWorld, segment.endWorld)
   615→    ), 0);
   616→  }
   617→  return detailPolylineLength(run.points || []);

   620→function detailPolylineMidpoint(points = []) {
   621→  if (!points.length) return { x: 0, y: 0 };
   622→  if (points.length === 1) return points[0];
   623→
   624→  const totalLength = detailPolylineLength(points);
   625→  if (totalLength <= 1e-6) {
   626→    return {
   627→      x: (points[0].x + points[points.length - 1].x) / 2,
   628→      y: (points[0].y + points[points.length - 1].y) / 2,
   629→    };
   630→  }
   631→
   632→  let remaining = totalLength / 2;
   633→  for (let index = 0; index < points.length - 1; index += 1) {
   634→    const start = points[index];
   635→    const end = points[index + 1];
   636→    const segmentLength = Math.hypot(end.x - start.x, end.y - start.y);
   637→    if (remaining <= segmentLength || index === points.length - 2) {
   638→      const ratio = segmentLength <= 1e-6 ? 0 : remaining / segmentLength;
   639→      return {
   640→        x: start.x + ((end.x - start.x) * ratio),
   641→        y: start.y + ((end.y - start.y) * ratio),
   642→      };
   643→    }
   644→    remaining -= segmentLength;
   645→  }
   646→
   647→  return points[points.length - 1];
   648→}

   686→function estimateDimensionBounds(figure) {
   687→  if (!figure) return null;
   688→
   689→  const points = [
   690→    figure.lineStart,
   691→    figure.lineEnd,
   692→    ...figure.extensionLines.flatMap((line) => [line.start, line.end]),
   693→    ...figure.arrowheads.flatMap((arrow) => [arrow.left, arrow.tip, arrow.right]),
   694→  ].filter(Boolean);
   695→  const fontSize = DRAWING_GRAPHICS.annotation.textSize;
   696→  const width = estimateTextWidth(figure.text?.value, fontSize) + DRAWING_GRAPHICS.annotation.textMaskPaddingX;
   697→  const height = fontSize + DRAWING_GRAPHICS.annotation.textMaskPaddingY;
   698→  const textPosition = figure.text?.position || { x: 0, y: 0 };
   699→  const textRadius = Math.sqrt((width * width) + (height * height)) / 2;
   700→
   701→  points.push(
   702→    { x: textPosition.x - textRadius, y: textPosition.y - textRadius },
   703→    { x: textPosition.x + textRadius, y: textPosition.y + textRadius }
   704→  );
   705→
   706→  return {
   707→    minX: Math.min(...points.map((point) => point.x)),
   708→    maxX: Math.max(...points.map((point) => point.x)),
   709→    minY: Math.min(...points.map((point) => point.y)),
   710→    maxY: Math.max(...points.map((point) => point.y)),
   711→  };
   712→}

   714→function estimateTagBounds(tag) {
   715→  if (!tag) return null;
   716→
   717→  const fontSize = Number(tag.fontSize || DETAIL_TAG_FONT_SIZE);
   718→  const lineHeight = Number(tag.lineHeight || DETAIL_TAG_LINE_HEIGHT);
   719→  const maskPaddingX = Number(tag.maskPaddingX ?? DETAIL_TAG_MASK_PADDING_X);
   720→  const maskPaddingY = Number(tag.maskPaddingY ?? DETAIL_TAG_MASK_PADDING_Y);
   721→  const maxLineLength = Math.max(...tag.textLines.map((line) => String(line || '').length), 1);
   722→  const width = (maxLineLength * fontSize * 0.58) + Math.max(maskPaddingX, fontSize * 0.5);
   723→  const height = Math.max(lineHeight, tag.textLines.length * lineHeight) + maskPaddingY;
   724→  const minX = tag.textAnchor === 'start'
   725→    ? tag.position.x
   726→    : (tag.textAnchor === 'end' ? tag.position.x - width : tag.position.x - (width / 2));
   727→  const maxX = tag.textAnchor === 'start'
   728→    ? tag.position.x + width
   729→    : (tag.textAnchor === 'end' ? tag.position.x : tag.position.x + (width / 2));
   730→  const minY = tag.position.y - (height / 2);
   731→  const maxY = tag.position.y + (height / 2);
   732→  const angle = Number(tag.angle || 0);
   733→  const corners = [
   734→    { x: minX, y: minY },
   735→    { x: maxX, y: minY },
   736→    { x: maxX, y: maxY },
   737→    { x: minX, y: maxY },
   738→  ].map((corner) => rotate(corner, tag.position, angle));
   739→
   740→  return {
   741→    minX: Math.min(...corners.map((point) => point.x)),
   742→    maxX: Math.max(...corners.map((point) => point.x)),
   743→    minY: Math.min(...corners.map((point) => point.y)),
   744→    maxY: Math.max(...corners.map((point) => point.y)),
   745→  };
   746→}

   748→function selectOutwardNormal(normal, midpoint, spanMidpoint) {
   749→  const preferred = midpoint.x < spanMidpoint
   750→    ? { x: -1, y: 0 }
   751→    : { x: 1, y: 0 };
   752→  const alignment = (normal.x * preferred.x) + (normal.y * preferred.y);
   753→
   754→  return alignment >= 0
   755→    ? normal
   756→    : { x: -normal.x, y: -normal.y };
   757→}

   759→function extendDetailBounds(detailBounds, dimensions = [], tags = []) {
   760→  const bounds = {
   761→    minX: detailBounds.minX,
   762→    maxX: detailBounds.maxX,
   763→    minY: -detailBounds.maxZ,
   764→    maxY: -detailBounds.minZ,
   765→  };
   766→
   767→  for (const figure of dimensions) {
   768→    const figureBounds = estimateDimensionBounds(figure);
   769→    if (!figureBounds) continue;
   770→    bounds.minX = Math.min(bounds.minX, figureBounds.minX);
   771→    bounds.maxX = Math.max(bounds.maxX, figureBounds.maxX);
   772→    bounds.minY = Math.min(bounds.minY, figureBounds.minY);
   773→    bounds.maxY = Math.max(bounds.maxY, figureBounds.maxY);
   774→  }
   775→
   776→  for (const tag of tags) {
   777→    const tagBounds = estimateTagBounds(tag);
   778→    if (!tagBounds) continue;
   779→    bounds.minX = Math.min(bounds.minX, tagBounds.minX);
   780→    bounds.maxX = Math.max(bounds.maxX, tagBounds.maxX);
   781→    bounds.minY = Math.min(bounds.minY, tagBounds.minY);
   782→    bounds.maxY = Math.max(bounds.maxY, tagBounds.maxY);
   783→  }
   784→
   785→  return {
   786→    minX: bounds.minX,
   787→    maxX: bounds.maxX,
   788→    minZ: -bounds.maxY,
   789→    maxZ: -bounds.minY,
   790→  };
   791→}

   793→function boundsOverlap(a, b, gap) {
   794→  if (!a || !b) return false;
   795→  const halfGap = gap / 2;
   796→  return (
   797→    a.minX - halfGap < b.maxX + halfGap &&
   798→    a.maxX + halfGap > b.minX - halfGap &&
   799→    a.minY - halfGap < b.maxY + halfGap &&
   800→    a.maxY + halfGap > b.minY - halfGap
   801→  );
   802→}

   804→function applyDetailTagCollisionOffsets(tags = [], fixedObstacles = []) {
   805→  const placed = fixedObstacles.filter(Boolean);
   806→
   807→  return [...tags]
   808→    .sort((a, b) => (b.priority || 0) - (a.priority || 0))
   809→    .map((tag) => {
   810→      for (const offset of DETAIL_TAG_OFFSETS) {
   811→        const candidate = {
   812→          ...tag,
   813→          position: {
   814→            x: tag.position.x + offset.x,
   815→            y: tag.position.y + offset.y,
   816→          },
   817→          leaderLine: tag.leaderLine
   818→            ? {
   819→                start: tag.leaderLine.start,
   820→                end: {
   821→                  x: tag.leaderLine.end.x + offset.x,
   822→                  y: tag.leaderLine.end.y + offset.y,
   823→                },
   824→              }
   825→            : null,
   826→        };
   827→        const candidateBounds = estimateTagBounds(candidate);
   828→        if (!candidateBounds) continue;
   829→        if (!placed.some((obs) => boundsOverlap(candidateBounds, obs, DETAIL_TAG_GAP))) {
   830→          placed.push(candidateBounds);
   831→          return candidate;
   832→        }
   833→      }
   834→
   835→      const bounds = estimateTagBounds(tag);
   836→      if (bounds) placed.push(bounds);
   837→      return tag;
   838→    });
   839→}

   841→function resolveDetailCrestPoint(roofOutline = []) {
   842→  if (!roofOutline.length) {
   843→    return { x: 0, z: 0 };
   844→  }
   845→
   846→  const maxZ = Math.max(...roofOutline.map((point) => point.z));
   847→  const crestPoints = roofOutline.filter((point) => Math.abs(point.z - maxZ) <= 1e-6);
   848→  const averageX = crestPoints.reduce((sum, point) => sum + point.x, 0) / Math.max(crestPoints.length, 1);
   849→
   850→  return {
   851→    x: averageX,
   852→    z: maxZ,
   853→  };
   854→}

  1029→function buildTrussDetailWebTags(lineSegments, instanceGeometry) {
  1030→  const spanMidpoint = (instanceGeometry?.metrics?.span || 0) / 2;
  1031→
  1032→  const webSegments = lineSegments
  1033→    .filter((segment) => segment.memberType === 'web' && segment.start && segment.end);
  1034→  const totalWebs = webSegments.length;
  1035→
  1036→  return webSegments
  1037→    .map((segment, index) => {
  1038→      const webIndex = index + 1;
  1039→
  1040→      const start = createDetailPoint(segment.start);
  1041→      const end = createDetailPoint(segment.end);
  1042→      const dx = end.x - start.x;
  1043→      const dy = end.y - start.y;
  1044→      const length = Math.hypot(dx, dy);
  1045→      if (length <= 1) return null;
  1046→
  1047→      const t = totalWebs <= 1 ? 0.5 : 0.3 + (index / (totalWebs - 1)) * 0.4;
  1048→      const anchor = {
  1049→        x: start.x + dx * t,
  1050→        y: start.y + dy * t,
  1051→      };
  1052→      let offset = { x: 0, y: 0 };
  1053→
  1054→      if (Math.abs(dx) <= 1e-6) {
  1055→        offset = {
  1056→          x: anchor.x < spanMidpoint ? -DETAIL_WEB_TAG_OFFSET : DETAIL_WEB_TAG_OFFSET,
  1057→          y: 0,
  1058→        };
  1059→      } else {
  1060→        const normal = { x: -dy / length, y: dx / length };
  1061→        const tangent = { x: dx / length, y: dy / length };
  1062→        const outwardNormal = selectOutwardNormal(normal, anchor, spanMidpoint);
  1063→        const tangentShift = (webIndex % 2 === 0 ? 1 : -1) * DETAIL_WEB_TAG_TANGENT_SHIFT;
  1064→        offset = {
  1065→          x: (outwardNormal.x * DETAIL_WEB_TAG_OFFSET) + (tangent.x * tangentShift),
  1066→          y: (outwardNormal.y * DETAIL_WEB_TAG_OFFSET) + (tangent.y * tangentShift),
  1067→        };
  1068→      }
  1069→
  1070→      return createDetailTag(
  1071→        `${instanceGeometry.instance.id}_detail_web_${webIndex}`,
  1072→        [`W${webIndex} ${formatMeasurement(length)}`],
  1073→        {
  1074→          x: anchor.x + offset.x,
  1075→          y: anchor.y + offset.y,
  1076→        },
  1077→        {
  1078→          angle: 0,
  1079→          sourceType: 'truss_web',
  1080→          sourceId: segment.id,
  1081→          measurementValue: length,
  1082→          leaderLine: {
  1083→            start: anchor,
  1084→            end: {
  1085→              x: anchor.x + (offset.x * 0.72),
  1086→              y: anchor.y + (offset.y * 0.72),
  1087→            },
  1088→          },
  1089→        }
  1090→      );
  1091→    })

  1095→function buildTopChordRuns(instanceGeometry) {
  1096→  return (instanceGeometry?.topChordRuns || [])
  1097→    .map((run) => {
  1098→      const points = (run.points || []).filter((point) => (
  1099→        Number.isFinite(point?.x) && Number.isFinite(point?.z)
  1100→      ));
  1101→      if (points.length < 2) return null;
  1102→      return {
  1103→        id: run.id || run.side || 'main',
  1104→        measurementCurve: run.measurementCurve || null,
  1105→        points,
  1106→        start: points[0],
  1107→        end: points[points.length - 1],
  1108→      };
  1109→    })
  1110→    .filter(Boolean);
  1111→}

  1113→function buildTrussDetailTopChordTags(instanceGeometry) {
  1114→  return buildTopChordRuns(instanceGeometry)
  1115→    .map((run, index) => {
  1116→      const midpointSample = sampleTopChordRunMidpoint(run);
  1117→      if (!midpointSample?.point || !midpointSample?.tangent) return null;
  1118→
  1119→      const detailTangent = {
  1120→        x: midpointSample.tangent.x,
  1121→        y: -midpointSample.tangent.z,
  1122→      };
  1123→      const tangentLength = Math.hypot(detailTangent.x, detailTangent.y);
  1124→      const length = getTopChordRunLength(run);
  1125→      if (length <= 1 || tangentLength <= 1e-6) return null;
  1126→
  1127→      const midpoint = {
  1128→        x: midpointSample.point.x,
  1129→        y: -midpointSample.point.z,
  1130→      };
  1131→      const outwardNormal = {
  1132→        x: detailTangent.y / tangentLength,
  1133→        y: -detailTangent.x / tangentLength,
  1134→      };
  1135→
  1136→      return createDetailTag(
  1137→        `${instanceGeometry.instance.id}_detail_top_chord_${run.id}`,
  1138→        [`TC${index + 1} ${formatMeasurement(length)}`],
  1139→        {
  1140→          x: midpoint.x + (outwardNormal.x * DETAIL_TOP_CHORD_TAG_OFFSET),
  1141→          y: midpoint.y + (outwardNormal.y * DETAIL_TOP_CHORD_TAG_OFFSET),
  1142→        },
  1143→        {
  1144→          angle: 0,
  1145→          sourceType: 'truss_top_chord',
  1146→          sourceId: run.id,
  1147→          measurementValue: length,
  1148→          priority: 1,
  1149→          leaderLine: {
  1150→            start: midpoint,
  1151→            end: {
  1152→              x: midpoint.x + (outwardNormal.x * DETAIL_TOP_CHORD_TAG_OFFSET * 0.78),
  1153→              y: midpoint.y + (outwardNormal.y * DETAIL_TOP_CHORD_TAG_OFFSET * 0.78),
  1154→            },
  1155→          },
  1156→        }
  1157→      );
  1158→    })
  1159→    .filter(Boolean);
  1160→}
