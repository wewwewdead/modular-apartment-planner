    27→function useUnits() {
    28→  const [unit, setUnit] = useState('mm');
    29→  const isMm = unit === 'mm';
    30→  return {
    31→    unit,
    32→    setUnit,
    33→    suffix: isMm ? 'mm' : 'm',
    34→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    35→    fromDisplay: (v) => isMm ? v : v * 1000,
    36→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    37→  };
    38→}

    40→function SheetExportMenu({ sheet, editorDispatch }) {
    41→  const [open, setOpen] = useState(false);
    42→  const rootRef = useRef(null);
    43→  const pngOptions = [
    44→    { label: 'PNG 150 DPI', dpi: 150 },
    45→    { label: 'PNG 300 DPI', dpi: 300 },
    46→    { label: 'PNG 600 DPI', dpi: 600 },
    47→  ];
    48→
    49→  useEffect(() => {
    50→    if (!open) return undefined;
    51→
    52→    const handleMouseDown = (event) => {
    53→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    54→        setOpen(false);
    55→      }
    56→    };
    57→
    58→    document.addEventListener('mousedown', handleMouseDown);
    59→    return () => document.removeEventListener('mousedown', handleMouseDown);
    60→  }, [open]);
    61→
    62→  const handleExport = async (kind, options = {}) => {
    63→    if (!sheet) return;
    64→    try {
    65→      if (kind === 'pdf') {
    66→        await exportActiveSheetAsPdf(
    67→          sheet.title || sheet.drawingName || 'sheet',
    68→          sheet.paperSize
    69→        );
    70→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    71→      } else {
    72→        await exportActiveSheetAsPng(
    73→          sheet.title || sheet.drawingName || 'sheet',
    74→          options
    75→        );
    76→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    77→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    78→      }
    79→    } catch (error) {
    80→      editorDispatch({
    81→        type: 'SET_STATUS_MESSAGE',
    82→        message: error.message || `${kind.toUpperCase()} export failed.`,
    83→      });
    84→    } finally {
    85→      setOpen(false);
    86→    }
    87→  };
    88→
    89→  return (
    90→    <div className={styles.exportMenu} ref={rootRef}>
    91→      <button
    92→        className={styles.exportMenuTrigger}
    93→        type="button"
    94→        onClick={() => setOpen((value) => !value)}
    95→      >
    96→        Export sheet
    97→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    98→      </button>
    99→      {open && (
   100→        <div className={styles.exportMenuList}>
   101→          <button
   102→            className={styles.exportMenuItem}
   103→            type="button"
   104→            onClick={() => handleExport('pdf')}
   105→          >
   106→            PDF
   107→          </button>
   108→          {pngOptions.map((option) => (
   109→            <button
   110→              key={option.dpi}
   111→              className={styles.exportMenuItem}
   112→              type="button"
   113→              onClick={() => handleExport('png', { dpi: option.dpi })}
   114→            >
   115→              {option.label}
   116→            </button>
   117→          ))}
   118→        </div>
   119→      )}
   120→    </div>
   121→  );
   122→}

   500→function LandingProperties({ landing, floor, dispatch, floorId, u }) {
   501→  const updateLanding = (updates) => {
   502→    dispatch({ type: 'LANDING_UPDATE', floorId, landing: { id: landing.id, ...updates } });
   503→  };
   504→
   505→  const stairs = floor?.stairs || [];
   506→  const autoElevation = computeLandingElevation(landing, stairs, 0);

  1074→export default function PropertiesPanel() {
  1075→  const { project, dispatch, duplicateFloor } = useProject();
  1076→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, dispatch: editorDispatch } = useEditor();
  1077→  const orderedFloors = getOrderedFloors(project);
  1078→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1079→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1080→  const u = useUnits();
  1081→
  1082→  const selectFloor = (floorId) => {
  1083→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1084→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1085→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1086→  };
  1087→
  1088→  const handleDuplicateFloor = (floorToDuplicate) => {
  1089→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1090→    if (!duplicatedFloor) return;
  1091→
  1092→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1093→    selectFloor(duplicatedFloor.id);
  1094→  };
  1095→
  1096→  const handleDeleteFloor = (floorToDelete) => {
  1097→    if (orderedFloors.length <= 1) return;
  1098→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1099→    if (!confirmed) return;
  1100→
  1101→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1102→    const fallbackFloor = remainingFloors[0] || null;
  1103→
  1104→    dispatch({
  1105→      type: 'FLOOR_DELETE',
  1106→      floorId: floorToDelete.id,
  1107→      fallbackFloorId: fallbackFloor?.id ?? null,
  1108→    });
  1109→
  1110→    if (fallbackFloor) {
  1111→      selectFloor(fallbackFloor.id);
  1112→    } else {
  1113→      editorDispatch({ type: 'DESELECT' });
  1114→    }
  1115→  };
  1116→
  1117→  const handleDelete = () => {
  1118→    if (!selectedId) return;
  1119→    if (selectedType === 'sheet') {
  1120→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1121→    } else if (selectedType === 'sheetViewport' && sheet) {
  1122→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1123→    } else if (selectedType === 'slab') {
  1124→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1125→    } else if (selectedType === 'wall') {
  1126→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1127→    } else if (selectedType === 'beam') {
  1128→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1129→    } else if (selectedType === 'sectionCut') {
  1130→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1131→    } else if (selectedType === 'annotation') {
  1132→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1133→    } else if (selectedType === 'stair') {
  1134→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1135→    } else if (selectedType === 'landing') {
  1136→      dispatch({ type: 'LANDING_DELETE', floorId: activeFloorId, landingId: selectedId });
  1137→    } else if (selectedType === 'door') {
  1138→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1139→    } else if (selectedType === 'window') {
  1140→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1141→    } else if (selectedType === 'column') {
  1142→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1143→    } else if (selectedType === 'room') {
  1144→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1145→    } else if (selectedType === 'floor') {
  1146→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1147→      if (targetFloor) {
  1148→        handleDeleteFloor(targetFloor);
  1149→        return;
  1150→      }
  1151→    }
  1152→    editorDispatch({ type: 'DESELECT' });
  1153→  };
  1154→
  1155→  let content = workspaceMode === 'sheet' && sheet
  1156→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
  1157→    : (
  1158→        <>
  1159→          {floor && (
  1160→            <FloorProperties
  1161→              floor={floor}
  1162→              dispatch={dispatch}
  1163→              onDuplicate={handleDuplicateFloor}
  1164→              onDelete={handleDeleteFloor}
  1165→              canDelete={orderedFloors.length > 1}
  1166→              u={u}
  1167→            />
  1168→          )}
  1169→          <ProjectSummary project={project} floor={floor} dispatch={dispatch} />
  1170→        </>
  1171→      );
  1172→
  1173→  if (selectedId && workspaceMode === 'sheet' && sheet) {
  1174→    if (selectedType === 'sheet') {
  1175→      if (sheet.id === selectedId) {
  1176→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;
  1177→      }
  1178→    } else if (selectedType === 'sheetViewport') {
  1179→      const viewport = (sheet.viewports || []).find((entry) => entry.id === selectedId);
  1180→      if (viewport) {
  1181→        content = <SheetViewportProperties sheet={sheet} viewport={viewport} project={project} dispatch={dispatch} u={u} />;
  1182→      }
  1183→    }
  1184→  } else if (selectedId && floor) {

  1187→      content = (

  1190→          dispatch={dispatch}
  1191→          onDuplicate={handleDuplicateFloor}

  1193→          canDelete={orderedFloors.length > 1}
  1194→          u={u}
  1195→        />
  1196→      );
  1197→    } else if (selectedType === 'slab') {

  1200→        content = <SlabProperties slab={slab} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1201→      }
  1202→    } else if (selectedType === 'wall') {
  1203→      const wall = floor.walls.find(w => w.id === selectedId);
  1204→      if (wall) {
  1205→        content = <WallProperties wall={wall} dispatch={dispatch} floorId={activeFloorId} u={u} />;

  1257→  return (

  1263→        >mm</button>

  1267→        >m</button>
  1268→      </div>
  1269→      {workspaceMode === 'sheet' && sheet && (

  1277→      )}
  1278→    </div>
  1279→  );
