    33→function useUnits() {
    34→  const [unit, setUnit] = useState('mm');
    35→  const isMm = unit === 'mm';
    36→  return {
    37→    unit,
    38→    setUnit,
    39→    suffix: isMm ? 'mm' : 'm',
    40→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    41→    fromDisplay: (v) => isMm ? v : v * 1000,
    42→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    43→  };
    44→}

    46→function SheetExportMenu({ sheet, editorDispatch }) {
    47→  const [open, setOpen] = useState(false);
    48→  const rootRef = useRef(null);
    49→  const pngOptions = [
    50→    { label: 'PNG 150 DPI', dpi: 150 },
    51→    { label: 'PNG 300 DPI', dpi: 300 },
    52→    { label: 'PNG 600 DPI', dpi: 600 },
    53→  ];
    54→
    55→  useEffect(() => {
    56→    if (!open) return undefined;
    57→
    58→    const handleMouseDown = (event) => {
    59→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    60→        setOpen(false);
    61→      }
    62→    };
    63→
    64→    document.addEventListener('mousedown', handleMouseDown);
    65→    return () => document.removeEventListener('mousedown', handleMouseDown);
    66→  }, [open]);
    67→
    68→  const handleExport = async (kind, options = {}) => {
    69→    if (!sheet) return;
    70→    try {
    71→      if (kind === 'pdf') {
    72→        await exportActiveSheetAsPdf(
    73→          sheet.title || sheet.drawingName || 'sheet',
    74→          sheet.paperSize
    75→        );
    76→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    77→      } else {
    78→        await exportActiveSheetAsPng(
    79→          sheet.title || sheet.drawingName || 'sheet',
    80→          options
    81→        );
    82→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    83→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    84→      }
    85→    } catch (error) {
    86→      editorDispatch({
    87→        type: 'SET_STATUS_MESSAGE',
    88→        message: error.message || `${kind.toUpperCase()} export failed.`,
    89→      });
    90→    } finally {
    91→      setOpen(false);
    92→    }
    93→  };
    94→
    95→  return (
    96→    <div className={styles.exportMenu} ref={rootRef}>
    97→      <button
    98→        className={styles.exportMenuTrigger}
    99→        type="button"
   100→        onClick={() => setOpen((value) => !value)}
   101→      >
   102→        Export sheet
   103→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   104→      </button>
   105→      {open && (
   106→        <div className={styles.exportMenuList}>
   107→          <button
   108→            className={styles.exportMenuItem}
   109→            type="button"
   110→            onClick={() => handleExport('pdf')}
   111→          >
   112→            PDF
   113→          </button>
   114→          {pngOptions.map((option) => (
   115→            <button
   116→              key={option.dpi}
   117→              className={styles.exportMenuItem}
   118→              type="button"
   119→              onClick={() => handleExport('png', { dpi: option.dpi })}
   120→            >
   121→              {option.label}
   122→            </button>
   123→          ))}
   124→        </div>
   125→      )}
   126→    </div>
   127→  );
   128→}

   900→function SheetProperties({ sheet, project, dispatch, editorDispatch, activeFloorId, u }) {
   901→  const updateSheet = (updates) => {
   902→    dispatch({ type: 'SHEET_UPDATE', sheet: { id: sheet.id, ...updates } });
   903→  };
   904→
   905→  const updateLayoutTemplate = (nextLayoutTemplate) => {
   906→    if (nextLayoutTemplate === sheet.layoutTemplate) return;
   907→
   908→    if (nextLayoutTemplate === 'manual') {
   909→      dispatch({
   910→        type: 'SHEET_UPDATE',
   911→        sheet: materializeSheetViewportFrames(project, sheet),
   912→      });
   913→      return;
   914→    }
   915→
   916→    updateSheet({
   917→      layoutTemplate: nextLayoutTemplate,
   918→      viewports: (sheet.viewports || []).map((viewport) => ({
   919→        ...viewport,
   920→        lockAutoLayout: false,
   921→      })),
   922→    });
   923→  };
   924→
   925→  const updateTitleBlock = (updates) => {
   926→    updateSheet({
   927→      titleBlock: {
   928→        ...(sheet.titleBlock || {}),
   929→        ...updates,
   930→      },
   931→    });
   932→  };
   933→
   934→  const updateRevisions = (nextRevisions) => {
   935→    updateSheet({ revisions: nextRevisions });
   936→  };
   937→
   938→  const addViewport = () => {
   939→    try {
   940→      const sourceFloorId = getDefaultActiveFloorId(project, activeFloorId);
   941→      const sourceView = 'plan';
   942→      const viewportBase = createSheetViewport(sourceView, sourceFloorId, {
   943→        scale: 100,
   944→        lockAutoLayout: sheet.layoutTemplate !== 'auto',
   945→      });
   946→
   947→      let rect;
   948→      try {
   949→        const source = resolveSheetViewportSource(project, viewportBase);
   950→        rect = getDefaultViewportRect(sheet, source, viewportBase.scale, { role: viewportBase.role });
   951→      } catch {
   952→        // Geometry resolution failed — use default dimensions
   953→        rect = { x: 20, y: 20, width: 160, height: 100 };
   954→      }
   955→
   956→      const viewport = { ...viewportBase, ...rect };
   957→      dispatch({ type: 'SHEET_VIEWPORT_ADD', sheetId: sheet.id, viewport });
   958→      editorDispatch({ type: 'SELECT_OBJECT', id: viewport.id, objectType: 'sheetViewport' });
   959→      editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Viewport added.' });

   996→          projectNameOverride: value,
   997→          titleBlock: {

  1001→        })}

  1375→export default function PropertiesPanel() {
  1376→  const { project, dispatch, duplicateFloor } = useProject();
  1377→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, dispatch: editorDispatch } = useEditor();
  1378→  const orderedFloors = getOrderedFloors(project);
  1379→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1380→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1381→  const u = useUnits();
  1382→
  1383→  const selectFloor = (floorId) => {
  1384→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1385→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1386→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1387→  };
  1388→
  1389→  const handleDuplicateFloor = (floorToDuplicate) => {
  1390→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1391→    if (!duplicatedFloor) return;
  1392→
  1393→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1394→    selectFloor(duplicatedFloor.id);
  1395→  };
  1396→
  1397→  const handleDeleteFloor = (floorToDelete) => {
  1398→    if (orderedFloors.length <= 1) return;
  1399→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1400→    if (!confirmed) return;
  1401→
  1402→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1403→    const fallbackFloor = remainingFloors[0] || null;
  1404→
  1405→    dispatch({
  1406→      type: 'FLOOR_DELETE',
  1407→      floorId: floorToDelete.id,
  1408→      fallbackFloorId: fallbackFloor?.id ?? null,
  1409→    });
  1410→
  1411→    if (fallbackFloor) {
  1412→      selectFloor(fallbackFloor.id);
  1413→    } else {
  1414→      editorDispatch({ type: 'DESELECT' });
  1415→    }
  1416→  };
  1417→
  1418→  const handleDelete = () => {
  1419→    if (!selectedId) return;
  1420→    if (selectedType === 'sheet') {
  1421→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1422→    } else if (selectedType === 'sheetViewport' && sheet) {
  1423→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1424→    } else if (selectedType === 'slab') {
  1425→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1426→    } else if (selectedType === 'wall') {
  1427→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1428→    } else if (selectedType === 'beam') {
  1429→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1430→    } else if (selectedType === 'sectionCut') {
  1431→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1432→    } else if (selectedType === 'annotation') {
  1433→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1434→    } else if (selectedType === 'stair') {
  1435→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1436→    } else if (selectedType === 'landing') {
  1437→      dispatch({ type: 'LANDING_DELETE', floorId: activeFloorId, landingId: selectedId });
  1438→    } else if (selectedType === 'door') {
  1439→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1440→    } else if (selectedType === 'window') {
  1441→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1442→    } else if (selectedType === 'column') {
  1443→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1444→    } else if (selectedType === 'fixture') {
  1445→      dispatch({ type: 'FIXTURE_DELETE', floorId: activeFloorId, fixtureId: selectedId });
  1446→    } else if (selectedType === 'room') {
  1447→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1448→    } else if (selectedType === 'floor') {
  1449→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1450→      if (targetFloor) {
  1451→        handleDeleteFloor(targetFloor);
  1452→        return;
  1453→      }
  1454→    }
  1455→    editorDispatch({ type: 'DESELECT' });
  1456→  };
  1457→
  1458→  let content = workspaceMode === 'sheet' && sheet
  1459→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
  1460→    : (
  1461→        <>
  1462→          {floor && (
  1463→            <FloorProperties
  1464→              floor={floor}
