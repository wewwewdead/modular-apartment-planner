   138тЖТ   138тЖТfunction normalizePlanVector(vector) {
   139тЖТ   139тЖТ  const magnitude = Math.sqrt((vector.x * vector.x) + (vector.y * vector.y));
   140тЖТ   140тЖТ  if (magnitude <= 1e-6) {
   141тЖТ   141тЖТ    return { x: 1, y: 0 };
   142тЖТ   142тЖТ  }
   143тЖТ   143тЖТ  return {
   144тЖТ   144тЖТ    x: vector.x / magnitude,
   145тЖТ   145тЖТ    y: vector.y / magnitude,
   146тЖТ   146тЖТ  };
   147тЖТ   147тЖТ}

   198тЖТ   198тЖТfunction buildInstancePurlinGeometry(trussSystem, instanceGeometry, purlinSystem) {
   199тЖТ   199тЖТ  if (!purlinSystem?.enabled) {
   200тЖТ   200тЖТ    return {
   201тЖТ   201тЖТ      purlinRuns: [],
   202тЖТ   202тЖТ      purlinMarkers: [],
   203тЖТ   203тЖТ      purlinSegments: [],
   204тЖТ   204тЖТ    };
   205тЖТ   205тЖТ  }
   206тЖТ   206тЖТ
   207тЖТ   207тЖТ  const attachmentGroups = new Map();
   208тЖТ   208тЖТ  for (const copy of instanceGeometry.copies) {
   209тЖТ   209тЖТ    for (const attachment of copy.purlinAttachments || []) {
   210тЖТ   210тЖТ      if (!attachmentGroups.has(attachment.id)) {
   211тЖТ   211тЖТ        attachmentGroups.set(attachment.id, []);
   212тЖТ   212тЖТ      }
   213тЖТ   213тЖТ      attachmentGroups.get(attachment.id).push({
   214тЖТ   214тЖТ        ...attachment,
   215тЖТ   215тЖТ        copyId: copy.id,
   216тЖТ   216тЖТ        copyIndex: copy.index,
   217тЖТ   217тЖТ      });
   218тЖТ   218тЖТ    }
   219тЖТ   219тЖТ  }
   220тЖТ   220тЖТ
   221тЖТ   221тЖТ  const layoutAxis = normalizePlanVector({
   222тЖТ   222тЖТ    x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   223тЖТ   223тЖТ    y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   224тЖТ   224тЖТ  });
   225тЖТ   225тЖТ  const purlinRuns = [];
   226тЖТ   226тЖТ  const purlinMarkers = [];
   227тЖТ   227тЖТ  const purlinSegments = [];
   228тЖТ   228тЖТ  const overhangStart = Math.max(Number(purlinSystem?.overhangStart || 0), 0);
   229тЖТ   229тЖТ  const overhangEnd = Math.max(Number(purlinSystem?.overhangEnd || 0), 0);
   230тЖТ   230тЖТ
   231тЖТ   231тЖТ  for (const [attachmentId, attachmentPoints] of attachmentGroups.entries()) {
   232тЖТ   232тЖТ    const sortedPoints = [...attachmentPoints].sort((a, b) => a.copyIndex - b.copyIndex);
   233тЖТ   233тЖТ    if (sortedPoints.length >= 2) {
   234тЖТ   234тЖТ      const firstPoint = sortedPoints[0];
   235тЖТ   235тЖТ      const lastPoint = sortedPoints[sortedPoints.length - 1];
   236тЖТ   236тЖТ      const runPoints = [
   237тЖТ   237тЖТ        ...(overhangStart > 1e-6
   238тЖТ   238тЖТ          ? [add(firstPoint.planPoint, scale(layoutAxis, -overhangStart))]
   239тЖТ   239тЖТ          : []),
   240тЖТ   240тЖТ        ...sortedPoints.map((point) => point.planPoint),
   241тЖТ   241тЖТ        ...(overhangEnd > 1e-6
   242тЖТ   242тЖТ          ? [add(lastPoint.planPoint, scale(layoutAxis, overhangEnd))]
   243тЖТ   243тЖТ          : []),
   244тЖТ   244тЖТ      ];
   245тЖТ   245тЖТ      const segments = sortedPoints.slice(0, -1).map((startPoint, index) => {
   246тЖТ   246тЖТ        const endPoint = sortedPoints[index + 1];
   247тЖТ   247тЖТ        const startWorld = index === 0
   248тЖТ   248тЖТ          ? offsetWorldPoint(startPoint.worldPoint, layoutAxis, -overhangStart)
   249тЖТ   249тЖТ          : startPoint.worldPoint;
   250тЖТ   250тЖТ        const endWorld = index === sortedPoints.length - 2
   251тЖТ   251тЖТ          ? offsetWorldPoint(endPoint.worldPoint, layoutAxis, overhangEnd)
   252тЖТ   252тЖТ          : endPoint.worldPoint;
   253тЖТ   253тЖТ
   254тЖТ   254тЖТ        return createPurlinSegment(
   255тЖТ   255тЖТ          `${instanceGeometry.instance.id}_${attachmentId}_${index}`,
   256тЖТ   256тЖТ          trussSystem,
   257тЖТ   257тЖТ          instanceGeometry.instance,
   258тЖТ   258тЖТ          startWorld,
   259тЖТ   259тЖТ          endWorld,
   260тЖТ   260тЖТ          purlinSystem,
   261тЖТ   261тЖТ          startPoint.side,
   262тЖТ   262тЖТ          [startPoint.copyId, endPoint.copyId]
   263тЖТ   263тЖТ        );
   264тЖТ   264тЖТ      }).filter(Boolean);
   265тЖТ   265тЖТ      purlinRuns.push({
   266тЖТ   266тЖТ        id: `${instanceGeometry.instance.id}_${attachmentId}_run`,
   267тЖТ   267тЖТ        side: firstPoint.side,
   268тЖТ   268тЖТ        points: runPoints,
   269тЖТ   269тЖТ        attachments: sortedPoints,
   270тЖТ   270тЖТ        segments,
   271тЖТ   271тЖТ      });
   272тЖТ   272тЖТ      purlinSegments.push(...segments);
   273тЖТ   273тЖТ      continue;
   274тЖТ   274тЖТ    }
   275тЖТ   275тЖТ
   276тЖТ   276тЖТ    if (sortedPoints.length === 1) {
   277тЖТ   277тЖТ      const marker = createPlanMarker(sortedPoints[0].planPoint, layoutAxis);
   278тЖТ   278тЖТ      purlinMarkers.push({
   279тЖТ   279тЖТ        id: `${instanceGeometry.instance.id}_${attachmentId}_marker`,
   280тЖТ   280тЖТ        side: sortedPoints[0].side,
   281тЖТ   281тЖТ        center: sortedPoints[0].planPoint,
   282тЖТ   282тЖТ        ...marker,
   283тЖТ   283тЖТ      });
   284тЖТ   284тЖТ    }
   285тЖТ   285тЖТ  }
   286тЖТ   286тЖТ
   287тЖТ   287тЖТ  return {
   288тЖТ   288тЖТ    purlinRuns,
   289тЖТ   289тЖТ    purlinMarkers,
   290тЖТ   290тЖТ    purlinSegments,
   291тЖТ   291тЖТ  };
   292тЖТ   292тЖТ}

   303тЖТ   303тЖТfunction transformCopyGeometry(copy, transform) {

   308тЖТ   308тЖТ  }));
   309тЖТ   309тЖТ  const members = copy.members.map((member) => ({

   318тЖТ   318тЖТ  }));
   319тЖТ   319тЖТ
   320тЖТ   320тЖТ  return {
   321тЖТ   321тЖТ    ...copy,
   322тЖТ   322тЖТ    origin: transformPlanPoint(copy.origin, transform),

   324тЖТ   324тЖТ    structuralEndPoint: transformPlanPoint(copy.structuralEndPoint, transform),
   325тЖТ   325тЖТ    overallStartPoint: transformPlanPoint(copy.overallStartPoint, transform),
   326тЖТ   326тЖТ    overallEndPoint: transformPlanPoint(copy.overallEndPoint, transform),
   327тЖТ   327тЖТ    worldNodes,

   333тЖТ   333тЖТfunction transformInstanceGeometry(instanceGeometry, transform, baseElevation) {
   334тЖТ   334тЖТ  const copies = instanceGeometry.copies.map((copy) => transformCopyGeometry(copy, transform));
   335тЖТ   335тЖТ  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   336тЖТ   336тЖТ
   337тЖТ   337тЖТ  return {
   338тЖТ   338тЖТ    ...instanceGeometry,
   339тЖТ   339тЖТ    copies,
   340тЖТ   340тЖТ    layoutLineStartPoint: transformPlanPoint(instanceGeometry.instance.startPoint, transform),
   341тЖТ   341тЖТ    layoutLineEndPoint: transformPlanPoint(instanceGeometry.instance.endPoint, transform),
   342тЖТ   342тЖТ    planBounds: createBounds(planPoints.map((point) => ({
   343тЖТ   343тЖТ      x: point.x,
   344тЖТ   344тЖТ      y: point.y,

   410тЖТ   410тЖТfunction createSystemTransform(trussSystem, rawInstances, baseElevation) {
   411тЖТ   411тЖТ  const rawPlanPoints = rawInstances.flatMap((instanceGeometry) => (
   412тЖТ   412тЖТ    instanceGeometry.copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint])
   413тЖТ   413тЖТ  ));
   414тЖТ   414тЖТ  const rawBounds = createBounds(rawPlanPoints.map((point) => ({
   415тЖТ   415тЖТ    x: point.x,
   416тЖТ   416тЖТ    y: point.y,
   417тЖТ   417тЖТ    elevation: baseElevation,
   418тЖТ   418тЖТ  })));
   419тЖТ   419тЖТ  const rawPivot = centerOfBounds(rawBounds);
   420тЖТ   420тЖТ  const primaryAxis = getSafeAxis(rawInstances[0]
   421тЖТ   421тЖТ    ? {
   422тЖТ   422тЖТ        x: rawInstances[0].layoutLineEndPoint.x - rawInstances[0].layoutLineStartPoint.x,
   423тЖТ   423тЖТ        y: rawInstances[0].layoutLineEndPoint.y - rawInstances[0].layoutLineStartPoint.y,
   424тЖТ   424тЖТ      }
   425тЖТ   425тЖТ    : { x: 1, y: 0 });
   426тЖТ   426тЖТ  const projectedAlongs = rawPlanPoints.map((point) => projectAlongAxis(point, rawPivot, primaryAxis));
   427тЖТ   427тЖТ  const startAlong = projectedAlongs.length ? Math.min(...projectedAlongs) : 0;
   428тЖТ   428тЖТ  const endAlong = projectedAlongs.length ? Math.max(...projectedAlongs) : 0;
   429тЖТ   429тЖТ  const rawLength = Math.max(endAlong - startAlong, 0);
   430тЖТ   430тЖТ  const resizable = rawLength > 1e-6 && rawInstances.every((instanceGeometry) => {
   431тЖТ   431тЖТ    const instanceAxis = getSafeAxis({
   432тЖТ   432тЖТ      x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   433тЖТ   433тЖТ      y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   434тЖТ   434тЖТ    });
   435тЖТ   435тЖТ    return Math.abs(Math.abs((instanceAxis.x * primaryAxis.x) + (instanceAxis.y * primaryAxis.y)) - 1) <= 1e-3;
   436тЖТ   436тЖТ  });
   437тЖТ   437тЖТ  const rotationDegrees = normalizeSignedRotationDegrees(trussSystem?.planRotationOffsetDegrees || 0);
   438тЖТ   438тЖТ  const translation = normalizePlanOffset(trussSystem?.planOffset);
   439тЖТ   439тЖТ  const lengthScale = normalizePlanLengthScale(trussSystem?.planLengthScale);
   440тЖТ   440тЖТ
   441тЖТ   441тЖТ  return {
   442тЖТ   442тЖТ    rawPivot,
   443тЖТ   443тЖТ    pivot: {
   444тЖТ   444тЖТ      x: rawPivot.x + translation.x,
   445тЖТ   445тЖТ      y: rawPivot.y + translation.y,
   446тЖТ   446тЖТ    },
   447тЖТ   447тЖТ    axis: primaryAxis,
   448тЖТ   448тЖТ    displayAxis: getDisplayedAxis({ axis: primaryAxis, rotationDegrees }),
   449тЖТ   449тЖТ    translation,
   450тЖТ   450тЖТ    lengthScale,
   451тЖТ   451тЖТ    rawLength,
   452тЖТ   452тЖТ    currentLength: rawLength * lengthScale,
   453тЖТ   453тЖТ    startAlong,
   454тЖТ   454тЖТ    endAlong,
   455тЖТ   455тЖТ    startHandlePoint: transformPlanPoint(
   456тЖТ   456тЖТ      {
   457тЖТ   457тЖТ        x: rawPivot.x + (primaryAxis.x * startAlong),
   458тЖТ   458тЖТ        y: rawPivot.y + (primaryAxis.y * startAlong),
   459тЖТ   459тЖТ      },
   460тЖТ   460тЖТ      {
   461тЖТ   461тЖТ        rawPivot,
   462тЖТ   462тЖТ        axis: primaryAxis,
   463тЖТ   463тЖТ        rotationDegrees,
   464тЖТ   464тЖТ        translation,
   465тЖТ   465тЖТ        lengthScale,
   466тЖТ   466тЖТ      }
   467тЖТ   467тЖТ    ),
   468тЖТ   468тЖТ    endHandlePoint: transformPlanPoint(
   469тЖТ   469тЖТ      {
   470тЖТ   470тЖТ        x: rawPivot.x + (primaryAxis.x * endAlong),
   471тЖТ   471тЖТ        y: rawPivot.y + (primaryAxis.y * endAlong),
   472тЖТ   472тЖТ      },
   473тЖТ   473тЖТ      {
   474тЖТ   474тЖТ        rawPivot,
   475тЖТ   475тЖТ        axis: primaryAxis,
   476тЖТ   476тЖТ        rotationDegrees,
   477тЖТ   477тЖТ        translation,
   478тЖТ   478тЖТ        lengthScale,
   479тЖТ   479тЖТ      }
   480тЖТ   480тЖТ    ),
   481тЖТ   481тЖТ    resizable,
   482тЖТ   482тЖТ    rotationDegrees,

   585тЖТ   585тЖТfunction detailPolylineMidpoint(points = []) {
   586тЖТ   586тЖТ  if (!points.length) return { x: 0, y: 0 };
   587тЖТ   587тЖТ  if (points.length === 1) return points[0];
   588тЖТ   588тЖТ
   589тЖТ   589тЖТ  const totalLength = detailPolylineLength(points);
   590тЖТ   590тЖТ  if (totalLength <= 1e-6) {
   591тЖТ   591тЖТ    return {
   592тЖТ   592тЖТ      x: (points[0].x + points[points.length - 1].x) / 2,
   593тЖТ   593тЖТ      y: (points[0].y + points[points.length - 1].y) / 2,
   594тЖТ   594тЖТ    };
   595тЖТ   595тЖТ  }
   596тЖТ   596тЖТ
   597тЖТ   597тЖТ  let remaining = totalLength / 2;
   598тЖТ   598тЖТ  for (let index = 0; index < points.length - 1; index += 1) {
   599тЖТ   599тЖТ    const start = points[index];
   600тЖТ   600тЖТ    const end = points[index + 1];
   601тЖТ   601тЖТ    const segmentLength = Math.hypot(end.x - start.x, end.y - start.y);
   602тЖТ   602тЖТ    if (remaining <= segmentLength || index === points.length - 2) {
   603тЖТ   603тЖТ      const ratio = segmentLength <= 1e-6 ? 0 : remaining / segmentLength;
   604тЖТ   604тЖТ      return {
   605тЖТ   605тЖТ        x: start.x + ((end.x - start.x) * ratio),
   606тЖТ   606тЖТ        y: start.y + ((end.y - start.y) * ratio),
   607тЖТ   607тЖТ      };
   608тЖТ   608тЖТ    }
   609тЖТ   609тЖТ    remaining -= segmentLength;
   610тЖТ   610тЖТ  }
   611тЖТ   611тЖТ
   612тЖТ   612тЖТ  return points[points.length - 1];
   613тЖТ   613тЖТ}

   684тЖТ   684тЖТ  const maskPaddingX = Number(tag.maskPaddingX ?? DETAIL_TAG_MASK_PADDING_X);
   685тЖТ   685тЖТ  const maskPaddingY = Number(tag.maskPaddingY ?? DETAIL_TAG_MASK_PADDING_Y);
   686тЖТ   686тЖТ  const maxLineLength = Math.max(...tag.textLines.map((line) => String(line || '').length), 1);
   687тЖТ   687тЖТ  const width = (maxLineLength * fontSize * 0.58) + Math.max(maskPaddingX, fontSize * 0.5);
   688тЖТ   688тЖТ  const height = Math.max(lineHeight, tag.textLines.length * lineHeight) + maskPaddingY;
   689тЖТ   689тЖТ  const minX = tag.textAnchor === 'start'
   690тЖТ   690тЖТ    ? tag.position.x
   691тЖТ   691тЖТ    : (tag.textAnchor === 'end' ? tag.position.x - width : tag.position.x - (width / 2));
   692тЖТ   692тЖТ  const maxX = tag.textAnchor === 'start'
   693тЖТ   693тЖТ    ? tag.position.x + width
   694тЖТ   694тЖТ    : (tag.textAnchor === 'end' ? tag.position.x : tag.position.x + (width / 2));
   695тЖТ   695тЖТ  const minY = tag.position.y - (height / 2);
   696тЖТ   696тЖТ  const maxY = tag.position.y + (height / 2);
   697тЖТ   697тЖТ  const angle = Number(tag.angle || 0);
   698тЖТ   698тЖТ  const corners = [
   699тЖТ   699тЖТ    { x: minX, y: minY },
   700тЖТ   700тЖТ    { x: maxX, y: minY },
   701тЖТ   701тЖТ    { x: maxX, y: maxY },
   702тЖТ   702тЖТ    { x: minX, y: maxY },
   703тЖТ   703тЖТ  ].map((corner) => rotate(corner, tag.position, angle));
   704тЖТ   704тЖТ
   705тЖТ   705тЖТ  return {
   706тЖТ   706тЖТ    minX: Math.min(...corners.map((point) => point.x)),
   707тЖТ   707тЖТ    maxX: Math.max(...corners.map((point) => point.x)),
   708тЖТ   708тЖТ    minY: Math.min(...corners.map((point) => point.y)),
   709тЖТ   709тЖТ    maxY: Math.max(...corners.map((point) => point.y)),
   710тЖТ   710тЖТ  };
   711тЖТ   711тЖТ}
   712тЖТ   712тЖТ
   713тЖТ   713тЖТfunction selectOutwardNormal(normal, midpoint, spanMidpoint) {
   714тЖТ   714тЖТ  const preferred = midpoint.x < spanMidpoint
   715тЖТ   715тЖТ    ? { x: -1, y: 0 }
   716тЖТ   716тЖТ    : { x: 1, y: 0 };
   717тЖТ   717тЖТ  const alignment = (normal.x * preferred.x) + (normal.y * preferred.y);
   718тЖТ   718тЖТ
   719тЖТ   719тЖТ  return alignment >= 0
   720тЖТ   720тЖТ    ? normal
   721тЖТ   721тЖТ    : { x: -normal.x, y: -normal.y };
   722тЖТ   722тЖТ}

   724тЖТ   724тЖТfunction extendDetailBounds(detailBounds, dimensions = [], tags = []) {
   725тЖТ   725тЖТ  const bounds = {
   726тЖТ   726тЖТ    minX: detailBounds.minX,
   727тЖТ   727тЖТ    maxX: detailBounds.maxX,
   728тЖТ   728тЖТ    minY: -detailBounds.maxZ,
   729тЖТ   729тЖТ    maxY: -detailBounds.minZ,
   730тЖТ   730тЖТ  };
   731тЖТ   731тЖТ
   732тЖТ   732тЖТ  for (const figure of dimensions) {
   733тЖТ   733тЖТ    const figureBounds = estimateDimensionBounds(figure);
   734тЖТ   734тЖТ    if (!figureBounds) continue;
   735тЖТ   735тЖТ    bounds.minX = Math.min(bounds.minX, figureBounds.minX);
   736тЖТ   736тЖТ    bounds.maxX = Math.max(bounds.maxX, figureBounds.maxX);
   737тЖТ   737тЖТ    bounds.minY = Math.min(bounds.minY, figureBounds.minY);
   738тЖТ   738тЖТ    bounds.maxY = Math.max(bounds.maxY, figureBounds.maxY);
   739тЖТ   739тЖТ  }
   740тЖТ   740тЖТ
   741тЖТ   741тЖТ  for (const tag of tags) {
   742тЖТ   742тЖТ    const tagBounds = estimateTagBounds(tag);
   743тЖТ   743тЖТ    if (!tagBounds) continue;
   744тЖТ   744тЖТ    bounds.minX = Math.min(bounds.minX, tagBounds.minX);
   745тЖТ   745тЖТ    bounds.maxX = Math.max(bounds.maxX, tagBounds.maxX);
   746тЖТ   746тЖТ    bounds.minY = Math.min(bounds.minY, tagBounds.minY);
   747тЖТ   747тЖТ    bounds.maxY = Math.max(bounds.maxY, tagBounds.maxY);
   748тЖТ   748тЖТ  }
   749тЖТ   749тЖТ
   750тЖТ   750тЖТ  return {
   751тЖТ   751тЖТ    minX: bounds.minX,
   752тЖТ   752тЖТ    maxX: bounds.maxX,
   753тЖТ   753тЖТ    minZ: -bounds.maxY,
   754тЖТ   754тЖТ    maxZ: -bounds.minY,
   755тЖТ   755тЖТ  };
   756тЖТ   756тЖТ}

   758тЖТ   758тЖТfunction resolveDetailCrestPoint(roofOutline = []) {
   759тЖТ   759тЖТ  if (!roofOutline.length) {
   760тЖТ   760тЖТ    return { x: 0, z: 0 };
   761тЖТ   761тЖТ  }
   762тЖТ   762тЖТ
   763тЖТ   763тЖТ  const maxZ = Math.max(...roofOutline.map((point) => point.z));
   764тЖТ   764тЖТ  const crestPoints = roofOutline.filter((point) => Math.abs(point.z - maxZ) <= 1e-6);
   765тЖТ   765тЖТ  const averageX = crestPoints.reduce((sum, point) => sum + point.x, 0) / Math.max(crestPoints.length, 1);
   766тЖТ   766тЖТ
   767тЖТ   767тЖТ  return {
   768тЖТ   768тЖТ    x: averageX,
   769тЖТ   769тЖТ    z: maxZ,
   770тЖТ   770тЖТ  };
   771тЖТ   771тЖТ}

   944тЖТ   944тЖТfunction buildTrussDetailWebTags(lineSegments, instanceGeometry) {
   945тЖТ   945тЖТ  const spanMidpoint = (instanceGeometry?.metrics?.span || 0) / 2;
   946тЖТ   946тЖТ  let webIndex = 0;
   947тЖТ   947тЖТ
   948тЖТ   948тЖТ  return lineSegments
   949тЖТ   949тЖТ    .filter((segment) => segment.memberType === 'web' && segment.start && segment.end)
   950тЖТ   950тЖТ    .map((segment) => {
   951тЖТ   951тЖТ      webIndex += 1;
   952тЖТ   952тЖТ
   953тЖТ   953тЖТ      const start = createDetailPoint(segment.start);
   954тЖТ   954тЖТ      const end = createDetailPoint(segment.end);
   955тЖТ   955тЖТ      const dx = end.x - start.x;
   956тЖТ   956тЖТ      const dy = end.y - start.y;
   957тЖТ   957тЖТ      const length = Math.hypot(dx, dy);
   958тЖТ   958тЖТ      if (length <= 1) return null;
   959тЖТ   959тЖТ
   960тЖТ   960тЖТ      const midpoint = {
   961тЖТ   961тЖТ        x: (start.x + end.x) / 2,
   962тЖТ   962тЖТ        y: (start.y + end.y) / 2,
   963тЖТ   963тЖТ      };
   964тЖТ   964тЖТ      let offset = { x: 0, y: 0 };
   965тЖТ   965тЖТ
   966тЖТ   966тЖТ      if (Math.abs(dx) <= 1e-6) {
   967тЖТ   967тЖТ        offset = {
   968тЖТ   968тЖТ          x: midpoint.x < spanMidpoint ? -DETAIL_WEB_TAG_OFFSET : DETAIL_WEB_TAG_OFFSET,
   969тЖТ   969тЖТ          y: 0,
   970тЖТ   970тЖТ        };
   971тЖТ   971тЖТ      } else {
   972тЖТ   972тЖТ        const normal = { x: -dy / length, y: dx / length };
   973тЖТ   973тЖТ        const tangent = { x: dx / length, y: dy / length };
   974тЖТ   974тЖТ        const outwardNormal = selectOutwardNormal(normal, midpoint, spanMidpoint);
   975тЖТ   975тЖТ        const tangentShift = (webIndex % 2 === 0 ? 1 : -1) * DETAIL_WEB_TAG_TANGENT_SHIFT;
   976тЖТ   976тЖТ        offset = {
   977тЖТ   977тЖТ          x: (outwardNormal.x * DETAIL_WEB_TAG_OFFSET) + (tangent.x * tangentShift),
   978тЖТ   978тЖТ          y: (outwardNormal.y * DETAIL_WEB_TAG_OFFSET) + (tangent.y * tangentShift),
   979тЖТ   979тЖТ        };
   980тЖТ   980тЖТ      }
   981тЖТ   981тЖТ
   982тЖТ   982тЖТ      return createDetailTag(
   983тЖТ   983тЖТ        `${instanceGeometry.instance.id}_detail_web_${webIndex}`,
   984тЖТ   984тЖТ        [`W${webIndex} ${formatMeasurement(length)}`],
   985тЖТ   985тЖТ        {
   986тЖТ   986тЖТ          x: midpoint.x + offset.x,
   987тЖТ   987тЖТ          y: midpoint.y + offset.y,
   988тЖТ   988тЖТ        },
   989тЖТ   989тЖТ        {
   990тЖТ   990тЖТ          angle: 0,
   991тЖТ   991тЖТ          sourceType: 'truss_web',
   992тЖТ   992тЖТ          sourceId: segment.id,
   993тЖТ   993тЖТ          measurementValue: length,
   994тЖТ   994тЖТ          leaderLine: {
   995тЖТ   995тЖТ            start: midpoint,
   996тЖТ   996тЖТ            end: {
   997тЖТ   997тЖТ              x: midpoint.x + (offset.x * 0.72),
