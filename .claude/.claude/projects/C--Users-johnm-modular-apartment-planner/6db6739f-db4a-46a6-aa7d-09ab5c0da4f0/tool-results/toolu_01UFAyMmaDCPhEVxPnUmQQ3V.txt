    37→function useUnits() {
    38→  const [unit, setUnit] = useState('mm');
    39→  const isMm = unit === 'mm';
    40→  return {
    41→    unit,
    42→    setUnit,
    43→    suffix: isMm ? 'mm' : 'm',
    44→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    45→    fromDisplay: (v) => isMm ? v : v * 1000,
    46→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    47→  };
    48→}

    50→function WallDrawingInput({ start, dispatch, editorDispatch, activeFloorId, u, selectNewWall }) {
    51→  const [nsDir, setNsDir] = useState('N');
    52→  const [ewDir, setEwDir] = useState('E');
    53→  const [degrees, setDegrees] = useState('');
    54→  const [minutes, setMinutes] = useState('');
    55→  const [lengthVal, setLengthVal] = useState('');
    56→  const lengthRef = useRef(null);
    57→
    58→  const handleConfirm = () => {
    59→    const deg = parseInt(degrees, 10) || 0;
    60→    const min = parseInt(minutes, 10) || 0;
    61→    const len = u.fromDisplay(parseFloat(lengthVal) || 0);
    62→
    63→    if (len < MIN_WALL_LENGTH) return;
    64→    if (deg < 0 || deg > 90 || min < 0 || min > 59) return;
    65→
    66→    const angle = surveyorBearingToSvgAngle(nsDir, deg, min, ewDir);
    67→    const endPt = endpointFromBearing(start, len, angle);
    68→    const wall = createWall(start, endPt);
    69→
    70→    dispatch({ type: 'WALL_ADD', floorId: activeFloorId, wall });
    71→    if (selectNewWall) {
    72→      editorDispatch({ type: 'SELECT_OBJECT', id: wall.id, objectType: 'wall' });
    73→    } else {
    74→      editorDispatch({
    75→        type: 'UPDATE_TOOL_STATE',
    76→        payload: { start: endPt, startAttachment: null, preview: null },
    77→      });
    78→    }
    79→
    80→    setDegrees('');
    81→    setMinutes('');
    82→    setLengthVal('');
    83→    setTimeout(() => lengthRef.current?.focus(), 0);
    84→  };
    85→
    86→  const handleKeyDown = (e) => {
    87→    if (e.key === 'Enter') {
    88→      e.preventDefault();
    89→      handleConfirm();
    90→    }
    91→  };
    92→
    93→  return (
    94→    <div>
    95→      {!selectNewWall && (
    96→        <>
    97→          <div className={styles.title}>Drawing Wall</div>
    98→          <div style={{ fontSize: 12, color: 'var(--color-text-secondary)', marginBottom: 'var(--space-sm)' }}>
    99→            Start: ({u.toDisplay(start.x)}, {u.toDisplay(start.y)}) {u.suffix}
   100→          </div>
   101→        </>
   102→      )}
   103→
   104→      <div className={styles.subtitle}>Bearing</div>
   105→      <div className={styles.bearingRow}>
   106→        <select className={styles.bearingSelect} value={nsDir} onChange={(e) => setNsDir(e.target.value)}>
   107→          <option value="N">N</option>
   108→          <option value="S">S</option>
   109→        </select>
   110→        <input
   111→          className={styles.bearingInput}
   112→          type="number"
   113→          min={0}
   114→          max={90}
   115→          placeholder="0"
   116→          value={degrees}
   117→          onChange={(e) => setDegrees(e.target.value)}
   118→          onKeyDown={handleKeyDown}
   119→        />
   120→        <span className={styles.bearingSymbol}>°</span>
   121→        <input
   122→          className={styles.bearingInput}
   123→          type="number"
   124→          min={0}
   125→          max={59}
   126→          placeholder="0"
   127→          value={minutes}
   128→          onChange={(e) => setMinutes(e.target.value)}
   129→          onKeyDown={handleKeyDown}
   130→        />
   131→        <span className={styles.bearingSymbol}>′</span>
   132→        <select className={styles.bearingSelect} value={ewDir} onChange={(e) => setEwDir(e.target.value)}>
   133→          <option value="E">E</option>
   134→          <option value="W">W</option>
   135→        </select>
   136→      </div>
   137→
   138→      <div className={styles.subtitle}>Length</div>
   139→      <input
   140→        ref={lengthRef}
   141→        className={styles.bearingInput}
   142→        style={{ width: '100%', textAlign: 'left', padding: '0 8px', marginBottom: 'var(--space-sm)' }}
   143→        type="number"
   144→        min={0}
   145→        step={u.step(100)}
   146→        placeholder={`0 ${u.suffix}`}
   147→        value={lengthVal}
   148→        onChange={(e) => setLengthVal(e.target.value)}
   149→        onKeyDown={handleKeyDown}
   150→      />
   151→
   152→      <button className={styles.confirmBtn} onClick={handleConfirm}>
   153→        Create Wall (Enter)
   154→      </button>
   155→    </div>
   156→  );
   157→}

   159→function SheetExportMenu({ sheet, editorDispatch }) {
   160→  const [open, setOpen] = useState(false);
   161→  const rootRef = useRef(null);
   162→  const pngOptions = [
   163→    { label: 'PNG 150 DPI', dpi: 150 },
   164→    { label: 'PNG 300 DPI', dpi: 300 },
   165→    { label: 'PNG 600 DPI', dpi: 600 },
   166→  ];
   167→
   168→  useEffect(() => {
   169→    if (!open) return undefined;
   170→
   171→    const handleMouseDown = (event) => {
   172→      if (rootRef.current && !rootRef.current.contains(event.target)) {
   173→        setOpen(false);
   174→      }
   175→    };
   176→
   177→    document.addEventListener('mousedown', handleMouseDown);
   178→    return () => document.removeEventListener('mousedown', handleMouseDown);
   179→  }, [open]);
   180→
   181→  const handleExport = async (kind, options = {}) => {
   182→    if (!sheet) return;
   183→    try {
   184→      if (kind === 'pdf') {
   185→        await exportActiveSheetAsPdf(
   186→          sheet.title || sheet.drawingName || 'sheet',
   187→          sheet.paperSize
   188→        );
   189→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
   190→      } else {
   191→        await exportActiveSheetAsPng(
   192→          sheet.title || sheet.drawingName || 'sheet',
   193→          options
   194→        );
   195→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
   196→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
   197→      }
   198→    } catch (error) {
   199→      editorDispatch({
   200→        type: 'SET_STATUS_MESSAGE',
   201→        message: error.message || `${kind.toUpperCase()} export failed.`,
   202→      });
   203→    } finally {
   204→      setOpen(false);
   205→    }
   206→  };
   207→
   208→  return (
   209→    <div className={styles.exportMenu} ref={rootRef}>
   210→      <button
   211→        className={styles.exportMenuTrigger}
   212→        type="button"
   213→        onClick={() => setOpen((value) => !value)}
   214→      >
   215→        Export sheet
   216→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   217→      </button>
   218→      {open && (
   219→        <div className={styles.exportMenuList}>
   220→          <button
   221→            className={styles.exportMenuItem}
   222→            type="button"
   223→            onClick={() => handleExport('pdf')}
   224→          >
   225→            PDF
   226→          </button>
   227→          {pngOptions.map((option) => (
   228→            <button
   229→              key={option.dpi}
   230→              className={styles.exportMenuItem}
   231→              type="button"
   232→              onClick={() => handleExport('png', { dpi: option.dpi })}
   233→            >
   234→              {option.label}
   235→            </button>
   236→          ))}
   237→        </div>
   238→      )}
   239→    </div>
   240→  );
   241→}

   243→function WallProperties({ wall, dispatch, editorDispatch, floorId, u }) {
   244→  const len = wallLength(wall);
   245→
   246→  const updateWall = (updates) => {
   247→    dispatch({ type: 'WALL_UPDATE', floorId, wall: { id: wall.id, ...updates } });
   248→  };
   249→
   250→  const updateWallLength = (value) => {
   251→    const resizedWall = resizeWallFromStart(wall, u.fromDisplay(value), MIN_WALL_LENGTH);
   252→    updateWall({ end: resizedWall.end });
   253→  };
   254→
   255→  return (
   256→    <div>
   257→      <div className={styles.title}>Wall</div>
   258→      <div className={styles.subtitle}>Start Point</div>
   259→      <InputField
   260→        label="X" type="number" suffix={u.suffix}
   261→        value={u.toDisplay(wall.start.x)}
   262→        onChange={(v) => updateWall({ start: { ...wall.start, x: u.fromDisplay(v) } })}
   263→      />
   264→      <InputField
   265→        label="Y" type="number" suffix={u.suffix}
   266→        value={u.toDisplay(wall.start.y)}
   267→        onChange={(v) => updateWall({ start: { ...wall.start, y: u.fromDisplay(v) } })}
   268→      />
   269→      <div className={styles.subtitle}>End Point</div>
   270→      <InputField
   271→        label="X" type="number" suffix={u.suffix}
   272→        value={u.toDisplay(wall.end.x)}
   273→        onChange={(v) => updateWall({ end: { ...wall.end, x: u.fromDisplay(v) } })}
   274→      />
   275→      <InputField
   276→        label="Y" type="number" suffix={u.suffix}
   277→        value={u.toDisplay(wall.end.y)}
   278→        onChange={(v) => updateWall({ end: { ...wall.end, y: u.fromDisplay(v) } })}
   279→      />
   280→      <div className={styles.subtitle}>Properties</div>

   282→        label="Thickness" type="number" suffix={u.suffix} step={u.step(10)}

   285→      />
   286→      <InputField
   287→        label="Height" type="number" suffix={u.suffix} step={u.step(100)}
   288→        value={u.toDisplay(wall.height)}

   301→        activeFloorId={floorId}
   302→        u={u}
   303→        selectNewWall

   535→          const newLen = u.fromDisplay(value);
   536→          if (newLen <= 0) return;
   537→          const s = annotation.startPoint;
   538→          const e = annotation.endPoint;
   539→          const mode = annotation.mode;
   540→          let newEnd;
   541→          if (mode === 'horizontal') {
   542→            const sign = e.x >= s.x ? 1 : -1;
   543→            newEnd = { ...e, x: s.x + sign * newLen };
   544→          } else if (mode === 'vertical') {
   545→            const sign = e.y >= s.y ? 1 : -1;
   546→            newEnd = { ...e, y: s.y + sign * newLen };
   547→          } else {
   548→            const dir = normalize(subtract(e, s));
   549→            newEnd = add(s, scale(dir, newLen));
   550→          }
   551→          updateAnnotation({ endPoint: newEnd });
   552→        }}
   553→      />
   554→    </div>
   555→  );
   556→}
