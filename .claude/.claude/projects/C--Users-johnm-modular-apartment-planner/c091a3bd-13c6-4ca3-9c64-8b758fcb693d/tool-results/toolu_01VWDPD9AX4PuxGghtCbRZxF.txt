    27→function useUnits() {
    28→  const [unit, setUnit] = useState('mm');
    29→  const isMm = unit === 'mm';
    30→  return {
    31→    unit,
    32→    setUnit,
    33→    suffix: isMm ? 'mm' : 'm',
    34→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    35→    fromDisplay: (v) => isMm ? v : v * 1000,
    36→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    37→  };
    38→}

    40→function SheetExportMenu({ sheet, editorDispatch }) {
    41→  const [open, setOpen] = useState(false);
    42→  const rootRef = useRef(null);
    43→  const pngOptions = [
    44→    { label: 'PNG 150 DPI', dpi: 150 },
    45→    { label: 'PNG 300 DPI', dpi: 300 },
    46→    { label: 'PNG 600 DPI', dpi: 600 },
    47→  ];
    48→
    49→  useEffect(() => {
    50→    if (!open) return undefined;
    51→
    52→    const handleMouseDown = (event) => {
    53→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    54→        setOpen(false);
    55→      }
    56→    };
    57→
    58→    document.addEventListener('mousedown', handleMouseDown);
    59→    return () => document.removeEventListener('mousedown', handleMouseDown);
    60→  }, [open]);
    61→
    62→  const handleExport = async (kind, options = {}) => {
    63→    if (!sheet) return;
    64→    try {
    65→      if (kind === 'pdf') {
    66→        await exportActiveSheetAsPdf(
    67→          sheet.title || sheet.drawingName || 'sheet',
    68→          sheet.paperSize
    69→        );
    70→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    71→      } else {
    72→        await exportActiveSheetAsPng(
    73→          sheet.title || sheet.drawingName || 'sheet',
    74→          options
    75→        );
    76→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    77→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    78→      }
    79→    } catch (error) {
    80→      editorDispatch({
    81→        type: 'SET_STATUS_MESSAGE',
    82→        message: error.message || `${kind.toUpperCase()} export failed.`,
    83→      });
    84→    } finally {
    85→      setOpen(false);
    86→    }
    87→  };
    88→
    89→  return (
    90→    <div className={styles.exportMenu} ref={rootRef}>
    91→      <button
    92→        className={styles.exportMenuTrigger}
    93→        type="button"
    94→        onClick={() => setOpen((value) => !value)}
    95→      >
    96→        Save sheet
    97→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    98→      </button>
    99→      {open && (
   100→        <div className={styles.exportMenuList}>
   101→          <button
   102→            className={styles.exportMenuItem}
   103→            type="button"
   104→            onClick={() => handleExport('pdf')}
   105→          >
   106→            PDF
   107→          </button>
   108→          {pngOptions.map((option) => (
   109→            <button
   110→              key={option.dpi}
   111→              className={styles.exportMenuItem}
   112→              type="button"
   113→              onClick={() => handleExport('png', { dpi: option.dpi })}
   114→            >
   115→              {option.label}
   116→            </button>
   117→          ))}
   118→        </div>
   119→      )}
   120→    </div>
   121→  );
   122→}

   501→function LandingProperties({ landing, floor, dispatch, floorId, u }) {
   502→  const updateLanding = (updates) => {
   503→    dispatch({ type: 'LANDING_UPDATE', floorId, landing: { id: landing.id, ...updates } });
   504→  };
   505→
   506→  const stairs = floor?.stairs || [];

   829→function SheetProperties({ sheet, project, dispatch, editorDispatch, activeFloorId, u }) {

   835→    try {
   836→      const sourceFloorId = getDefaultActiveFloorId(project, activeFloorId);

   845→        // Geometry resolution failed — use default dimensions

   974→    </div>

  1069→export default function PropertiesPanel() {
  1070→  const { project, dispatch, duplicateFloor } = useProject();
  1071→  const { selectedId, selectedType, activeFloorId, activeSheetId, workspaceMode, dispatch: editorDispatch } = useEditor();
  1072→  const orderedFloors = getOrderedFloors(project);
  1073→  const floor = orderedFloors.find((entry) => entry.id === activeFloorId) || null;
  1074→  const sheet = (project.sheets || []).find((entry) => entry.id === activeSheetId) || null;
  1075→  const u = useUnits();
  1076→
  1077→  const selectFloor = (floorId) => {
  1078→    editorDispatch({ type: 'SET_WORKSPACE_MODE', workspaceMode: 'model' });
  1079→    editorDispatch({ type: 'SET_ACTIVE_FLOOR', floorId });
  1080→    editorDispatch({ type: 'SELECT_OBJECT', id: floorId, objectType: 'floor' });
  1081→  };
  1082→
  1083→  const handleDuplicateFloor = (floorToDuplicate) => {
  1084→    const duplicatedFloor = duplicateFloor(floorToDuplicate.id);
  1085→    if (!duplicatedFloor) return;
  1086→
  1087→    dispatch({ type: 'FLOOR_DUPLICATE', floor: duplicatedFloor });
  1088→    selectFloor(duplicatedFloor.id);
  1089→  };
  1090→
  1091→  const handleDeleteFloor = (floorToDelete) => {
  1092→    if (orderedFloors.length <= 1) return;
  1093→    const confirmed = window.confirm(`Delete "${floorToDelete.name}"?`);
  1094→    if (!confirmed) return;
  1095→
  1096→    const remainingFloors = orderedFloors.filter((entry) => entry.id !== floorToDelete.id);
  1097→    const fallbackFloor = remainingFloors[0] || null;
  1098→
  1099→    dispatch({
  1100→      type: 'FLOOR_DELETE',
  1101→      floorId: floorToDelete.id,
  1102→      fallbackFloorId: fallbackFloor?.id ?? null,
  1103→    });
  1104→
  1105→    if (fallbackFloor) {
  1106→      selectFloor(fallbackFloor.id);
  1107→    } else {
  1108→      editorDispatch({ type: 'DESELECT' });
  1109→    }
  1110→  };
  1111→
  1112→  const handleDelete = () => {
  1113→    if (!selectedId) return;
  1114→    if (selectedType === 'sheet') {
  1115→      dispatch({ type: 'SHEET_DELETE', sheetId: selectedId });
  1116→    } else if (selectedType === 'sheetViewport' && sheet) {
  1117→      dispatch({ type: 'SHEET_VIEWPORT_DELETE', sheetId: sheet.id, viewportId: selectedId });
  1118→    } else if (selectedType === 'slab') {
  1119→      dispatch({ type: 'SLAB_DELETE', floorId: activeFloorId, slabId: selectedId });
  1120→    } else if (selectedType === 'wall') {
  1121→      dispatch({ type: 'WALL_DELETE', floorId: activeFloorId, wallId: selectedId });
  1122→    } else if (selectedType === 'beam') {
  1123→      dispatch({ type: 'BEAM_DELETE', floorId: activeFloorId, beamId: selectedId });
  1124→    } else if (selectedType === 'sectionCut') {
  1125→      dispatch({ type: 'SECTION_DELETE', floorId: activeFloorId, sectionId: selectedId });
  1126→    } else if (selectedType === 'annotation') {
  1127→      dispatch({ type: 'ANNOTATION_DELETE', floorId: activeFloorId, annotationId: selectedId });
  1128→    } else if (selectedType === 'stair') {
  1129→      dispatch({ type: 'STAIR_DELETE', floorId: activeFloorId, stairId: selectedId });
  1130→    } else if (selectedType === 'landing') {
  1131→      dispatch({ type: 'LANDING_DELETE', floorId: activeFloorId, landingId: selectedId });
  1132→    } else if (selectedType === 'door') {
  1133→      dispatch({ type: 'DOOR_DELETE', floorId: activeFloorId, doorId: selectedId });
  1134→    } else if (selectedType === 'window') {
  1135→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
  1136→    } else if (selectedType === 'column') {
  1137→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
  1138→    } else if (selectedType === 'room') {
  1139→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
  1140→    } else if (selectedType === 'floor') {
  1141→      const targetFloor = orderedFloors.find((entry) => entry.id === selectedId) || null;
  1142→      if (targetFloor) {
  1143→        handleDeleteFloor(targetFloor);
  1144→        return;
  1145→      }
  1146→    }
  1147→    editorDispatch({ type: 'DESELECT' });
  1148→  };
  1149→
  1150→  let content = workspaceMode === 'sheet' && sheet
  1151→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
  1152→    : (
  1153→        <>
  1154→          {floor && (
  1155→            <FloorProperties
  1156→              floor={floor}
  1157→              dispatch={dispatch}
  1158→              onDuplicate={handleDuplicateFloor}
  1159→              onDelete={handleDeleteFloor}
  1160→              canDelete={orderedFloors.length > 1}
  1161→              u={u}
  1162→            />
  1163→          )}
  1164→          <ProjectSummary project={project} floor={floor} dispatch={dispatch} />
  1165→        </>
  1166→      );
  1167→
  1168→  if (selectedId && workspaceMode === 'sheet' && sheet) {
  1169→    if (selectedType === 'sheet') {
  1170→      if (sheet.id === selectedId) {
  1171→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;
  1172→      }
  1173→    } else if (selectedType === 'sheetViewport') {
  1174→      const viewport = (sheet.viewports || []).find((entry) => entry.id === selectedId);
  1175→      if (viewport) {
  1176→        content = <SheetViewportProperties sheet={sheet} viewport={viewport} project={project} dispatch={dispatch} u={u} />;
  1177→      }
  1178→    }
  1179→  } else if (selectedId && floor) {
  1180→    if (selectedType === 'floor') {
  1181→      const selectedFloor = orderedFloors.find((entry) => entry.id === selectedId) || floor;
  1182→      content = (
  1183→        <FloorProperties
  1184→          floor={selectedFloor}
  1185→          dispatch={dispatch}
  1186→          onDuplicate={handleDuplicateFloor}
  1187→          onDelete={handleDeleteFloor}
  1188→          canDelete={orderedFloors.length > 1}
  1189→          u={u}
  1190→        />
  1191→      );
  1192→    } else if (selectedType === 'slab') {
  1193→      const slab = (floor.slabs || []).find(s => s.id === selectedId) || null;
  1194→      if (slab) {
  1195→        content = <SlabProperties slab={slab} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;

  1197→    } else if (selectedType === 'wall') {
  1198→      const wall = floor.walls.find(w => w.id === selectedId);
  1199→      if (wall) {

  1259→        >mm</button>

  1263→        >m</button>
  1264→      </div>
  1265→      {workspaceMode === 'sheet' && sheet && (
