    49→  return `trussChord_${normalizedMaterial}`;

    56→function createBounds(points = []) {
    57→  if (!points.length) {
    58→    return {
    59→      minX: 0,
    60→      maxX: 1000,
    61→      minY: 0,
    62→      maxY: 1000,
    63→      minElevation: 0,
    64→      maxElevation: 1000,
    65→    };
    66→  }
    67→
    68→  return {
    69→    minX: Math.min(...points.map((point) => point.x)),
    70→    maxX: Math.max(...points.map((point) => point.x)),
    71→    minY: Math.min(...points.map((point) => point.y)),
    72→    maxY: Math.max(...points.map((point) => point.y)),
    73→    minElevation: Math.min(...points.map((point) => point.elevation ?? 0)),
    74→    maxElevation: Math.max(...points.map((point) => point.elevation ?? 0)),
    75→  };
    76→}

    78→function centerOfBounds(bounds) {
    79→  return {
    80→    x: (bounds.minX + bounds.maxX) / 2,
    81→    y: (bounds.minY + bounds.maxY) / 2,
    82→  };
    83→}

    85→function pointFromLocal(structuralStartPoint, spanDirection, localPoint, baseElevation) {
    86→  const planPoint = add(structuralStartPoint, scale(spanDirection, localPoint.x));
    87→  return {
    88→    x: planPoint.x,
    89→    y: planPoint.y,
    90→    elevation: baseElevation + localPoint.z,
    91→  };
    92→}

    94→function createCopyGeometry(trussSystem, instance, profile, layout, origin, index) {
    95→  const structuralStartPoint = add(origin, scale(layout.spanDirection, -(profile.metrics.span / 2)));
    96→  const structuralEndPoint = add(origin, scale(layout.spanDirection, profile.metrics.span / 2));
    97→  const overallStartPoint = add(structuralStartPoint, scale(layout.spanDirection, -profile.metrics.overhangStart));
    98→  const overallEndPoint = add(structuralEndPoint, scale(layout.spanDirection, profile.metrics.overhangEnd));
    99→  const nodeMap = new Map(profile.nodes.map((node) => [node.id, node]));
   100→  const baseElevation = trussSystem.baseElevation || 0;
   101→
   102→  const worldNodes = profile.nodes.map((node) => ({
   103→    ...node,
   104→    planPoint: add(structuralStartPoint, scale(layout.spanDirection, node.x)),
   105→    worldPoint: pointFromLocal(structuralStartPoint, layout.spanDirection, node, baseElevation),
   106→  }));
   107→  const members = profile.members.map((member) => {
   108→    const startNode = nodeMap.get(member.startNodeId);
   109→    const endNode = nodeMap.get(member.endNodeId);
   110→    return {
   111→      ...member,
   112→      startNode,
   113→      endNode,
   114→      startWorld: pointFromLocal(structuralStartPoint, layout.spanDirection, startNode, baseElevation),
   115→      endWorld: pointFromLocal(structuralStartPoint, layout.spanDirection, endNode, baseElevation),
   116→    };
   117→  });
   118→  const purlinAttachments = (profile.purlinAttachments || []).map((attachment) => ({
   119→    ...attachment,
   120→    planPoint: add(structuralStartPoint, scale(layout.spanDirection, attachment.localPoint.x)),
   121→    worldPoint: pointFromLocal(structuralStartPoint, layout.spanDirection, attachment.localPoint, baseElevation),
   122→  }));
   123→
   124→  return {
   125→    id: `${instance.id}_copy_${index}`,
   126→    index,
   127→    origin,
   128→    structuralStartPoint,
   129→    structuralEndPoint,
   130→    overallStartPoint,
   131→    overallEndPoint,
   132→    worldNodes,
   133→    members,
   134→    purlinAttachments,
   135→  };
   136→}

   138→function normalizePlanVector(vector) {
   139→  const magnitude = Math.sqrt((vector.x * vector.x) + (vector.y * vector.y));
   140→  if (magnitude <= 1e-6) {
   141→    return { x: 1, y: 0 };
   142→  }
   143→  return {
   144→    x: vector.x / magnitude,
   145→    y: vector.y / magnitude,
   146→  };
   147→}

   149→function createPlanMarker(centerPoint, axis, length = DEFAULT_PURLIN_MARKER_LENGTH) {
   150→  return {
   151→    start: add(centerPoint, scale(axis, -(length / 2))),
   152→    end: add(centerPoint, scale(axis, length / 2)),
   153→  };
   154→}

   156→function createPurlinSegment(id, trussSystem, instance, startWorld, endWorld, purlinSystem, side = null, copyIds = []) {
   157→  const material = normalizeMaterial(purlinSystem?.material || 'timber');
   158→  return {
   159→    id,
   160→    side,
   161→    copyIds,
   162→    startWorld,
   163→    endWorld,
   164→    thickness: Math.max(purlinSystem?.depth || 1, 1),
   165→    crossSection: {
   166→      height: Math.max(purlinSystem?.depth || 1, 1),
   167→      width: Math.max(purlinSystem?.width || 1, 1),
   168→    },
   169→    materialKey: getPurlinMaterialKey(material),
   170→    metadata: {
   171→      sourceId: instance.id,
   172→      trussSystemId: trussSystem.id,
   173→      floorId: trussSystem.floorId,
   174→      memberType: 'purlin',
   175→      purlinSide: side,
   176→      material,
   177→    },
   178→    bounds: {
   179→      minX: Math.min(startWorld.x, endWorld.x),
   180→      maxX: Math.max(startWorld.x, endWorld.x),
   181→      minY: Math.min(startWorld.y, endWorld.y),
   182→      maxY: Math.max(startWorld.y, endWorld.y),
   183→      minElevation: Math.min(startWorld.elevation, endWorld.elevation),
   184→      maxElevation: Math.max(startWorld.elevation, endWorld.elevation),
   185→    },
   186→  };
   187→}

   189→function offsetWorldPoint(worldPoint, axis, distance) {
   190→  if (!worldPoint || Math.abs(distance) <= 1e-6) return worldPoint;
   191→  return {
   192→    ...worldPoint,
   193→    x: worldPoint.x + (axis.x * distance),
   194→    y: worldPoint.y + (axis.y * distance),
   195→  };
   196→}

   198→function buildInstancePurlinGeometry(trussSystem, instanceGeometry, purlinSystem) {
   199→  if (!purlinSystem?.enabled) {
   200→    return {
   201→      purlinRuns: [],
   202→      purlinMarkers: [],
   203→      purlinSegments: [],
   204→    };
   205→  }
   206→
   207→  const attachmentGroups = new Map();
   208→  for (const copy of instanceGeometry.copies) {
   209→    for (const attachment of copy.purlinAttachments || []) {
   210→      if (!attachmentGroups.has(attachment.id)) {
   211→        attachmentGroups.set(attachment.id, []);
   212→      }
   213→      attachmentGroups.get(attachment.id).push({
   214→        ...attachment,
   215→        copyId: copy.id,
   216→        copyIndex: copy.index,
   217→      });
   218→    }
   219→  }
   220→
   221→  const layoutAxis = normalizePlanVector({
   222→    x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   223→    y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   224→  });
   225→  const purlinRuns = [];
   226→  const purlinMarkers = [];
   227→  const purlinSegments = [];
   228→  const overhangStart = Math.max(Number(purlinSystem?.overhangStart || 0), 0);
   229→  const overhangEnd = Math.max(Number(purlinSystem?.overhangEnd || 0), 0);
   230→
   231→  for (const [attachmentId, attachmentPoints] of attachmentGroups.entries()) {
   232→    const sortedPoints = [...attachmentPoints].sort((a, b) => a.copyIndex - b.copyIndex);
   233→    if (sortedPoints.length >= 2) {
   234→      const firstPoint = sortedPoints[0];
   235→      const lastPoint = sortedPoints[sortedPoints.length - 1];
   236→      const runPoints = [
   237→        ...(overhangStart > 1e-6
   238→          ? [add(firstPoint.planPoint, scale(layoutAxis, -overhangStart))]
   239→          : []),
   240→        ...sortedPoints.map((point) => point.planPoint),
   241→        ...(overhangEnd > 1e-6
   242→          ? [add(lastPoint.planPoint, scale(layoutAxis, overhangEnd))]
   243→          : []),
   244→      ];
   245→      const segments = sortedPoints.slice(0, -1).map((startPoint, index) => {
   246→        const endPoint = sortedPoints[index + 1];
   247→        const startWorld = index === 0
   248→          ? offsetWorldPoint(startPoint.worldPoint, layoutAxis, -overhangStart)
   249→          : startPoint.worldPoint;
   250→        const endWorld = index === sortedPoints.length - 2
   251→          ? offsetWorldPoint(endPoint.worldPoint, layoutAxis, overhangEnd)
   252→          : endPoint.worldPoint;
   253→
   254→        return createPurlinSegment(
   255→          `${instanceGeometry.instance.id}_${attachmentId}_${index}`,
   256→          trussSystem,
   257→          instanceGeometry.instance,
   258→          startWorld,
   259→          endWorld,
   260→          purlinSystem,
   261→          startPoint.side,
   262→          [startPoint.copyId, endPoint.copyId]
   263→        );
   264→      }).filter(Boolean);
   265→      purlinRuns.push({
   266→        id: `${instanceGeometry.instance.id}_${attachmentId}_run`,
   267→        side: firstPoint.side,
   268→        points: runPoints,
   269→        attachments: sortedPoints,
   270→        segments,
   271→      });
   272→      purlinSegments.push(...segments);
   273→      continue;
   274→    }
   275→
   276→    if (sortedPoints.length === 1) {
   277→      const marker = createPlanMarker(sortedPoints[0].planPoint, layoutAxis);
   278→      purlinMarkers.push({
   279→        id: `${instanceGeometry.instance.id}_${attachmentId}_marker`,
   280→        side: sortedPoints[0].side,
   281→        center: sortedPoints[0].planPoint,
   282→        ...marker,
   283→      });
   284→    }
   285→  }
   286→
   287→  return {
   288→    purlinRuns,
   289→    purlinMarkers,
   290→    purlinSegments,
   291→  };
   292→}

   295→  const rotatedPlanPoint = transformPlanPoint(worldPoint, transform);
   296→  return {
   297→    ...worldPoint,
   298→    x: rotatedPlanPoint.x,
   299→    y: rotatedPlanPoint.y,

   303→function transformCopyGeometry(copy, transform) {
   304→  const worldNodes = copy.worldNodes.map((node) => ({
   305→    ...node,
   306→    planPoint: transformPlanPoint(node.planPoint, transform),
   307→    worldPoint: rotateWorldPoint(node.worldPoint, transform),
   308→  }));
   309→  const members = copy.members.map((member) => ({
   310→    ...member,
   311→    startWorld: rotateWorldPoint(member.startWorld, transform),
   312→    endWorld: rotateWorldPoint(member.endWorld, transform),
   313→  }));
   314→  const purlinAttachments = (copy.purlinAttachments || []).map((attachment) => ({
   315→    ...attachment,
   316→    planPoint: transformPlanPoint(attachment.planPoint, transform),
   317→    worldPoint: rotateWorldPoint(attachment.worldPoint, transform),
   318→  }));
   319→
   320→  return {
   321→    ...copy,
   322→    origin: transformPlanPoint(copy.origin, transform),
   323→    structuralStartPoint: transformPlanPoint(copy.structuralStartPoint, transform),
   324→    structuralEndPoint: transformPlanPoint(copy.structuralEndPoint, transform),
   325→    overallStartPoint: transformPlanPoint(copy.overallStartPoint, transform),
   326→    overallEndPoint: transformPlanPoint(copy.overallEndPoint, transform),
   327→    worldNodes,
   328→    members,
   329→    purlinAttachments,
   330→  };
   331→}

   333→function transformInstanceGeometry(instanceGeometry, transform, baseElevation) {
   334→  const copies = instanceGeometry.copies.map((copy) => transformCopyGeometry(copy, transform));
   335→  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   336→
   337→  return {
   338→    ...instanceGeometry,
   339→    copies,
   340→    layoutLineStartPoint: transformPlanPoint(instanceGeometry.instance.startPoint, transform),
   341→    layoutLineEndPoint: transformPlanPoint(instanceGeometry.instance.endPoint, transform),
   342→    planBounds: createBounds(planPoints.map((point) => ({
   343→      x: point.x,
   344→      y: point.y,
   345→      elevation: baseElevation,
   346→    }))),
   347→  };
   348→}

   350→function buildRawTrussInstanceGeometry(trussSystem, instance) {
   351→  const layout = resolveTrussLayout(instance);
   352→  const profile = buildTrussProfile(instance);
   353→  const purlinAttachments = buildTrussPurlinAttachments(profile, trussSystem?.purlinSystem);
   354→  const copies = buildTrussCopyOrigins(instance).map((copy) => {
   355→    const layoutOffset = (
   356→      ((copy.origin.x ?? 0) - (instance.startPoint?.x ?? 0)) * layout.axis.x
   357→    ) + (
   358→      ((copy.origin.y ?? 0) - (instance.startPoint?.y ?? 0)) * layout.axis.y
   359→    );
   360→    const copyProfile = buildTrussProfile(instance, undefined, {
   361→      layoutOffset,
   362→      supportLength: layout.supportLength,
   363→    });
   364→    const copyPurlinAttachments = buildTrussPurlinAttachments(copyProfile, trussSystem?.purlinSystem);
   365→
   366→    return createCopyGeometry(trussSystem, instance, {
   367→      ...copyProfile,
   368→      purlinAttachments: copyPurlinAttachments,
   369→    }, layout, copy.origin, copy.index);
   370→  });
   371→
   372→  const planPoints = copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint]);
   373→  const detailPoints = [
   374→    ...profile.roofOutline.map((point) => ({ x: point.x, y: 0, elevation: point.z })),
   375→    ...profile.bottomOutline.map((point) => ({ x: point.x, y: 0, elevation: point.z })),
   376→    ...purlinAttachments.map((attachment) => ({
   377→      x: attachment.localPoint.x,
   378→      y: 0,
   379→      elevation: attachment.localPoint.z,
   380→    })),
   381→  ];
   382→
   383→  return {
   384→    instance,
   385→    trussType: profile.trussType,
   386→    metrics: profile.metrics,
   387→    nodes: profile.nodes,
   388→    members: profile.members,
   389→    roofOutline: profile.roofOutline,
   390→    bottomOutline: profile.bottomOutline,
   391→    topChordRuns: profile.topChordRuns || [],
   392→    purlinAttachments,
   393→    copies,
   394→    layoutLineStartPoint: instance.startPoint,
   395→    layoutLineEndPoint: instance.endPoint,
   396→    planBounds: createBounds(planPoints.map((point) => ({
   397→      x: point.x,
   398→      y: point.y,
   399→      elevation: trussSystem.baseElevation || 0,
   400→    }))),
   401→    detailBounds: {
   402→      minX: Math.min(...detailPoints.map((point) => point.x)) - DETAIL_PADDING_X,
   403→      maxX: Math.max(...detailPoints.map((point) => point.x)) + DETAIL_PADDING_X,
   404→      minZ: Math.min(...detailPoints.map((point) => point.elevation), 0) - DETAIL_PADDING_Z,
   405→      maxZ: Math.max(...detailPoints.map((point) => point.elevation), 0) + DETAIL_PADDING_Z,
   406→    },
   407→  };
   408→}

   410→function createSystemTransform(trussSystem, rawInstances, baseElevation) {
   411→  const rawPlanPoints = rawInstances.flatMap((instanceGeometry) => (
   412→    instanceGeometry.copies.flatMap((copy) => [copy.overallStartPoint, copy.overallEndPoint])
   413→  ));
   414→  const rawBounds = createBounds(rawPlanPoints.map((point) => ({
   415→    x: point.x,
   416→    y: point.y,
   417→    elevation: baseElevation,
   418→  })));
   419→  const rawPivot = centerOfBounds(rawBounds);
   420→  const primaryAxis = getSafeAxis(rawInstances[0]
   421→    ? {
   422→        x: rawInstances[0].layoutLineEndPoint.x - rawInstances[0].layoutLineStartPoint.x,
   423→        y: rawInstances[0].layoutLineEndPoint.y - rawInstances[0].layoutLineStartPoint.y,
   424→      }
   425→    : { x: 1, y: 0 });
   426→  const projectedAlongs = rawPlanPoints.map((point) => projectAlongAxis(point, rawPivot, primaryAxis));
   427→  const startAlong = projectedAlongs.length ? Math.min(...projectedAlongs) : 0;
   428→  const endAlong = projectedAlongs.length ? Math.max(...projectedAlongs) : 0;
   429→  const rawLength = Math.max(endAlong - startAlong, 0);
   430→  const resizable = rawLength > 1e-6 && rawInstances.every((instanceGeometry) => {
   431→    const instanceAxis = getSafeAxis({
   432→      x: instanceGeometry.layoutLineEndPoint.x - instanceGeometry.layoutLineStartPoint.x,
   433→      y: instanceGeometry.layoutLineEndPoint.y - instanceGeometry.layoutLineStartPoint.y,
   434→    });
   435→    return Math.abs(Math.abs((instanceAxis.x * primaryAxis.x) + (instanceAxis.y * primaryAxis.y)) - 1) <= 1e-3;
   436→  });
   437→  const rotationDegrees = normalizeSignedRotationDegrees(trussSystem?.planRotationOffsetDegrees || 0);
   438→  const translation = normalizePlanOffset(trussSystem?.planOffset);
   439→  const lengthScale = normalizePlanLengthScale(trussSystem?.planLengthScale);
   440→
   441→  return {
   442→    rawPivot,
   443→    pivot: {
   444→      x: rawPivot.x + translation.x,
   445→      y: rawPivot.y + translation.y,
   446→    },
   447→    axis: primaryAxis,
   448→    displayAxis: getDisplayedAxis({ axis: primaryAxis, rotationDegrees }),
   449→    translation,
   450→    lengthScale,
   451→    rawLength,
   452→    currentLength: rawLength * lengthScale,
   453→    startAlong,
   454→    endAlong,
   455→    startHandlePoint: transformPlanPoint(
   456→      {
   457→        x: rawPivot.x + (primaryAxis.x * startAlong),
   458→        y: rawPivot.y + (primaryAxis.y * startAlong),
   459→      },
   460→      {
   461→        rawPivot,
   462→        axis: primaryAxis,
   463→        rotationDegrees,
   464→        translation,
   465→        lengthScale,
   466→      }
   467→    ),
   468→    endHandlePoint: transformPlanPoint(
   469→      {
   470→        x: rawPivot.x + (primaryAxis.x * endAlong),
   471→        y: rawPivot.y + (primaryAxis.y * endAlong),
   472→      },
   473→      {
   474→        rawPivot,
   475→        axis: primaryAxis,
   476→        rotationDegrees,
   477→        translation,
   478→        lengthScale,
   479→      }
   480→    ),
   481→    resizable,
   482→    rotationDegrees,
   483→  };
   484→}

   558→function detailPolylineLength(points = []) {
   559→  let total = 0;
   560→  for (let index = 0; index < points.length - 1; index += 1) {
   561→    total += Math.hypot(points[index + 1].x - points[index].x, points[index + 1].y - points[index].y);
   562→  }
   563→  return total;
   564→}
   565→
   566→function worldSegmentLength(startWorld, endWorld) {
   567→  if (!startWorld || !endWorld) return 0;
   568→  return Math.hypot(
   569→    endWorld.x - startWorld.x,
   570→    endWorld.y - startWorld.y,
   571→    (endWorld.elevation || 0) - (startWorld.elevation || 0)
   572→  );
   573→}

   575→function measurePurlinRunLength(run = null) {
   576→  if (!run) return 0;
   577→  if (run.segments?.length) {
   578→    return run.segments.reduce((total, segment) => (
   579→      total + worldSegmentLength(segment.startWorld, segment.endWorld)
   580→    ), 0);
   581→  }
   582→  return detailPolylineLength(run.points || []);
   583→}

   585→function detailPolylineMidpoint(points = []) {
   586→  if (!points.length) return { x: 0, y: 0 };
   587→  if (points.length === 1) return points[0];
   588→
   589→  const totalLength = detailPolylineLength(points);
   590→  if (totalLength <= 1e-6) {
   591→    return {
   592→      x: (points[0].x + points[points.length - 1].x) / 2,
   593→      y: (points[0].y + points[points.length - 1].y) / 2,
   594→    };
   595→  }
   596→
   597→  let remaining = totalLength / 2;
   598→  for (let index = 0; index < points.length - 1; index += 1) {
   599→    const start = points[index];
   600→    const end = points[index + 1];
   601→    const segmentLength = Math.hypot(end.x - start.x, end.y - start.y);
   602→    if (remaining <= segmentLength || index === points.length - 2) {
   603→      const ratio = segmentLength <= 1e-6 ? 0 : remaining / segmentLength;
   604→      return {
   605→        x: start.x + ((end.x - start.x) * ratio),
   606→        y: start.y + ((end.y - start.y) * ratio),
   607→      };
   608→    }
   609→    remaining -= segmentLength;
   610→  }
   611→
   612→  return points[points.length - 1];
   613→}

   648→  return Math.max(fontSize * 1.2, String(text || '').length * fontSize * 0.58);

   651→function estimateDimensionBounds(figure) {
   652→  if (!figure) return null;
   653→
   654→  const points = [
   655→    figure.lineStart,
   656→    figure.lineEnd,
   657→    ...figure.extensionLines.flatMap((line) => [line.start, line.end]),
   658→    ...figure.arrowheads.flatMap((arrow) => [arrow.left, arrow.tip, arrow.right]),
   659→  ].filter(Boolean);
   660→  const fontSize = DRAWING_GRAPHICS.annotation.textSize;
   661→  const width = estimateTextWidth(figure.text?.value, fontSize) + DRAWING_GRAPHICS.annotation.textMaskPaddingX;
   662→  const height = fontSize + DRAWING_GRAPHICS.annotation.textMaskPaddingY;
   663→  const textPosition = figure.text?.position || { x: 0, y: 0 };
   664→  const textRadius = Math.sqrt((width * width) + (height * height)) / 2;
   665→
   666→  points.push(
   667→    { x: textPosition.x - textRadius, y: textPosition.y - textRadius },
   668→    { x: textPosition.x + textRadius, y: textPosition.y + textRadius }
   669→  );
   670→
   671→  return {
   672→    minX: Math.min(...points.map((point) => point.x)),
   673→    maxX: Math.max(...points.map((point) => point.x)),
   674→    minY: Math.min(...points.map((point) => point.y)),
   675→    maxY: Math.max(...points.map((point) => point.y)),
   676→  };
   677→}

   679→function estimateTagBounds(tag) {
   680→  if (!tag) return null;
   681→
   682→  const fontSize = Number(tag.fontSize || DETAIL_TAG_FONT_SIZE);
   683→  const lineHeight = Number(tag.lineHeight || DETAIL_TAG_LINE_HEIGHT);
   684→  const maskPaddingX = Number(tag.maskPaddingX ?? DETAIL_TAG_MASK_PADDING_X);
   685→  const maskPaddingY = Number(tag.maskPaddingY ?? DETAIL_TAG_MASK_PADDING_Y);
   686→  const maxLineLength = Math.max(...tag.textLines.map((line) => String(line || '').length), 1);
   687→  const width = (maxLineLength * fontSize * 0.58) + Math.max(maskPaddingX, fontSize * 0.5);
   688→  const height = Math.max(lineHeight, tag.textLines.length * lineHeight) + maskPaddingY;
   689→  const minX = tag.textAnchor === 'start'
   690→    ? tag.position.x
   691→    : (tag.textAnchor === 'end' ? tag.position.x - width : tag.position.x - (width / 2));
   692→  const maxX = tag.textAnchor === 'start'
   693→    ? tag.position.x + width
   694→    : (tag.textAnchor === 'end' ? tag.position.x : tag.position.x + (width / 2));
   695→  const minY = tag.position.y - (height / 2);
   696→  const maxY = tag.position.y + (height / 2);
   697→  const angle = Number(tag.angle || 0);
   698→  const corners = [
   699→    { x: minX, y: minY },
   700→    { x: maxX, y: minY },
   701→    { x: maxX, y: maxY },
   702→    { x: minX, y: maxY },
   703→  ].map((corner) => rotate(corner, tag.position, angle));
   704→
   705→  return {
   706→    minX: Math.min(...corners.map((point) => point.x)),
   707→    maxX: Math.max(...corners.map((point) => point.x)),
   708→    minY: Math.min(...corners.map((point) => point.y)),
   709→    maxY: Math.max(...corners.map((point) => point.y)),
   710→  };
   711→}

   713→function selectOutwardNormal(normal, midpoint, spanMidpoint) {
   714→  const preferred = midpoint.x < spanMidpoint
   715→    ? { x: -1, y: 0 }
   716→    : { x: 1, y: 0 };
   717→  const alignment = (normal.x * preferred.x) + (normal.y * preferred.y);
   718→
   719→  return alignment >= 0
   720→    ? normal
   721→    : { x: -normal.x, y: -normal.y };
   722→}

   724→function extendDetailBounds(detailBounds, dimensions = [], tags = []) {
   725→  const bounds = {
   726→    minX: detailBounds.minX,
   727→    maxX: detailBounds.maxX,
   728→    minY: -detailBounds.maxZ,
   729→    maxY: -detailBounds.minZ,
   730→  };
   731→
   732→  for (const figure of dimensions) {
   733→    const figureBounds = estimateDimensionBounds(figure);
   734→    if (!figureBounds) continue;
   735→    bounds.minX = Math.min(bounds.minX, figureBounds.minX);
   736→    bounds.maxX = Math.max(bounds.maxX, figureBounds.maxX);
   737→    bounds.minY = Math.min(bounds.minY, figureBounds.minY);
   738→    bounds.maxY = Math.max(bounds.maxY, figureBounds.maxY);
   739→  }
   740→
   741→  for (const tag of tags) {
   742→    const tagBounds = estimateTagBounds(tag);
   743→    if (!tagBounds) continue;
   744→    bounds.minX = Math.min(bounds.minX, tagBounds.minX);
   745→    bounds.maxX = Math.max(bounds.maxX, tagBounds.maxX);
   746→    bounds.minY = Math.min(bounds.minY, tagBounds.minY);
   747→    bounds.maxY = Math.max(bounds.maxY, tagBounds.maxY);
   748→  }
   749→
   750→  return {
   751→    minX: bounds.minX,
   752→    maxX: bounds.maxX,
   753→    minZ: -bounds.maxY,
   754→    maxZ: -bounds.minY,
   755→  };
   756→}

   758→function resolveDetailCrestPoint(roofOutline = []) {
   759→  if (!roofOutline.length) {
   760→    return { x: 0, z: 0 };
   761→  }
   762→
   763→  const maxZ = Math.max(...roofOutline.map((point) => point.z));
   764→  const crestPoints = roofOutline.filter((point) => Math.abs(point.z - maxZ) <= 1e-6);
   765→  const averageX = crestPoints.reduce((sum, point) => sum + point.x, 0) / Math.max(crestPoints.length, 1);
   766→
   767→  return {
   768→    x: averageX,
   769→    z: maxZ,
   770→  };
   771→}

   944→function buildTrussDetailWebTags(lineSegments, instanceGeometry) {
   945→  const spanMidpoint = (instanceGeometry?.metrics?.span || 0) / 2;
   946→  let webIndex = 0;
   947→
   948→  return lineSegments
   949→    .filter((segment) => segment.memberType === 'web' && segment.start && segment.end)
   950→    .map((segment) => {
   951→      webIndex += 1;
   952→
   953→      const start = createDetailPoint(segment.start);
   954→      const end = createDetailPoint(segment.end);
   955→      const dx = end.x - start.x;
   956→      const dy = end.y - start.y;
   957→      const length = Math.hypot(dx, dy);
   958→      if (length <= 1) return null;
   959→
   960→      const midpoint = {
   961→        x: (start.x + end.x) / 2,
   962→        y: (start.y + end.y) / 2,
   963→      };
   964→      let offset = { x: 0, y: 0 };
   965→
   966→      if (Math.abs(dx) <= 1e-6) {
   967→        offset = {
   968→          x: midpoint.x < spanMidpoint ? -DETAIL_WEB_TAG_OFFSET : DETAIL_WEB_TAG_OFFSET,
   969→          y: 0,
   970→        };
   971→      } else {
   972→        const normal = { x: -dy / length, y: dx / length };
   973→        const tangent = { x: dx / length, y: dy / length };
   974→        const outwardNormal = selectOutwardNormal(normal, midpoint, spanMidpoint);
   975→        const tangentShift = (webIndex % 2 === 0 ? 1 : -1) * DETAIL_WEB_TAG_TANGENT_SHIFT;
   976→        offset = {
   977→          x: (outwardNormal.x * DETAIL_WEB_TAG_OFFSET) + (tangent.x * tangentShift),
   978→          y: (outwardNormal.y * DETAIL_WEB_TAG_OFFSET) + (tangent.y * tangentShift),
   979→        };
   980→      }
   981→
   982→      return createDetailTag(
   983→        `${instanceGeometry.instance.id}_detail_web_${webIndex}`,
   984→        [`W${webIndex} ${formatMeasurement(length)}`],
   985→        {
   986→          x: midpoint.x + offset.x,
   987→          y: midpoint.y + offset.y,
   988→        },
   989→        {
   990→          angle: 0,
   991→          sourceType: 'truss_web',
   992→          sourceId: segment.id,
   993→          measurementValue: length,
   994→          leaderLine: {
   995→            start: midpoint,
   996→            end: {
   997→              x: midpoint.x + (offset.x * 0.72),
   998→              y: midpoint.y + (offset.y * 0.72),
   999→            },
  1000→          },
  1001→        }
  1002→      );
  1003→    })
