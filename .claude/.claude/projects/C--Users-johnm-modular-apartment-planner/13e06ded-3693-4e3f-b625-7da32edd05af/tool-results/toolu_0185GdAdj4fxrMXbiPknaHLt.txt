    36→function useUnits() {
    37→  const [unit, setUnit] = useState('mm');
    38→  const isMm = unit === 'mm';
    39→  return {
    40→    unit,
    41→    setUnit,
    42→    suffix: isMm ? 'mm' : 'm',
    43→    toDisplay: (mm) => isMm ? +mm.toFixed(1) : +(mm / 1000).toFixed(4),
    44→    fromDisplay: (v) => isMm ? v : v * 1000,
    45→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    46→  };
    47→}

    49→function WallDrawingInput({ start, dispatch, editorDispatch, activeFloorId, u, selectNewWall }) {
    50→  const [nsDir, setNsDir] = useState('N');
    51→  const [ewDir, setEwDir] = useState('E');
    52→  const [degrees, setDegrees] = useState('');
    53→  const [minutes, setMinutes] = useState('');
    54→  const [lengthVal, setLengthVal] = useState('');
    55→  const lengthRef = useRef(null);
    56→
    57→  const handleConfirm = () => {
    58→    const deg = parseInt(degrees, 10) || 0;
    59→    const min = parseInt(minutes, 10) || 0;
    60→    const len = u.fromDisplay(parseFloat(lengthVal) || 0);
    61→
    62→    if (len < MIN_WALL_LENGTH) return;
    63→    if (deg < 0 || deg > 90 || min < 0 || min > 59) return;
    64→
    65→    const angle = surveyorBearingToSvgAngle(nsDir, deg, min, ewDir);
    66→    const endPt = endpointFromBearing(start, len, angle);
    67→    const wall = createWall(start, endPt);
    68→
    69→    dispatch({ type: 'WALL_ADD', floorId: activeFloorId, wall });
    70→    if (selectNewWall) {
    71→      editorDispatch({ type: 'SELECT_OBJECT', id: wall.id, objectType: 'wall' });
    72→    } else {
    73→      editorDispatch({
    74→        type: 'UPDATE_TOOL_STATE',
    75→        payload: { start: endPt, startAttachment: null, preview: null },
    76→      });
    77→    }
    78→
    79→    setDegrees('');
    80→    setMinutes('');
    81→    setLengthVal('');
    82→    setTimeout(() => lengthRef.current?.focus(), 0);
    83→  };
    84→
    85→  const handleKeyDown = (e) => {
    86→    if (e.key === 'Enter') {
    87→      e.preventDefault();
    88→      handleConfirm();
    89→    }
    90→  };
    91→
    92→  return (
    93→    <div>
    94→      {!selectNewWall && (
    95→        <>
    96→          <div className={styles.title}>Drawing Wall</div>
    97→          <div style={{ fontSize: 12, color: 'var(--color-text-secondary)', marginBottom: 'var(--space-sm)' }}>
    98→            Start: ({u.toDisplay(start.x)}, {u.toDisplay(start.y)}) {u.suffix}
    99→          </div>
   100→        </>
   101→      )}
   102→
   103→      <div className={styles.subtitle}>Bearing</div>
   104→      <div className={styles.bearingRow}>
   105→        <select className={styles.bearingSelect} value={nsDir} onChange={(e) => setNsDir(e.target.value)}>
   106→          <option value="N">N</option>
   107→          <option value="S">S</option>
   108→        </select>
   109→        <input
   110→          className={styles.bearingInput}
   111→          type="number"
   112→          min={0}
   113→          max={90}
   114→          placeholder="0"
   115→          value={degrees}
   116→          onChange={(e) => setDegrees(e.target.value)}
   117→          onKeyDown={handleKeyDown}
   118→        />
   119→        <span className={styles.bearingSymbol}>°</span>
   120→        <input
   121→          className={styles.bearingInput}
   122→          type="number"
   123→          min={0}
   124→          max={59}
   125→          placeholder="0"
   126→          value={minutes}
   127→          onChange={(e) => setMinutes(e.target.value)}
   128→          onKeyDown={handleKeyDown}
   129→        />
   130→        <span className={styles.bearingSymbol}>′</span>
   131→        <select className={styles.bearingSelect} value={ewDir} onChange={(e) => setEwDir(e.target.value)}>
   132→          <option value="E">E</option>
   133→          <option value="W">W</option>
   134→        </select>
   135→      </div>
   136→
   137→      <div className={styles.subtitle}>Length</div>
   138→      <input
   139→        ref={lengthRef}
   140→        className={styles.bearingInput}
   141→        style={{ width: '100%', textAlign: 'left', padding: '0 8px', marginBottom: 'var(--space-sm)' }}
   142→        type="number"
   143→        min={0}
   144→        step={u.step(100)}
   145→        placeholder={`0 ${u.suffix}`}
   146→        value={lengthVal}
   147→        onChange={(e) => setLengthVal(e.target.value)}
   148→        onKeyDown={handleKeyDown}
   149→      />
   150→
   151→      <button className={styles.confirmBtn} onClick={handleConfirm}>
   152→        Create Wall (Enter)
   153→      </button>
   154→    </div>
   155→  );
   156→}

   158→function SheetExportMenu({ sheet, editorDispatch }) {
   159→  const [open, setOpen] = useState(false);
   160→  const rootRef = useRef(null);
   161→  const pngOptions = [
   162→    { label: 'PNG 150 DPI', dpi: 150 },
   163→    { label: 'PNG 300 DPI', dpi: 300 },
   164→    { label: 'PNG 600 DPI', dpi: 600 },
   165→  ];
   166→
   167→  useEffect(() => {
   168→    if (!open) return undefined;
   169→
   170→    const handleMouseDown = (event) => {
   171→      if (rootRef.current && !rootRef.current.contains(event.target)) {
   172→        setOpen(false);
   173→      }
   174→    };
   175→
   176→    document.addEventListener('mousedown', handleMouseDown);
   177→    return () => document.removeEventListener('mousedown', handleMouseDown);
   178→  }, [open]);
   179→
   180→  const handleExport = async (kind, options = {}) => {
   181→    if (!sheet) return;
   182→    try {
   183→      if (kind === 'pdf') {
   184→        await exportActiveSheetAsPdf(
   185→          sheet.title || sheet.drawingName || 'sheet',
   186→          sheet.paperSize
   187→        );
   188→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
   189→      } else {
   190→        await exportActiveSheetAsPng(
   191→          sheet.title || sheet.drawingName || 'sheet',
   192→          options
   193→        );
   194→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
   195→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
   196→      }
   197→    } catch (error) {
   198→      editorDispatch({
   199→        type: 'SET_STATUS_MESSAGE',
   200→        message: error.message || `${kind.toUpperCase()} export failed.`,
   201→      });
   202→    } finally {
   203→      setOpen(false);
   204→    }
   205→  };
   206→
   207→  return (
   208→    <div className={styles.exportMenu} ref={rootRef}>
   209→      <button
   210→        className={styles.exportMenuTrigger}
   211→        type="button"
   212→        onClick={() => setOpen((value) => !value)}
   213→      >
   214→        Export sheet
   215→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
   216→      </button>
   217→      {open && (
   218→        <div className={styles.exportMenuList}>
   219→          <button
   220→            className={styles.exportMenuItem}
   221→            type="button"
   222→            onClick={() => handleExport('pdf')}
   223→          >
   224→            PDF
   225→          </button>
   226→          {pngOptions.map((option) => (
   227→            <button
   228→              key={option.dpi}
   229→              className={styles.exportMenuItem}
   230→              type="button"
   231→              onClick={() => handleExport('png', { dpi: option.dpi })}
   232→            >
   233→              {option.label}
   234→            </button>
   235→          ))}
   236→        </div>
   237→      )}
   238→    </div>
   239→  );
   240→}

   242→function WallProperties({ wall, dispatch, editorDispatch, floorId, u }) {
   243→  const len = wallLength(wall);
   244→
   245→  const updateWall = (updates) => {
   246→    dispatch({ type: 'WALL_UPDATE', floorId, wall: { id: wall.id, ...updates } });
   247→  };
   248→
   249→  const updateWallLength = (value) => {
   250→    const resizedWall = resizeWallFromStart(wall, u.fromDisplay(value), MIN_WALL_LENGTH);
   251→    updateWall({ end: resizedWall.end });
   252→  };
   253→
   254→  return (
   255→    <div>
   256→      <div className={styles.title}>Wall</div>
   257→      <div className={styles.subtitle}>Start Point</div>
   258→      <InputField
   259→        label="X" type="number" suffix={u.suffix}
   260→        value={u.toDisplay(wall.start.x)}
   261→        onChange={(v) => updateWall({ start: { ...wall.start, x: u.fromDisplay(v) } })}
   262→      />
   263→      <InputField
   264→        label="Y" type="number" suffix={u.suffix}
   265→        value={u.toDisplay(wall.start.y)}
   266→        onChange={(v) => updateWall({ start: { ...wall.start, y: u.fromDisplay(v) } })}
   267→      />
   268→      <div className={styles.subtitle}>End Point</div>
   269→      <InputField
   270→        label="X" type="number" suffix={u.suffix}
   271→        value={u.toDisplay(wall.end.x)}
   272→        onChange={(v) => updateWall({ end: { ...wall.end, x: u.fromDisplay(v) } })}
   273→      />
   274→      <InputField
   275→        label="Y" type="number" suffix={u.suffix}
   276→        value={u.toDisplay(wall.end.y)}
   277→        onChange={(v) => updateWall({ end: { ...wall.end, y: u.fromDisplay(v) } })}
   278→      />
   279→      <div className={styles.subtitle}>Properties</div>
   280→      <InputField
   281→        label="Thickness" type="number" suffix={u.suffix} step={u.step(10)}
   282→        value={u.toDisplay(wall.thickness)}
   283→        onChange={(v) => updateWall({ thickness: Math.max(50, u.fromDisplay(v)) })}
   284→      />
   285→      <InputField
   286→        label="Height" type="number" suffix={u.suffix} step={u.step(100)}
   287→        value={u.toDisplay(wall.height)}
   288→        onChange={(v) => updateWall({ height: Math.max(100, u.fromDisplay(v)) })}
   289→      />
   290→      <InputField
   291→        label="Length" type="number" suffix={u.suffix} step={u.step(100)}
   292→        value={u.toDisplay(len)}
   293→        onChange={updateWallLength}
   294→      />
   295→      <div className={styles.subtitle}>Continue from End</div>
   296→      <WallDrawingInput
   297→        start={wall.end}
   298→        dispatch={dispatch}
   299→        editorDispatch={editorDispatch}
   300→        activeFloorId={floorId}
   301→        u={u}
   302→        selectNewWall
   303→      />
   304→    </div>
   305→  );
   306→}
