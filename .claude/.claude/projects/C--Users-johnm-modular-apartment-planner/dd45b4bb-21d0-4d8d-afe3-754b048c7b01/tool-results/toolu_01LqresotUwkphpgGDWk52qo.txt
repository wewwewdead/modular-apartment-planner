    24→function useUnits() {
    25→  const [unit, setUnit] = useState('mm');
    26→  const isMm = unit === 'mm';
    27→  return {
    28→    unit,
    29→    setUnit,
    30→    suffix: isMm ? 'mm' : 'm',
    31→    toDisplay: (mm) => isMm ? Math.round(mm) : +(mm / 1000).toFixed(3),
    32→    fromDisplay: (v) => isMm ? v : v * 1000,
    33→    step: (mmStep) => isMm ? mmStep : mmStep / 1000,
    34→  };
    35→}

    37→function SheetExportMenu({ sheet, editorDispatch }) {
    38→  const [open, setOpen] = useState(false);
    39→  const rootRef = useRef(null);
    40→  const pngOptions = [
    41→    { label: 'PNG 150 DPI', dpi: 150 },
    42→    { label: 'PNG 300 DPI', dpi: 300 },
    43→    { label: 'PNG 600 DPI', dpi: 600 },
    44→  ];
    45→
    46→  useEffect(() => {
    47→    if (!open) return undefined;
    48→
    49→    const handleMouseDown = (event) => {
    50→      if (rootRef.current && !rootRef.current.contains(event.target)) {
    51→        setOpen(false);
    52→      }
    53→    };
    54→
    55→    document.addEventListener('mousedown', handleMouseDown);
    56→    return () => document.removeEventListener('mousedown', handleMouseDown);
    57→  }, [open]);
    58→
    59→  const handleExport = async (kind, options = {}) => {
    60→    if (!sheet) return;
    61→    try {
    62→      if (kind === 'pdf') {
    63→        await exportActiveSheetAsPdf(
    64→          sheet.title || sheet.drawingName || 'sheet',
    65→          sheet.paperSize
    66→        );
    67→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: 'Exported sheet as PDF.' });
    68→      } else {
    69→        await exportActiveSheetAsPng(
    70→          sheet.title || sheet.drawingName || 'sheet',
    71→          options
    72→        );
    73→        const dpiLabel = options.dpi ? ` (${options.dpi} DPI)` : '';
    74→        editorDispatch({ type: 'SET_STATUS_MESSAGE', message: `Exported sheet as PNG${dpiLabel}.` });
    75→      }
    76→    } catch (error) {
    77→      editorDispatch({
    78→        type: 'SET_STATUS_MESSAGE',
    79→        message: error.message || `${kind.toUpperCase()} export failed.`,
    80→      });
    81→    } finally {
    82→      setOpen(false);
    83→    }
    84→  };
    85→
    86→  return (
    87→    <div className={styles.exportMenu} ref={rootRef}>
    88→      <button
    89→        className={styles.exportMenuTrigger}
    90→        type="button"
    91→        onClick={() => setOpen((value) => !value)}
    92→      >
    93→        Save sheet
    94→        <span className={styles.exportMenuChevron}>{open ? '▴' : '▾'}</span>
    95→      </button>
    96→      {open && (
    97→        <div className={styles.exportMenuList}>
    98→          <button
    99→            className={styles.exportMenuItem}
   100→            type="button"
   101→            onClick={() => handleExport('pdf')}
   102→          >
   103→            PDF
   104→          </button>
   105→          {pngOptions.map((option) => (
   106→            <button
   107→              key={option.dpi}
   108→              className={styles.exportMenuItem}
   109→              type="button"
   110→              onClick={() => handleExport('png', { dpi: option.dpi })}
   111→            >
   112→              {option.label}
   113→            </button>
   114→          ))}
   115→        </div>
   116→      )}
   117→    </div>
   118→  );
   119→}

   961→      dispatch({ type: 'WINDOW_DELETE', floorId: activeFloorId, windowId: selectedId });
   962→    } else if (selectedType === 'column') {
   963→      dispatch({ type: 'COLUMN_DELETE', floorId: activeFloorId, columnId: selectedId });
   964→    } else if (selectedType === 'room') {
   965→      dispatch({ type: 'ROOM_DELETE', floorId: activeFloorId, roomId: selectedId });
   966→    }
   967→    editorDispatch({ type: 'DESELECT' });
   968→  };
   969→
   970→  let content = workspaceMode === 'sheet' && sheet
   971→    ? <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />
   972→    : <ProjectSummary project={project} floor={floor} dispatch={dispatch} />;
   973→
   974→  if (selectedId && workspaceMode === 'sheet' && sheet) {
   975→    if (selectedType === 'sheet') {
   976→      if (sheet.id === selectedId) {
   977→        content = <SheetProperties sheet={sheet} project={project} dispatch={dispatch} editorDispatch={editorDispatch} activeFloorId={activeFloorId} u={u} />;
   978→      }
   979→    } else if (selectedType === 'sheetViewport') {
   980→      const viewport = (sheet.viewports || []).find((entry) => entry.id === selectedId);
   981→      if (viewport) {
   982→        content = <SheetViewportProperties sheet={sheet} viewport={viewport} project={project} dispatch={dispatch} u={u} />;
   983→      }
   984→    }
   985→  } else if (selectedId && floor) {
   986→    if (selectedType === 'slab') {
   987→      const slab = (floor.slabs || []).find(s => s.id === selectedId) || null;
   988→      if (slab) {
   989→        content = <SlabProperties slab={slab} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
   990→      }
   991→    } else if (selectedType === 'wall') {
   992→      const wall = floor.walls.find(w => w.id === selectedId);
   993→      if (wall) {
   994→        content = <WallProperties wall={wall} dispatch={dispatch} floorId={activeFloorId} u={u} />;
   995→      }
   996→    } else if (selectedType === 'beam') {
   997→      const beam = (floor.beams || []).find(b => b.id === selectedId);
   998→      if (beam) {
   999→        content = <BeamProperties beam={beam} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1000→      }
  1001→    } else if (selectedType === 'sectionCut') {
  1002→      const sectionCut = (floor.sectionCuts || []).find(s => s.id === selectedId) || null;
  1003→      if (sectionCut) {
  1004→        content = <SectionCutProperties sectionCut={sectionCut} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
  1005→      }
  1006→    } else if (selectedType === 'annotation') {
  1007→      const annotation = (floor.annotations || []).find((entry) => entry.id === selectedId);
  1008→      if (annotation) {
  1009→        content = <AnnotationProperties annotation={annotation} floor={floor} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1010→      }
  1011→    } else if (selectedType === 'stair') {
  1012→      const stair = (floor.stairs || []).find(entry => entry.id === selectedId);
  1013→      if (stair) {
  1014→        content = <StairProperties stair={stair} project={project} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1015→      }
  1016→    } else if (selectedType === 'door') {
  1017→      const door = floor.doors.find(d => d.id === selectedId);
  1018→      const wall = door ? floor.walls.find(w => w.id === door.wallId) : null;
  1019→      if (door) {
  1020→        content = <DoorProperties door={door} wall={wall} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
  1021→      }
  1022→    } else if (selectedType === 'window') {
  1023→      const win = floor.windows.find(w => w.id === selectedId);
  1024→      const wall = win ? floor.walls.find(w => w.id === win.wallId) : null;
  1025→      if (win) {
  1026→        content = <WindowProperties window={win} wall={wall} dispatch={dispatch} floorId={activeFloorId} u={u} />;
  1027→      }
  1028→    } else if (selectedType === 'column') {
  1029→      const column = (floor.columns || []).find(c => c.id === selectedId);
  1030→      if (column) {
  1031→        content = <ColumnProperties column={column} floor={floor} dispatch={dispatch} floorId={activeFloorId} editorDispatch={editorDispatch} u={u} />;
  1032→      }
  1033→    } else if (selectedType === 'room') {
  1034→      const room = floor.rooms.find(r => r.id === selectedId);
  1035→      if (room) {
  1036→        content = <RoomProperties room={room} dispatch={dispatch} floorId={activeFloorId} />;
  1037→      }
  1038→    }
  1039→  }
  1040→
  1041→  return (
  1042→    <div className={styles.panel}>
  1043→      <div className={styles.unitToggle}>
  1044→        <span>Units:</span>
  1045→        <button
  1046→          className={`${styles.unitBtn} ${u.unit === 'mm' ? styles.unitBtnActive : ''}`}
  1047→          onClick={() => u.setUnit('mm')}
  1048→        >mm</button>
  1049→        <button
  1050→          className={`${styles.unitBtn} ${u.unit === 'm' ? styles.unitBtnActive : ''}`}
  1051→          onClick={() => u.setUnit('m')}
  1052→        >m</button>
  1053→      </div>
  1054→      {workspaceMode === 'sheet' && sheet && (
  1055→        <SheetExportMenu sheet={sheet} editorDispatch={editorDispatch} />
  1056→      )}
  1057→      {content}
  1058→      {selectedId && (
  1059→        <button className={styles.deleteBtn} onClick={handleDelete}>
  1060→          Delete {selectedType === 'sheetViewport' ? 'viewport' : selectedType}
  1061→        </button>
  1062→      )}
  1063→    </div>
  1064→  );
  1065→}
