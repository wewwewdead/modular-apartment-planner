    91→const constrainViewport = (candidate, stageWidth, stageHeight) => {
    92→    const scale = clamp(candidate?.scale || 1, MIN_VIEWPORT_SCALE, MAX_VIEWPORT_SCALE);
    93→    const scaledWidth = stageWidth * scale;
    94→    const scaledHeight = stageHeight * scale;
    95→    let minX = Math.min(0, stageWidth - scaledWidth);
    96→    let maxX = 0;
    97→    let minY = Math.min(0, stageHeight - scaledHeight);
    98→    let maxY = 0;
    99→
   100→    if(scaledWidth <= stageWidth){
   101→        const centeredX = (stageWidth - scaledWidth) / 2;
   102→        minX = centeredX;
   103→        maxX = centeredX;
   104→    }
   105→    if(scaledHeight <= stageHeight){
   106→        const centeredY = (stageHeight - scaledHeight) / 2;
   107→        minY = centeredY;
   108→        maxY = centeredY;
   109→    }
   110→
   111→    return {
   112→        scale,
   113→        x: clamp(candidate?.x || 0, minX, maxX),
   114→        y: clamp(candidate?.y || 0, minY, maxY)
   115→    };

   118→const getTouchGeometry = (nativeEvent, stage) => {
   119→    const touches = nativeEvent?.touches;
   120→    if(!stage || !touches || touches.length < 2){
   121→        return null;
   122→    }
   123→
   124→    const bounds = stage.container()?.getBoundingClientRect();
   125→    if(!bounds){
   126→        return null;
   127→    }
   128→
   129→    const first = touches[0];
   130→    const second = touches[1];
   131→    const firstPoint = {x: first.clientX - bounds.left, y: first.clientY - bounds.top};
   132→    const secondPoint = {x: second.clientX - bounds.left, y: second.clientY - bounds.top};
   133→
   134→    return {
   135→        center: {x: (firstPoint.x + secondPoint.x) / 2, y: (firstPoint.y + secondPoint.y) / 2},
   136→        distance: Math.hypot(secondPoint.x - firstPoint.x, secondPoint.y - firstPoint.y)
   137→    };

   142→const CanvasMinimap = ({viewport, stageWidth, stageHeight, viewportWidth, viewportHeight, fitScale, isDark, updateViewport}) => {
   143→    const containerRef = useRef(null);
   144→    const dragRef = useRef(null);
   145→
   146→    const miniH = Math.round(MINIMAP_WIDTH * stageHeight / stageWidth);
   147→    const rectW = MINIMAP_WIDTH / viewport.scale;
   148→    const rectH = miniH / viewport.scale;
   149→    const rectX = -viewport.x * MINIMAP_WIDTH / (viewportWidth * viewport.scale);
   150→    const rectY = -viewport.y * miniH / (viewportHeight * viewport.scale);
   151→
   152→    const minimapToViewport = useCallback((mx, my) => {
   153→        const cx = mx * stageWidth / MINIMAP_WIDTH;
   154→        const cy = my * stageHeight / miniH;
   155→        return {
   156→            scale: viewport.scale,
   157→            x: viewportWidth / 2 - cx * fitScale * viewport.scale,
   158→            y: viewportHeight / 2 - cy * fitScale * viewport.scale
   159→        };
   160→    }, [viewport.scale, stageWidth, stageHeight, miniH, viewportWidth, viewportHeight, fitScale]);
   161→
   162→    const getClientPos = (e) => {
   163→        const touch = e.touches?.[0] || e.changedTouches?.[0];
   164→        return touch ? {x: touch.clientX, y: touch.clientY} : {x: e.clientX, y: e.clientY};
   165→    };
   166→
   167→    const handleBackgroundClick = (e) => {
   168→        if(dragRef.current?.didDrag) return;
   169→        const bounds = containerRef.current?.getBoundingClientRect();
   170→        if(!bounds) return;
   171→        const pos = getClientPos(e);
   172→        const mx = pos.x - bounds.left;
   173→        const my = pos.y - bounds.top;
   174→        updateViewport(minimapToViewport(mx, my));
   175→    };
   176→
   177→    const handleRectPointerDown = (e) => {
   178→        e.stopPropagation();
   179→        if(e.cancelable) e.preventDefault();
   180→        const startPos = getClientPos(e);
   181→        const startRectX = rectX;
   182→        const startRectY = rectY;
   183→        const isTouch = e.type === "touchstart";
   184→        dragRef.current = {didDrag: false};
   185→
   186→        const onMove = (moveEvent) => {
   187→            if(moveEvent.cancelable) moveEvent.preventDefault();
   188→            dragRef.current.didDrag = true;
   189→            const pos = getClientPos(moveEvent);
   190→            const dx = pos.x - startPos.x;
   191→            const dy = pos.y - startPos.y;
   192→            const newRectX = startRectX + dx;
   193→            const newRectY = startRectY + dy;
   194→            updateViewport({
   195→                scale: viewport.scale,
   196→                x: -newRectX * viewportWidth * viewport.scale / MINIMAP_WIDTH,
   197→                y: -newRectY * viewportHeight * viewport.scale / miniH
   198→            });
   199→        };
   200→
   201→        const onUp = () => {
   202→            document.removeEventListener(isTouch ? "touchmove" : "mousemove", onMove);
   203→            document.removeEventListener(isTouch ? "touchend" : "mouseup", onUp);
   204→            setTimeout(() => { dragRef.current = null; }, 0);
   205→        };
   206→
   207→        document.addEventListener(isTouch ? "touchmove" : "mousemove", onMove, {passive: false});
   208→        document.addEventListener(isTouch ? "touchend" : "mouseup", onUp);
   209→    };
   210→
   211→    return (
   212→        <div
   213→            ref={containerRef}
   214→            className={`${styles.minimapContainer} ${isDark ? styles.minimapDark : ""}`}
   215→            style={{width: MINIMAP_WIDTH, height: miniH}}
   216→            onClick={handleBackgroundClick}
   217→            onTouchEnd={(e) => { e.stopPropagation(); handleBackgroundClick(e); }}
   218→        >
   219→            <div
   220→                className={styles.minimapViewport}
   221→                style={{
   222→                    left: clamp(rectX, 0, MINIMAP_WIDTH - Math.max(rectW, 4)),
   223→                    top: clamp(rectY, 0, miniH - Math.max(rectH, 4)),
   224→                    width: Math.max(rectW, 4),
   225→                    height: Math.max(rectH, 4)
   226→                }}
   227→                onMouseDown={handleRectPointerDown}
   228→                onTouchStart={handleRectPointerDown}
   229→            />
   230→        </div>
   231→    );
   232→};
