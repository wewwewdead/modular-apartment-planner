   219→   219→    }

   221→   221→    const mergedById = new Map();

   225→   225→            continue;
   226→   226→        }

   584→   584→export const getProfileMediaService = async(userId, limit = PROFILE_MEDIA_LIMIT_DEFAULT, cursor = null) => {
   585→   585→    if(!userId){
   586→   586→        throw {status: 400, error: 'userId is undefined'};
   587→   587→    }
   588→   588→
   589→   589→    const parsedLimit = Number(limit);
   590→   590→    if(Number.isNaN(parsedLimit) || parsedLimit < 1 || parsedLimit > PROFILE_MEDIA_LIMIT_MAX){
   591→   591→        throw {status: 400, error: `limit should be an integer between 1 and ${PROFILE_MEDIA_LIMIT_MAX}`};
   592→   592→    }
   593→   593→
   594→   594→    const cursorState = parseMediaCursor(cursor);
   595→   595→    const offsets = {...cursorState.offsets};
   596→   596→    const exhausted = {...cursorState.exhausted};
   597→   597→    const pending = Array.isArray(cursorState.pending) ? [...cursorState.pending] : [];
   598→   598→    const unavailableBuckets = [];
   599→   599→    const maxIterations = 6;
   600→   600→    let iterationCount = 0;
   601→   601→
   602→   602→    while(
   603→   603→        pending.length < parsedLimit &&
   604→   604→        PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]) &&
   605→   605→        iterationCount < maxIterations
   606→   606→    ){
   607→   607→        iterationCount += 1;
   608→   608→        const activeBuckets = PROFILE_MEDIA_BUCKETS.filter((bucket) => !exhausted[bucket]);
   609→   609→        const chunkResults = await Promise.all(
   610→   610→            activeBuckets.map((bucket) => listMediaChunkForBucket(bucket, userId, offsets[bucket] || 0, parsedLimit))
   611→   611→        );
   612→   612→
   613→   613→        chunkResults.forEach((result) => {
   614→   614→            offsets[result.bucket] = result.nextOffset;
   615→   615→            exhausted[result.bucket] = result.exhausted;
   616→   616→
   617→   617→            if(result.error){
   618→   618→                console.error(`supabase storage list error on ${result.bucket}:`, result.error);
   619→   619→                unavailableBuckets.push(result.bucket);
   620→   620→                return;
   621→   621→            }
   622→   622→
   623→   623→            pending.push(...result.items);
   624→   624→        });
   625→   625→
   626→   626→        if(chunkResults.every((result) => result.items.length === 0)){
   627→   627→            break;
   628→   628→        }
   629→   629→    }
   630→   630→
   631→   631→    pending.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
   632→   632→
   633→   633→    const selectedItems = pending.slice(0, parsedLimit);
   634→   634→    const remainingItems = pending.slice(parsedLimit);
   635→   635→    const hydratedSelectedItems = await hydrateMediaItems(selectedItems);
   636→   636→
   637→   637→    // If a URL cannot be created for some selected items, try to fill from remaining items.
   638→   638→    if(hydratedSelectedItems.length < parsedLimit && remainingItems.length > 0){
   639→   639→        const neededItemsCount = parsedLimit - hydratedSelectedItems.length;
   640→   640→        const fallbackCandidates = remainingItems.splice(0, neededItemsCount);
   641→   641→        const hydratedFallbackItems = await hydrateMediaItems(fallbackCandidates);
   642→   642→        hydratedSelectedItems.push(...hydratedFallbackItems);
   643→   643→    }
   644→   644→
   645→   645→    const hasMore = remainingItems.length > 0 || PROFILE_MEDIA_BUCKETS.some((bucket) => !exhausted[bucket]);
   646→   646→    const dedupedUnavailableBuckets = [...new Set(unavailableBuckets)];
   647→   647→
   648→   648→    const nextCursor = hasMore
   649→   649→        ? encodeMediaCursor({
   650→   650→            offsets: offsets,
   651→   651→            exhausted: exhausted,
   652→   652→            pending: remainingItems
   653→   653→        })
   654→   654→        : null;
   655→   655→
   656→   656→    return {
   657→   657→        data: hydratedSelectedItems,
   658→   658→        hasMore: hasMore,
   659→   659→        nextCursor: nextCursor,
   660→   660→        unavailableBuckets: dedupedUnavailableBuckets
   661→   661→    };
   662→   662→}
